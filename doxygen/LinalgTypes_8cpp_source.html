38 Type mlir::linalg::LinalgDialect::parseType(DialectAsmParser &parser) const {
39  // Parse the main keyword for the type.
40  StringRef keyword;
41  if (parser.parseKeyword(&keyword))
42  return Type();
43  MLIRContext *context = getContext();
44 
45  // Handle 'range' types.
46  if (keyword == "range")
47  return RangeType::get(context);
48 
49  parser.emitError(parser.getNameLoc(), "unknown Linalg type: " + keyword);
50  return Type();
51 }
