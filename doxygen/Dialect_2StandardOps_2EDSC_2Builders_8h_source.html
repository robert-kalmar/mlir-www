19 /// Base class for MemRefBoundsCapture and VectorBoundsCapture.
20 class BoundsCapture {
21 public:
22  unsigned rank() const { return lbs.size(); }
23  Value lb(unsigned idx) { return lbs[idx]; }
24  Value ub(unsigned idx) { return ubs[idx]; }
25  int64_t step(unsigned idx) { return steps[idx]; }
26  std::tuple<Value, Value, int64_t> range(unsigned idx) {
27  return std::make_tuple(lbs[idx], ubs[idx], steps[idx]);
28  }
29  void swapRanges(unsigned i, unsigned j) {
30  if (i == j)
31  return;
32  std::swap(lbs[i], lbs[j]);
33  std::swap(ubs[i], ubs[j]);
34  std::swap(steps[i], steps[j]);
35  }
36 
37  ArrayRef<Value> getLbs() { return lbs; }
38  ArrayRef<Value> getUbs() { return ubs; }
39  ArrayRef<int64_t> getSteps() { return steps; }
40 
41 protected:
42  SmallVector<Value, 8> lbs;
43  SmallVector<Value, 8> ubs;
44  SmallVector<int64_t, 8> steps;
45 };

47 /// A MemRefBoundsCapture represents the information required to step through a
48 /// MemRef. It has placeholders for non-contiguous tensors that fit within the
49 /// Fortran subarray model.
50 /// At the moment it can only capture a MemRef with an identity layout map.
51 // TODO: Support MemRefs with layoutMaps.
52 class MemRefBoundsCapture : public BoundsCapture {
53 public:
54  explicit MemRefBoundsCapture(Value v);
55  MemRefBoundsCapture(const MemRefBoundsCapture &) = default;
56  MemRefBoundsCapture &operator=(const MemRefBoundsCapture &) = default;
57 
58  unsigned fastestVarying() const { return rank() - 1; }
59 
60 private:
61  Value base;
62 };
