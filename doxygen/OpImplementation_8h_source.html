214 /// The OpAsmParser has methods for interacting with the asm parser: parsing
215 /// things from it, emitting errors etc. It has an intentionally high-level API
216 /// that is designed to reduce/constrain syntax innovation in individual
217 /// operations.
218 ///
219 /// For example, consider an op like this:
220 ///
221 /// %x = load %p[%1, %2] : memref<...>
222 ///
223 /// The "%x = load" tokens are already parsed and therefore invisible to the
224 /// custom op parser. This can be supported by calling `parseOperandList` to
225 /// parse the %p, then calling `parseOperandList` with a `SquareDelimiter` to
226 /// parse the indices, then calling `parseColonTypeList` to parse the result
227 /// type.
228 ///

511  /// Resolve a list of operands to SSA values, emitting an error on failure, or

514  ParseResult resolveOperands(ArrayRef<OperandType> operands, Type type,

522  /// Resolve a list of operands and a list of operand types to SSA values,
523  /// emitting an error and returning failure, or appending the results
524  /// to the list on success.
525  ParseResult resolveOperands(ArrayRef<OperandType> operands,
526  ArrayRef<Type> types, llvm::SMLoc loc,
527  SmallVectorImpl<Value> &result) {
528  if (operands.size() != types.size())
529  return emitError(loc)
530  << operands.size() << " operands present, but expected "
531  << types.size();
532 
533  for (unsigned i = 0, e = operands.size(); i != e; ++i)
534  if (resolveOperand(operands[i], types[i], result))
535  return failure();
536  return success();
537  }
538  template <typename Operands>
539  ParseResult resolveOperands(Operands &&operands, Type type, llvm::SMLoc loc,
540  SmallVectorImpl<Value> &result) {
541  return resolveOperands(std::forward<Operands>(operands),
542  ArrayRef<Type>(type), loc, result);
543  }
544  template <typename Operands, typename Types>
545  std::enable_if_t<!std::is_convertible<Types, Type>::value, ParseResult>
546  resolveOperands(Operands &&operands, Types &&types, llvm::SMLoc loc,
547  SmallVectorImpl<Value> &result) {
548  size_t operandSize = std::distance(operands.begin(), operands.end());
549  size_t typeSize = std::distance(types.begin(), types.end());
550  if (operandSize != typeSize)
551  return emitError(loc)
552  << operandSize << " operands present, but expected " << typeSize;
553 
554  for (auto it : llvm::zip(operands, types))
555  if (resolveOperand(std::get<0>(it), std::get<1>(it), result))
556  return failure();
557  return success();
558  }
