223 /// Get the number of bits require to store a value of the given shaped type.
224 /// Compute the value recursively since tensors are allowed to have vectors as
225 /// elements.
226 int64_t ShapedType::getSizeInBits() const {
227  assert(hasStaticShape() &&
228  "cannot get the bit size of an aggregate with a dynamic shape");
229 
230  auto elementType = getElementType();
231  if (elementType.isIntOrFloat())
232  return elementType.getIntOrFloatBitWidth() * getNumElements();
233 
234  if (auto complexType = elementType.dyn_cast<ComplexType>()) {
235  elementType = complexType.getElementType();
236  return elementType.getIntOrFloatBitWidth() * getNumElements() * 2;
237  }
238 
239  // Tensors can have vectors and other tensors as elements, other shaped types
240  // cannot.
241  assert(isa<TensorType>() && "unsupported element type");
242  assert((elementType.isa<VectorType, TensorType>()) &&
243  "unsupported tensor element type");
244  return getNumElements() * elementType.cast<ShapedType>().getSizeInBits();
245 }

401 /// Get or create a new MemRefType defined by the arguments. If the resulting
402 /// type would be ill-formed, return nullptr. If the location is provided,
403 /// emit detailed error messages. To emit errors when the location is unknown,
404 /// pass in an instance of UnknownLoc.
405 MemRefType MemRefType::getImpl(ArrayRef<int64_t> shape, Type elementType,
406  ArrayRef<AffineMap> affineMapComposition,
407  unsigned memorySpace,
408  Optional<Location> location) {
409  auto *context = elementType.getContext();
410 
411  if (!BaseMemRefType::isValidElementType(elementType))
412  return emitOptionalError(location, "invalid memref element type"),
413  MemRefType();
414 
415  for (int64_t s : shape) {
416  // Negative sizes are not allowed except for `-1` that means dynamic size.
417  if (s < -1)
418  return emitOptionalError(location, "invalid memref size"), MemRefType();
419  }
420 
421  // Check that the structure of the composition is valid, i.e. that each
422  // subsequent affine map has as many inputs as the previous map has results.
423  // Take the dimensionality of the MemRef for the first map.
424  auto dim = shape.size();
425  unsigned i = 0;
426  for (const auto &affineMap : affineMapComposition) {
427  if (affineMap.getNumDims() != dim) {
428  if (location)
429  emitError(*location)
430  << "memref affine map dimension mismatch between "
431  << (i == 0 ? Twine("memref rank") : "affine map " + Twine(i))
432  << " and affine map" << i + 1 << ": " << dim
433  << " != " << affineMap.getNumDims();
434  return nullptr;
435  }
436 
437  dim = affineMap.getNumResults();
438  ++i;
439  }
440 
441  // Drop identity maps from the composition.
442  // This may lead to the composition becoming empty, which is interpreted as an
443  // implicit identity.
444  SmallVector<AffineMap, 2> cleanedAffineMapComposition;
445  for (const auto &map : affineMapComposition) {
446  if (map.isIdentity())
447  continue;
448  cleanedAffineMapComposition.push_back(map);
449  }
450 
451  return Base::get(context, StandardTypes::MemRef, shape, elementType,
452  cleanedAffineMapComposition, memorySpace);
453 }

492 // Fallback cases for terminal dim/sym/cst that are not part of a binary op (
493 // i.e. single term). Accumulate the AffineExpr into the existing one.
494 static void extractStridesFromTerm(AffineExpr e,
495  AffineExpr multiplicativeFactor,
496  MutableArrayRef<AffineExpr> strides,
497  AffineExpr &offset) {
498  if (auto dim = e.dyn_cast<AffineDimExpr>())
499  strides[dim.getPosition()] =
500  strides[dim.getPosition()] + multiplicativeFactor;
501  else
502  offset = offset + e * multiplicativeFactor;
503 }
504 
505 /// Takes a single AffineExpr `e` and populates the `strides` array with the
506 /// strides expressions for each dim position.
507 /// The convention is that the strides for dimensions d0, .. dn appear in
508 /// order to make indexing intuitive into the result.
509 static LogicalResult extractStrides(AffineExpr e,
510  AffineExpr multiplicativeFactor,
511  MutableArrayRef<AffineExpr> strides,
512  AffineExpr &offset) {
513  auto bin = e.dyn_cast<AffineBinaryOpExpr>();
514  if (!bin) {
515  extractStridesFromTerm(e, multiplicativeFactor, strides, offset);
516  return success();
517  }
518 
519  if (bin.getKind() == AffineExprKind::CeilDiv ||
520  bin.getKind() == AffineExprKind::FloorDiv ||
521  bin.getKind() == AffineExprKind::Mod)
522  return failure();
523 
524  if (bin.getKind() == AffineExprKind::Mul) {
525  auto dim = bin.getLHS().dyn_cast<AffineDimExpr>();
526  if (dim) {
527  strides[dim.getPosition()] =
528  strides[dim.getPosition()] + bin.getRHS() * multiplicativeFactor;
529  return success();
530  }
531  // LHS and RHS may both contain complex expressions of dims. Try one path
532  // and if it fails try the other. This is guaranteed to succeed because
533  // only one path may have a `dim`, otherwise this is not an AffineExpr in
534  // the first place.
535  if (bin.getLHS().isSymbolicOrConstant())
536  return extractStrides(bin.getRHS(), multiplicativeFactor * bin.getLHS(),
537  strides, offset);
538  return extractStrides(bin.getLHS(), multiplicativeFactor * bin.getRHS(),
539  strides, offset);
540  }
541 
542  if (bin.getKind() == AffineExprKind::Add) {
543  auto res1 =
544  extractStrides(bin.getLHS(), multiplicativeFactor, strides, offset);
545  auto res2 =
546  extractStrides(bin.getRHS(), multiplicativeFactor, strides, offset);
547  return success(succeeded(res1) && succeeded(res2));
548  }
549 
550  llvm_unreachable("unexpected binary operation");
551 }

553 LogicalResult mlir::getStridesAndOffset(MemRefType t,
554  SmallVectorImpl<AffineExpr> &strides,
555  AffineExpr &offset) {
556  auto affineMaps = t.getAffineMaps();
557  // For now strides are only computed on a single affine map with a single
558  // result (i.e. the closed subset of linearization maps that are compatible
559  // with striding semantics).
560  // TODO: support more forms on a per-need basis.
561  if (affineMaps.size() > 1)
562  return failure();
563  if (affineMaps.size() == 1 && affineMaps[0].getNumResults() != 1)
564  return failure();
565 
566  auto zero = getAffineConstantExpr(0, t.getContext());
567  auto one = getAffineConstantExpr(1, t.getContext());
568  offset = zero;
569  strides.assign(t.getRank(), zero);
570 
571  AffineMap m;
572  if (!affineMaps.empty()) {
573  m = affineMaps.front();
574  assert(!m.isIdentity() && "unexpected identity map");
575  }
576 
577  // Canonical case for empty map.
578  if (!m) {
579  // 0-D corner case, offset is already 0.
580  if (t.getRank() == 0)
581  return success();
582  auto stridedExpr =
583  makeCanonicalStridedLayoutExpr(t.getShape(), t.getContext());
584  if (succeeded(extractStrides(stridedExpr, one, strides, offset)))
585  return success();
586  assert(false && "unexpected failure: extract strides in canonical layout");
587  }
588 
589  // Non-canonical case requires more work.
590  auto stridedExpr =
591  simplifyAffineExpr(m.getResult(0), m.getNumDims(), m.getNumSymbols());
592  if (failed(extractStrides(stridedExpr, one, strides, offset))) {
593  offset = AffineExpr();
594  strides.clear();
595  return failure();
596  }
597 
598  // Simplify results to allow folding to constants and simple checks.
599  unsigned numDims = m.getNumDims();
600  unsigned numSymbols = m.getNumSymbols();
601  offset = simplifyAffineExpr(offset, numDims, numSymbols);
602  for (auto &stride : strides)
603  stride = simplifyAffineExpr(stride, numDims, numSymbols);
604 
605  /// In practice, a strided memref must be internally non-aliasing. Test
606  /// against 0 as a proxy.
607  /// TODO: static cases can have more advanced checks.
608  /// TODO: dynamic cases would require a way to compare symbolic
609  /// expressions and would probably need an affine set context propagated
610  /// everywhere.
611  if (llvm::any_of(strides, [](AffineExpr e) {
612  return e == getAffineConstantExpr(0, e.getContext());
613  })) {
614  offset = AffineExpr();
615  strides.clear();
616  return failure();
617  }
618 
619  return success();
620 }

622 LogicalResult mlir::getStridesAndOffset(MemRefType t,
623  SmallVectorImpl<int64_t> &strides,
624  int64_t &offset) {
625  AffineExpr offsetExpr;
626  SmallVector<AffineExpr, 4> strideExprs;
627  if (failed(::getStridesAndOffset(t, strideExprs, offsetExpr)))
628  return failure();
629  if (auto cst = offsetExpr.dyn_cast<AffineConstantExpr>())
630  offset = cst.getValue();
631  else
632  offset = ShapedType::kDynamicStrideOrOffset;
633  for (auto e : strideExprs) {
634  if (auto c = e.dyn_cast<AffineConstantExpr>())
635  strides.push_back(c.getValue());
636  else
637  strides.push_back(ShapedType::kDynamicStrideOrOffset);
638  }
639  return success();
640 }

674 AffineMap mlir::makeStridedLinearLayoutMap(ArrayRef<int64_t> strides,
675  int64_t offset,
676  MLIRContext *context) {
677  AffineExpr expr;
678  unsigned nSymbols = 0;
679 
680  // AffineExpr for offset.
681  // Static case.
682  if (offset != MemRefType::getDynamicStrideOrOffset()) {
683  auto cst = getAffineConstantExpr(offset, context);
684  expr = cst;
685  } else {
686  // Dynamic case, new symbol for the offset.
687  auto sym = getAffineSymbolExpr(nSymbols++, context);
688  expr = sym;
689  }
690 
691  // AffineExpr for strides.
692  for (auto en : llvm::enumerate(strides)) {
693  auto dim = en.index();
694  auto stride = en.value();
695  assert(stride != 0 && "Invalid stride specification");
696  auto d = getAffineDimExpr(dim, context);
697  AffineExpr mult;
698  // Static case.
699  if (stride != MemRefType::getDynamicStrideOrOffset())
700  mult = getAffineConstantExpr(stride, context);
701  else
702  // Dynamic case, new symbol for each new stride.
703  mult = getAffineSymbolExpr(nSymbols++, context);
704  expr = expr + d * mult;
705  }
706 
707  return AffineMap::get(strides.size(), nSymbols, expr);
708 }

710 /// Return a version of `t` with identity layout if it can be determined
711 /// statically that the layout is the canonical contiguous strided layout.
712 /// Otherwise pass `t`'s layout into `simplifyAffineMap` and return a copy of
713 /// `t` with simplified layout.
714 /// If `t` has multiple layout maps or a multi-result layout, just return `t`.
715 MemRefType mlir::canonicalizeStridedLayout(MemRefType t) {
716  auto affineMaps = t.getAffineMaps();
717  // Already in canonical form.
718  if (affineMaps.empty())
719  return t;
720 
721  // Can't reduce to canonical identity form, return in canonical form.
722  if (affineMaps.size() > 1 || affineMaps[0].getNumResults() > 1)
723  return t;
724 
725  // If the canonical strided layout for the sizes of `t` is equal to the
726  // simplified layout of `t` we can just return an empty layout. Otherwise,
727  // just simplify the existing layout.
728  AffineExpr expr =
729  makeCanonicalStridedLayoutExpr(t.getShape(), t.getContext());
730  auto m = affineMaps[0];
731  auto simplifiedLayoutExpr =
732  simplifyAffineExpr(m.getResult(0), m.getNumDims(), m.getNumSymbols());
733  if (expr != simplifiedLayoutExpr)
734  return MemRefType::Builder(t).setAffineMaps({AffineMap::get(
735  m.getNumDims(), m.getNumSymbols(), simplifiedLayoutExpr)});
736  return MemRefType::Builder(t).setAffineMaps({});
737 }

739 AffineExpr mlir::makeCanonicalStridedLayoutExpr(ArrayRef<int64_t> sizes,
740  ArrayRef<AffineExpr> exprs,
741  MLIRContext *context) {
742  // Size 0 corner case is useful for canonicalizations.
743  if (llvm::is_contained(sizes, 0))
744  return getAffineConstantExpr(0, context);
745 
746  auto maps = AffineMap::inferFromExprList(exprs);
747  assert(!maps.empty() && "Expected one non-empty map");
748  unsigned numDims = maps[0].getNumDims(), nSymbols = maps[0].getNumSymbols();
749 
750  AffineExpr expr;
751  bool dynamicPoisonBit = false;
752  int64_t runningSize = 1;
753  for (auto en : llvm::zip(llvm::reverse(exprs), llvm::reverse(sizes))) {
754  int64_t size = std::get<1>(en);
755  // Degenerate case, no size =-> no stride
756  if (size == 0)
757  continue;
758  AffineExpr dimExpr = std::get<0>(en);
759  AffineExpr stride = dynamicPoisonBit
760  ? getAffineSymbolExpr(nSymbols++, context)
761  : getAffineConstantExpr(runningSize, context);
762  expr = expr ? expr + dimExpr * stride : dimExpr * stride;
763  if (size > 0)
764  runningSize *= size;
765  else
766  dynamicPoisonBit = true;
767  }
768  return simplifyAffineExpr(expr, numDims, nSymbols);
769 }
