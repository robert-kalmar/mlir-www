94 /// Given the 'indices' of an load/store operation where the memref is a result
95 /// of a subview op, returns the indices w.r.t to the source memref of the
96 /// subview op. For example
97 ///
98 /// %0 = ... : memref<12x42xf32>
99 /// %1 = subview %0[%arg0, %arg1][][%stride1, %stride2] : memref<12x42xf32> to
100 /// memref<4x4xf32, offset=?, strides=[?, ?]>
101 /// %2 = load %1[%i1, %i2] : memref<4x4xf32, offset=?, strides=[?, ?]>
102 ///
103 /// could be folded into
104 ///
105 /// %2 = load %0[%arg0 + %i1 * %stride1][%arg1 + %i2 * %stride2] :
106 /// memref<12x42xf32>
107 static LogicalResult
108 resolveSourceIndices(Location loc, PatternRewriter &rewriter,
109  SubViewOp subViewOp, ValueRange indices,
110  SmallVectorImpl<Value> &sourceIndices) {
111  // TODO: Aborting when the offsets are static. There might be a way to fold
112  // the subview op with load even if the offsets have been canonicalized
113  // away.
114  SmallVector<Value, 4> opOffsets = subViewOp.getOrCreateOffsets(rewriter, loc);
115  SmallVector<Value, 4> opStrides = subViewOp.getOrCreateStrides(rewriter, loc);
116  assert(opOffsets.size() == indices.size() &&
117  "expected as many indices as rank of subview op result type");
118  assert(opStrides.size() == indices.size() &&
119  "expected as many indices as rank of subview op result type");
120 
121  // New indices for the load are the current indices * subview_stride +
122  // subview_offset.
123  sourceIndices.resize(indices.size());
124  for (auto index : llvm::enumerate(indices)) {
125  auto offset = opOffsets[index.index()];
126  auto stride = opStrides[index.index()];
127  auto mul = rewriter.create<MulIOp>(loc, index.value(), stride);
128  sourceIndices[index.index()] =
129  rewriter.create<AddIOp>(loc, offset, mul).getResult();
130  }
131  return success();
132 }

138 template <typename OpTy>
139 LogicalResult
140 LoadOpOfSubViewFolder<OpTy>::matchAndRewrite(OpTy loadOp,
141  PatternRewriter &rewriter) const {
142  auto subViewOp = loadOp.memref().template getDefiningOp<SubViewOp>();
143  if (!subViewOp) {
144  return failure();
145  }
146  SmallVector<Value, 4> sourceIndices;
147  if (failed(resolveSourceIndices(loadOp.getLoc(), rewriter, subViewOp,
148  loadOp.indices(), sourceIndices)))
149  return failure();
150 
151  replaceOp(loadOp, subViewOp, sourceIndices, rewriter);
152  return success();
153 }
