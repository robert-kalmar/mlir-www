33 /// Perform the actions on the input file indicated by the command line flags
34 /// within the specified context.
35 ///
36 /// This typically parses the main source file, runs zero or more optimization
37 /// passes, then prints the output.
38 ///
39 static LogicalResult performActions(raw_ostream &os, bool verifyDiagnostics,
40  bool verifyPasses, SourceMgr &sourceMgr,
41  MLIRContext *context,
42  const PassPipelineCLParser &passPipeline) {
43  // Disable multi-threading when parsing the input file. This removes the
44  // unnecessary/costly context synchronization when parsing.
45  bool wasThreadingEnabled = context->isMultithreadingEnabled();
46  context->disableMultithreading();
47 
48  // Parse the input file and reset the context threading state.
49  OwningModuleRef module(parseSourceFile(sourceMgr, context));
50  context->enableMultithreading(wasThreadingEnabled);
51  if (!module)
52  return failure();
53 
54  // Apply any pass manager command line options.
55  PassManager pm(context, verifyPasses);
56  applyPassManagerCLOptions(pm);
57 
58  // Build the provided pipeline.
59  if (failed(passPipeline.addToPipeline(pm)))
60  return failure();
61 
62  // Run the pipeline.
63  if (failed(pm.run(*module)))
64  return failure();
65 
66  // Print the output.
67  module->print(os);
68  os << '\n';
69  return success();
70 }

72 /// Parses the memory buffer. If successfully, run a series of passes against
73 /// it and print the result.
74 static LogicalResult processBuffer(raw_ostream &os,
75  std::unique_ptr<MemoryBuffer> ownedBuffer,
76  bool verifyDiagnostics, bool verifyPasses,
77  bool allowUnregisteredDialects,
78  const PassPipelineCLParser &passPipeline) {
79  // Tell sourceMgr about this buffer, which is what the parser will pick up.
80  SourceMgr sourceMgr;
81  sourceMgr.AddNewSourceBuffer(std::move(ownedBuffer), SMLoc());
82 
83  // Parse the input file.
84  MLIRContext context;
85  context.allowUnregisteredDialects(allowUnregisteredDialects);
86  context.printOpOnDiagnostic(!verifyDiagnostics);
87 
88  // If we are in verify diagnostics mode then we have a lot of work to do,
89  // otherwise just perform the actions without worrying about it.
90  if (!verifyDiagnostics) {
91  SourceMgrDiagnosticHandler sourceMgrHandler(sourceMgr, &context);
92  return performActions(os, verifyDiagnostics, verifyPasses, sourceMgr,
93  &context, passPipeline);
94  }
95 
96  SourceMgrDiagnosticVerifierHandler sourceMgrHandler(sourceMgr, &context);
97 
98  // Do any processing requested by command line flags. We don't care whether
99  // these actions succeed or fail, we only care what diagnostics they produce
100  // and whether they match our expectations.
101  performActions(os, verifyDiagnostics, verifyPasses, sourceMgr, &context,
102  passPipeline);
103 
104  // Verify the diagnostic handler to make sure that each of the diagnostics
105  // matched.
106  return sourceMgrHandler.verify();
107 }
