87 /// Constructs a permutation map of invariant memref indices to vector
88 /// dimension.
89 ///
90 /// If no index is found to be invariant, 0 is added to the permutation_map and
91 /// corresponds to a vector broadcast along that dimension.
92 ///
93 /// The implementation uses the knowledge of the mapping of loops to
94 /// vector dimension. `loopToVectorDim` carries this information as a map with:
95 /// - keys representing "vectorized enclosing loops";
96 /// - values representing the corresponding vector dimension.
97 /// Note that loopToVectorDim is a whole function map from which only enclosing
98 /// loop information is extracted.
99 ///
100 /// Prerequisites: `opInst` is a vectorizable load or store operation (i.e. at
101 /// most one invariant index along each AffineForOp of `loopToVectorDim`).
102 ///
103 /// Example 1:
104 /// The following MLIR snippet:
105 ///
106 /// ```mlir
107 /// affine.for %i3 = 0 to %0 {
108 /// affine.for %i4 = 0 to %1 {
109 /// affine.for %i5 = 0 to %2 {
110 /// %a5 = load %arg0[%i4, %i5, %i3] : memref<?x?x?xf32>
111 /// }}}
112 /// ```
113 ///
114 /// may vectorize with {permutation_map: (d0, d1, d2) -> (d2, d1)} into:
115 ///
116 /// ```mlir
117 /// affine.for %i3 = 0 to %0 step 32 {
118 /// affine.for %i4 = 0 to %1 {
119 /// affine.for %i5 = 0 to %2 step 256 {
120 /// %4 = vector.transfer_read %arg0, %i4, %i5, %i3
121 /// {permutation_map: (d0, d1, d2) -> (d2, d1)} :
122 /// (memref<?x?x?xf32>, index, index) -> vector<32x256xf32>
123 /// }}}
124 /// ```
125 ///
126 /// Meaning that vector.transfer_read will be responsible for reading the slice:
127 /// `%arg0[%i4, %i5:%15+256, %i3:%i3+32]` into vector<32x256xf32>.
128 ///
129 /// Example 2:
130 /// The following MLIR snippet:
131 ///
132 /// ```mlir
133 /// %cst0 = constant 0 : index
134 /// affine.for %i0 = 0 to %0 {
135 /// %a0 = load %arg0[%cst0, %cst0] : memref<?x?xf32>
136 /// }
137 /// ```
138 ///
139 /// may vectorize with {permutation_map: (d0) -> (0)} into:
140 ///
141 /// ```mlir
142 /// affine.for %i0 = 0 to %0 step 128 {
143 /// %3 = vector.transfer_read %arg0, %c0_0, %c0_0
144 /// {permutation_map: (d0, d1) -> (0)} :
145 /// (memref<?x?xf32>, index, index) -> vector<128xf32>
146 /// }
147 /// ````
148 ///
149 /// Meaning that vector.transfer_read will be responsible of reading the slice
150 /// `%arg0[%c0, %c0]` into vector<128xf32> which needs a 1-D vector broadcast.
151 ///
152 AffineMap
153 makePermutationMap(Operation *op, ArrayRef<Value> indices,
154  const DenseMap<Operation *, unsigned> &loopToVectorDim);
