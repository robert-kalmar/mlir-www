195 //----------------------------------------------------------------------------//
196 // Creation and setting of LLVM's identified struct types
197 
198 LLVMType LLVMType::createStructTy(MLIRContext *context,
199  ArrayRef<LLVMType> elements,
200  Optional<StringRef> name, bool isPacked) {
201  assert(name.hasValue() &&
202  "identified structs with no identifier not supported");
203  StringRef stringNameBase = name.getValueOr("");
204  std::string stringName = stringNameBase.str();
205  unsigned counter = 0;
206  do {
207  auto type = LLVMStructType::getIdentified(context, stringName);
208  if (type.isInitialized() || failed(type.setBody(elements, isPacked))) {
209  counter += 1;
210  stringName =
211  (Twine(stringNameBase) + "." + std::to_string(counter)).str();
212  continue;
213  }
214  return type;
215  } while (true);
216 }

218 LLVMType LLVMType::setStructTyBody(LLVMType structType,
219  ArrayRef<LLVMType> elements, bool isPacked) {
220  LogicalResult couldSet =
221  structType.cast<LLVMStructType>().setBody(elements, isPacked);
222  assert(succeeded(couldSet) && "failed to set the body");
223  (void)couldSet;
224  return structType;
225 }
