28 /// Implement a predecessor iterator for blocks. This works by walking the use
29 /// lists of the blocks. The entries on this list are the BlockOperands that
30 /// are embedded into terminator operations. From the operand, we can get the
31 /// terminator that contains it, and its parent block is the predecessor.
32 class PredecessorIterator final
33  : public llvm::mapped_iterator<ValueUseIterator<BlockOperand>,
34  Block *(*)(BlockOperand &)> {
35  static Block *unwrap(BlockOperand &value);
36 
37 public:
38  using reference = Block *;
39 
40  /// Initializes the operand type iterator to the specified operand iterator.
41  PredecessorIterator(ValueUseIterator<BlockOperand> it)
42  : llvm::mapped_iterator<ValueUseIterator<BlockOperand>,
43  Block *(*)(BlockOperand &)>(it, &unwrap) {}
44  explicit PredecessorIterator(BlockOperand *operand)
45  : PredecessorIterator(ValueUseIterator<BlockOperand>(operand)) {}

122 /// Provide support for hashing successor ranges.
123 template <>
124 struct DenseMapInfo<mlir::SuccessorRange> {
125  static mlir::SuccessorRange getEmptyKey() {
126  auto *pointer = llvm::DenseMapInfo<mlir::BlockOperand *>::getEmptyKey();
127  return mlir::SuccessorRange(pointer, 0);
128  }
129  static mlir::SuccessorRange getTombstoneKey() {
130  auto *pointer = llvm::DenseMapInfo<mlir::BlockOperand *>::getTombstoneKey();
131  return mlir::SuccessorRange(pointer, 0);
132  }
133  static unsigned getHashValue(mlir::SuccessorRange value) {
134  return llvm::hash_combine_range(value.begin(), value.end());
135  }
136  static bool isEqual(mlir::SuccessorRange lhs, mlir::SuccessorRange rhs) {
137  if (rhs.getBase() == getEmptyKey().getBase())
138  return lhs.getBase() == getEmptyKey().getBase();
139  if (rhs.getBase() == getTombstoneKey().getBase())
140  return lhs.getBase() == getTombstoneKey().getBase();
141  return lhs == rhs;
142  }
143 };
