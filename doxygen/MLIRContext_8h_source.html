27 /// MLIRContext is the top-level object for a collection of MLIR modules. It
28 /// holds immortal uniqued objects like types, and the tables used to unique
29 /// them.
30 ///
31 /// MLIRContext gets a redundant "MLIR" prefix because otherwise it ends up with
32 /// a very generic name ("Context") and because it is uncommon for clients to
33 /// interact with it.
34 ///
35 class MLIRContext {
36 public:
37  explicit MLIRContext();
38  ~MLIRContext();
39 
40  /// Return information about all registered IR dialects.
41  std::vector<Dialect *> getRegisteredDialects();
42 
43  /// Get a registered IR dialect with the given namespace. If an exact match is
44  /// not found, then return nullptr.
45  Dialect *getRegisteredDialect(StringRef name);
46 
47  /// Get a registered IR dialect for the given derived dialect type. The
48  /// derived type must provide a static 'getDialectNamespace' method.
49  template <typename T> T *getRegisteredDialect() {
50  return static_cast<T *>(getRegisteredDialect(T::getDialectNamespace()));
51  }
52 
53  /// Get (or create) a dialect for the given derived dialect type. The derived
54  /// type must provide a static 'getDialectNamespace' method.
55  template <typename T>
56  T *getOrCreateDialect() {
57  return static_cast<T *>(getOrCreateDialect(
58  T::getDialectNamespace(), TypeID::get<T>(), [this]() {
59  std::unique_ptr<T> dialect(new T(this));
60  dialect->dialectID = TypeID::get<T>();
61  return dialect;
62  }));
63  }
64 
65  /// Return true if we allow to create operation for unregistered dialects.
66  bool allowsUnregisteredDialects();
67 
68  /// Enables creating operations in unregistered dialects.
69  void allowUnregisteredDialects(bool allow = true);
70 
71  /// Return true if multi-threading is enabled by the context.
72  bool isMultithreadingEnabled();
73 
74  /// Set the flag specifying if multi-threading is disabled by the context.
75  void disableMultithreading(bool disable = true);
76  void enableMultithreading(bool enable = true) {
77  disableMultithreading(!enable);
78  }

82  bool shouldPrintOpOnDiagnostic();
83

85  /// emitted via Operation::emit.
86  void printOpOnDiagnostic(bool enable);
87 
88  /// Return true if we should attach the current stacktrace to diagnostics when

90  bool shouldPrintStackTraceOnDiagnostic();
91 
92  /// Set the flag specifying if we should attach the current stacktrace when
93  /// emitting diagnostics.
94  void printStackTraceOnDiagnostic(bool enable);
95 
96  /// Return information about all registered operations. This isn't very
97  /// efficient: typically you should ask the operations about their properties
98  /// directly.
99  std::vector<AbstractOperation *> getRegisteredOperations();
100 
101  /// Return true if this operation name is registered in this context.
102  bool isOperationRegistered(StringRef name);
103 
104  // This is effectively private given that only MLIRContext.cpp can see the
105  // MLIRContextImpl type.
106  MLIRContextImpl &getImpl() { return *impl; }
107 
108  /// Returns the diagnostic engine for this context.
109  DiagnosticEngine &getDiagEngine();
110 
111  /// Returns the storage uniquer used for creating affine constructs.
112  StorageUniquer &getAffineUniquer();
113 
114  /// Returns the storage uniquer used for constructing type storage instances.
115  /// This should not be used directly.
116  StorageUniquer &getTypeUniquer();
117 
118  /// Returns the storage uniquer used for constructing attribute storage
119  /// instances. This should not be used directly.
120  StorageUniquer &getAttributeUniquer();
121 
122 private:
123  const std::unique_ptr<MLIRContextImpl> impl;
124 
125  /// Get a dialect for the provided namespace and TypeID: abort the program if
126  /// a dialect exist for this namespace with different TypeID. Returns a
127  /// pointer to the dialect owned by the context.
128  Dialect *getOrCreateDialect(StringRef dialectNamespace, TypeID dialectID,
129  function_ref<std::unique_ptr<Dialect>()> ctor);
130 
131  MLIRContext(const MLIRContext &) = delete;
132  void operator=(const MLIRContext &) = delete;
133 };
