35 /// Entry point for unrolling declarative pattern rewrites.
36 /// `op` is unrolled to the `targetShape` as follows, for each of its operands:
37 /// 1. the unrolled type `unrolledVectorType` and number of unrolled instances
38 /// `numUnrolledInstances` are computed from the `targetShape`. For now it is
39 /// assumed the unrolling factors divide the vector sizes.
40 /// 2. a fakeFork cast op is inserted that takes the operand and returns
41 /// `numUnrolledInstances` results of type `unrolledVectorType`.
42 /// 3. the original op is cloned `numUnrolledInstances` times, once for each
43 /// result of the fakeFork cast op.
44 /// 4. a fakeJoin cast op takes all these results and merges them into a
45 /// single aggregate vector result whose size matches the original
46 /// non-unrolled op operand types.
47 ///
48 /// Example:
49 ///
50 /// opA(operand0, operand1) // numUnrolledInstances = 3
51 ///
52 /// operand0 operand1
53 /// | |
54 /// fork fork
55 /// <----------gather all fork ops --------->
56 /// /|\ /|\
57 /// f00 f01 f02 f10 f11 f12
58 /// <---------- clone op 3 times --------->
59 /// opA0(f00, f10), opA1(f01, f11), opA2(f02, f12)
60 /// \ | /
61 /// <-------------------- join ------------------------->
62 ///
63 /// Other local patterns then kick in iteratively (including DCE) and compose
64 /// until all the fakeFork and fakeJoin ops are removed.
65 ///
66 /// This will be extended in the future to support more advanced use cases than
67 /// simple pointwise ops.
68 SmallVector<Value, 1> unrollSingleResultVectorOp(OpBuilder &builder,
69  Operation *op,
70  ArrayRef<int64_t> targetShape);

72 /// Pattern to apply `unrollSingleResultVectorOp` to a `targetShape`
73 /// declaratively.
74 template <typename OpTy>
75 struct UnrollVectorPattern : public OpRewritePattern<OpTy> {
76  using FilterConstraintType = std::function<LogicalResult(OpTy op)>;
77  UnrollVectorPattern(
78  ArrayRef<int64_t> targetShape, MLIRContext *context,
79  FilterConstraintType constraint = [](OpTy op) { return success(); })
80  : OpRewritePattern<OpTy>(context),
81  targetShape(targetShape.begin(), targetShape.end()),
82  filter(constraint) {}
83  LogicalResult matchAndRewrite(OpTy op,
84  PatternRewriter &rewriter) const override {
85  if (failed(filter(op)))
86  return failure();
87  auto unrollableVectorOp =
88  dyn_cast<VectorUnrollOpInterface>(op.getOperation());
89  if (!unrollableVectorOp)
90  return failure();
91  auto maybeUnrollShape = unrollableVectorOp.getShapeForUnroll();
92  if (!maybeUnrollShape)
93  return failure();
94  auto maybeShapeRatio = shapeRatio(*maybeUnrollShape, targetShape);
95  if (!maybeShapeRatio ||
96  llvm::all_of(*maybeShapeRatio, [](int64_t v) { return v == 1; }))
97  return failure();
98  if (op.getOperation()->getNumResults() != 1)
99  return failure();
100  auto resultVector = unrollSingleResultVectorOp(rewriter, op, targetShape);
101  if (resultVector.size() != 1)
102  return failure();
103  rewriter.replaceOp(op, resultVector.front());
104  return success();
105  }
106 
107 private:
108  SmallVector<int64_t, 4> targetShape;
109  FilterConstraintType filter;
110 };

112 /// Split a vector.transfer operation into an unmasked fastpath and a slowpath.
113 /// If `ifOp` is not null and the result is `success, the `ifOp` points to the
114 /// newly created conditional upon function return.
115 /// To accomodate for the fact that the original vector.transfer indexing may be
116 /// arbitrary and the slow path indexes @[0...0] in the temporary buffer, the
117 /// scf.if op returns a view and values of type index.
118 /// At this time, only vector.transfer_read case is implemented.
119 ///
120 /// Example (a 2-D vector.transfer_read):
121 /// ```
122 /// %1 = vector.transfer_read %0[...], %pad : memref<A...>, vector<...>
123 /// ```
124 /// is transformed into:
125 /// ```
126 /// %1:3 = scf.if (%inBounds) {
127 /// // fastpath, direct cast
128 /// memref_cast %A: memref<A...> to compatibleMemRefType
129 /// scf.yield %view : compatibleMemRefType, index, index
130 /// } else {
131 /// // slowpath, masked vector.transfer or linalg.copy.
132 /// memref_cast %alloc: memref<B...> to compatibleMemRefType
133 /// scf.yield %4 : compatibleMemRefType, index, index
134 // }
135 /// %0 = vector.transfer_read %1#0[%1#1, %1#2] {masked = [false ... false]}
136 /// ```
137 /// where `alloc` is a top of the function alloca'ed buffer of one vector.
138 ///
139 /// Preconditions:
140 /// 1. `xferOp.permutation_map()` must be a minor identity map
141 /// 2. the rank of the `xferOp.memref()` and the rank of the `xferOp.vector()`
142 /// must be equal. This will be relaxed in the future but requires
143 /// rank-reducing subviews.
144 LogicalResult
145 splitFullAndPartialTransferPrecondition(VectorTransferOpInterface xferOp);
146 LogicalResult splitFullAndPartialTransfer(
147  OpBuilder &b, VectorTransferOpInterface xferOp,
148  VectorTransformsOptions options = VectorTransformsOptions(),
149  scf::IfOp *ifOp = nullptr);
