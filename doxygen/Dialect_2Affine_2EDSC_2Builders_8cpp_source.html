17 void mlir::edsc::affineLoopNestBuilder(
18  ValueRange lbs, ValueRange ubs, ArrayRef<int64_t> steps,
19  function_ref<void(ValueRange)> bodyBuilderFn) {
20  assert(ScopedContext::getContext() && "EDSC ScopedContext not set up");
21 
22  // Wrap the body builder function into an interface compatible with the main
23  // builder.
24  auto wrappedBuilderFn = [&](OpBuilder &nestedBuilder, Location nestedLoc,
25  ValueRange ivs) {
26  ScopedContext context(nestedBuilder, nestedLoc);
27  bodyBuilderFn(ivs);
28  };
29  function_ref<void(OpBuilder &, Location, ValueRange)> wrapper;
30  if (bodyBuilderFn)
31  wrapper = wrappedBuilderFn;
32 
33  // Extract the builder, location and construct the loop nest.
34  OpBuilder &builder = ScopedContext::getBuilderRef();
35  Location loc = ScopedContext::getLocation();
36  buildAffineLoopNest(builder, loc, lbs, ubs, steps, wrapper);
37 }

39 void mlir::edsc::affineLoopBuilder(ValueRange lbs, ValueRange ubs, int64_t step,
40  function_ref<void(Value)> bodyBuilderFn) {
41  // Fetch the builder and location.
42  assert(ScopedContext::getContext() && "EDSC ScopedContext not set up");
43  OpBuilder &builder = ScopedContext::getBuilderRef();
44  Location loc = ScopedContext::getLocation();
45 
46  // Create the actual loop and call the body builder, if provided, after
47  // updating the scoped context.
48  builder.create<AffineForOp>(
49  loc, lbs, builder.getMultiDimIdentityMap(lbs.size()), ubs,
50  builder.getMultiDimIdentityMap(ubs.size()), step,
51  [&](OpBuilder &nestedBuilder, Location nestedLoc, Value iv) {
52  if (bodyBuilderFn) {
53  ScopedContext nestedContext(nestedBuilder, nestedLoc);
54  OpBuilder::InsertionGuard guard(nestedBuilder);
55  bodyBuilderFn(iv);
56  }
57  nestedBuilder.create<AffineYieldOp>(nestedLoc);
58  });
59 }

61 static std::pair<AffineExpr, Value>
62 categorizeValueByAffineType(MLIRContext *context, Value val, unsigned &numDims,
63  unsigned &numSymbols) {
64  AffineExpr d;
65  Value resultVal = nullptr;
66  if (auto constant = val.getDefiningOp<ConstantIndexOp>()) {
67  d = getAffineConstantExpr(constant.getValue(), context);
68  } else if (isValidSymbol(val) && !isValidDim(val)) {
69  d = getAffineSymbolExpr(numSymbols++, context);
70  resultVal = val;
71  } else {
72  d = getAffineDimExpr(numDims++, context);
73  resultVal = val;
74  }
75  return std::make_pair(d, resultVal);
76 }

78 static Value createBinaryIndexHandle(
79  Value lhs, Value rhs,
80  function_ref<AffineExpr(AffineExpr, AffineExpr)> affCombiner) {
81  MLIRContext *context = ScopedContext::getContext();
82  unsigned numDims = 0, numSymbols = 0;
83  AffineExpr d0, d1;
84  Value v0, v1;
85  std::tie(d0, v0) =
86  categorizeValueByAffineType(context, lhs, numDims, numSymbols);
87  std::tie(d1, v1) =
88  categorizeValueByAffineType(context, rhs, numDims, numSymbols);
89  SmallVector<Value, 2> operands;
90  if (v0)
91  operands.push_back(v0);
92  if (v1)
93  operands.push_back(v1);
94  auto map = AffineMap::get(numDims, numSymbols, affCombiner(d0, d1));
95 
96  // TODO: createOrFold when available.
97  Operation *op =
98  makeComposedAffineApply(ScopedContext::getBuilderRef(),
99  ScopedContext::getLocation(), map, operands)
100  .getOperation();
101  assert(op->getNumResults() == 1 && "Expected single result AffineApply");
102  return op->getResult(0);
103 }

105 template <typename IOp, typename FOp>
106 static Value createBinaryHandle(
107  Value lhs, Value rhs,
108  function_ref<AffineExpr(AffineExpr, AffineExpr)> affCombiner) {
109  auto thisType = lhs.getType();
110  auto thatType = rhs.getType();
111  assert(thisType == thatType && "cannot mix types in operators");
112  (void)thisType;
113  (void)thatType;
114  if (thisType.isIndex()) {
115  return createBinaryIndexHandle(lhs, rhs, affCombiner);
116  } else if (thisType.isSignlessInteger()) {
117  return ValueBuilder<IOp>(lhs, rhs);
118  } else if (thisType.isa<FloatType>()) {
119  return ValueBuilder<FOp>(lhs, rhs);
120  } else if (thisType.isa<VectorType, TensorType>()) {
121  auto aggregateType = thisType.cast<ShapedType>();
122  if (aggregateType.getElementType().isSignlessInteger())
123  return ValueBuilder<IOp>(lhs, rhs);
124  else if (aggregateType.getElementType().isa<FloatType>())
125  return ValueBuilder<FOp>(lhs, rhs);
126  }
127  llvm_unreachable("failed to create a Value");
128 }

184 static Value createIComparisonExpr(CmpIPredicate predicate, Value lhs,
185  Value rhs) {
186  auto lhsType = lhs.getType();
187  auto rhsType = rhs.getType();
188  (void)lhsType;
189  (void)rhsType;
190  assert(lhsType == rhsType && "cannot mix types in operators");
191  assert((lhsType.isa<IndexType>() || lhsType.isSignlessInteger()) &&
192  "only integer comparisons are supported");
193 
194  return ScopedContext::getBuilderRef().create<CmpIOp>(
195  ScopedContext::getLocation(), predicate, lhs, rhs);
196 }
