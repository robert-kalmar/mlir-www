48 /// A pre-order depth-first visitor function for processing basic blocks.
49 ///
50 /// Visits the basic blocks starting from the given `headerBlock` in pre-order
51 /// depth-first manner and calls `blockHandler` on each block. Skips handling
52 /// blocks in the `skipBlocks` list. If `skipHeader` is true, `blockHandler`
53 /// will not be invoked in `headerBlock` but still handles all `headerBlock`'s
54 /// successors.
55 ///
56 /// SPIR-V spec "2.16.1. Universal Validation Rules" requires that "the order
57 /// of blocks in a function must satisfy the rule that blocks appear before
58 /// all blocks they dominate." This can be achieved by a pre-order CFG
59 /// traversal algorithm. To make the serialization output more logical and
60 /// readable to human, we perform depth-first CFG traversal and delay the
61 /// serialization of the merge block and the continue block, if exists, until
62 /// after all other blocks have been processed.
63 static LogicalResult visitInPrettyBlockOrder(
64  Block *headerBlock, function_ref<LogicalResult(Block *)> blockHandler,
65  bool skipHeader = false, ArrayRef<Block *> skipBlocks = {}) {
66  llvm::df_iterator_default_set<Block *, 4> doneBlocks;
67  doneBlocks.insert(skipBlocks.begin(), skipBlocks.end());
68 
69  for (Block *block : llvm::depth_first_ext(headerBlock, doneBlocks)) {
70  if (skipHeader && block == headerBlock)
71  continue;
72  if (failed(blockHandler(block)))
73  return failure();
74  }
75  return success();
76 }

88 /// Given a predecessor `block` for a block with arguments, returns the block
89 /// that should be used as the parent block for SPIR-V OpPhi instructions
90 /// corresponding to the block arguments.
91 static Block *getPhiIncomingBlock(Block *block) {
92  // If the predecessor block in question is the entry block for a spv.loop,
93  // we jump to this spv.loop from its enclosing block.
94  if (block->isEntryBlock()) {
95  if (auto loopOp = dyn_cast<spirv::LoopOp>(block->getParentOp())) {
96  // Then the incoming parent block for OpPhi should be the merge block of
97  // the structured control flow op before this loop.
98  Operation *op = loopOp.getOperation();
99  while ((op = op->getPrevNode()) != nullptr)
100  if (Block *incomingBlock = getStructuredControlFlowOpMergeBlock(op))
101  return incomingBlock;
102  // Or the enclosing block itself if no structured control flow ops
103  // exists before this loop.
104  return loopOp.getOperation()->getBlock();
105  }
106  }
107 
108  // Otherwise, we jump from the given predecessor block. Try to see if there is
109  // a structured control flow op inside it.
110  for (Operation &op : llvm::reverse(block->getOperations())) {
111  if (Block *incomingBlock = getStructuredControlFlowOpMergeBlock(&op))
112  return incomingBlock;
113  }
114  return block;
115 }

463  /// Map from values used in OpPhi instructions to their offset in the
464  /// `functions` section.
465  ///
466  /// When processing a block with arguments, we need to emit OpPhi
467  /// instructions to record the predecessor block <id>s and the values they
468  /// send to the block in question. But it's not guaranteed all values are
469  /// visited and thus assigned result <id>s. So we need this list to capture
470  /// the offsets into `functions` where a value is used so that we can fix it
471  /// up later after processing all the blocks in a function.
472  ///
473  /// More concretely, say if we are visiting the following blocks:
474  ///
475  /// ```mlir
476  /// ^phi(%arg0: i32):
477  /// ...
478  /// ^parent1:
479  /// ...
480  /// spv.Branch ^phi(%val0: i32)
481  /// ^parent2:
482  /// ...
483  /// spv.Branch ^phi(%val1: i32)
484  /// ```
485  ///
486  /// When we are serializing the `^phi` block, we need to emit at the beginning
487  /// of the block OpPhi instructions which has the following parameters:
488  ///
489  /// OpPhi id-for-i32 id-for-%arg0 id-for-%val0 id-for-^parent1
490  /// id-for-%val1 id-for-^parent2
491  ///
492  /// But we don't know the <id> for %val0 and %val1 yet. One way is to visit
493  /// all the blocks twice and use the first visit to assign an <id> to each
494  /// value. But it's paying the overheads just for OpPhi emission. Instead,
495  /// we still visit the blocks once for emission. When we emit the OpPhi
496  /// instructions, we use 0 as a placeholder for the <id>s for %val0 and %val1.
497  /// At the same time, we record their offsets in the emitted binary (which is
498  /// placed inside `functions`) here. And then after emitting all blocks, we
499  /// replace the dummy <id> 0 with the real result <id> by overwriting
500  /// `functions[offset]`.
501  DenseMap<Value, SmallVector<size_t, 1>> deferredPhiValues;

509 LogicalResult Serializer::serialize() {
510  LLVM_DEBUG(llvm::dbgs() << "+++ starting serialization +++\n");
511 
512  if (failed(module.verify()))
513  return failure();
514 
515  // TODO: handle the other sections
516  processCapability();
517  processExtension();
518  processMemoryModel();
519  processDebugInfo();
520 
521  // Iterate over the module body to serialize it. Assumptions are that there is
522  // only one basic block in the moduleOp
523  for (auto &op : module.getBlock()) {
524  if (failed(processOperation(&op))) {
525  return failure();
526  }
527  }
528 
529  LLVM_DEBUG(llvm::dbgs() << "+++ completed serialization +++\n");
530  return success();
531 }

663 LogicalResult Serializer::processDecoration(Location loc, uint32_t resultID,
664  NamedAttribute attr) {
665  auto attrName = attr.first.strref();
666  auto decorationName = llvm::convertToCamelFromSnakeCase(attrName, true);
667  auto decoration = spirv::symbolizeDecoration(decorationName);
668  if (!decoration) {
669  return emitError(
670  loc, "non-argument attributes expected to have snake-case-ified "
671  "decoration name, unhandled attribute with name : ")
672  << attrName;
673  }
674  SmallVector<uint32_t, 1> args;
675  switch (decoration.getValue()) {
676  case spirv::Decoration::Binding:
677  case spirv::Decoration::DescriptorSet:
678  case spirv::Decoration::Location:
679  if (auto intAttr = attr.second.dyn_cast<IntegerAttr>()) {
680  args.push_back(intAttr.getValue().getZExtValue());
681  break;
682  }
683  return emitError(loc, "expected integer attribute for ") << attrName;
684  case spirv::Decoration::BuiltIn:
685  if (auto strAttr = attr.second.dyn_cast<StringAttr>()) {
686  auto enumVal = spirv::symbolizeBuiltIn(strAttr.getValue());
687  if (enumVal) {
688  args.push_back(static_cast<uint32_t>(enumVal.getValue()));
689  break;
690  }
691  return emitError(loc, "invalid ")
692  << attrName << " attribute " << strAttr.getValue();
693  }
694  return emitError(loc, "expected string attribute for ") << attrName;
695  case spirv::Decoration::Flat:
696  case spirv::Decoration::NoPerspective:
697  if (auto unitAttr = attr.second.dyn_cast<UnitAttr>()) {
698  // For unit attributes, the args list has no values so we do nothing
699  break;
700  }
701  return emitError(loc, "expected unit attribute for ") << attrName;
702  default:
703  return emitError(loc, "unhandled decoration ") << decorationName;
704  }
705  return emitDecoration(resultID, decoration.getValue(), args);
706 }

754 LogicalResult Serializer::processFuncOp(spirv::FuncOp op) {
755  LLVM_DEBUG(llvm::dbgs() << "-- start function '" << op.getName() << "' --\n");
756  assert(functionHeader.empty() && functionBody.empty());
757 
758  uint32_t fnTypeID = 0;
759  // Generate type of the function.
760  processType(op.getLoc(), op.getType(), fnTypeID);
761 
762  // Add the function definition.
763  SmallVector<uint32_t, 4> operands;
764  uint32_t resTypeID = 0;
765  auto resultTypes = op.getType().getResults();
766  if (resultTypes.size() > 1) {
767  return op.emitError("cannot serialize function with multiple return types");
768  }
769  if (failed(processType(op.getLoc(),
770  (resultTypes.empty() ? getVoidType() : resultTypes[0]),
771  resTypeID))) {
772  return failure();
773  }
774  operands.push_back(resTypeID);
775  auto funcID = getOrCreateFunctionID(op.getName());
776  operands.push_back(funcID);
777  // TODO: Support other function control options.
778  operands.push_back(static_cast<uint32_t>(spirv::FunctionControl::None));
779  operands.push_back(fnTypeID);
780  encodeInstructionInto(functionHeader, spirv::Opcode::OpFunction, operands);
781 
782  // Add function name.
783  if (failed(processName(funcID, op.getName()))) {
784  return failure();
785  }
786 
787  // Declare the parameters.
788  for (auto arg : op.getArguments()) {
789  uint32_t argTypeID = 0;
790  if (failed(processType(op.getLoc(), arg.getType(), argTypeID))) {
791  return failure();
792  }
793  auto argValueID = getNextID();
794  valueIDMap[arg] = argValueID;
795  encodeInstructionInto(functionHeader, spirv::Opcode::OpFunctionParameter,
796  {argTypeID, argValueID});
797  }
798 
799  // Process the body.
800  if (op.isExternal()) {
801  return op.emitError("external function is unhandled");
802  }
803 
804  // Some instructions (e.g., OpVariable) in a function must be in the first
805  // block in the function. These instructions will be put in functionHeader.
806  // Thus, we put the label in functionHeader first, and omit it from the first
807  // block.
808  encodeInstructionInto(functionHeader, spirv::Opcode::OpLabel,
809  {getOrCreateBlockID(&op.front())});
810  processBlock(&op.front(), /*omitLabel=*/true);
811  if (failed(visitInPrettyBlockOrder(
812  &op.front(), [&](Block *block) { return processBlock(block); },
813  /*skipHeader=*/true))) {
814  return failure();
815  }
816 
817  // There might be OpPhi instructions who have value references needing to fix.
818  for (auto deferredValue : deferredPhiValues) {
819  Value value = deferredValue.first;
820  uint32_t id = getValueID(value);
821  LLVM_DEBUG(llvm::dbgs() << "[phi] fix reference of value " << value
822  << " to id = " << id << '\n');
823  assert(id && "OpPhi references undefined value!");
824  for (size_t offset : deferredValue.second)
825  functionBody[offset] = id;
826  }
827  deferredPhiValues.clear();
828 
829  LLVM_DEBUG(llvm::dbgs() << "-- completed function '" << op.getName()
830  << "' --\n");
831  // Insert OpFunctionEnd.
832  if (failed(encodeInstructionInto(functionBody, spirv::Opcode::OpFunctionEnd,
833  {}))) {
834  return failure();
835  }
836 
837  functions.append(functionHeader.begin(), functionHeader.end());
838  functions.append(functionBody.begin(), functionBody.end());
839  functionHeader.clear();
840  functionBody.clear();
841 
842  return success();
843 }

845 LogicalResult Serializer::processVariableOp(spirv::VariableOp op) {
846  SmallVector<uint32_t, 4> operands;
847  SmallVector<StringRef, 2> elidedAttrs;
848  uint32_t resultID = 0;
849  uint32_t resultTypeID = 0;
850  if (failed(processType(op.getLoc(), op.getType(), resultTypeID))) {
851  return failure();
852  }
853  operands.push_back(resultTypeID);
854  resultID = getNextID();
855  valueIDMap[op.getResult()] = resultID;
856  operands.push_back(resultID);
857  auto attr = op.getAttr(spirv::attributeName<spirv::StorageClass>());
858  if (attr) {
859  operands.push_back(static_cast<uint32_t>(
860  attr.cast<IntegerAttr>().getValue().getZExtValue()));
861  }
862  elidedAttrs.push_back(spirv::attributeName<spirv::StorageClass>());
863  for (auto arg : op.getODSOperands(0)) {
864  auto argID = getValueID(arg);
865  if (!argID) {
866  return emitError(op.getLoc(), "operand 0 has a use before def");
867  }
868  operands.push_back(argID);
869  }
870  emitDebugLine(functionHeader, op.getLoc());
871  encodeInstructionInto(functionHeader, spirv::Opcode::OpVariable, operands);
872  for (auto attr : op.getAttrs()) {
873  if (llvm::any_of(elidedAttrs,
874  [&](StringRef elided) { return attr.first == elided; })) {
875  continue;
876  }
877  if (failed(processDecoration(op.getLoc(), resultID, attr))) {
878  return failure();
879  }
880  }
881  return success();
882 }

884 LogicalResult
885 Serializer::processGlobalVariableOp(spirv::GlobalVariableOp varOp) {
886  // Get TypeID.
887  uint32_t resultTypeID = 0;
888  SmallVector<StringRef, 4> elidedAttrs;
889  if (failed(processType(varOp.getLoc(), varOp.type(), resultTypeID))) {
890  return failure();
891  }
892 
893  if (isInterfaceStructPtrType(varOp.type())) {
894  auto structType = varOp.type()
895  .cast<spirv::PointerType>()
896  .getPointeeType()
897  .cast<spirv::StructType>();
898  if (failed(
899  emitDecoration(getTypeID(structType), spirv::Decoration::Block))) {
900  return varOp.emitError("cannot decorate ")
901  << structType << " with Block decoration";
902  }
903  }
904 
905  elidedAttrs.push_back("type");
906  SmallVector<uint32_t, 4> operands;
907  operands.push_back(resultTypeID);
908  auto resultID = getNextID();
909 
910  // Encode the name.
911  auto varName = varOp.sym_name();
912  elidedAttrs.push_back(SymbolTable::getSymbolAttrName());
913  if (failed(processName(resultID, varName))) {
914  return failure();
915  }
916  globalVarIDMap[varName] = resultID;
917  operands.push_back(resultID);
918 
919  // Encode StorageClass.
920  operands.push_back(static_cast<uint32_t>(varOp.storageClass()));
921 
922  // Encode initialization.
923  if (auto initializer = varOp.initializer()) {
924  auto initializerID = getVariableID(initializer.getValue());
925  if (!initializerID) {
926  return emitError(varOp.getLoc(),
927  "invalid usage of undefined variable as initializer");
928  }
929  operands.push_back(initializerID);
930  elidedAttrs.push_back("initializer");
931  }
932 
933  emitDebugLine(typesGlobalValues, varOp.getLoc());
934  if (failed(encodeInstructionInto(typesGlobalValues, spirv::Opcode::OpVariable,
935  operands))) {
936  elidedAttrs.push_back("initializer");
937  return failure();
938  }
939 
940  // Encode decorations.
941  for (auto attr : varOp.getAttrs()) {
942  if (llvm::any_of(elidedAttrs,
943  [&](StringRef elided) { return attr.first == elided; })) {
944  continue;
945  }
946  if (failed(processDecoration(varOp.getLoc(), resultID, attr))) {
947  return failure();
948  }
949  }
950  return success();
951 }

995 LogicalResult
996 Serializer::prepareBasicType(Location loc, Type type, uint32_t resultID,
997  spirv::Opcode &typeEnum,
998  SmallVectorImpl<uint32_t> &operands) {
999  if (isVoidType(type)) {
1000  typeEnum = spirv::Opcode::OpTypeVoid;
1001  return success();
1002  }
1003 
1004  if (auto intType = type.dyn_cast<IntegerType>()) {
1005  if (intType.getWidth() == 1) {
1006  typeEnum = spirv::Opcode::OpTypeBool;
1007  return success();
1008  }
1009 
1010  typeEnum = spirv::Opcode::OpTypeInt;
1011  operands.push_back(intType.getWidth());
1012  // SPIR-V OpTypeInt "Signedness specifies whether there are signed semantics
1013  // to preserve or validate.
1014  // 0 indicates unsigned, or no signedness semantics
1015  // 1 indicates signed semantics."
1016  operands.push_back(intType.isSigned() ? 1 : 0);
1017  return success();
1018  }
1019 
1020  if (auto floatType = type.dyn_cast<FloatType>()) {
1021  typeEnum = spirv::Opcode::OpTypeFloat;
1022  operands.push_back(floatType.getWidth());
1023  return success();
1024  }
1025 
1026  if (auto vectorType = type.dyn_cast<VectorType>()) {
1027  uint32_t elementTypeID = 0;
1028  if (failed(processType(loc, vectorType.getElementType(), elementTypeID))) {
1029  return failure();
1030  }
1031  typeEnum = spirv::Opcode::OpTypeVector;
1032  operands.push_back(elementTypeID);
1033  operands.push_back(vectorType.getNumElements());
1034  return success();
1035  }
1036 
1037  if (auto arrayType = type.dyn_cast<spirv::ArrayType>()) {
1038  typeEnum = spirv::Opcode::OpTypeArray;
1039  uint32_t elementTypeID = 0;
1040  if (failed(processType(loc, arrayType.getElementType(), elementTypeID))) {
1041  return failure();
1042  }
1043  operands.push_back(elementTypeID);
1044  if (auto elementCountID = prepareConstantInt(
1045  loc, mlirBuilder.getI32IntegerAttr(arrayType.getNumElements()))) {
1046  operands.push_back(elementCountID);
1047  }
1048  return processTypeDecoration(loc, arrayType, resultID);
1049  }
1050 
1051  if (auto ptrType = type.dyn_cast<spirv::PointerType>()) {
1052  uint32_t pointeeTypeID = 0;
1053  if (failed(processType(loc, ptrType.getPointeeType(), pointeeTypeID))) {
1054  return failure();
1055  }
1056  typeEnum = spirv::Opcode::OpTypePointer;
1057  operands.push_back(static_cast<uint32_t>(ptrType.getStorageClass()));
1058  operands.push_back(pointeeTypeID);
1059  return success();
1060  }
1061 
1062  if (auto runtimeArrayType = type.dyn_cast<spirv::RuntimeArrayType>()) {
1063  uint32_t elementTypeID = 0;
1064  if (failed(processType(loc, runtimeArrayType.getElementType(),
1065  elementTypeID))) {
1066  return failure();
1067  }
1068  typeEnum = spirv::Opcode::OpTypeRuntimeArray;
1069  operands.push_back(elementTypeID);
1070  return processTypeDecoration(loc, runtimeArrayType, resultID);
1071  }
1072 
1073  if (auto structType = type.dyn_cast<spirv::StructType>()) {
1074  bool hasOffset = structType.hasOffset();
1075  for (auto elementIndex :
1076  llvm::seq<uint32_t>(0, structType.getNumElements())) {
1077  uint32_t elementTypeID = 0;
1078  if (failed(processType(loc, structType.getElementType(elementIndex),
1079  elementTypeID))) {
1080  return failure();
1081  }
1082  operands.push_back(elementTypeID);
1083  if (hasOffset) {
1084  // Decorate each struct member with an offset
1085  spirv::StructType::MemberDecorationInfo offsetDecoration{
1086  elementIndex, /*hasValue=*/1, spirv::Decoration::Offset,
1087  static_cast<uint32_t>(structType.getMemberOffset(elementIndex))};
1088  if (failed(processMemberDecoration(resultID, offsetDecoration))) {
1089  return emitError(loc, "cannot decorate ")
1090  << elementIndex << "-th member of " << structType
1091  << " with its offset";
1092  }
1093  }
1094  }
1095  SmallVector<spirv::StructType::MemberDecorationInfo, 4> memberDecorations;
1096  structType.getMemberDecorations(memberDecorations);
1097  for (auto &memberDecoration : memberDecorations) {
1098  if (failed(processMemberDecoration(resultID, memberDecoration))) {
1099  return emitError(loc, "cannot decorate ")
1100  << static_cast<uint32_t>(memberDecoration.memberIndex)
1101  << "-th member of " << structType << " with "
1102  << stringifyDecoration(memberDecoration.decoration);
1103  }
1104  }
1105  typeEnum = spirv::Opcode::OpTypeStruct;
1106  return success();
1107  }
1108 
1109  if (auto cooperativeMatrixType =
1110  type.dyn_cast<spirv::CooperativeMatrixNVType>()) {
1111  uint32_t elementTypeID = 0;
1112  if (failed(processType(loc, cooperativeMatrixType.getElementType(),
1113  elementTypeID))) {
1114  return failure();
1115  }
1116  typeEnum = spirv::Opcode::OpTypeCooperativeMatrixNV;
1117  auto getConstantOp = [&](uint32_t id) {
1118  auto attr = IntegerAttr::get(IntegerType::get(32, type.getContext()), id);
1119  return prepareConstantInt(loc, attr);
1120  };
1121  operands.push_back(elementTypeID);
1122  operands.push_back(
1123  getConstantOp(static_cast<uint32_t>(cooperativeMatrixType.getScope())));
1124  operands.push_back(getConstantOp(cooperativeMatrixType.getRows()));
1125  operands.push_back(getConstantOp(cooperativeMatrixType.getColumns()));
1126  return success();
1127  }
1128 
1129  if (auto matrixType = type.dyn_cast<spirv::MatrixType>()) {
1130  uint32_t elementTypeID = 0;
1131  if (failed(processType(loc, matrixType.getColumnType(), elementTypeID))) {
1132  return failure();
1133  }
1134  typeEnum = spirv::Opcode::OpTypeMatrix;
1135  operands.push_back(elementTypeID);
1136  operands.push_back(matrixType.getNumColumns());
1137  return success();
1138  }
1139 
1140  // TODO: Handle other types.
1141  return emitError(loc, "unhandled type in serialization: ") << type;
1142 }

1450 Serializer::processBlock(Block *block, bool omitLabel,
1451  function_ref<void()> actionBeforeTerminator) {
1452  LLVM_DEBUG(llvm::dbgs() << "processing block " << block << ":\n");
1453  LLVM_DEBUG(block->print(llvm::dbgs()));
1454  LLVM_DEBUG(llvm::dbgs() << '\n');
1455  if (!omitLabel) {
1456  uint32_t blockID = getOrCreateBlockID(block);
1457  LLVM_DEBUG(llvm::dbgs()
1458  << "[block] " << block << " (id = " << blockID << ")\n");
1459 
1460  // Emit OpLabel for this block.
1461  encodeInstructionInto(functionBody, spirv::Opcode::OpLabel, {blockID});
1462  }
1463 
1464  // Emit OpPhi instructions for block arguments, if any.
1465  if (failed(emitPhiForBlockArguments(block)))
1466  return failure();
1467 
1468  // Process each op in this block except the terminator.
1469  for (auto &op : llvm::make_range(block->begin(), std::prev(block->end()))) {
1470  if (failed(processOperation(&op)))
1471  return failure();
1472  }
1473 
1474  // Process the terminator.
1475  if (actionBeforeTerminator)
1476  actionBeforeTerminator();
1477  if (failed(processOperation(&block->back())))
1478  return failure();
1479 
1480  return success();

1483 LogicalResult Serializer::emitPhiForBlockArguments(Block *block) {
1484  // Nothing to do if this block has no arguments or it's the entry block, which
1485  // always has the same arguments as the function signature.
1486  if (block->args_empty() || block->isEntryBlock())
1487  return success();
1488 
1489  // If the block has arguments, we need to create SPIR-V OpPhi instructions.
1490  // A SPIR-V OpPhi instruction is of the syntax:
1491  // OpPhi | result type | result <id> | (value <id>, parent block <id>) pair
1492  // So we need to collect all predecessor blocks and the arguments they send
1493  // to this block.
1494  SmallVector<std::pair<Block *, Operation::operand_iterator>, 4> predecessors;
1495  for (Block *predecessor : block->getPredecessors()) {
1496  auto *terminator = predecessor->getTerminator();
1497  // The predecessor here is the immediate one according to MLIR's IR
1498  // structure. It does not directly map to the incoming parent block for the
1499  // OpPhi instructions at SPIR-V binary level. This is because structured
1500  // control flow ops are serialized to multiple SPIR-V blocks. If there is a
1501  // spv.selection/spv.loop op in the MLIR predecessor block, the branch op
1502  // jumping to the OpPhi's block then resides in the previous structured
1503  // control flow op's merge block.
1504  predecessor = getPhiIncomingBlock(predecessor);
1505  if (auto branchOp = dyn_cast<spirv::BranchOp>(terminator)) {
1506  predecessors.emplace_back(predecessor, branchOp.operand_begin());
1507  } else {
1508  return terminator->emitError("unimplemented terminator for Phi creation");
1509  }
1510  }
1511 
1512  // Then create OpPhi instruction for each of the block argument.
1513  for (auto argIndex : llvm::seq<unsigned>(0, block->getNumArguments())) {
1514  BlockArgument arg = block->getArgument(argIndex);
1515 
1516  // Get the type <id> and result <id> for this OpPhi instruction.
1517  uint32_t phiTypeID = 0;
1518  if (failed(processType(arg.getLoc(), arg.getType(), phiTypeID)))
1519  return failure();
1520  uint32_t phiID = getNextID();
1521 
1522  LLVM_DEBUG(llvm::dbgs() << "[phi] for block argument #" << argIndex << ' '
1523  << arg << " (id = " << phiID << ")\n");
1524 
1525  // Prepare the (value <id>, parent block <id>) pairs.
1526  SmallVector<uint32_t, 8> phiArgs;
1527  phiArgs.push_back(phiTypeID);
1528  phiArgs.push_back(phiID);
1529 
1530  for (auto predIndex : llvm::seq<unsigned>(0, predecessors.size())) {
1531  Value value = *(predecessors[predIndex].second + argIndex);
1532  uint32_t predBlockId = getOrCreateBlockID(predecessors[predIndex].first);
1533  LLVM_DEBUG(llvm::dbgs() << "[phi] use predecessor (id = " << predBlockId
1534  << ") value " << value << ' ');
1535  // Each pair is a value <id> ...
1536  uint32_t valueId = getValueID(value);
1537  if (valueId == 0) {
1538  // The op generating this value hasn't been visited yet so we don't have
1539  // an <id> assigned yet. Record this to fix up later.
1540  LLVM_DEBUG(llvm::dbgs() << "(need to fix)\n");
1541  deferredPhiValues[value].push_back(functionBody.size() + 1 +
1542  phiArgs.size());
1543  } else {
1544  LLVM_DEBUG(llvm::dbgs() << "(id = " << valueId << ")\n");
1545  }
1546  phiArgs.push_back(valueId);
1547  // ... and a parent block <id>.
1548  phiArgs.push_back(predBlockId);
1549  }
1550 
1551  encodeInstructionInto(functionBody, spirv::Opcode::OpPhi, phiArgs);
1552  valueIDMap[arg] = phiID;
1553  }
1554 
1555  return success();
1556 }

1558 LogicalResult Serializer::processSelectionOp(spirv::SelectionOp selectionOp) {
1559  // Assign <id>s to all blocks so that branches inside the SelectionOp can
1560  // resolve properly.
1561  auto &body = selectionOp.body();
1562  for (Block &block : body)
1563  getOrCreateBlockID(&block);
1564 
1565  auto *headerBlock = selectionOp.getHeaderBlock();
1566  auto *mergeBlock = selectionOp.getMergeBlock();
1567  auto mergeID = getBlockID(mergeBlock);
1568  auto loc = selectionOp.getLoc();
1569 
1570  // Emit the selection header block, which dominates all other blocks, first.
1571  // We need to emit an OpSelectionMerge instruction before the selection header
1572  // block's terminator.
1573  auto emitSelectionMerge = [&]() {
1574  emitDebugLine(functionBody, loc);
1575  lastProcessedWasMergeInst = true;
1576  encodeInstructionInto(
1577  functionBody, spirv::Opcode::OpSelectionMerge,
1578  {mergeID, static_cast<uint32_t>(selectionOp.selection_control())});
1579  };
1580  // For structured selection, we cannot have blocks in the selection construct
1581  // branching to the selection header block. Entering the selection (and
1582  // reaching the selection header) must be from the block containing the
1583  // spv.selection op. If there are ops ahead of the spv.selection op in the
1584  // block, we can "merge" them into the selection header. So here we don't need
1585  // to emit a separate block; just continue with the existing block.
1586  if (failed(processBlock(headerBlock, /*omitLabel=*/true, emitSelectionMerge)))
1587  return failure();
1588 
1589  // Process all blocks with a depth-first visitor starting from the header
1590  // block. The selection header block and merge block are skipped by this
1591  // visitor.
1592  if (failed(visitInPrettyBlockOrder(
1593  headerBlock, [&](Block *block) { return processBlock(block); },
1594  /*skipHeader=*/true, /*skipBlocks=*/{mergeBlock})))
1595  return failure();
1596 
1597  // There is nothing to do for the merge block in the selection, which just
1598  // contains a spv._merge op, itself. But we need to have an OpLabel
1599  // instruction to start a new SPIR-V block for ops following this SelectionOp.
1600  // The block should use the <id> for the merge block.
1601  return encodeInstructionInto(functionBody, spirv::Opcode::OpLabel, {mergeID});
1602 }

1604 LogicalResult Serializer::processLoopOp(spirv::LoopOp loopOp) {
1605  // Assign <id>s to all blocks so that branches inside the LoopOp can resolve
1606  // properly. We don't need to assign for the entry block, which is just for
1607  // satisfying MLIR region's structural requirement.
1608  auto &body = loopOp.body();
1609  for (Block &block :
1610  llvm::make_range(std::next(body.begin(), 1), body.end())) {
1611  getOrCreateBlockID(&block);
1612  }
1613  auto *headerBlock = loopOp.getHeaderBlock();
1614  auto *continueBlock = loopOp.getContinueBlock();
1615  auto *mergeBlock = loopOp.getMergeBlock();
1616  auto headerID = getBlockID(headerBlock);
1617  auto continueID = getBlockID(continueBlock);
1618  auto mergeID = getBlockID(mergeBlock);
1619  auto loc = loopOp.getLoc();
1620 
1621  // This LoopOp is in some MLIR block with preceding and following ops. In the
1622  // binary format, it should reside in separate SPIR-V blocks from its
1623  // preceding and following ops. So we need to emit unconditional branches to
1624  // jump to this LoopOp's SPIR-V blocks and jumping back to the normal flow
1625  // afterwards.
1626  encodeInstructionInto(functionBody, spirv::Opcode::OpBranch, {headerID});
1627 
1628  // LoopOp's entry block is just there for satisfying MLIR's structural
1629  // requirements so we omit it and start serialization from the loop header
1630  // block.
1631 
1632  // Emit the loop header block, which dominates all other blocks, first. We
1633  // need to emit an OpLoopMerge instruction before the loop header block's
1634  // terminator.
1635  auto emitLoopMerge = [&]() {
1636  emitDebugLine(functionBody, loc);
1637  lastProcessedWasMergeInst = true;
1638  encodeInstructionInto(
1639  functionBody, spirv::Opcode::OpLoopMerge,
1640  {mergeID, continueID, static_cast<uint32_t>(loopOp.loop_control())});
1641  };
1642  if (failed(processBlock(headerBlock, /*omitLabel=*/false, emitLoopMerge)))
1643  return failure();
1644 
1645  // Process all blocks with a depth-first visitor starting from the header
1646  // block. The loop header block, loop continue block, and loop merge block are
1647  // skipped by this visitor and handled later in this function.
1648  if (failed(visitInPrettyBlockOrder(
1649  headerBlock, [&](Block *block) { return processBlock(block); },
1650  /*skipHeader=*/true, /*skipBlocks=*/{continueBlock, mergeBlock})))
1651  return failure();
1652 
1653  // We have handled all other blocks. Now get to the loop continue block.
1654  if (failed(processBlock(continueBlock)))
1655  return failure();
1656 
1657  // There is nothing to do for the merge block in the loop, which just contains
1658  // a spv._merge op, itself. But we need to have an OpLabel instruction to
1659  // start a new SPIR-V block for ops following this LoopOp. The block should
1660  // use the <id> for the merge block.
1661  return encodeInstructionInto(functionBody, spirv::Opcode::OpLabel, {mergeID});
1662 }

1747 LogicalResult Serializer::processOperation(Operation *opInst) {
1748  LLVM_DEBUG(llvm::dbgs() << "[op] '" << opInst->getName() << "'\n");
1749 
1750  // First dispatch the ops that do not directly mirror an instruction from
1751  // the SPIR-V spec.
1752  return TypeSwitch<Operation *, LogicalResult>(opInst)
1753  .Case([&](spirv::AddressOfOp op) { return processAddressOfOp(op); })
1754  .Case([&](spirv::BranchOp op) { return processBranchOp(op); })
1755  .Case([&](spirv::BranchConditionalOp op) {
1756  return processBranchConditionalOp(op);
1757  })
1758  .Case([&](spirv::ConstantOp op) { return processConstantOp(op); })
1759  .Case([&](spirv::FuncOp op) { return processFuncOp(op); })
1760  .Case([&](spirv::GlobalVariableOp op) {
1761  return processGlobalVariableOp(op);
1762  })
1763  .Case([&](spirv::LoopOp op) { return processLoopOp(op); })
1764  .Case([&](spirv::ModuleEndOp) { return success(); })
1765  .Case([&](spirv::ReferenceOfOp op) { return processReferenceOfOp(op); })
1766  .Case([&](spirv::SelectionOp op) { return processSelectionOp(op); })
1767  .Case([&](spirv::SpecConstantOp op) { return processSpecConstantOp(op); })
1768  .Case([&](spirv::UndefOp op) { return processUndefOp(op); })
1769  .Case([&](spirv::VariableOp op) { return processVariableOp(op); })
1770 
1771  // Then handle all the ops that directly mirror SPIR-V instructions with
1772  // auto-generated methods.
1773  .Default(
1774  [&](Operation *op) { return dispatchToAutogenSerialization(op); });
1775 }
