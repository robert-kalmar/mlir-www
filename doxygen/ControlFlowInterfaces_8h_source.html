49 /// This class represents a successor of a region. A region successor can either
50 /// be another region, or the parent operation. If the successor is a region,
51 /// this class represents the destination region, as well as a set of arguments
52 /// from that region that will be populated by values from the current region.
53 /// If the successor is the parent operation, this class represents an optional
54 /// set of results that will be populated by values from the current region.
55 ///
56 /// This interface assumes that the values from the current region that are used
57 /// to populate the successor inputs are the operands of the return-like
58 /// terminator operations in the blocks within this region.
59 class RegionSuccessor {
60 public:
61  /// Initialize a successor that branches to another region of the parent
62  /// operation.
63  RegionSuccessor(Region *region, Block::BlockArgListType regionInputs = {})
64  : region(region), inputs(regionInputs) {}
65  /// Initialize a successor that branches back to/out of the parent operation.
66  RegionSuccessor(Optional<Operation::result_range> results = {})
67  : region(nullptr), inputs(results ? ValueRange(*results) : ValueRange()) {
68  }
69 
70  /// Return the given region successor. Returns nullptr if the successor is the
71  /// parent operation.
72  Region *getSuccessor() const { return region; }
73 
74  /// Return true if the successor is the parent operation.
75  bool isParent() const { return region == nullptr; }
76 
77  /// Return the inputs to the successor that are remapped by the exit values of
78  /// the current region.
79  ValueRange getSuccessorInputs() const { return inputs; }
80 
81 private:
82  Region *region;
83  ValueRange inputs;
84 };

90 namespace OpTrait {
91 /// This trait indicates that a terminator operation is "return-like". This
92 /// means that it exits its current region and forwards its operands as "exit"
93 /// values to the parent region. Operations with this trait are not permitted to
94 /// contain successors or produce results.
95 template <typename ConcreteType>
96 struct ReturnLike : public TraitBase<ConcreteType, ReturnLike> {
97  static LogicalResult verifyTrait(Operation *op) {
98  static_assert(ConcreteType::template hasTrait<IsTerminator>(),
99  "expected operation to be a terminator");
100  static_assert(ConcreteType::template hasTrait<ZeroResult>(),
101  "expected operation to have zero results");
102  static_assert(ConcreteType::template hasTrait<ZeroSuccessor>(),
103  "expected operation to have zero successors");
104  return success();
105  }
106 };
107 } // namespace OpTrait
