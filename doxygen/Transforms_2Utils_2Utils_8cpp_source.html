45 // Perform the replacement in `op`.
46 LogicalResult mlir::replaceAllMemRefUsesWith(Value oldMemRef, Value newMemRef,
47  Operation *op,
48  ArrayRef<Value> extraIndices,
49  AffineMap indexRemap,
50  ArrayRef<Value> extraOperands,
51  ArrayRef<Value> symbolOperands,
52  bool allowNonDereferencingOps) {
53  unsigned newMemRefRank = newMemRef.getType().cast<MemRefType>().getRank();
54  (void)newMemRefRank; // unused in opt mode
55  unsigned oldMemRefRank = oldMemRef.getType().cast<MemRefType>().getRank();
56  (void)oldMemRefRank; // unused in opt mode
57  if (indexRemap) {
58  assert(indexRemap.getNumSymbols() == symbolOperands.size() &&
59  "symbolic operand count mismatch");
60  assert(indexRemap.getNumInputs() ==
61  extraOperands.size() + oldMemRefRank + symbolOperands.size());
62  assert(indexRemap.getNumResults() + extraIndices.size() == newMemRefRank);
63  } else {
64  assert(oldMemRefRank + extraIndices.size() == newMemRefRank);
65  }
66 
67  // Assert same elemental type.
68  assert(oldMemRef.getType().cast<MemRefType>().getElementType() ==
69  newMemRef.getType().cast<MemRefType>().getElementType());
70 
71  SmallVector<unsigned, 2> usePositions;
72  for (const auto &opEntry : llvm::enumerate(op->getOperands())) {
73  if (opEntry.value() == oldMemRef)
74  usePositions.push_back(opEntry.index());
75  }
76 
77  // If memref doesn't appear, nothing to do.
78  if (usePositions.empty())
79  return success();
80 
81  if (usePositions.size() > 1) {
82  // TODO: extend it for this case when needed (rare).
83  assert(false && "multiple dereferencing uses in a single op not supported");
84  return failure();
85  }
86 
87  unsigned memRefOperandPos = usePositions.front();
88 
89  OpBuilder builder(op);
90  // The following checks if op is dereferencing memref and performs the access
91  // index rewrites.
92  if (!isMemRefDereferencingOp(*op)) {
93  if (!allowNonDereferencingOps)
94  // Failure: memref used in a non-dereferencing context (potentially
95  // escapes); no replacement in these cases unless allowNonDereferencingOps
96  // is set.
97  return failure();
98  op->setOperand(memRefOperandPos, newMemRef);
99  return success();
100  }
101  // Perform index rewrites for the dereferencing op and then replace the op
102  NamedAttribute oldMapAttrPair = getAffineMapAttrForMemRef(op, oldMemRef);
103  AffineMap oldMap = oldMapAttrPair.second.cast<AffineMapAttr>().getValue();
104  unsigned oldMapNumInputs = oldMap.getNumInputs();
105  SmallVector<Value, 4> oldMapOperands(
106  op->operand_begin() + memRefOperandPos + 1,
107  op->operand_begin() + memRefOperandPos + 1 + oldMapNumInputs);
108 
109  // Apply 'oldMemRefOperands = oldMap(oldMapOperands)'.
110  SmallVector<Value, 4> oldMemRefOperands;
111  SmallVector<Value, 4> affineApplyOps;
112  oldMemRefOperands.reserve(oldMemRefRank);
113  if (oldMap != builder.getMultiDimIdentityMap(oldMap.getNumDims())) {
114  for (auto resultExpr : oldMap.getResults()) {
115  auto singleResMap = AffineMap::get(oldMap.getNumDims(),
116  oldMap.getNumSymbols(), resultExpr);
117  auto afOp = builder.create<AffineApplyOp>(op->getLoc(), singleResMap,
118  oldMapOperands);
119  oldMemRefOperands.push_back(afOp);
120  affineApplyOps.push_back(afOp);
121  }
122  } else {
123  oldMemRefOperands.assign(oldMapOperands.begin(), oldMapOperands.end());
124  }
125 
126  // Construct new indices as a remap of the old ones if a remapping has been
127  // provided. The indices of a memref come right after it, i.e.,
128  // at position memRefOperandPos + 1.
129  SmallVector<Value, 4> remapOperands;
130  remapOperands.reserve(extraOperands.size() + oldMemRefRank +
131  symbolOperands.size());
132  remapOperands.append(extraOperands.begin(), extraOperands.end());
133  remapOperands.append(oldMemRefOperands.begin(), oldMemRefOperands.end());
134  remapOperands.append(symbolOperands.begin(), symbolOperands.end());
135 
136  SmallVector<Value, 4> remapOutputs;
137  remapOutputs.reserve(oldMemRefRank);
138 
139  if (indexRemap &&
140  indexRemap != builder.getMultiDimIdentityMap(indexRemap.getNumDims())) {
141  // Remapped indices.
142  for (auto resultExpr : indexRemap.getResults()) {
143  auto singleResMap = AffineMap::get(
144  indexRemap.getNumDims(), indexRemap.getNumSymbols(), resultExpr);
145  auto afOp = builder.create<AffineApplyOp>(op->getLoc(), singleResMap,
146  remapOperands);
147  remapOutputs.push_back(afOp);
148  affineApplyOps.push_back(afOp);
149  }
150  } else {
151  // No remapping specified.
152  remapOutputs.assign(remapOperands.begin(), remapOperands.end());
153  }
154 
155  SmallVector<Value, 4> newMapOperands;
156  newMapOperands.reserve(newMemRefRank);
157 
158  // Prepend 'extraIndices' in 'newMapOperands'.
159  for (Value extraIndex : extraIndices) {
160  assert(extraIndex.getDefiningOp()->getNumResults() == 1 &&
161  "single result op's expected to generate these indices");
162  assert((isValidDim(extraIndex) || isValidSymbol(extraIndex)) &&
163  "invalid memory op index");
164  newMapOperands.push_back(extraIndex);
165  }
166 
167  // Append 'remapOutputs' to 'newMapOperands'.
168  newMapOperands.append(remapOutputs.begin(), remapOutputs.end());
169 
170  // Create new fully composed AffineMap for new op to be created.
171  assert(newMapOperands.size() == newMemRefRank);
172  auto newMap = builder.getMultiDimIdentityMap(newMemRefRank);
173  // TODO: Avoid creating/deleting temporary AffineApplyOps here.
174  fullyComposeAffineMapAndOperands(&newMap, &newMapOperands);
175  newMap = simplifyAffineMap(newMap);
176  canonicalizeMapAndOperands(&newMap, &newMapOperands);
177  // Remove any affine.apply's that became dead as a result of composition.
178  for (Value value : affineApplyOps)
179  if (value.use_empty())
180  value.getDefiningOp()->erase();
181 
182  OperationState state(op->getLoc(), op->getName());
183  // Construct the new operation using this memref.
184  state.operands.reserve(op->getNumOperands() + extraIndices.size());
185  // Insert the non-memref operands.
186  state.operands.append(op->operand_begin(),
187  op->operand_begin() + memRefOperandPos);
188  // Insert the new memref value.
189  state.operands.push_back(newMemRef);
190 
191  // Insert the new memref map operands.
192  state.operands.append(newMapOperands.begin(), newMapOperands.end());
193 
194  // Insert the remaining operands unmodified.
195  state.operands.append(op->operand_begin() + memRefOperandPos + 1 +
196  oldMapNumInputs,
197  op->operand_end());
198 
199  // Result types don't change. Both memref's are of the same elemental type.
200  state.types.reserve(op->getNumResults());
201  for (auto result : op->getResults())
202  state.types.push_back(result.getType());
203 
204  // Add attribute for 'newMap', other Attributes do not change.
205  auto newMapAttr = AffineMapAttr::get(newMap);
206  for (auto namedAttr : op->getAttrs()) {
207  if (namedAttr.first == oldMapAttrPair.first)
208  state.attributes.push_back({namedAttr.first, newMapAttr});
209  else
210  state.attributes.push_back(namedAttr);
211  }
212 
213  // Create the new operation.
214  auto *repOp = builder.createOperation(state);
215  op->replaceAllUsesWith(repOp);
216  op->erase();
217 
218  return success();
219 }

221 LogicalResult mlir::replaceAllMemRefUsesWith(
222  Value oldMemRef, Value newMemRef, ArrayRef<Value> extraIndices,
223  AffineMap indexRemap, ArrayRef<Value> extraOperands,
224  ArrayRef<Value> symbolOperands, Operation *domInstFilter,
225  Operation *postDomInstFilter, bool allowNonDereferencingOps,
226  bool replaceInDeallocOp) {
227  unsigned newMemRefRank = newMemRef.getType().cast<MemRefType>().getRank();
228  (void)newMemRefRank; // unused in opt mode
229  unsigned oldMemRefRank = oldMemRef.getType().cast<MemRefType>().getRank();
230  (void)oldMemRefRank;
231  if (indexRemap) {
232  assert(indexRemap.getNumSymbols() == symbolOperands.size() &&
233  "symbol operand count mismatch");
234  assert(indexRemap.getNumInputs() ==
235  extraOperands.size() + oldMemRefRank + symbolOperands.size());
236  assert(indexRemap.getNumResults() + extraIndices.size() == newMemRefRank);
237  } else {
238  assert(oldMemRefRank + extraIndices.size() == newMemRefRank);
239  }
240 
241  // Assert same elemental type.
242  assert(oldMemRef.getType().cast<MemRefType>().getElementType() ==
243  newMemRef.getType().cast<MemRefType>().getElementType());
244 
245  std::unique_ptr<DominanceInfo> domInfo;
246  std::unique_ptr<PostDominanceInfo> postDomInfo;
247  if (domInstFilter)
248  domInfo = std::make_unique<DominanceInfo>(
249  domInstFilter->getParentOfType<FuncOp>());
250 
251  if (postDomInstFilter)
252  postDomInfo = std::make_unique<PostDominanceInfo>(
253  postDomInstFilter->getParentOfType<FuncOp>());
254 
255  // Walk all uses of old memref; collect ops to perform replacement. We use a
256  // DenseSet since an operation could potentially have multiple uses of a
257  // memref (although rare), and the replacement later is going to erase ops.
258  DenseSet<Operation *> opsToReplace;
259  for (auto *op : oldMemRef.getUsers()) {
260  // Skip this use if it's not dominated by domInstFilter.
261  if (domInstFilter && !domInfo->dominates(domInstFilter, op))
262  continue;
263 
264  // Skip this use if it's not post-dominated by postDomInstFilter.
265  if (postDomInstFilter && !postDomInfo->postDominates(postDomInstFilter, op))
266  continue;
267 
268  // Skip dealloc's - no replacement is necessary, and a memref replacement
269  // at other uses doesn't hurt these dealloc's.
270  if (isa<DeallocOp>(op) && !replaceInDeallocOp)
271  continue;
272 
273  // Check if the memref was used in a non-dereferencing context. It is fine
274  // for the memref to be used in a non-dereferencing way outside of the
275  // region where this replacement is happening.
276  if (!isMemRefDereferencingOp(*op)) {
277  // Currently we support the following non-dereferencing types to be a
278  // candidate for replacement: Dealloc and CallOp.
279  // TODO: Add support for other kinds of ops.
280  if (!allowNonDereferencingOps)
281  return failure();
282  if (!(isa<DeallocOp, CallOp>(*op)))
283  return failure();
284  }
285 
286  // We'll first collect and then replace --- since replacement erases the op
287  // that has the use, and that op could be postDomFilter or domFilter itself!
288  opsToReplace.insert(op);
289  }
290 
291  for (auto *op : opsToReplace) {
292  if (failed(replaceAllMemRefUsesWith(
293  oldMemRef, newMemRef, op, extraIndices, indexRemap, extraOperands,
294  symbolOperands, allowNonDereferencingOps)))
295  llvm_unreachable("memref replacement guaranteed to succeed here");
296  }
297 
298  return success();
299 }

301 /// Given an operation, inserts one or more single result affine
302 /// apply operations, results of which are exclusively used by this operation
303 /// operation. The operands of these newly created affine apply ops are
304 /// guaranteed to be loop iterators or terminal symbols of a function.
305 ///
306 /// Before
307 ///
308 /// affine.for %i = 0 to #map(%N)
309 /// %idx = affine.apply (d0) -> (d0 mod 2) (%i)
310 /// "send"(%idx, %A, ...)
311 /// "compute"(%idx)
312 ///
313 /// After
314 ///
315 /// affine.for %i = 0 to #map(%N)
316 /// %idx = affine.apply (d0) -> (d0 mod 2) (%i)
317 /// "send"(%idx, %A, ...)
318 /// %idx_ = affine.apply (d0) -> (d0 mod 2) (%i)
319 /// "compute"(%idx_)
320 ///
321 /// This allows applying different transformations on send and compute (for eg.
322 /// different shifts/delays).
323 ///
324 /// Returns nullptr either if none of opInst's operands were the result of an
325 /// affine.apply and thus there was no affine computation slice to create, or if
326 /// all the affine.apply op's supplying operands to this opInst did not have any
327 /// uses besides this opInst; otherwise returns the list of affine.apply
328 /// operations created in output argument `sliceOps`.
329 void mlir::createAffineComputationSlice(
330  Operation *opInst, SmallVectorImpl<AffineApplyOp> *sliceOps) {
331  // Collect all operands that are results of affine apply ops.
332  SmallVector<Value, 4> subOperands;
333  subOperands.reserve(opInst->getNumOperands());
334  for (auto operand : opInst->getOperands())
335  if (isa_and_nonnull<AffineApplyOp>(operand.getDefiningOp()))
336  subOperands.push_back(operand);
337 
338  // Gather sequence of AffineApplyOps reachable from 'subOperands'.
339  SmallVector<Operation *, 4> affineApplyOps;
340  getReachableAffineApplyOps(subOperands, affineApplyOps);
341  // Skip transforming if there are no affine maps to compose.
342  if (affineApplyOps.empty())
343  return;
344 
345  // Check if all uses of the affine apply op's lie only in this op op, in
346  // which case there would be nothing to do.
347  bool localized = true;
348  for (auto *op : affineApplyOps) {
349  for (auto result : op->getResults()) {
350  for (auto *user : result.getUsers()) {
351  if (user != opInst) {
352  localized = false;
353  break;
354  }
355  }
356  }
357  }
358  if (localized)
359  return;
360 
361  OpBuilder builder(opInst);
362  SmallVector<Value, 4> composedOpOperands(subOperands);
363  auto composedMap = builder.getMultiDimIdentityMap(composedOpOperands.size());
364  fullyComposeAffineMapAndOperands(&composedMap, &composedOpOperands);
365 
366  // Create an affine.apply for each of the map results.
367  sliceOps->reserve(composedMap.getNumResults());
368  for (auto resultExpr : composedMap.getResults()) {
369  auto singleResMap = AffineMap::get(composedMap.getNumDims(),
370  composedMap.getNumSymbols(), resultExpr);
371  sliceOps->push_back(builder.create<AffineApplyOp>(
372  opInst->getLoc(), singleResMap, composedOpOperands));
373  }
374 
375  // Construct the new operands that include the results from the composed
376  // affine apply op above instead of existing ones (subOperands). So, they
377  // differ from opInst's operands only for those operands in 'subOperands', for
378  // which they will be replaced by the corresponding one from 'sliceOps'.
379  SmallVector<Value, 4> newOperands(opInst->getOperands());
380  for (unsigned i = 0, e = newOperands.size(); i < e; i++) {
381  // Replace the subOperands from among the new operands.
382  unsigned j, f;
383  for (j = 0, f = subOperands.size(); j < f; j++) {
384  if (newOperands[i] == subOperands[j])
385  break;
386  }
387  if (j < subOperands.size()) {
388  newOperands[i] = (*sliceOps)[j];
389  }
390  }
391  for (unsigned idx = 0, e = newOperands.size(); idx < e; idx++) {
392  opInst->setOperand(idx, newOperands[idx]);
393  }
394 }

396 // TODO: Currently works for static memrefs with a single layout map.
397 LogicalResult mlir::normalizeMemRef(AllocOp allocOp) {
398  MemRefType memrefType = allocOp.getType();
399  OpBuilder b(allocOp);
400 
401  // Fetch a new memref type after normalizing the old memref to have an
402  // identity map layout.
403  MemRefType newMemRefType =
404  normalizeMemRefType(memrefType, b, allocOp.getNumSymbolicOperands());
405  if (newMemRefType == memrefType)
406  // Either memrefType already had an identity map or the map couldn't be
407  // transformed to an identity map.
408  return failure();
409 
410  Value oldMemRef = allocOp.getResult();
411 
412  SmallVector<Value, 4> symbolOperands(allocOp.getSymbolicOperands());
413  AllocOp newAlloc = b.create<AllocOp>(allocOp.getLoc(), newMemRefType,
414  llvm::None, allocOp.alignmentAttr());
415  AffineMap layoutMap = memrefType.getAffineMaps().front();
416  // Replace all uses of the old memref.
417  if (failed(replaceAllMemRefUsesWith(oldMemRef, /*newMemRef=*/newAlloc,
418  /*extraIndices=*/{},
419  /*indexRemap=*/layoutMap,
420  /*extraOperands=*/{},
421  /*symbolOperands=*/symbolOperands,
422  /*domInstFilter=*/nullptr,
423  /*postDomInstFilter=*/nullptr,
424  /*allowDereferencingOps=*/true))) {
425  // If it failed (due to escapes for example), bail out.
426  newAlloc.erase();
427  return failure();
428  }
429  // Replace any uses of the original alloc op and erase it. All remaining uses
430  // have to be dealloc's; RAMUW above would've failed otherwise.
431  assert(llvm::all_of(oldMemRef.getUsers(),
432  [](Operation *op) { return isa<DeallocOp>(op); }));
433  oldMemRef.replaceAllUsesWith(newAlloc);
434  allocOp.erase();
435  return success();
436 }

438 MemRefType mlir::normalizeMemRefType(MemRefType memrefType, OpBuilder b,
439  unsigned numSymbolicOperands) {
440  unsigned rank = memrefType.getRank();
441  if (rank == 0)
442  return memrefType;
443 
444  ArrayRef<AffineMap> layoutMaps = memrefType.getAffineMaps();
445  if (layoutMaps.empty() ||
446  layoutMaps.front() == b.getMultiDimIdentityMap(rank)) {
447  // Either no maps is associated with this memref or this memref has
448  // a trivial (identity) map.
449  return memrefType;
450  }
451 
452  // We don't do any checks for one-to-one'ness; we assume that it is
453  // one-to-one.
454 
455  // TODO: Only for static memref's for now.
456  if (memrefType.getNumDynamicDims() > 0)
457  return memrefType;
458 
459  // We have a single map that is not an identity map. Create a new memref
460  // with the right shape and an identity layout map.
461  ArrayRef<int64_t> shape = memrefType.getShape();
462  // FlatAffineConstraint may later on use symbolicOperands.
463  FlatAffineConstraints fac(rank, numSymbolicOperands);
464  for (unsigned d = 0; d < rank; ++d) {
465  fac.addConstantLowerBound(d, 0);
466  fac.addConstantUpperBound(d, shape[d] - 1);
467  }
468  // We compose this map with the original index (logical) space to derive
469  // the upper bounds for the new index space.
470  AffineMap layoutMap = layoutMaps.front();
471  unsigned newRank = layoutMap.getNumResults();
472  if (failed(fac.composeMatchingMap(layoutMap)))
473  return memrefType;
474  // TODO: Handle semi-affine maps.
475  // Project out the old data dimensions.
476  fac.projectOut(newRank, fac.getNumIds() - newRank - fac.getNumLocalIds());
477  SmallVector<int64_t, 4> newShape(newRank);
478  for (unsigned d = 0; d < newRank; ++d) {
479  // The lower bound for the shape is always zero.
480  auto ubConst = fac.getConstantUpperBound(d);
481  // For a static memref and an affine map with no symbols, this is
482  // always bounded.
483  assert(ubConst.hasValue() && "should always have an upper bound");
484  if (ubConst.getValue() < 0)
485  // This is due to an invalid map that maps to a negative space.
486  return memrefType;
487  newShape[d] = ubConst.getValue() + 1;
488  }
489 
490  // Create the new memref type after trivializing the old layout map.
491  MemRefType newMemRefType =
492  MemRefType::Builder(memrefType)
493  .setShape(newShape)
494  .setAffineMaps(b.getMultiDimIdentityMap(newRank));
495 
496  return newMemRefType;
497 }
