153 static ParseResult parseGenericOp(OpAsmParser &parser, OperationState &result) {
154  SmallVector<OpAsmParser::OperandType, 8> operandsInfo, regionOperandsInfo;
155  DictionaryAttr dictAttr;
156  // Parse the core linalg traits that must check into a dictAttr.
157  // The name is unimportant as we will overwrite result.attributes.
158  // The core linalg traits must contain the information necessary to pass the
159  // verifier.
160  if (parser.parseAttribute(dictAttr, "_", result.attributes))
161  return failure();
162  result.attributes.assign(dictAttr.getValue().begin(),
163  dictAttr.getValue().end());
164 
165  // Optional attributes may be added.
166  if (parser.parseOptionalAttrDict(result.attributes) ||
167  parser.parseOperandList(operandsInfo))
168  return failure();
169 
170  Region &region = *result.addRegion();
171  SmallVector<Type, 8> operandTypes, regionTypes;
172  if (parser.parseRegion(region, regionOperandsInfo, regionTypes))
173  return failure();
174  if (parser.parseColonTypeList(operandTypes))
175  return failure();
176  // Generic ops may specify that a subset of its outputs are tensors. Such
177  // outputs are specified in the result type.
178  SmallVector<Type, 8> tensorResultTypes;
179  if (parser.parseOptionalArrowTypeList(tensorResultTypes))
180  return failure();
181  if (!tensorResultTypes.empty())
182  result.addTypes(tensorResultTypes);
183  return parser.resolveOperands(operandsInfo, operandTypes,
184  parser.getCurrentLocation(), result.operands);
185 }

193 template <typename GenericOpType>
194 LogicalResult BlockArgsVerifier<GenericOpType>::verify(GenericOpType op,
195  Block &block) {
196  auto nOperands = op.getNumOperands();
197  if (block.getNumArguments() != nOperands)
198  return op.emitOpError("expected number of block arguments to match number "
199  "of operands");
200 
201  // Note: the number and type of yield values are checked in the YieldOp.
202  auto nInputViews = op.getNumInputs();
203  for (unsigned i = 0; i < nOperands; ++i) {
204  auto viewType = op.getShapedType(i);
205  if (viewType.getElementType() != block.getArgument(i).getType())
206  return op.emitOpError("expected block argument ")
207  << (i + 1) << " of the same type as elemental type of "
208  << ((i < nInputViews) ? "input " : "output ")
209  << "operand: " << viewType;
210  }
211  return success();
212 }

217  auto nInputViews = op.getNumInputs();
218  auto nLoops = op.getNumLoops();

240  << "operand: " << viewType;

246 template <typename GenericOpType>
247 static LogicalResult verifyGenericOp(GenericOpType op) {
248  auto nInputViews = op.getNumInputs();
249  auto nLoops = op.getNumLoops();
250  auto nInputsAndOutputBuffers = op.getNumInputsAndOutputBuffers();
251  if (nInputsAndOutputBuffers != llvm::size(op.views()))
252  return op.emitOpError("expected exactly ")
253  << nInputsAndOutputBuffers
254  << " inputs (tensor or buffer) and output buffer operands";
255 
256  auto &region = op.region();
257  if (!llvm::hasSingleElement(region))
258  return op.emitOpError("expected region with 1 block");
259  if (failed(BlockArgsVerifier<GenericOpType>::verify(op, region.front())))
260  return failure();
261 
262  auto attr = op.template getAttrOfType<IntegerAttr>("symbol_source");
263  int64_t targetRank = 0;
264  if (attr) {
265  unsigned index = attr.getInt();
266  if (index >= op.getNumOperands())
267  return op.emitOpError("symbol_source index out of range");
268  targetRank = op.getShapedType(index).getRank();
269  }
270 
271  SmallVector<AffineMap, 4> indexingMaps;
272  indexingMaps.reserve(op.indexing_maps().size());
273  for (auto en : llvm::enumerate(op.indexing_maps())) {
274  auto idx = en.index();
275  auto m = en.value().template cast<AffineMapAttr>().getValue();
276  indexingMaps.push_back(m); // Save reference to map for further checks.
277  auto view = (idx < nInputViews) ? op.getInputShapedType(idx)
278  : op.getOutputShapedType(idx - nInputViews);
279 
280  if (m.getNumSymbols() != targetRank)
281  return op.emitOpError("expected the number of symbols in indexing_map #")
282  << idx << " to match target rank";
283 
284  if (m.getNumDims() != nLoops)
285  return op.emitOpError("expected indexing_map #")
286  << idx << " to have " << nLoops
287  << " dim(s) to match the number of loops";
288 
289  if (m.getNumResults() != view.getRank())
290  return op.emitOpError("expected indexing_map #")
291  << idx << " results to match view rank: " << view;
292  }
293 
294  auto concatMap = concatAffineMaps(indexingMaps);
295  // TODO: Bound inference for maps with symbols
296  if (!concatMap.getNumSymbols() && !inversePermutation(concatMap))
297  return op.emitOpError("expected the concatenation of maps in indexing_map "
298  "to be invertible");
299 
300  return success();
301 }

310 /// Collapse reassociation maps that are used in pair of reshape ops where one
311 /// is a producer and other is the consumer. Only valid to use this method when
312 /// both the producer and consumer are collapsing dimensions or both are
313 /// expanding dimensions.
314 ///
315 /// For example,
316 /// mapsProducer = [affine_map<(d0, d1, d2, d3, d4) -> (d0, d1)>,
317 /// affine_map<(d0, d1, d2, d3, d4) -> (d2)>,
318 /// affine_map<(d0, d1, d2, d3, d4) -> (d3, d4)>]
319 /// mapsConsumer = [affine_map<(d0, d1, d2) -> (d0, d1)>,
320 /// affine_map<(d0, d1, d2) -> (d2)>]
321 ///
322 /// is folded into
323 ///
324 /// result = [affine_map<(d0, d1, d2, d3, d4) -> (d0, d1, d2)>,
325 /// affine_map<(d0, d1, d2, d3, d4) -> (d3, d4)>]
326 static ArrayAttr collapseReassociationMaps(ArrayRef<AffineMap> mapsProducer,
327  ArrayRef<AffineMap> mapsConsumer,
328  MLIRContext *context) {
329  if (mapsProducer.empty() || mapsConsumer.empty() ||
330  mapsProducer[0].getNumDims() < mapsConsumer[0].getNumDims() ||
331  mapsProducer.size() != mapsConsumer[0].getNumDims())
332  return nullptr;
333  unsigned numLhsDims = mapsProducer[0].getNumDims();
334  unsigned currDim = 0;
335  SmallVector<AffineExpr, 4> reassociations;
336  SmallVector<Attribute, 4> reassociationMaps;
337  for (AffineMap rhs : mapsConsumer) {
338  for (AffineExpr rhsExpr : rhs.getResults()) {
339  AffineDimExpr dimExpr = rhsExpr.cast<AffineDimExpr>();
340  for (int i = 0, e = mapsProducer[dimExpr.getPosition()].getNumResults();
341  i < e; ++i) {
342  reassociations.push_back(getAffineDimExpr(currDim++, context));
343  }
344  }
345  reassociationMaps.push_back(AffineMapAttr::get(AffineMap::get(
346  numLhsDims, /*numSymbols =*/0, reassociations, context)));
347  reassociations.clear();
348  }
349  return ArrayAttr::get(reassociationMaps, context);
350 }

353 /// Pattern to collapse producer/consumer reshape ops that are both collapsing
354 /// dimensions or are both expanding dimensions.
355 template <typename ReshapeOpTy>
356 struct CollapseReshapeOps : public OpRewritePattern<ReshapeOpTy> {
357  using OpRewritePattern<ReshapeOpTy>::OpRewritePattern;
358  LogicalResult matchAndRewrite(ReshapeOpTy reshapeOp,
359  PatternRewriter &rewriter) const override {
360  auto srcReshapeOp = reshapeOp.src().template getDefiningOp<ReshapeOpTy>();
361  if (!srcReshapeOp)
362  return failure();
363 
364  auto areReshapeOpsFoldable = [](ShapedType largerType,
365  ShapedType intermediateType,
366  ShapedType smallerType) -> bool {
367  return largerType.getRank() > intermediateType.getRank() &&
368  intermediateType.getRank() > smallerType.getRank() &&
369  smallerType.getRank() > 0;
370  };
371  // Check if producer and consumer are both expanding dims.
372  if (areReshapeOpsFoldable(reshapeOp.getResultType(), reshapeOp.getSrcType(),
373  srcReshapeOp.getSrcType())) {
374  rewriter.replaceOpWithNewOp<ReshapeOpTy>(
375  reshapeOp, reshapeOp.getResultType(), srcReshapeOp.src(),
376  collapseReassociationMaps(reshapeOp.getReassociationMaps(),
377  srcReshapeOp.getReassociationMaps(),
378  rewriter.getContext()));
379  return success();
380  }
381  // Check if producer and consumer are both collapsing dims.
382  else if (areReshapeOpsFoldable(srcReshapeOp.getSrcType(),
383  reshapeOp.getSrcType(),
384  reshapeOp.getResultType())) {
385  rewriter.replaceOpWithNewOp<ReshapeOpTy>(
386  reshapeOp, reshapeOp.getResultType(), srcReshapeOp.src(),
387  collapseReassociationMaps(srcReshapeOp.getReassociationMaps(),
388  reshapeOp.getReassociationMaps(),
389  rewriter.getContext()));
390  return success();
391  }
392  return failure();
393  }
394 };

398 static OpFoldResult foldReshapeOp(ReshapeOpTy reshapeOp) {
399  // Fold producer-consumer reshape ops that where the operand type of the
400  // producer is same as the return type of the consumer. This can only be
401  // verified if the shapes in question are static.

411 /// Return true if the reassociation specification is valid, false otherwise.
412 /// When false, the `invalidIndex` integer pointer is optionally filled with the
413 /// index of the offending reassociation map.
414 static bool isReassociationValid(ArrayRef<AffineMap> reassociation,
415  int *invalidIndex = nullptr) {
416  if (reassociation.empty())
417  return true;
418  unsigned nDims = reassociation[0].getNumDims();
419  unsigned nextExpectedDim = 0;
420  for (auto it : llvm::enumerate(reassociation)) {
421  auto m = it.value();
422  if (m.getNumDims() != nDims || m.getNumSymbols() != 0) {
423  if (invalidIndex)
424  *invalidIndex = it.index();
425  return false;
426  }
427  for (auto e : m.getResults()) {
428  auto d = e.dyn_cast<AffineDimExpr>();
429  if (!d || d.getPosition() != nextExpectedDim++) {
430  if (invalidIndex)
431  *invalidIndex = it.index();
432  return false;
433  }
434  }
435  }
436  if (nextExpectedDim != nDims) {
437  if (invalidIndex)
438  *invalidIndex = reassociation.size() - 1;
439  return false;
440  }
441  return true;
442 }

444 /// Detect whether memref dims [dim, dim + extent) can be reshaped without
445 /// copies.
446 static bool isReshapableDimBand(unsigned dim, unsigned extent,
447  ArrayRef<int64_t> sizes,
448  ArrayRef<AffineExpr> strides) {
449  assert(sizes.size() == strides.size() && "mismatched ranks");
450  // off by 1 indexing to avoid out of bounds
451  // V
452  for (auto idx = dim, e = dim + extent; idx + 1 < e; ++idx) {
453  // Only bands of static shapes are reshapable. This is due to the fact that
454  // there is no relation between dynamic sizes and dynamic strides: we do not
455  // have enough information to know whether a "-1" size corresponds to the
456  // proper symbol in the AffineExpr of a stride.
457  if (ShapedType::isDynamic(sizes[dim + 1]))
458  return false;
459  // TODO: Refine this by passing the proper nDims and nSymbols so we can
460  // simplify on the fly and catch more reshapable cases.
461  if (strides[idx] != strides[idx + 1] * sizes[idx + 1])
462  return false;
463  }
464  return true;
465 }

467 /// Compute the MemRefType obtained by applying the `reassociation` (which is
468 /// expected to be valid) to `type`.
469 /// If `type` is Contiguous MemRefType, this always produce a contiguous
470 /// MemRefType.
471 static MemRefType
472 computeReshapeCollapsedType(MemRefType type,
473  ArrayRef<AffineMap> reassociation) {
474  auto sizes = type.getShape();
475  AffineExpr offset;
476  SmallVector<AffineExpr, 4> strides;
477  auto status = getStridesAndOffset(type, strides, offset);
478  (void)status;
479  assert(succeeded(status) && "expected strided memref");
480 
481  SmallVector<int64_t, 4> newSizes;
482  newSizes.reserve(reassociation.size());
483  SmallVector<AffineExpr, 4> newStrides;
484  newStrides.reserve(reassociation.size());
485 
486  // Use the fact that reassociation is valid to simplify the logic: only use
487  // each map's rank.
488  assert(isReassociationValid(reassociation) && "invalid reassociation");
489  unsigned currentDim = 0;
490  for (AffineMap m : reassociation) {
491  unsigned dim = m.getNumResults();
492  int64_t size = 1;
493  AffineExpr stride = strides[currentDim + dim - 1];
494  if (!isReshapableDimBand(currentDim, dim, sizes, strides)) {
495  size = ShapedType::kDynamicSize;
496  stride = AffineExpr();
497  } else {
498  for (unsigned d = 0; d < dim; ++d)
499  size *= sizes[currentDim + d];
500  }
501  newSizes.push_back(size);
502  newStrides.push_back(stride);
503  currentDim += dim;
504  }
505 
506  // Early-exit: if `type` is contiguous, the result must be contiguous.
507  if (canonicalizeStridedLayout(type).getAffineMaps().empty())
508  return MemRefType::Builder(type).setShape(newSizes).setAffineMaps({});
509 
510  // Convert back to int64_t because we don't have enough information to create
511  // new strided layouts from AffineExpr only. This corresponds to a case where
512  // copies may be necessary.
513  int64_t intOffset = ShapedType::kDynamicStrideOrOffset;
514  if (auto o = offset.dyn_cast<AffineConstantExpr>())
515  intOffset = o.getValue();
516  SmallVector<int64_t, 4> intStrides;
517  intStrides.reserve(strides.size());
518  for (auto stride : newStrides) {
519  if (auto cst = stride.dyn_cast_or_null<AffineConstantExpr>())
520  intStrides.push_back(cst.getValue());
521  else
522  intStrides.push_back(ShapedType::kDynamicStrideOrOffset);
523  }
524  auto layout =
525  makeStridedLinearLayoutMap(intStrides, intOffset, type.getContext());
526  return canonicalizeStridedLayout(
527  MemRefType::Builder(type).setShape(newSizes).setAffineMaps({layout}));
528 }

605 // Common verifier for reshape-like types. Fills `expandedType` and
606 // `collapsedType` with the proper `src` or `result` type.
607 template <typename Op, typename T>
608 static LogicalResult verifyReshapeLikeTypes(Op op, T &expandedType,
609  T &collapsedType) {
610  expandedType = op.getSrcType();
611  collapsedType = op.getResultType();
612  unsigned expandedRank = expandedType.getRank();
613  unsigned collapsedRank = collapsedType.getRank();
614  bool isCollapse = expandedRank > collapsedRank;
615  if (!isCollapse) {
616  std::swap(expandedRank, collapsedRank);
617  std::swap(expandedType, collapsedType);
618  }
619  if (expandedRank == 0)
620  return op.emitOpError("expected non-zero memref ranks");
621  if (expandedRank == collapsedRank)
622  return op.emitOpError("expected to collapse or expand dims");
623 
624  if (collapsedRank == 0) {
625  // If collapsed rank is 0, then expanded type must be static shaped and of
626  // sizes 1.
627  if (llvm::any_of(expandedType.getShape(),
628  [](int64_t dim) -> bool { return dim != 1; }))
629  return op.emitOpError(
630  "invalid to reshape tensor/memref with non-unit extent dimensions to "
631  "zero-rank tensor/memref");
632  return success();
633  }
634  if (collapsedRank != op.reassociation().size())
635  return op.emitOpError("expected rank of the collapsed type(")
636  << collapsedRank << ") to be the number of reassociation maps("
637  << op.reassociation().size() << ")";
638  auto maps = getAffineMaps(op.reassociation());
639  for (auto it : llvm::enumerate(maps))
640  if (it.value().getNumDims() != expandedRank)
641  return op.emitOpError("expected reassociation map #")
642  << it.index() << " of same rank as expanded memref("
643  << expandedRank << "), but got " << it.value().getNumDims();
644  int invalidIdx = 0;
645  if (!isReassociationValid(maps, &invalidIdx))
646  return op.emitOpError("expected reassociation map #")
647  << invalidIdx << " to be valid and contiguous";
648  return success();
649 }

672 /// Compute the RankedTensorType obtained by applying `reassociation` to `type`.
673 static RankedTensorType
674 computeTensorReshapeCollapsedType(RankedTensorType type,
675  ArrayRef<AffineMap> reassociation) {
676  auto shape = type.getShape();
677  SmallVector<int64_t, 4> newShape;
678  newShape.reserve(reassociation.size());
679 
680  // Use the fact that reassociation is valid to simplify the logic: only use
681  // each map's rank.
682  assert(isReassociationValid(reassociation) && "invalid reassociation");
683  unsigned currentDim = 0;
684  for (AffineMap m : reassociation) {
685  unsigned dim = m.getNumResults();
686  auto band = shape.slice(currentDim, dim);
687  int64_t size = 1;
688  if (llvm::is_contained(band, ShapedType::kDynamicSize))
689  size = ShapedType::kDynamicSize;
690  else
691  for (unsigned d = 0; d < dim; ++d)
692  size *= shape[currentDim + d];
693  newShape.push_back(size);
694  currentDim += dim;
695  }
696 
697  return RankedTensorType::get(newShape, type.getElementType());
698 }

1225  OperationState &result) {
1226  SmallVector<OpAsmParser::OperandType, 8> operandsInfo;
1227 
1228  // Optional attributes may be added.
1229  if (parser.parseOperandList(operandsInfo) ||
1230  parser.parseOptionalAttrDict(result.attributes))
1231  return failure();
1232

1234  if (parser.parseColon() || parser.parseLParen() ||
1235  parser.parseTypeList(operandTypes) || parser.parseRParen())
1236  return failure();

1241  if (parser.parseOptionalArrowTypeList(tensorResultTypes))

1246

1248  // the number of expected arguments for the current operation.
1249  auto parsedArgs = operandsInfo.size();
1250  auto expectedArgs = NamedStructuredOpType::getNumInputs() +
1251  NamedStructuredOpType::getNumOutputs();

1253  return parser.emitError(parser.getNameLoc(),
1254  "expects " + std::to_string(expectedArgs) +

1256  std::to_string(parsedArgs));
1257 
1258  buildNamedStructuredOpRegionAndAttributes<NamedStructuredOpType>(
1259  parser.getBuilder(), result, operandTypes, tensorResultTypes);
1260

1262  parser.getCurrentLocation(), result.operands);
