53  Builder builder) {
54  // If the source number uses less active bits than the target bitwidth, then
55  // it should be safe to convert.
56  if (srcAttr.getValue().isIntN(dstType.getWidth()))

77 /// Converts the given `srcAttr` to a new attribute of the given `dstType`.
78 /// Returns null attribute if `dstType` is not 32-bit or conversion fails.
79 static FloatAttr convertFloatAttr(FloatAttr srcAttr, FloatType dstType,
80  Builder builder) {
81  // Only support converting to float for now.
82  if (!dstType.isF32())
83  return FloatAttr();
84 
85  // Try to convert the source floating-point number to single precision.
86  APFloat dstVal = srcAttr.getValue();
87  bool losesInfo = false;
88  APFloat::opStatus status =
89  dstVal.convert(APFloat::IEEEsingle(), APFloat::rmTowardZero, &losesInfo);
90  if (status != APFloat::opOK || losesInfo) {
91  LLVM_DEBUG(llvm::dbgs()
92  << srcAttr << " illegal: cannot fit into converted type '"
93  << dstType << "'\n");
94  return FloatAttr();
95  }
96 
97  return builder.getF32FloatAttr(dstVal.convertToFloat());
98 }

100 /// Returns signed remainder for `lhs` and `rhs` and lets the result follow
101 /// the sign of `signOperand`.
102 ///
103 /// Note that this is needed for Vulkan. Per the Vulkan's SPIR-V environment
104 /// spec, "for the OpSRem and OpSMod instructions, if either operand is negative
105 /// the result is undefined." So we cannot directly use spv.SRem/spv.SMod
106 /// if either operand can be negative. Emulate it via spv.UMod.
107 static Value emulateSignedRemainder(Location loc, Value lhs, Value rhs,
108  Value signOperand, OpBuilder &builder) {
109  assert(lhs.getType() == rhs.getType());
110  assert(lhs == signOperand || rhs == signOperand);
111 
112  Type type = lhs.getType();
113 
114  // Calculate the remainder with spv.UMod.
115  Value lhsAbs = builder.create<spirv::GLSLSAbsOp>(loc, type, lhs);
116  Value rhsAbs = builder.create<spirv::GLSLSAbsOp>(loc, type, rhs);
117  Value abs = builder.create<spirv::UModOp>(loc, lhsAbs, rhsAbs);
118 
119  // Fix the sign.
120  Value isPositive;
121  if (lhs == signOperand)
122  isPositive = builder.create<spirv::IEqualOp>(loc, lhs, lhsAbs);
123  else
124  isPositive = builder.create<spirv::IEqualOp>(loc, rhs, rhsAbs);
125  Value absNegate = builder.create<spirv::SNegateOp>(loc, type, abs);
126  return builder.create<spirv::SelectOp>(loc, type, isPositive, abs, absNegate);
127 }

129 /// Returns the offset of the value in `targetBits` representation.
130 ///
131 /// `srcIdx` is an index into a 1-D array with each element having `sourceBits`.
132 /// It's assumed to be non-negative.
133 ///
134 /// When accessing an element in the array treating as having elements of
135 /// `targetBits`, multiple values are loaded in the same time. The method
136 /// returns the offset where the `srcIdx` locates in the value. For example, if
137 /// `sourceBits` equals to 8 and `targetBits` equals to 32, the x-th element is
138 /// located at (x % 4) * 8. Because there are four elements in one i32, and one
139 /// element has 8 bits.
140 static Value getOffsetForBitwidth(Location loc, Value srcIdx, int sourceBits,
141  int targetBits, OpBuilder &builder) {
142  assert(targetBits % sourceBits == 0);
143  IntegerType targetType = builder.getIntegerType(targetBits);
144  IntegerAttr idxAttr =
145  builder.getIntegerAttr(targetType, targetBits / sourceBits);
146  auto idx = builder.create<spirv::ConstantOp>(loc, targetType, idxAttr);
147  IntegerAttr srcBitsAttr = builder.getIntegerAttr(targetType, sourceBits);
148  auto srcBitsValue =
149  builder.create<spirv::ConstantOp>(loc, targetType, srcBitsAttr);
150  auto m = builder.create<spirv::UModOp>(loc, srcIdx, idx);
151  return builder.create<spirv::IMulOp>(loc, targetType, m, srcBitsValue);
152 }

154 /// Returns an adjusted spirv::AccessChainOp. Based on the
155 /// extension/capabilities, certain integer bitwidths `sourceBits` might not be
156 /// supported. During conversion if a memref of an unsupported type is used,
157 /// load/stores to this memref need to be modified to use a supported higher
158 /// bitwidth `targetBits` and extracting the required bits. For an accessing a
159 /// 1D array (spv.array or spv.rt_array), the last index is modified to load the
160 /// bits needed. The extraction of the actual bits needed are handled
161 /// separately. Note that this only works for a 1-D tensor.
162 static Value adjustAccessChainForBitwidth(SPIRVTypeConverter &typeConverter,
163  spirv::AccessChainOp op,
164  int sourceBits, int targetBits,
165  OpBuilder &builder) {
166  assert(targetBits % sourceBits == 0);
167  const auto loc = op.getLoc();
168  IntegerType targetType = builder.getIntegerType(targetBits);
169  IntegerAttr attr =
170  builder.getIntegerAttr(targetType, targetBits / sourceBits);
171  auto idx = builder.create<spirv::ConstantOp>(loc, targetType, attr);
172  auto lastDim = op.getOperation()->getOperand(op.getNumOperands() - 1);
173  auto indices = llvm::to_vector<4>(op.indices());
174  // There are two elements if this is a 1-D tensor.
175  assert(indices.size() == 2);
176  indices.back() = builder.create<spirv::SDivOp>(loc, lastDim, idx);
177  Type t = typeConverter.convertType(op.component_ptr().getType());
178  return builder.create<spirv::AccessChainOp>(loc, t, op.base_ptr(), indices);
179 }

220 /// Returns true if the allocations of type `t` can be lowered to SPIR-V.
221 static bool isAllocationSupported(MemRefType t) {
222  // Currently only support workgroup local memory allocations with static
223  // shape and int or float or vector of int or float element type.
224  if (!(t.hasStaticShape() &&
225  SPIRVTypeConverter::getMemorySpaceForStorageClass(
226  spirv::StorageClass::Workgroup) == t.getMemorySpace()))
227  return false;
228  Type elementType = t.getElementType();
229  if (auto vecType = elementType.dyn_cast<VectorType>())
230  elementType = vecType.getElementType();
231  return elementType.isIntOrFloat();
232 }

263 /// Converts an allocation operation to SPIR-V. Currently only supports lowering
264 /// to Workgroup memory when the size is constant. Note that this pattern needs
265 /// to be applied in a pass that runs at least at spv.module scope since it wil
266 /// ladd global variables into the spv.module.
267 class AllocOpPattern final : public SPIRVOpLowering<AllocOp> {
268 public:
269  using SPIRVOpLowering<AllocOp>::SPIRVOpLowering;
270 
271  LogicalResult
272  matchAndRewrite(AllocOp operation, ArrayRef<Value> operands,
273  ConversionPatternRewriter &rewriter) const override {
274  MemRefType allocType = operation.getType();
275  if (!isAllocationSupported(allocType))
276  return operation.emitError("unhandled allocation type");
277 
278  // Get the SPIR-V type for the allocation.
279  Type spirvType = typeConverter.convertType(allocType);
280 
281  // Insert spv.globalVariable for this allocation.
282  Operation *parent =
283  SymbolTable::getNearestSymbolTable(operation.getParentOp());
284  if (!parent)
285  return failure();
286  Location loc = operation.getLoc();
287  spirv::GlobalVariableOp varOp;
288  {
289  OpBuilder::InsertionGuard guard(rewriter);
290  Block &entryBlock = *parent->getRegion(0).begin();
291  rewriter.setInsertionPointToStart(&entryBlock);
292  auto varOps = entryBlock.getOps<spirv::GlobalVariableOp>();
293  std::string varName =
294  std::string("__workgroup_mem__") +
295  std::to_string(std::distance(varOps.begin(), varOps.end()));
296  varOp = rewriter.create<spirv::GlobalVariableOp>(
297  loc, TypeAttr::get(spirvType), varName,
298  /*initializer = */ nullptr);
299  }
300 
301  // Get pointer to global variable at the current scope.
302  rewriter.replaceOpWithNewOp<spirv::AddressOfOp>(operation, varOp);
303  return success();
304  }
305 };

616  // attribute conversion.

633  dstElemType.cast<IntegerType>(), rewriter);
634  if (!dstAttr)
635  return failure();
636  elements.push_back(dstAttr);
637  }
638  }

641  // attributes; element attributes only works with standard types. So we need
642  // to prepare another converted standard types for the destination elements
643  // attribute.
644  if (dstAttrType.isa<RankedTensorType>())
645  dstAttrType = RankedTensorType::get(dstAttrType.getShape(), dstElemType);
646  else
647  dstAttrType = VectorType::get(dstAttrType.getShape(), dstElemType);
648 
649  dstElementsAttr = DenseElementsAttr::get(dstAttrType, elements);

653  dstElementsAttr);

825 LogicalResult
826 IntLoadOpPattern::matchAndRewrite(LoadOp loadOp, ArrayRef<Value> operands,
827  ConversionPatternRewriter &rewriter) const {
828  LoadOpAdaptor loadOperands(operands);
829  auto loc = loadOp.getLoc();
830  auto memrefType = loadOp.memref().getType().cast<MemRefType>();
831  if (!memrefType.getElementType().isSignlessInteger())
832  return failure();
833  spirv::AccessChainOp accessChainOp =
834  spirv::getElementPtr(typeConverter, memrefType, loadOperands.memref(),
835  loadOperands.indices(), loc, rewriter);
836 
837  int srcBits = memrefType.getElementType().getIntOrFloatBitWidth();
838  auto dstType = typeConverter.convertType(memrefType)
839  .cast<spirv::PointerType>()
840  .getPointeeType()
841  .cast<spirv::StructType>()
842  .getElementType(0)
843  .cast<spirv::ArrayType>()
844  .getElementType();
845  int dstBits = dstType.getIntOrFloatBitWidth();
846  assert(dstBits % srcBits == 0);
847 
848  // If the rewrited load op has the same bit width, use the loading value
849  // directly.
850  if (srcBits == dstBits) {
851  rewriter.replaceOpWithNewOp<spirv::LoadOp>(loadOp,
852  accessChainOp.getResult());
853  return success();
854  }
855 
856  // Assume that getElementPtr() works linearizely. If it's a scalar, the method
857  // still returns a linearized accessing. If the accessing is not linearized,
858  // there will be offset issues.
859  assert(accessChainOp.indices().size() == 2);
860  Value adjustedPtr = adjustAccessChainForBitwidth(typeConverter, accessChainOp,
861  srcBits, dstBits, rewriter);
862  Value spvLoadOp = rewriter.create<spirv::LoadOp>(
863  loc, dstType, adjustedPtr,
864  loadOp.getAttrOfType<IntegerAttr>(
865  spirv::attributeName<spirv::MemoryAccess>()),
866  loadOp.getAttrOfType<IntegerAttr>("alignment"));
867 
868  // Shift the bits to the rightmost.
869  // ____XXXX________ -> ____________XXXX
870  Value lastDim = accessChainOp.getOperation()->getOperand(
871  accessChainOp.getNumOperands() - 1);
872  Value offset = getOffsetForBitwidth(loc, lastDim, srcBits, dstBits, rewriter);
873  Value result = rewriter.create<spirv::ShiftRightArithmeticOp>(
874  loc, spvLoadOp.getType(), spvLoadOp, offset);
875 
876  // Apply the mask to extract corresponding bits.
877  Value mask = rewriter.create<spirv::ConstantOp>(
878  loc, dstType, rewriter.getIntegerAttr(dstType, (1 << srcBits) - 1));
879  result = rewriter.create<spirv::BitwiseAndOp>(loc, dstType, result, mask);
880 
881  // Apply sign extension on the loading value unconditionally. The signedness
882  // semantic is carried in the operator itself, we relies other pattern to
883  // handle the casting.
884  IntegerAttr shiftValueAttr =
885  rewriter.getIntegerAttr(dstType, dstBits - srcBits);
886  Value shiftValue =
887  rewriter.create<spirv::ConstantOp>(loc, dstType, shiftValueAttr);
888  result = rewriter.create<spirv::ShiftLeftLogicalOp>(loc, dstType, result,
889  shiftValue);
890  result = rewriter.create<spirv::ShiftRightArithmeticOp>(loc, dstType, result,
891  shiftValue);
892  rewriter.replaceOp(loadOp, result);
893 
894  assert(accessChainOp.use_empty());
895  rewriter.eraseOp(accessChainOp);
896 
897  return success();
898 }

946 LogicalResult
947 IntStoreOpPattern::matchAndRewrite(StoreOp storeOp, ArrayRef<Value> operands,
948  ConversionPatternRewriter &rewriter) const {
949  StoreOpAdaptor storeOperands(operands);
950  auto memrefType = storeOp.memref().getType().cast<MemRefType>();
951  if (!memrefType.getElementType().isSignlessInteger())
952  return failure();
953 
954  auto loc = storeOp.getLoc();
955  spirv::AccessChainOp accessChainOp =
956  spirv::getElementPtr(typeConverter, memrefType, storeOperands.memref(),
957  storeOperands.indices(), loc, rewriter);
958  int srcBits = memrefType.getElementType().getIntOrFloatBitWidth();
959  auto dstType = typeConverter.convertType(memrefType)
960  .cast<spirv::PointerType>()
961  .getPointeeType()
962  .cast<spirv::StructType>()
963  .getElementType(0)
964  .cast<spirv::ArrayType>()
965  .getElementType();
966  int dstBits = dstType.getIntOrFloatBitWidth();
967  assert(dstBits % srcBits == 0);
968 
969  if (srcBits == dstBits) {
970  rewriter.replaceOpWithNewOp<spirv::StoreOp>(
971  storeOp, accessChainOp.getResult(), storeOperands.value());
972  return success();
973  }
974 
975  // Since there are multi threads in the processing, the emulation will be done
976  // with atomic operations. E.g., if the storing value is i8, rewrite the
977  // StoreOp to
978  // 1) load a 32-bit integer
979  // 2) clear 8 bits in the loading value
980  // 3) store 32-bit value back
981  // 4) load a 32-bit integer
982  // 5) modify 8 bits in the loading value
983  // 6) store 32-bit value back
984  // The step 1 to step 3 are done by AtomicAnd as one atomic step, and the step
985  // 4 to step 6 are done by AtomicOr as another atomic step.
986  assert(accessChainOp.indices().size() == 2);
987  Value lastDim = accessChainOp.getOperation()->getOperand(
988  accessChainOp.getNumOperands() - 1);
989  Value offset = getOffsetForBitwidth(loc, lastDim, srcBits, dstBits, rewriter);
990 
991  // Create a mask to clear the destination. E.g., if it is the second i8 in
992  // i32, 0xFFFF00FF is created.
993  Value mask = rewriter.create<spirv::ConstantOp>(
994  loc, dstType, rewriter.getIntegerAttr(dstType, (1 << srcBits) - 1));
995  Value clearBitsMask =
996  rewriter.create<spirv::ShiftLeftLogicalOp>(loc, dstType, mask, offset);
997  clearBitsMask = rewriter.create<spirv::NotOp>(loc, dstType, clearBitsMask);
998 
999  Value storeVal =
1000  shiftValue(loc, storeOperands.value(), offset, mask, dstBits, rewriter);
1001  Value adjustedPtr = adjustAccessChainForBitwidth(typeConverter, accessChainOp,
1002  srcBits, dstBits, rewriter);
1003  Optional<spirv::Scope> scope = getAtomicOpScope(memrefType);
1004  if (!scope)
1005  return failure();
1006  Value result = rewriter.create<spirv::AtomicAndOp>(
1007  loc, dstType, adjustedPtr, *scope, spirv::MemorySemantics::AcquireRelease,
1008  clearBitsMask);
1009  result = rewriter.create<spirv::AtomicOrOp>(
1010  loc, dstType, adjustedPtr, *scope, spirv::MemorySemantics::AcquireRelease,
1011  storeVal);
1012 
1013  // The AtomicOrOp has no side effect. Since it is already inserted, we can
1014  // just remove the original StoreOp. Note that rewriter.replaceOp()
1015  // doesn't work because it only accepts that the numbers of result are the
1016  // same.
1017  rewriter.eraseOp(storeOp);
1018 
1019  assert(accessChainOp.use_empty());
1020  rewriter.eraseOp(accessChainOp);
1021 
1022  return success();
1023 }
