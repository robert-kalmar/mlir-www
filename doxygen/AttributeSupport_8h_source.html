28 /// This class contains all of the static information common to all instances of
29 /// a registered Attribute.
30 class AbstractAttribute {
31 public:
32  /// Look up the specified abstract attribute in the MLIRContext and return a
33  /// reference to it.
34  static const AbstractAttribute &lookup(TypeID typeID, MLIRContext *context);
35 
36  /// This method is used by Dialect objects when they register the list of
37  /// attributes they contain.
38  template <typename T> static AbstractAttribute get(Dialect &dialect) {
39  return AbstractAttribute(dialect, T::getInterfaceMap(), T::getTypeID());
40  }
41 
42  /// Return the dialect this attribute was registered to.
43  Dialect &getDialect() const { return const_cast<Dialect &>(dialect); }
44 
45  /// Returns an instance of the concept object for the given interface if it
46  /// was registered to this attribute, null otherwise. This should not be used
47  /// directly.
48  template <typename T> typename T::Concept *getInterface() const {
49  return interfaceMap.lookup<T>();
50  }
51 
52  /// Return the unique identifier representing the concrete attribute class.
53  TypeID getTypeID() const { return typeID; }
54 
55 private:
56  AbstractAttribute(Dialect &dialect, detail::InterfaceMap &&interfaceMap,
57  TypeID typeID)
58  : dialect(dialect), interfaceMap(std::move(interfaceMap)),
59  typeID(typeID) {}
60 
61  /// This is the dialect that this attribute was registered to.
62  Dialect &dialect;
63 
64  /// This is a collection of the interfaces registered to this attribute.
65  detail::InterfaceMap interfaceMap;
66 
67  /// The unique identifier of the derived Attribute class.
68  TypeID typeID;
69 };

79 /// Base storage class appearing in an attribute. Derived storage classes should
80 /// only be constructed within the context of the AttributeUniquer.
81 class AttributeStorage : public StorageUniquer::BaseStorage {
82  friend detail::AttributeUniquer;
83  friend StorageUniquer;
84 
85 public:
86  /// Get the type of this attribute.
87  Type getType() const;
88 
89  /// Return the abstract descriptor for this attribute.
90  const AbstractAttribute &getAbstractAttribute() const {
91  assert(abstractAttribute && "Malformed attribute storage object.");
92  return *abstractAttribute;
93  }
94 
95 protected:
96  /// Construct a new attribute storage instance with the given type.
97  /// Note: All attributes require a valid type. If no type is provided here,
98  /// the type of the attribute will automatically default to NoneType
99  /// upon initialization in the uniquer.
100  AttributeStorage(Type type);
101  AttributeStorage();
102 
103  /// Set the type of this attribute.
104  void setType(Type type);
105 
106  // Set the abstract attribute for this storage instance. This is used by the
107  // AttributeUniquer when initializing a newly constructed storage object.
108  void initialize(const AbstractAttribute &abstractAttr) {
109  abstractAttribute = &abstractAttr;
110  }
111 
112 private:
113  /// The abstract descriptor for this attribute.
114  const AbstractAttribute *abstractAttribute;
115 
116  /// The opaque type of the attribute value.

136 // A utility class to get, or create, unique instances of attributes within an
137 // MLIRContext. This class manages all creation and uniquing of attributes.
138 class AttributeUniquer {
139 public:
140  /// Get an uniqued instance of attribute T.
141  template <typename T, typename... Args>
142  static T get(MLIRContext *ctx, unsigned kind, Args &&... args) {
143  return ctx->getAttributeUniquer().get<typename T::ImplType>(
144  T::getTypeID(),
145  [ctx](AttributeStorage *storage) {
146  initializeAttributeStorage(storage, ctx, T::getTypeID());
147  },
148  kind, std::forward<Args>(args)...);
149  }
150 
151  template <typename T, typename... Args>
152  static LogicalResult mutate(MLIRContext *ctx, typename T::ImplType *impl,
153  Args &&...args) {
154  assert(impl && "cannot mutate null attribute");
155  return ctx->getAttributeUniquer().mutate(T::getTypeID(), impl,
156  std::forward<Args>(args)...);
157  }
158 
159 private:
160  /// Initialize the given attribute storage instance.
161  static void initializeAttributeStorage(AttributeStorage *storage,
162  MLIRContext *ctx, TypeID attrID);
163 };
