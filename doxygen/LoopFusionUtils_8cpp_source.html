36 // Gathers all load and store memref accesses in 'opA' into 'values', where
37 // 'values[memref] == true' for each store operation.
38 static void getLoadAndStoreMemRefAccesses(Operation *opA,
39  DenseMap<Value, bool> &values) {
40  opA->walk([&](Operation *op) {
41  if (auto loadOp = dyn_cast<AffineReadOpInterface>(op)) {
42  if (values.count(loadOp.getMemRef()) == 0)
43  values[loadOp.getMemRef()] = false;
44  } else if (auto storeOp = dyn_cast<AffineWriteOpInterface>(op)) {
45  values[storeOp.getMemRef()] = true;
46  }
47  });
48 }

50 // Returns true if 'op' is a load or store operation which access an memref
51 // accessed 'values' and at least one of the access is a store operation.
52 // Returns false otherwise.
53 static bool isDependentLoadOrStoreOp(Operation *op,
54  DenseMap<Value, bool> &values) {
55  if (auto loadOp = dyn_cast<AffineReadOpInterface>(op)) {
56  return values.count(loadOp.getMemRef()) > 0 &&
57  values[loadOp.getMemRef()] == true;
58  } else if (auto storeOp = dyn_cast<AffineWriteOpInterface>(op)) {
59  return values.count(storeOp.getMemRef()) > 0;
60  }
61  return false;
62 }

64 // Returns the first operation in range ('opA', 'opB') which has a data
65 // dependence on 'opA'. Returns 'nullptr' of no dependence exists.
66 static Operation *getFirstDependentOpInRange(Operation *opA, Operation *opB) {
67  // Record memref values from all loads/store in loop nest rooted at 'opA'.
68  // Map from memref value to bool which is true if store, false otherwise.
69  DenseMap<Value, bool> values;
70  getLoadAndStoreMemRefAccesses(opA, values);
71 
72  // For each 'opX' in block in range ('opA', 'opB'), check if there is a data
73  // dependence from 'opA' to 'opX' ('opA' and 'opX' access the same memref
74  // and at least one of the accesses is a store).
75  Operation *firstDepOp = nullptr;
76  for (Block::iterator it = std::next(Block::iterator(opA));
77  it != Block::iterator(opB); ++it) {
78  Operation *opX = &(*it);
79  opX->walk([&](Operation *op) {
80  if (!firstDepOp && isDependentLoadOrStoreOp(op, values))
81  firstDepOp = opX;
82  });
83  if (firstDepOp)
84  break;
85  }
86  return firstDepOp;
87 }

89 // Returns the last operation 'opX' in range ('opA', 'opB'), for which there
90 // exists a data dependence from 'opX' to 'opB'.
91 // Returns 'nullptr' of no dependence exists.
92 static Operation *getLastDependentOpInRange(Operation *opA, Operation *opB) {
93  // Record memref values from all loads/store in loop nest rooted at 'opB'.
94  // Map from memref value to bool which is true if store, false otherwise.
95  DenseMap<Value, bool> values;
96  getLoadAndStoreMemRefAccesses(opB, values);
97 
98  // For each 'opX' in block in range ('opA', 'opB') in reverse order,
99  // check if there is a data dependence from 'opX' to 'opB':
100  // *) 'opX' and 'opB' access the same memref and at least one of the accesses
101  // is a store.
102  // *) 'opX' produces an SSA Value which is used by 'opB'.
103  Operation *lastDepOp = nullptr;
104  for (Block::reverse_iterator it = std::next(Block::reverse_iterator(opB));
105  it != Block::reverse_iterator(opA); ++it) {
106  Operation *opX = &(*it);
107  opX->walk([&](Operation *op) {
108  if (isa<AffineReadOpInterface, AffineWriteOpInterface>(op)) {
109  if (isDependentLoadOrStoreOp(op, values)) {
110  lastDepOp = opX;
111  return WalkResult::interrupt();
112  }
113  return WalkResult::advance();
114  }
115  for (auto value : op->getResults()) {
116  for (Operation *user : value.getUsers()) {
117  SmallVector<AffineForOp, 4> loops;
118  // Check if any loop in loop nest surrounding 'user' is 'opB'.
119  getLoopIVs(*user, &loops);
120  if (llvm::is_contained(loops, cast<AffineForOp>(opB))) {
121  lastDepOp = opX;
122  return WalkResult::interrupt();
123  }
124  }
125  }
126  return WalkResult::advance();
127  });
128  if (lastDepOp)
129  break;
130  }
131  return lastDepOp;
132 }

134 // Computes and returns an insertion point operation, before which the
135 // the fused <srcForOp, dstForOp> loop nest can be inserted while preserving
136 // dependences. Returns nullptr if no such insertion point is found.
137 static Operation *getFusedLoopNestInsertionPoint(AffineForOp srcForOp,
138  AffineForOp dstForOp) {
139  bool isSrcForOpBeforeDstForOp =
140  srcForOp.getOperation()->isBeforeInBlock(dstForOp.getOperation());
141  auto forOpA = isSrcForOpBeforeDstForOp ? srcForOp : dstForOp;
142  auto forOpB = isSrcForOpBeforeDstForOp ? dstForOp : srcForOp;
143 
144  auto *firstDepOpA =
145  getFirstDependentOpInRange(forOpA.getOperation(), forOpB.getOperation());
146  auto *lastDepOpB =
147  getLastDependentOpInRange(forOpA.getOperation(), forOpB.getOperation());
148  // Block:
149  // ...
150  // |-- opA
151  // | ...
152  // | lastDepOpB --|
153  // | ... |
154  // |-> firstDepOpA |
155  // ... |
156  // opB <---------
157  //
158  // Valid insertion point range: (lastDepOpB, firstDepOpA)
159  //
160  if (firstDepOpA != nullptr) {
161  if (lastDepOpB != nullptr) {
162  if (firstDepOpA->isBeforeInBlock(lastDepOpB) || firstDepOpA == lastDepOpB)
163  // No valid insertion point exists which preserves dependences.
164  return nullptr;
165  }
166  // Return insertion point in valid range closest to 'opB'.
167  // TODO: Consider other insertion points in valid range.
168  return firstDepOpA;
169  }
170  // No dependences from 'opA' to operation in range ('opA', 'opB'), return
171  // 'opB' insertion point.
172  return forOpB.getOperation();
173 }

175 // Gathers all load and store ops in loop nest rooted at 'forOp' into
176 // 'loadAndStoreOps'.
177 static bool
178 gatherLoadsAndStores(AffineForOp forOp,
179  SmallVectorImpl<Operation *> &loadAndStoreOps) {
180  bool hasIfOp = false;
181  forOp.walk([&](Operation *op) {
182  if (isa<AffineReadOpInterface, AffineWriteOpInterface>(op))
183  loadAndStoreOps.push_back(op);
184  else if (isa<AffineIfOp>(op))
185  hasIfOp = true;
186  });
187  return !hasIfOp;
188 }

190 // TODO: Prevent fusion of loop nests with side-effecting operations.
191 FusionResult mlir::canFuseLoops(AffineForOp srcForOp, AffineForOp dstForOp,
192  unsigned dstLoopDepth,
193  ComputationSliceState *srcSlice) {
194  // Return 'failure' if 'dstLoopDepth == 0'.
195  if (dstLoopDepth == 0) {
196  LLVM_DEBUG(llvm::dbgs() << "Cannot fuse loop nests at depth 0\n.");
197  return FusionResult::FailPrecondition;
198  }
199  // Return 'failure' if 'srcForOp' and 'dstForOp' are not in the same block.
200  auto *block = srcForOp.getOperation()->getBlock();
201  if (block != dstForOp.getOperation()->getBlock()) {
202  LLVM_DEBUG(llvm::dbgs() << "Cannot fuse loop nests in different blocks\n.");
203  return FusionResult::FailPrecondition;
204  }
205 
206  // Return 'failure' if no valid insertion point for fused loop nest in 'block'
207  // exists which would preserve dependences.
208  if (!getFusedLoopNestInsertionPoint(srcForOp, dstForOp)) {
209  LLVM_DEBUG(llvm::dbgs() << "Fusion would violate dependences in block\n.");
210  return FusionResult::FailBlockDependence;
211  }
212 
213  // Check if 'srcForOp' precedes 'dstForOp' in 'block'.
214  bool isSrcForOpBeforeDstForOp =
215  srcForOp.getOperation()->isBeforeInBlock(dstForOp.getOperation());
216  // 'forOpA' executes before 'forOpB' in 'block'.
217  auto forOpA = isSrcForOpBeforeDstForOp ? srcForOp : dstForOp;
218  auto forOpB = isSrcForOpBeforeDstForOp ? dstForOp : srcForOp;
219 
220  // Gather all load and store from 'forOpA' which precedes 'forOpB' in 'block'.
221  SmallVector<Operation *, 4> opsA;
222  if (!gatherLoadsAndStores(forOpA, opsA)) {
223  LLVM_DEBUG(llvm::dbgs() << "Fusing loops with affine.if unsupported.\n.");
224  return FusionResult::FailPrecondition;
225  }
226 
227  // Gather all load and store from 'forOpB' which succeeds 'forOpA' in 'block'.
228  SmallVector<Operation *, 4> opsB;
229  if (!gatherLoadsAndStores(forOpB, opsB)) {
230  LLVM_DEBUG(llvm::dbgs() << "Fusing loops with affine.if unsupported.\n.");
231  return FusionResult::FailPrecondition;
232  }
233 
234  // Calculate the number of common loops surrounding 'srcForOp' and 'dstForOp'.
235  unsigned numCommonLoops = mlir::getNumCommonSurroundingLoops(
236  *srcForOp.getOperation(), *dstForOp.getOperation());
237 
238  // Compute union of computation slices computed between all pairs of ops
239  // from 'forOpA' and 'forOpB'.
240  if (failed(mlir::computeSliceUnion(opsA, opsB, dstLoopDepth, numCommonLoops,
241  isSrcForOpBeforeDstForOp, srcSlice))) {
242  LLVM_DEBUG(llvm::dbgs() << "computeSliceUnion failed\n");
243  return FusionResult::FailPrecondition;
244  }
245 
246  return FusionResult::Success;
247 }

249 /// Fuses 'srcForOp' into 'dstForOp' with destination loop block insertion point
250 /// and source slice loop bounds specified in 'srcSlice'.
251 void mlir::fuseLoops(AffineForOp srcForOp, AffineForOp dstForOp,
252  ComputationSliceState *srcSlice) {
253  // Clone 'srcForOp' into 'dstForOp' at 'srcSlice->insertPoint'.
254  OpBuilder b(srcSlice->insertPoint->getBlock(), srcSlice->insertPoint);
255  BlockAndValueMapping mapper;
256  b.clone(*srcForOp, mapper);
257 
258  // Update 'sliceLoopNest' upper and lower bounds from computed 'srcSlice'.
259  SmallVector<AffineForOp, 4> sliceLoops;
260  for (unsigned i = 0, e = srcSlice->ivs.size(); i < e; ++i) {
261  auto loopIV = mapper.lookupOrNull(srcSlice->ivs[i]);
262  if (!loopIV)
263  continue;
264  auto forOp = getForInductionVarOwner(loopIV);
265  sliceLoops.push_back(forOp);
266  if (AffineMap lbMap = srcSlice->lbs[i])
267  forOp.setLowerBound(srcSlice->lbOperands[i], lbMap);
268  if (AffineMap ubMap = srcSlice->ubs[i])
269  forOp.setUpperBound(srcSlice->ubOperands[i], ubMap);
270  }
271 
272  // Promote any single iteration slice loops.
273  for (AffineForOp forOp : sliceLoops)
274  promoteIfSingleIteration(forOp);
275 }

277 /// Collect loop nest statistics (eg. loop trip count and operation count)
278 /// in 'stats' for loop nest rooted at 'forOp'. Returns true on success,
279 /// returns false otherwise.
280 bool mlir::getLoopNestStats(AffineForOp forOpRoot, LoopNestStats *stats) {
281  auto walkResult = forOpRoot.walk([&](AffineForOp forOp) {
282  auto *childForOp = forOp.getOperation();
283  auto *parentForOp = forOp.getParentOp();
284  if (!llvm::isa<FuncOp>(parentForOp)) {
285  if (!isa<AffineForOp>(parentForOp)) {
286  LLVM_DEBUG(llvm::dbgs() << "Expected parent AffineForOp");
287  return WalkResult::interrupt();
288  }
289  // Add mapping to 'forOp' from its parent AffineForOp.
290  stats->loopMap[parentForOp].push_back(forOp);
291  }
292 
293  // Record the number of op operations in the body of 'forOp'.
294  unsigned count = 0;
295  stats->opCountMap[childForOp] = 0;
296  for (auto &op : *forOp.getBody()) {
297  if (!isa<AffineForOp, AffineIfOp>(op))
298  ++count;
299  }
300  stats->opCountMap[childForOp] = count;
301 
302  // Record trip count for 'forOp'. Set flag if trip count is not
303  // constant.
304  Optional<uint64_t> maybeConstTripCount = getConstantTripCount(forOp);
305  if (!maybeConstTripCount.hasValue()) {
306  // Currently only constant trip count loop nests are supported.
307  LLVM_DEBUG(llvm::dbgs() << "Non-constant trip count unsupported");
308  return WalkResult::interrupt();
309  }
310 
311  stats->tripCountMap[childForOp] = maybeConstTripCount.getValue();
312  return WalkResult::advance();
313  });
314  return !walkResult.wasInterrupted();
315 }

317 // Computes the total cost of the loop nest rooted at 'forOp'.
318 // Currently, the total cost is computed by counting the total operation
319 // instance count (i.e. total number of operations in the loop bodyloop
320 // operation count * loop trip count) for the entire loop nest.
321 // If 'tripCountOverrideMap' is non-null, overrides the trip count for loops
322 // specified in the map when computing the total op instance count.
323 // NOTEs: 1) This is used to compute the cost of computation slices, which are
324 // sliced along the iteration dimension, and thus reduce the trip count.
325 // If 'computeCostMap' is non-null, the total op count for forOps specified
326 // in the map is increased (not overridden) by adding the op count from the
327 // map to the existing op count for the for loop. This is done before
328 // multiplying by the loop's trip count, and is used to model the cost of
329 // inserting a sliced loop nest of known cost into the loop's body.
330 // 2) This is also used to compute the cost of fusing a slice of some loop nest
331 // within another loop.
332 static int64_t getComputeCostHelper(
333  Operation *forOp, LoopNestStats &stats,
334  llvm::SmallDenseMap<Operation *, uint64_t, 8> *tripCountOverrideMap,
335  DenseMap<Operation *, int64_t> *computeCostMap) {
336  // 'opCount' is the total number operations in one iteration of 'forOp' body,
337  // minus terminator op which is a no-op.
338  int64_t opCount = stats.opCountMap[forOp] - 1;
339  if (stats.loopMap.count(forOp) > 0) {
340  for (auto childForOp : stats.loopMap[forOp]) {
341  opCount += getComputeCostHelper(childForOp.getOperation(), stats,
342  tripCountOverrideMap, computeCostMap);
343  }
344  }
345  // Add in additional op instances from slice (if specified in map).
346  if (computeCostMap != nullptr) {
347  auto it = computeCostMap->find(forOp);
348  if (it != computeCostMap->end()) {
349  opCount += it->second;
350  }
351  }
352  // Override trip count (if specified in map).
353  int64_t tripCount = stats.tripCountMap[forOp];
354  if (tripCountOverrideMap != nullptr) {
355  auto it = tripCountOverrideMap->find(forOp);
356  if (it != tripCountOverrideMap->end()) {
357  tripCount = it->second;
358  }
359  }
360  // Returns the total number of dynamic instances of operations in loop body.
361  return tripCount * opCount;
362 }

365 static Optional<uint64_t> getConstDifference(AffineMap lbMap, AffineMap ubMap) {
366  assert(lbMap.getNumResults() == 1 && "expected single result bound map");
367  assert(ubMap.getNumResults() == 1 && "expected single result bound map");
368  assert(lbMap.getNumDims() == ubMap.getNumDims());
369  assert(lbMap.getNumSymbols() == ubMap.getNumSymbols());
370  AffineExpr lbExpr(lbMap.getResult(0));
371  AffineExpr ubExpr(ubMap.getResult(0));
372  auto loopSpanExpr = simplifyAffineExpr(ubExpr - lbExpr, lbMap.getNumDims(),
373  lbMap.getNumSymbols());
374  auto cExpr = loopSpanExpr.dyn_cast<AffineConstantExpr>();
375  if (!cExpr)
376  return None;
377  return cExpr.getValue();
378 }

390 // Builds a map 'tripCountMap' from AffineForOp to constant trip count for loop
391 // nest surrounding represented by slice loop bounds in 'slice'.
392 // Returns true on success, false otherwise (if a non-constant trip count
393 // was encountered).
394 // TODO: Make this work with non-unit step loops.
395 static bool buildSliceTripCountMap(
396  ComputationSliceState *slice,
397  llvm::SmallDenseMap<Operation *, uint64_t, 8> *tripCountMap) {
398  unsigned numSrcLoopIVs = slice->ivs.size();
399  // Populate map from AffineForOp -> trip count
400  for (unsigned i = 0; i < numSrcLoopIVs; ++i) {
401  AffineForOp forOp = getForInductionVarOwner(slice->ivs[i]);
402  auto *op = forOp.getOperation();
403  AffineMap lbMap = slice->lbs[i];
404  AffineMap ubMap = slice->ubs[i];
405  if (lbMap == AffineMap() || ubMap == AffineMap()) {
406  // The iteration of src loop IV 'i' was not sliced. Use full loop bounds.
407  if (forOp.hasConstantLowerBound() && forOp.hasConstantUpperBound()) {
408  (*tripCountMap)[op] =
409  forOp.getConstantUpperBound() - forOp.getConstantLowerBound();
410  continue;
411  }
412  Optional<uint64_t> maybeConstTripCount = getConstantTripCount(forOp);
413  if (maybeConstTripCount.hasValue()) {
414  (*tripCountMap)[op] = maybeConstTripCount.getValue();
415  continue;
416  }
417  return false;
418  }
419  Optional<uint64_t> tripCount = getConstDifference(lbMap, ubMap);
420  // Slice bounds are created with a constant ub - lb difference.
421  if (!tripCount.hasValue())
422  return false;
423  (*tripCountMap)[op] = tripCount.getValue();
424  }
425  return true;
426 }

438 /// Computes and returns in 'computeCost', the total compute cost of fusing the
439 /// 'slice' of the loop nest rooted at 'srcForOp' into 'dstForOp'. Currently,
440 /// the total cost is computed by counting the total operation instance count
441 /// (i.e. total number of operations in the loop body * loop trip count) for
442 /// the entire loop nest.
443 bool mlir::getFusionComputeCost(AffineForOp srcForOp, LoopNestStats &srcStats,
444  AffineForOp dstForOp, LoopNestStats &dstStats,
445  ComputationSliceState *slice,
446  int64_t *computeCost) {
447  llvm::SmallDenseMap<Operation *, uint64_t, 8> sliceTripCountMap;
448  DenseMap<Operation *, int64_t> computeCostMap;
449 
450  // Build trip count map for computation slice.
451  if (!buildSliceTripCountMap(slice, &sliceTripCountMap))
452  return false;
453  // Checks whether a store to load forwarding will happen.
454  int64_t sliceIterationCount = getSliceIterationCount(sliceTripCountMap);
455  assert(sliceIterationCount > 0);
456  bool storeLoadFwdGuaranteed = (sliceIterationCount == 1);
457  auto *insertPointParent = slice->insertPoint->getParentOp();
458 
459  // The store and loads to this memref will disappear.
460  // TODO: Add load coalescing to memref data flow opt pass.
461  if (storeLoadFwdGuaranteed) {
462  // Subtract from operation count the loads/store we expect load/store
463  // forwarding to remove.
464  unsigned storeCount = 0;
465  llvm::SmallDenseSet<Value, 4> storeMemrefs;
466  srcForOp.walk([&](Operation *op) {
467  if (auto storeOp = dyn_cast<AffineWriteOpInterface>(op)) {
468  storeMemrefs.insert(storeOp.getMemRef());
469  ++storeCount;
470  }
471  });
472  // Subtract out any store ops in single-iteration src slice loop nest.
473  if (storeCount > 0)
474  computeCostMap[insertPointParent] = -storeCount;
475  // Subtract out any load users of 'storeMemrefs' nested below
476  // 'insertPointParent'.
477  for (auto value : storeMemrefs) {
478  for (auto *user : value.getUsers()) {
479  if (auto loadOp = dyn_cast<AffineReadOpInterface>(user)) {
480  SmallVector<AffineForOp, 4> loops;
481  // Check if any loop in loop nest surrounding 'user' is
482  // 'insertPointParent'.
483  getLoopIVs(*user, &loops);
484  if (llvm::is_contained(loops, cast<AffineForOp>(insertPointParent))) {
485  if (auto forOp =
486  dyn_cast_or_null<AffineForOp>(user->getParentOp())) {
487  if (computeCostMap.count(forOp) == 0)
488  computeCostMap[forOp] = 0;
489  computeCostMap[forOp] -= 1;
490  }
491  }
492  }
493  }
494  }
495  }
496 
497  // Compute op instance count for the src loop nest with iteration slicing.
498  int64_t sliceComputeCost = getComputeCostHelper(
499  srcForOp.getOperation(), srcStats, &sliceTripCountMap, &computeCostMap);
500 
501  // Compute cost of fusion for this depth.
502  computeCostMap[insertPointParent] = sliceComputeCost;
503 
504  *computeCost =
505  getComputeCostHelper(dstForOp.getOperation(), dstStats,
506  /*tripCountOverrideMap=*/nullptr, &computeCostMap);
507  return true;
508 }
