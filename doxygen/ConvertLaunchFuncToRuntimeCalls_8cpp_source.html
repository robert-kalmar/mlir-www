123 /// A rewrite patter to convert gpu.launch_func operations into a sequence of
124 /// GPU runtime calls. Currently it supports CUDA and ROCm (HIP).
125 ///
126 /// In essence, a gpu.launch_func operations gets compiled into the following
127 /// sequence of runtime calls:
128 ///
129 /// * moduleLoad -- loads the module given the cubin / hsaco data
130 /// * moduleGetFunction -- gets a handle to the actual kernel function
131 /// * getStreamHelper -- initializes a new compute stream on GPU
132 /// * launchKernel -- launches the kernel on a stream
133 /// * streamSynchronize -- waits for operations on the stream to finish
134 ///
135 /// Intermediate data structures are allocated on the stack.
136 class ConvertLaunchFuncOpToGpuRuntimeCallPattern
137  : public ConvertOpToGpuRuntimeCallPattern<gpu::LaunchFuncOp> {
138 public:
139  ConvertLaunchFuncOpToGpuRuntimeCallPattern(LLVMTypeConverter &typeConverter,
140  StringRef gpuBinaryAnnotation)
141  : ConvertOpToGpuRuntimeCallPattern<gpu::LaunchFuncOp>(typeConverter),
142  gpuBinaryAnnotation(gpuBinaryAnnotation) {}
143 
144 private:
145  void addParamToArray(OpBuilder &builder, Location loc, Value param,
146  Value array, unsigned pos, Value one) const;
147  Value generateParamsArray(gpu::LaunchFuncOp launchOp, unsigned numArguments,
148  OpBuilder &builder) const;
149  Value generateKernelNameConstant(StringRef moduleName, StringRef name,
150  Location loc, OpBuilder &builder) const;
151 
152  LogicalResult
153  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
154  ConversionPatternRewriter &rewriter) const override;
155 
156  llvm::SmallString<32> gpuBinaryAnnotation;
157 };

197 /// Emits the IR with the following structure:
198 ///
199 /// %data = llvm.alloca 1 x type-of(<param>)
200 /// llvm.store <param>, %data
201 /// %typeErased = llvm.bitcast %data to !llvm<"i8*">
202 /// %addr = llvm.getelementptr <array>[<pos>]
203 /// llvm.store %typeErased, %addr
204 ///
205 /// This is necessary to construct the array of arguments passed to the kernel
206 /// function as accepted by cuLaunchKernel, i.e. as a void** that points to
207 /// array of stack-allocated type-erased pointers to the actual arguments.
208 void ConvertLaunchFuncOpToGpuRuntimeCallPattern::addParamToArray(
209  OpBuilder &builder, Location loc, Value param, Value array, unsigned pos,
210  Value one) const {
211  auto memLocation = builder.create<LLVM::AllocaOp>(
212  loc, param.getType().cast<LLVM::LLVMType>().getPointerTo(), one,
213  /*alignment=*/0);
214  builder.create<LLVM::StoreOp>(loc, param, memLocation);
215  auto casted =
216  builder.create<LLVM::BitcastOp>(loc, llvmPointerType, memLocation);
217 
218  auto index = builder.create<LLVM::ConstantOp>(loc, llvmInt32Type,
219  builder.getI32IntegerAttr(pos));
220  auto gep = builder.create<LLVM::GEPOp>(loc, llvmPointerPointerType, array,
221  index.getResult());
222  builder.create<LLVM::StoreOp>(loc, casted, gep);
223 }

225 // Generates a parameters array to be used with a CUDA / ROCm (HIP) kernel
226 // launch call. The arguments are extracted from the launchOp.
227 // The generated code is essentially as follows:
228 //
229 // %array = alloca(numparams * sizeof(void *))
230 // for (i : [0, NumKernelOperands))
231 // %array[i] = cast<void*>(KernelOperand[i])
232 // return %array
233 Value ConvertLaunchFuncOpToGpuRuntimeCallPattern::generateParamsArray(
234  gpu::LaunchFuncOp launchOp, unsigned numArguments,
235  OpBuilder &builder) const {
236  auto numKernelOperands = launchOp.getNumKernelOperands();
237  Location loc = launchOp.getLoc();
238  auto one = builder.create<LLVM::ConstantOp>(loc, llvmInt32Type,
239  builder.getI32IntegerAttr(1));
240  auto arraySize = builder.create<LLVM::ConstantOp>(
241  loc, llvmInt32Type, builder.getI32IntegerAttr(numArguments));
242  auto array = builder.create<LLVM::AllocaOp>(loc, llvmPointerPointerType,
243  arraySize, /*alignment=*/0);
244 
245  unsigned pos = 0;
246  for (unsigned idx = 0; idx < numKernelOperands; ++idx) {
247  auto operand = launchOp.getKernelOperand(idx);
248  auto llvmType = operand.getType().cast<LLVM::LLVMType>();
249 
250  // Assume all struct arguments come from MemRef. If this assumption does not
251  // hold anymore then we `launchOp` to lower from MemRefType and not after
252  // LLVMConversion has taken place and the MemRef information is lost.
253  if (!llvmType.isStructTy()) {
254  addParamToArray(builder, loc, operand, array, pos++, one);
255  continue;
256  }
257 
258  // Put individual components of a memref descriptor into the flat argument
259  // list. We cannot use unpackMemref from LLVM lowering here because we have
260  // no access to MemRefType that had been lowered away.
261  for (int32_t j = 0, ej = llvmType.getStructNumElements(); j < ej; ++j) {
262  auto elemType = llvmType.getStructElementType(j);
263  if (elemType.isArrayTy()) {
264  for (int32_t k = 0, ek = elemType.getArrayNumElements(); k < ek; ++k) {
265  Value elem = builder.create<LLVM::ExtractValueOp>(
266  loc, elemType.getArrayElementType(), operand,
267  builder.getI32ArrayAttr({j, k}));
268  addParamToArray(builder, loc, elem, array, pos++, one);
269  }
270  } else {
271  assert((elemType.isIntegerTy() || elemType.isFloatTy() ||
272  elemType.isDoubleTy() || elemType.isPointerTy()) &&
273  "expected scalar type");
274  Value strct = builder.create<LLVM::ExtractValueOp>(
275  loc, elemType, operand, builder.getI32ArrayAttr(j));
276  addParamToArray(builder, loc, strct, array, pos++, one);
277  }
278  }
279  }
280 
281  return array;
282 }

284 // Generates an LLVM IR dialect global that contains the name of the given
285 // kernel function as a C string, and returns a pointer to its beginning.
286 // The code is essentially:
287 //
288 // llvm.global constant @kernel_name("function_name\00")
289 // func(...) {
290 // %0 = llvm.addressof @kernel_name
291 // %1 = llvm.constant (0 : index)
292 // %2 = llvm.getelementptr %0[%1, %1] : !llvm<"i8*">
293 // }
294 Value ConvertLaunchFuncOpToGpuRuntimeCallPattern::generateKernelNameConstant(
295  StringRef moduleName, StringRef name, Location loc,
296  OpBuilder &builder) const {
297  // Make sure the trailing zero is included in the constant.
298  std::vector<char> kernelName(name.begin(), name.end());
299  kernelName.push_back('\0');
300 
301  std::string globalName =
302  std::string(llvm::formatv("{0}_{1}_kernel_name", moduleName, name));
303  return LLVM::createGlobalString(
304  loc, builder, globalName, StringRef(kernelName.data(), kernelName.size()),
305  LLVM::Linkage::Internal);
306 }

308 // Emits LLVM IR to launch a kernel function. Expects the module that contains
309 // the compiled kernel function as a cubin in the 'nvvm.cubin' attribute, or a
310 // hsaco in the 'rocdl.hsaco' attribute of the kernel function in the IR.
311 //
312 // %0 = call %binarygetter
313 // %1 = call %moduleLoad(%0)
314 // %2 = <see generateKernelNameConstant>
315 // %3 = call %moduleGetFunction(%1, %2)
316 // %4 = call %streamCreate()
317 // %5 = <see generateParamsArray>
318 // call %launchKernel(%3, <launchOp operands 0..5>, 0, %4, %5, nullptr)
319 // call %streamSynchronize(%4)
320 LogicalResult ConvertLaunchFuncOpToGpuRuntimeCallPattern::matchAndRewrite(
321  Operation *op, ArrayRef<Value> operands,
322  ConversionPatternRewriter &rewriter) const {
323  Location loc = op->getLoc();
324  auto launchOp = cast<gpu::LaunchFuncOp>(op);
325  auto moduleOp = op->getParentOfType<ModuleOp>();
326 
327  // Create an LLVM global with CUBIN extracted from the kernel annotation and
328  // obtain a pointer to the first byte in it.
329  auto kernelModule =
330  moduleOp.lookupSymbol<gpu::GPUModuleOp>(launchOp.getKernelModuleName());
331  assert(kernelModule && "expected a kernel module");
332 
333  auto binaryAttr = kernelModule.getAttrOfType<StringAttr>(gpuBinaryAnnotation);
334  if (!binaryAttr) {
335  kernelModule.emitOpError()
336  << "missing " << gpuBinaryAnnotation << " attribute";
337  return failure();
338  }
339 
340  SmallString<128> nameBuffer(kernelModule.getName());
341  nameBuffer.append(kGpuBinaryStorageSuffix);
342  Value data =
343  LLVM::createGlobalString(loc, rewriter, nameBuffer.str(),
344  binaryAttr.getValue(), LLVM::Linkage::Internal);
345 
346  auto module = moduleLoadCallBuilder.create(loc, rewriter, data);
347  // Get the function from the module. The name corresponds to the name of
348  // the kernel function.
349  auto kernelName = generateKernelNameConstant(
350  launchOp.getKernelModuleName(), launchOp.getKernelName(), loc, rewriter);
351  auto function = moduleGetFunctionCallBuilder.create(
352  loc, rewriter, {module.getResult(0), kernelName});
353  // Grab the global stream needed for execution.
354  auto stream = streamCreateCallBuilder.create(loc, rewriter, {});
355 
356  // Get the launch target.
357  auto gpuFuncOp = SymbolTable::lookupNearestSymbolFrom<LLVM::LLVMFuncOp>(
358  launchOp, launchOp.kernel());
359  if (!gpuFuncOp) {
360  launchOp.emitOpError() << "corresponding kernel function not found";
361  return failure();
362  }
363  // Build array of kernel parameters.
364  auto kernelParams =
365  generateParamsArray(launchOp, gpuFuncOp.getNumArguments(), rewriter);
366 
367  // Invoke the function with required arguments.
368  auto zero = rewriter.create<LLVM::ConstantOp>(loc, llvmInt32Type,
369  rewriter.getI32IntegerAttr(0));
370  auto nullpointer =
371  rewriter.create<LLVM::IntToPtrOp>(loc, llvmPointerPointerType, zero);
372  launchKernelCallBuilder.create(
373  loc, rewriter,
374  {function.getResult(0), launchOp.gridSizeX(), launchOp.gridSizeY(),
375  launchOp.gridSizeZ(), launchOp.blockSizeX(), launchOp.blockSizeY(),
376  launchOp.blockSizeZ(), zero, /* sharedMemBytes */
377  stream.getResult(0), /* stream */
378  kernelParams, /* kernel params */
379  nullpointer /* extra */});
380  streamSynchronizeCallBuilder.create(loc, rewriter, stream.getResult(0));
381 
382  rewriter.eraseOp(op);
383  return success();
384 }
