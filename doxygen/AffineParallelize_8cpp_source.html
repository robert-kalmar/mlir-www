37 void AffineParallelize::runOnFunction() {
38  FuncOp f = getFunction();
39  SmallVector<AffineForOp, 8> parallelizableLoops;
40  f.walk([&](AffineForOp loop) {
41  if (isLoopParallel(loop))
42  parallelizableLoops.push_back(loop);
43  });
44  for (AffineForOp loop : parallelizableLoops)
45  affineParallelize(loop);
46 }
