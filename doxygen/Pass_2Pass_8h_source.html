266 /// Pass to transform an operation of a specific type.
267 ///
268 /// Operation passes must not:
269 /// - modify any other operations within the parent region, as other threads
270 /// may be manipulating them concurrently.
271 /// - modify any state within the parent operation, this includes adding
272 /// additional operations.
273 ///
274 /// Derived function passes are expected to provide the following:
275 /// - A 'void runOnOperation()' method.
276 /// - A 'StringRef getName() const' method.
277 /// - A 'std::unique_ptr<Pass> clonePass() const' method.
278 template <typename OpT = void> class OperationPass : public Pass {
279 protected:
280  OperationPass(TypeID passID) : Pass(passID, OpT::getOperationName()) {}
281 
282  /// Support isa/dyn_cast functionality.
283  static bool classof(const Pass *pass) {
284  return pass->getOpName() == OpT::getOperationName();
285  }
286 
287  /// Return the current operation being transformed.
288  OpT getOperation() { return cast<OpT>(Pass::getOperation()); }
289 };
290 
291 /// Pass to transform an operation.
292 ///
293 /// Operation passes must not:
294 /// - modify any other operations within the parent region, as other threads
295 /// may be manipulating them concurrently.
296 /// - modify any state within the parent operation, this includes adding
297 /// additional operations.
298 ///
299 /// Derived function passes are expected to provide the following:
300 /// - A 'void runOnOperation()' method.
301 /// - A 'StringRef getName() const' method.
302 /// - A 'std::unique_ptr<Pass> clonePass() const' method.
303 template <> class OperationPass<void> : public Pass {
304 protected:
305  OperationPass(TypeID passID) : Pass(passID) {}
306 };

308 /// A model for providing function pass specific utilities.
309 ///
310 /// Derived function passes are expected to provide the following:
311 /// - A 'void runOnFunction()' method.
312 /// - A 'StringRef getName() const' method.
313 /// - A 'std::unique_ptr<Pass> clonePass() const' method.
314 class FunctionPass : public OperationPass<FuncOp> {
315 public:
316  using OperationPass<FuncOp>::OperationPass;
317 
318  /// The polymorphic API that runs the pass over the currently held function.
319  virtual void runOnFunction() = 0;
320 
321  /// The polymorphic API that runs the pass over the currently held operation.
322  void runOnOperation() final {
323  if (!getFunction().isExternal())
324  runOnFunction();
325  }
326 
327  /// Return the current function being transformed.
328  FuncOp getFunction() { return this->getOperation(); }
329 };

331 /// This class provides a CRTP wrapper around a base pass class to define
332 /// several necessary utility methods. This should only be used for passes that
333 /// are not suitably represented using the declarative pass specification(i.e.
334 /// tablegen backend).
335 template <typename PassT, typename BaseT> class PassWrapper : public BaseT {
336 public:
337  /// Support isa/dyn_cast functionality for the derived pass class.
338  static bool classof(const Pass *pass) {
339  return pass->getTypeID() == TypeID::get<PassT>();
340  }
341 
342 protected:
343  PassWrapper() : BaseT(TypeID::get<PassT>()) {}
344 
345  /// Returns the derived pass name.
346  StringRef getName() const override { return llvm::getTypeName<PassT>(); }
347 
348  /// A clone method to create a copy of this pass.
349  std::unique_ptr<Pass> clonePass() const override {
350  return std::make_unique<PassT>(*static_cast<const PassT *>(this));
351  }
352 };
