37 /// Collect a set of vector slices transformation patterns:
38 /// ExtractSlicesOpLowering, InsertSlicesOpLowering
39 /// Useful for clients that want to express all vector "slices"
40 /// ops in terms of more elementary vector "slice" ops. If all

48 /// Enum to control the lowering of `vector.contract` operations.
49 enum class VectorContractLowering {
50  /// Progressively lower to finer grained `vector.contract` and dot-products.
51  Dot = 0,
52  /// Lower to `vector.matrix_multiply`, maps 1-1 to LLVM matrix intrinsics.
53  Matmul = 1,
54  /// Lower to `vector.outerproduct`.
55  OuterProduct = 2,
56 };
57 /// Enum to control the lowering of `vector.transpose` operations.
58 enum class VectorTransposeLowering {
59  /// Lower transpose into element-wise extract and inserts.
60  EltWise = 0,
61  /// Lower 2-D transpose to `vector.flat_transpose`, maps 1-1 to LLVM matrix
62  /// intrinsics.
63  Flat = 1,
64 };
65 /// Enum to control the splitting of `vector.transfer` operations into masked
66 /// and unmasked variants.
67 enum class VectorTransferSplit {
68  /// Do not split vector transfer operations.
69  None = 0,
70  /// Split using masked + unmasked vector.transfer operations.
71  VectorTransfer = 1,
72  /// Split using a unmasked vector.transfer + linalg.fill + linalg.copy
73  /// operations.
74  LinalgCopy = 2,
75  /// Do not split vector transfer operation but instead mark it as "unmasked".
76  ForceUnmasked = 3
77 };
