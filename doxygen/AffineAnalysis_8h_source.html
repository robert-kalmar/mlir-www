75 // DependenceComponent contains state about the direction of a dependence as an
76 // interval [lb, ub] for an AffineForOp.
77 // Distance vectors components are represented by the interval [lb, ub] with
78 // lb == ub.
79 // Direction vectors components are represented by the interval [lb, ub] with
80 // lb < ub. Note that ub/lb == None means unbounded.
81 struct DependenceComponent {
82  // The AffineForOp Operation associated with this dependence component.
83  Operation *op;
84  // The lower bound of the dependence distance.
85  Optional<int64_t> lb;
86  // The upper bound of the dependence distance (inclusive).
87  Optional<int64_t> ub;
88  DependenceComponent() : lb(llvm::None), ub(llvm::None) {}
89 };

91 /// Checks whether two accesses to the same memref access the same element.
92 /// Each access is specified using the MemRefAccess structure, which contains
93 /// the operation, indices and memref associated with the access. Returns
94 /// 'NoDependence' if it can be determined conclusively that the accesses do not
95 /// access the same memref element. If 'allowRAR' is true, will consider
96 /// read-after-read dependences (typically used by applications trying to
97 /// optimize input reuse).
98 // TODO: Wrap 'dependenceConstraints' and 'dependenceComponents' into a single
99 // struct.
100 // TODO: Make 'dependenceConstraints' optional arg.
101 struct DependenceResult {
102  enum ResultEnum {
103  HasDependence, // A dependence exists between 'srcAccess' and 'dstAccess'.
104  NoDependence, // No dependence exists between 'srcAccess' and 'dstAccess'.
105  Failure, // Dependence check failed due to unsupported cases.
106  } value;
107  DependenceResult(ResultEnum v) : value(v) {}
108 };
