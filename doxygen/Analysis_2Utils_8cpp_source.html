31 /// Populates 'loops' with IVs of the loops surrounding 'op' ordered from
32 /// the outermost 'affine.for' operation to the innermost one.
33 void mlir::getLoopIVs(Operation &op, SmallVectorImpl<AffineForOp> *loops) {
34  auto *currOp = op.getParentOp();
35  AffineForOp currAffineForOp;
36  // Traverse up the hierarchy collecting all 'affine.for' operation while
37  // skipping over 'affine.if' operations.
38  while (currOp && ((currAffineForOp = dyn_cast<AffineForOp>(currOp)) ||
39  isa<AffineIfOp>(currOp))) {
40  if (currAffineForOp)
41  loops->push_back(currAffineForOp);
42  currOp = currOp->getParentOp();
43  }
44  std::reverse(loops->begin(), loops->end());
45 }

48 LogicalResult
49 ComputationSliceState::getAsConstraints(FlatAffineConstraints *cst) {
50  assert(!lbOperands.empty());
51  // Adds src 'ivs' as dimension identifiers in 'cst'.
52  unsigned numDims = ivs.size();
53  // Adds operands (dst ivs and symbols) as symbols in 'cst'.
54  unsigned numSymbols = lbOperands[0].size();
55 
56  SmallVector<Value, 4> values(ivs);
57  // Append 'ivs' then 'operands' to 'values'.
58  values.append(lbOperands[0].begin(), lbOperands[0].end());
59  cst->reset(numDims, numSymbols, 0, values);
60 
61  // Add loop bound constraints for values which are loop IVs and equality
62  // constraints for symbols which are constants.
63  for (const auto &value : values) {
64  assert(cst->containsId(value) && "value expected to be present");
65  if (isValidSymbol(value)) {
66  // Check if the symbol is a constant.
67  if (auto cOp = value.getDefiningOp<ConstantIndexOp>())
68  cst->setIdToConstant(value, cOp.getValue());
69  } else if (auto loop = getForInductionVarOwner(value)) {
70  if (failed(cst->addAffineForOpDomain(loop)))
71  return failure();
72  }
73  }
74 
75  // Add slices bounds on 'ivs' using maps 'lbs'/'ubs' with 'lbOperands[0]'
76  LogicalResult ret = cst->addSliceBounds(ivs, lbs, ubs, lbOperands[0]);
77  assert(succeeded(ret) &&
78  "should not fail as we never have semi-affine slice maps");
79  (void)ret;
80  return success();
81 }

95 Optional<int64_t> MemRefRegion::getConstantBoundingSizeAndShape(
96  SmallVectorImpl<int64_t> *shape, std::vector<SmallVector<int64_t, 4>> *lbs,
97  SmallVectorImpl<int64_t> *lbDivisors) const {
98  auto memRefType = memref.getType().cast<MemRefType>();
99  unsigned rank = memRefType.getRank();
100  if (shape)
101  shape->reserve(rank);
102 
103  assert(rank == cst.getNumDimIds() && "inconsistent memref region");
104 
105  // Use a copy of the region constraints that has upper/lower bounds for each
106  // memref dimension with static size added to guard against potential
107  // over-approximation from projection or union bounding box. We may not add
108  // this on the region itself since they might just be redundant constraints
109  // that will need non-trivials means to eliminate.
110  FlatAffineConstraints cstWithShapeBounds(cst);
111  for (unsigned r = 0; r < rank; r++) {
112  cstWithShapeBounds.addConstantLowerBound(r, 0);
113  int64_t dimSize = memRefType.getDimSize(r);
114  if (ShapedType::isDynamic(dimSize))
115  continue;
116  cstWithShapeBounds.addConstantUpperBound(r, dimSize - 1);
117  }
118 
119  // Find a constant upper bound on the extent of this memref region along each
120  // dimension.
121  int64_t numElements = 1;
122  int64_t diffConstant;
123  int64_t lbDivisor;
124  for (unsigned d = 0; d < rank; d++) {
125  SmallVector<int64_t, 4> lb;
126  Optional<int64_t> diff =
127  cstWithShapeBounds.getConstantBoundOnDimSize(d, &lb, &lbDivisor);
128  if (diff.hasValue()) {
129  diffConstant = diff.getValue();
130  assert(lbDivisor > 0);
131  } else {
132  // If no constant bound is found, then it can always be bound by the
133  // memref's dim size if the latter has a constant size along this dim.
134  auto dimSize = memRefType.getDimSize(d);
135  if (dimSize == -1)
136  return None;
137  diffConstant = dimSize;
138  // Lower bound becomes 0.
139  lb.resize(cstWithShapeBounds.getNumSymbolIds() + 1, 0);
140  lbDivisor = 1;
141  }
142  numElements *= diffConstant;
143  if (lbs) {
144  lbs->push_back(lb);
145  assert(lbDivisors && "both lbs and lbDivisor or none");
146  lbDivisors->push_back(lbDivisor);
147  }
148  if (shape) {
149  shape->push_back(diffConstant);
150  }
151  }
152  return numElements;
153 }

179 /// Computes the memory region accessed by this memref with the region
180 /// represented as constraints symbolic/parametric in 'loopDepth' loops
181 /// surrounding opInst and any additional Function symbols.
182 // For example, the memref region for this load operation at loopDepth = 1 will
183 // be as below:
184 //
185 // affine.for %i = 0 to 32 {
186 // affine.for %ii = %i to (d0) -> (d0 + 8) (%i) {
187 // load %A[%ii]
188 // }
189 // }
190 //
191 // region: {memref = %A, write = false, {%i <= m0 <= %i + 7} }
192 // The last field is a 2-d FlatAffineConstraints symbolic in %i.
193 //
194 // TODO: extend this to any other memref dereferencing ops
195 // (dma_start, dma_wait).
196 LogicalResult MemRefRegion::compute(Operation *op, unsigned loopDepth,
197  ComputationSliceState *sliceState,
198  bool addMemRefDimBounds) {
199  assert((isa<AffineReadOpInterface, AffineWriteOpInterface>(op)) &&
200  "affine read/write op expected");
201 
202  MemRefAccess access(op);
203  memref = access.memref;
204  write = access.isStore();
205 
206  unsigned rank = access.getRank();
207 
208  LLVM_DEBUG(llvm::dbgs() << "MemRefRegion::compute: " << *op
209  << "depth: " << loopDepth << "\n";);
210 
211  // 0-d memrefs.
212  if (rank == 0) {
213  SmallVector<AffineForOp, 4> ivs;
214  getLoopIVs(*op, &ivs);
215  assert(loopDepth <= ivs.size() && "invalid 'loopDepth'");
216  // The first 'loopDepth' IVs are symbols for this region.
217  ivs.resize(loopDepth);
218  SmallVector<Value, 4> regionSymbols;
219  extractForInductionVars(ivs, &regionSymbols);
220  // A 0-d memref has a 0-d region.
221  cst.reset(rank, loopDepth, /*numLocals=*/0, regionSymbols);
222  return success();
223  }
224 
225  // Build the constraints for this region.
226  AffineValueMap accessValueMap;
227  access.getAccessMap(&accessValueMap);
228  AffineMap accessMap = accessValueMap.getAffineMap();
229 
230  unsigned numDims = accessMap.getNumDims();
231  unsigned numSymbols = accessMap.getNumSymbols();
232  unsigned numOperands = accessValueMap.getNumOperands();
233  // Merge operands with slice operands.
234  SmallVector<Value, 4> operands;
235  operands.resize(numOperands);
236  for (unsigned i = 0; i < numOperands; ++i)
237  operands[i] = accessValueMap.getOperand(i);
238 
239  if (sliceState != nullptr) {
240  operands.reserve(operands.size() + sliceState->lbOperands[0].size());
241  // Append slice operands to 'operands' as symbols.
242  for (auto extraOperand : sliceState->lbOperands[0]) {
243  if (!llvm::is_contained(operands, extraOperand)) {
244  operands.push_back(extraOperand);
245  numSymbols++;
246  }
247  }
248  }
249  // We'll first associate the dims and symbols of the access map to the dims
250  // and symbols resp. of cst. This will change below once cst is
251  // fully constructed out.
252  cst.reset(numDims, numSymbols, 0, operands);
253 
254  // Add equality constraints.
255  // Add inequalities for loop lower/upper bounds.
256  for (unsigned i = 0; i < numDims + numSymbols; ++i) {
257  auto operand = operands[i];
258  if (auto loop = getForInductionVarOwner(operand)) {
259  // Note that cst can now have more dimensions than accessMap if the
260  // bounds expressions involve outer loops or other symbols.
261  // TODO: rewrite this to use getInstIndexSet; this way
262  // conditionals will be handled when the latter supports it.
263  if (failed(cst.addAffineForOpDomain(loop)))
264  return failure();
265  } else {
266  // Has to be a valid symbol.
267  auto symbol = operand;
268  assert(isValidSymbol(symbol));
269  // Check if the symbol is a constant.
270  if (auto *op = symbol.getDefiningOp()) {
271  if (auto constOp = dyn_cast<ConstantIndexOp>(op)) {
272  cst.setIdToConstant(symbol, constOp.getValue());
273  }
274  }
275  }
276  }
277 
278  // Add lower/upper bounds on loop IVs using bounds from 'sliceState'.
279  if (sliceState != nullptr) {
280  // Add dim and symbol slice operands.
281  for (auto operand : sliceState->lbOperands[0]) {
282  cst.addInductionVarOrTerminalSymbol(operand);
283  }
284  // Add upper/lower bounds from 'sliceState' to 'cst'.
285  LogicalResult ret =
286  cst.addSliceBounds(sliceState->ivs, sliceState->lbs, sliceState->ubs,
287  sliceState->lbOperands[0]);
288  assert(succeeded(ret) &&
289  "should not fail as we never have semi-affine slice maps");
290  (void)ret;
291  }
292 
293  // Add access function equalities to connect loop IVs to data dimensions.
294  if (failed(cst.composeMap(&accessValueMap))) {
295  op->emitError("getMemRefRegion: compose affine map failed");
296  LLVM_DEBUG(accessValueMap.getAffineMap().dump());
297  return failure();
298  }
299 
300  // Set all identifiers appearing after the first 'rank' identifiers as
301  // symbolic identifiers - so that the ones corresponding to the memref
302  // dimensions are the dimensional identifiers for the memref region.
303  cst.setDimSymbolSeparation(cst.getNumDimAndSymbolIds() - rank);
304 
305  // Eliminate any loop IVs other than the outermost 'loopDepth' IVs, on which
306  // this memref region is symbolic.
307  SmallVector<AffineForOp, 4> enclosingIVs;
308  getLoopIVs(*op, &enclosingIVs);
309  assert(loopDepth <= enclosingIVs.size() && "invalid loop depth");
310  enclosingIVs.resize(loopDepth);
311  SmallVector<Value, 4> ids;
312  cst.getIdValues(cst.getNumDimIds(), cst.getNumDimAndSymbolIds(), &ids);
313  for (auto id : ids) {
314  AffineForOp iv;
315  if ((iv = getForInductionVarOwner(id)) &&
316  llvm::is_contained(enclosingIVs, iv) == false) {
317  cst.projectOut(id);
318  }
319  }
320 
321  // Project out any local variables (these would have been added for any
322  // mod/divs).
323  cst.projectOut(cst.getNumDimAndSymbolIds(), cst.getNumLocalIds());
324 
325  // Constant fold any symbolic identifiers.
326  cst.constantFoldIdRange(/*pos=*/cst.getNumDimIds(),
327  /*num=*/cst.getNumSymbolIds());
328 
329  assert(cst.getNumDimIds() == rank && "unexpected MemRefRegion format");
330 
331  // Add upper/lower bounds for each memref dimension with static size
332  // to guard against potential over-approximation from projection.
333  // TODO: Support dynamic memref dimensions.
334  if (addMemRefDimBounds) {
335  auto memRefType = memref.getType().cast<MemRefType>();
336  for (unsigned r = 0; r < rank; r++) {
337  cst.addConstantLowerBound(/*pos=*/r, /*lb=*/0);
338  if (memRefType.isDynamicDim(r))
339  continue;
340  cst.addConstantUpperBound(/*pos=*/r, memRefType.getDimSize(r) - 1);
341  }
342  }
343  cst.removeTrivialRedundancy();
344 
345  LLVM_DEBUG(llvm::dbgs() << "Memory region:\n");
346  LLVM_DEBUG(cst.dump());
347  return success();
348 }

351  auto elementType = memRefType.getElementType();

408 template <typename LoadOrStoreOp>
409 LogicalResult mlir::boundCheckLoadOrStoreOp(LoadOrStoreOp loadOrStoreOp,
410  bool emitError) {
411  static_assert(llvm::is_one_of<LoadOrStoreOp, AffineReadOpInterface,
412  AffineWriteOpInterface>::value,
413  "argument should be either a AffineReadOpInterface or a "
414  "AffineWriteOpInterface");
415 
416  Operation *op = loadOrStoreOp.getOperation();
417  MemRefRegion region(op->getLoc());
418  if (failed(region.compute(op, /*loopDepth=*/0, /*sliceState=*/nullptr,
419  /*addMemRefDimBounds=*/false)))
420  return success();
421 
422  LLVM_DEBUG(llvm::dbgs() << "Memory region");
423  LLVM_DEBUG(region.getConstraints()->dump());
424 
425  bool outOfBounds = false;
426  unsigned rank = loadOrStoreOp.getMemRefType().getRank();
427 
428  // For each dimension, check for out of bounds.
429  for (unsigned r = 0; r < rank; r++) {
430  FlatAffineConstraints ucst(*region.getConstraints());
431 
432  // Intersect memory region with constraint capturing out of bounds (both out
433  // of upper and out of lower), and check if the constraint system is
434  // feasible. If it is, there is at least one point out of bounds.
435  SmallVector<int64_t, 4> ineq(rank + 1, 0);
436  int64_t dimSize = loadOrStoreOp.getMemRefType().getDimSize(r);
437  // TODO: handle dynamic dim sizes.
438  if (dimSize == -1)
439  continue;
440 
441  // Check for overflow: d_i >= memref dim size.
442  ucst.addConstantLowerBound(r, dimSize);
443  outOfBounds = !ucst.isEmpty();
444  if (outOfBounds && emitError) {
445  loadOrStoreOp.emitOpError()
446  << "memref out of upper bound access along dimension #" << (r + 1);
447  }
448 
449  // Check for a negative index.
450  FlatAffineConstraints lcst(*region.getConstraints());
451  std::fill(ineq.begin(), ineq.end(), 0);
452  // d_i <= -1;
453  lcst.addConstantUpperBound(r, -1);
454  outOfBounds = !lcst.isEmpty();
455  if (outOfBounds && emitError) {
456  loadOrStoreOp.emitOpError()
457  << "memref out of lower bound access along dimension #" << (r + 1);
458  }
459  }
460  return failure(outOfBounds);
461 }

469 // Returns in 'positions' the Block positions of 'op' in each ancestor

485 // Returns the Operation in a possibly nested set of Blocks, where the
486 // position of the operation is represented by 'positions', which has a
487 // Block position for each level of nesting.
488 static Operation *getInstAtPosition(ArrayRef<unsigned> positions,
489  unsigned level, Block *block) {
490  unsigned i = 0;
491  for (auto &op : *block) {
492  if (i != positions[level]) {
493  ++i;
494  continue;
495  }
496  if (level == positions.size() - 1)
497  return &op;
498  if (auto childAffineForOp = dyn_cast<AffineForOp>(op))
499  return getInstAtPosition(positions, level + 1,
500  childAffineForOp.getBody());
501 
502  for (auto &region : op.getRegions()) {
503  for (auto &b : region)
504  if (auto *ret = getInstAtPosition(positions, level + 1, &b))
505  return ret;
506  }
507  return nullptr;
508  }
509  return nullptr;
510 }

556 /// Computes in 'sliceUnion' the union of all slice bounds computed at
557 /// 'loopDepth' between all dependent pairs of ops in 'opsA' and 'opsB'.
558 /// Returns 'Success' if union was computed, 'failure' otherwise.
559 LogicalResult mlir::computeSliceUnion(ArrayRef<Operation *> opsA,
560  ArrayRef<Operation *> opsB,
561  unsigned loopDepth,
562  unsigned numCommonLoops,
563  bool isBackwardSlice,
564  ComputationSliceState *sliceUnion) {
565  // Compute the union of slice bounds between all pairs in 'opsA' and
566  // 'opsB' in 'sliceUnionCst'.
567  FlatAffineConstraints sliceUnionCst;
568  assert(sliceUnionCst.getNumDimAndSymbolIds() == 0);
569  std::vector<std::pair<Operation *, Operation *>> dependentOpPairs;
570  for (unsigned i = 0, numOpsA = opsA.size(); i < numOpsA; ++i) {
571  MemRefAccess srcAccess(opsA[i]);
572  for (unsigned j = 0, numOpsB = opsB.size(); j < numOpsB; ++j) {
573  MemRefAccess dstAccess(opsB[j]);
574  if (srcAccess.memref != dstAccess.memref)
575  continue;
576  // Check if 'loopDepth' exceeds nesting depth of src/dst ops.
577  if ((!isBackwardSlice && loopDepth > getNestingDepth(opsA[i])) ||
578  (isBackwardSlice && loopDepth > getNestingDepth(opsB[j]))) {
579  LLVM_DEBUG(llvm::dbgs() << "Invalid loop depth\n");
580  return failure();
581  }
582 
583  bool readReadAccesses = isa<AffineReadOpInterface>(srcAccess.opInst) &&
584  isa<AffineReadOpInterface>(dstAccess.opInst);
585  FlatAffineConstraints dependenceConstraints;
586  // Check dependence between 'srcAccess' and 'dstAccess'.
587  DependenceResult result = checkMemrefAccessDependence(

625  if (!sliceUnionCst.areIdsAlignedWithOther(tmpSliceCst)) {

628  // system.

639  // id alignment to constraint systems. This can occur if one constraint

641  // to unionBoundingBox below expects constraints for each Loop IV, even
642  // if they are the unsliced full loop bounds added here.
643  if (failed(addMissingLoopIVBounds(sliceUnionIVs, &sliceUnionCst)))
644  return failure();

646  return failure();

649  if (sliceUnionCst.getNumLocalIds() > 0 ||

651  failed(sliceUnionCst.unionBoundingBox(tmpSliceCst))) {
652  LLVM_DEBUG(llvm::dbgs()

654  return failure();
655  }
656  }
657  }

659  // Empty union.

661  return failure();

664  SmallVector<Operation *, 4> ops;
665  for (auto &dep : dependentOpPairs) {
666  ops.push_back(isBackwardSlice ? dep.second : dep.first);
667  }

670  getInnermostCommonLoopDepth(ops, surroundingLoops);

672  LLVM_DEBUG(llvm::dbgs() << "Exceeds max loop depth\n");
673  return failure();
674  }

677  unsigned numSliceLoopIVs = sliceUnionCst.getNumDimIds();
678

680  sliceUnionCst.convertLoopIVSymbolsToDims();

686  sliceUnionCst.getSliceBounds(/*offset=*/0, numSliceLoopIVs,
687  opsA[0]->getContext(), &sliceUnion->lbs,
688  &sliceUnion->ubs);

691  SmallVector<Value, 4> sliceBoundOperands;
692  sliceUnionCst.getIdValues(numSliceLoopIVs,
693  sliceUnionCst.getNumDimAndSymbolIds(),

697  sliceUnion->ivs.clear();

701  sliceUnion->insertPoint =

704  : std::prev(surroundingLoops[loopDepth - 1].getBody()->end());

707  // canonicalization.
708  sliceUnion->lbOperands.resize(numSliceLoopIVs, sliceBoundOperands);
709  sliceUnion->ubOperands.resize(numSliceLoopIVs, sliceBoundOperands);
710  return success();

713 const char *const kSliceFusionBarrierAttrName = "slice_fusion_barrier";
714 // Computes slice bounds by projecting out any loop IVs from
715 // 'dependenceConstraints' at depth greater than 'loopDepth', and computes slice
716 // bounds in 'sliceState' which represent the one loop nest's IVs in terms of
717 // the other loop nest's IVs, symbols and constants (using 'isBackwardsSlice').
718 void mlir::getComputationSliceState(
719  Operation *depSourceOp, Operation *depSinkOp,
720  FlatAffineConstraints *dependenceConstraints, unsigned loopDepth,
721  bool isBackwardSlice, ComputationSliceState *sliceState) {
722  // Get loop nest surrounding src operation.
723  SmallVector<AffineForOp, 4> srcLoopIVs;
724  getLoopIVs(*depSourceOp, &srcLoopIVs);
725  unsigned numSrcLoopIVs = srcLoopIVs.size();
726 
727  // Get loop nest surrounding dst operation.
728  SmallVector<AffineForOp, 4> dstLoopIVs;
729  getLoopIVs(*depSinkOp, &dstLoopIVs);
730  unsigned numDstLoopIVs = dstLoopIVs.size();
731 
732  assert((!isBackwardSlice && loopDepth <= numSrcLoopIVs) ||
733  (isBackwardSlice && loopDepth <= numDstLoopIVs));
734 
735  // Project out dimensions other than those up to 'loopDepth'.
736  unsigned pos = isBackwardSlice ? numSrcLoopIVs + loopDepth : loopDepth;
737  unsigned num =
738  isBackwardSlice ? numDstLoopIVs - loopDepth : numSrcLoopIVs - loopDepth;
739  dependenceConstraints->projectOut(pos, num);
740 
741  // Add slice loop IV values to 'sliceState'.
742  unsigned offset = isBackwardSlice ? 0 : loopDepth;
743  unsigned numSliceLoopIVs = isBackwardSlice ? numSrcLoopIVs : numDstLoopIVs;
744  dependenceConstraints->getIdValues(offset, offset + numSliceLoopIVs,
745  &sliceState->ivs);
746 
747  // Set up lower/upper bound affine maps for the slice.
748  sliceState->lbs.resize(numSliceLoopIVs, AffineMap());
749  sliceState->ubs.resize(numSliceLoopIVs, AffineMap());
750 
751  // Get bounds for slice IVs in terms of other IVs, symbols, and constants.
752  dependenceConstraints->getSliceBounds(offset, numSliceLoopIVs,
753  depSourceOp->getContext(),
754  &sliceState->lbs, &sliceState->ubs);
755 
756  // Set up bound operands for the slice's lower and upper bounds.
757  SmallVector<Value, 4> sliceBoundOperands;
758  unsigned numDimsAndSymbols = dependenceConstraints->getNumDimAndSymbolIds();
759  for (unsigned i = 0; i < numDimsAndSymbols; ++i) {
760  if (i < offset || i >= offset + numSliceLoopIVs) {
761  sliceBoundOperands.push_back(dependenceConstraints->getIdValue(i));
762  }
763  }
764 
765  // Give each bound its own copy of 'sliceBoundOperands' for subsequent
766  // canonicalization.
767  sliceState->lbOperands.resize(numSliceLoopIVs, sliceBoundOperands);
768  sliceState->ubOperands.resize(numSliceLoopIVs, sliceBoundOperands);
769 
770  // Set destination loop nest insertion point to block start at 'dstLoopDepth'.
771  sliceState->insertPoint =
772  isBackwardSlice ? dstLoopIVs[loopDepth - 1].getBody()->begin()
773  : std::prev(srcLoopIVs[loopDepth - 1].getBody()->end());
774 
775  llvm::SmallDenseSet<Value, 8> sequentialLoops;
776  if (isa<AffineReadOpInterface>(depSourceOp) &&
777  isa<AffineReadOpInterface>(depSinkOp)) {
778  // For read-read access pairs, clear any slice bounds on sequential loops.
779  // Get sequential loops in loop nest rooted at 'srcLoopIVs[0]'.
780  getSequentialLoops(isBackwardSlice ? srcLoopIVs[0] : dstLoopIVs[0],
781  &sequentialLoops);
782  }
783  // Clear all sliced loop bounds beginning at the first sequential loop, or
784  // first loop with a slice fusion barrier attribute..
785  // TODO: Use MemRef read/write regions instead of
786  // using 'kSliceFusionBarrierAttrName'.
787  auto getSliceLoop = [&](unsigned i) {
788  return isBackwardSlice ? srcLoopIVs[i] : dstLoopIVs[i];
789  };
790  for (unsigned i = 0; i < numSliceLoopIVs; ++i) {
791  Value iv = getSliceLoop(i).getInductionVar();
792  if (sequentialLoops.count(iv) == 0 &&
793  getSliceLoop(i).getAttr(kSliceFusionBarrierAttrName) == nullptr)
794  continue;
795  for (unsigned j = i; j < numSliceLoopIVs; ++j) {
796  sliceState->lbs[j] = AffineMap();
797  sliceState->ubs[j] = AffineMap();
798  }
799  break;
800  }
801 }

803 /// Creates a computation slice of the loop nest surrounding 'srcOpInst',
804 /// updates the slice loop bounds with any non-null bound maps specified in
805 /// 'sliceState', and inserts this slice into the loop nest surrounding
806 /// 'dstOpInst' at loop depth 'dstLoopDepth'.
807 // TODO: extend the slicing utility to compute slices that
808 // aren't necessarily a one-to-one relation b/w the source and destination. The
809 // relation between the source and destination could be many-to-many in general.
810 // TODO: the slice computation is incorrect in the cases
811 // where the dependence from the source to the destination does not cover the
812 // entire destination index set. Subtract out the dependent destination
813 // iterations from destination index set and check for emptiness --- this is one
814 // solution.
815 AffineForOp
816 mlir::insertBackwardComputationSlice(Operation *srcOpInst, Operation *dstOpInst,
817  unsigned dstLoopDepth,
818  ComputationSliceState *sliceState) {
819  // Get loop nest surrounding src operation.
820  SmallVector<AffineForOp, 4> srcLoopIVs;
821  getLoopIVs(*srcOpInst, &srcLoopIVs);
822  unsigned numSrcLoopIVs = srcLoopIVs.size();
823 
824  // Get loop nest surrounding dst operation.
825  SmallVector<AffineForOp, 4> dstLoopIVs;
826  getLoopIVs(*dstOpInst, &dstLoopIVs);
827  unsigned dstLoopIVsSize = dstLoopIVs.size();
828  if (dstLoopDepth > dstLoopIVsSize) {
829  dstOpInst->emitError("invalid destination loop depth");
830  return AffineForOp();
831  }
832 
833  // Find the op block positions of 'srcOpInst' within 'srcLoopIVs'.
834  SmallVector<unsigned, 4> positions;
835  // TODO: This code is incorrect since srcLoopIVs can be 0-d.
836  findInstPosition(srcOpInst, srcLoopIVs[0].getOperation()->getBlock(),
837  &positions);
838 
839  // Clone src loop nest and insert it a the beginning of the operation block
840  // of the loop at 'dstLoopDepth' in 'dstLoopIVs'.
841  auto dstAffineForOp = dstLoopIVs[dstLoopDepth - 1];
842  OpBuilder b(dstAffineForOp.getBody(), dstAffineForOp.getBody()->begin());
843  auto sliceLoopNest =
844  cast<AffineForOp>(b.clone(*srcLoopIVs[0].getOperation()));
845 
846  Operation *sliceInst =
847  getInstAtPosition(positions, /*level=*/0, sliceLoopNest.getBody());
848  // Get loop nest surrounding 'sliceInst'.
849  SmallVector<AffineForOp, 4> sliceSurroundingLoops;
850  getLoopIVs(*sliceInst, &sliceSurroundingLoops);
851 
852  // Sanity check.
853  unsigned sliceSurroundingLoopsSize = sliceSurroundingLoops.size();
854  (void)sliceSurroundingLoopsSize;
855  assert(dstLoopDepth + numSrcLoopIVs >= sliceSurroundingLoopsSize);
856  unsigned sliceLoopLimit = dstLoopDepth + numSrcLoopIVs;
857  (void)sliceLoopLimit;
858  assert(sliceLoopLimit >= sliceSurroundingLoopsSize);
859 
860  // Update loop bounds for loops in 'sliceLoopNest'.
861  for (unsigned i = 0; i < numSrcLoopIVs; ++i) {
862  auto forOp = sliceSurroundingLoops[dstLoopDepth + i];
863  if (AffineMap lbMap = sliceState->lbs[i])
864  forOp.setLowerBound(sliceState->lbOperands[i], lbMap);
865  if (AffineMap ubMap = sliceState->ubs[i])
866  forOp.setUpperBound(sliceState->ubOperands[i], ubMap);
867  }
868  return sliceLoopNest;
869 }

951 static Optional<int64_t> getMemoryFootprintBytes(Block &block,
952  Block::iterator start,
953  Block::iterator end,
954  int memorySpace) {
955  SmallDenseMap<Value, std::unique_ptr<MemRefRegion>, 4> regions;
956 
957  // Walk this 'affine.for' operation to gather all memory regions.
958  auto result = block.walk(start, end, [&](Operation *opInst) -> WalkResult {
959  if (!isa<AffineReadOpInterface, AffineWriteOpInterface>(opInst)) {
960  // Neither load nor a store op.
961  return WalkResult::advance();
962  }
963 
964  // Compute the memref region symbolic in any IVs enclosing this block.
965  auto region = std::make_unique<MemRefRegion>(opInst->getLoc());
966  if (failed(
967  region->compute(opInst,
968  /*loopDepth=*/getNestingDepth(&*block.begin())))) {
969  return opInst->emitError("error obtaining memory region\n");
970  }
971 
972  auto it = regions.find(region->memref);
973  if (it == regions.end()) {
974  regions[region->memref] = std::move(region);
975  } else if (failed(it->second->unionBoundingBox(*region))) {
976  return opInst->emitWarning(
977  "getMemoryFootprintBytes: unable to perform a union on a memory "
978  "region");
979  }
980  return WalkResult::advance();
981  });
982  if (result.wasInterrupted())
983  return None;
984 
985  int64_t totalSizeInBytes = 0;
986  for (const auto &region : regions) {
987  Optional<int64_t> size = region.second->getRegionSize();
988  if (!size.hasValue())
989  return None;
990  totalSizeInBytes += size.getValue();
991  }
992  return totalSizeInBytes;
993 }

1033  unsigned depth = getNestingDepth(forOp) + 1;
