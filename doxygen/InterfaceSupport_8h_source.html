27 /// This class represents an abstract interface. An interface is a simplified
28 /// mechanism for attaching concept based polymorphism to a class hierarchy. An
29 /// interace is comprised of two components:
30 /// * The derived interface class: This is what users interact with, and invoke
31 /// methods on.
32 /// * An interface `Trait` class: This is the class that is attached to the
33 /// object implementing the interface. It is the mechanism with which models
34 /// are specialized.
35 ///
36 /// Derived interfaces types must provide the following template types:
37 /// * ConcreteType: The CRTP derived type.
38 /// * ValueT: The opaque type the derived interface operates on. For example
39 /// `Operation*` for operation interfaces, or `Attribute` for
40 /// attribute interfaces.
41 /// * Traits: A class that contains definitions for a 'Concept' and a 'Model'
42 /// class. The 'Concept' class defines an abstract virtual interface,
43 /// where as the 'Model' class implements this interface for a
44 /// specific derived T type. Both of these classes *must* not contain
45 /// non-static data. A simple example is shown below:
46 ///
47 /// ```c++
48 /// struct ExampleInterfaceTraits {
49 /// struct Concept {
50 /// virtual unsigned getNumInputs(T t) const = 0;
51 /// };
52 /// template <typename DerivedT> class Model {
53 /// unsigned getNumInputs(T t) const final {
54 /// return cast<DerivedT>(t).getNumInputs();
55 /// }
56 /// };
57 /// };
58 /// ```
59 ///
60 /// * BaseType: A desired base type for the interface. This is a class that
61 /// provides that provides specific functionality for the `ValueT`
62 /// value. For instance the specific `Op` that will wrap the
63 /// `Operation*` for an `OpInterface`.
64 /// * BaseTrait: The base type for the interface trait. This is the base class
65 /// to use for the interface trait that will be attached to each
66 /// instance of `ValueT` that implements this interface.
67 ///
68 template <typename ConcreteType, typename ValueT, typename Traits,
69  typename BaseType,
70  template <typename, template <typename> class> class BaseTrait>
71 class Interface : public BaseType {
72 public:
73  using Concept = typename Traits::Concept;
74  template <typename T> using Model = typename Traits::template Model<T>;
75  using InterfaceBase =
76  Interface<ConcreteType, ValueT, Traits, BaseType, BaseTrait>;
77 
78  Interface(ValueT t = ValueT())
79  : BaseType(t), impl(t ? ConcreteType::getInterfaceFor(t) : nullptr) {
80  assert((!t || impl) &&
81  "instantiating an interface with an unregistered operation");
82  }
83 
84  /// Support 'classof' by checking if the given object defines the concrete
85  /// interface.
86  static bool classof(ValueT t) { return ConcreteType::getInterfaceFor(t); }
87 
88  /// Define an accessor for the ID of this interface.
89  static TypeID getInterfaceID() { return TypeID::get<ConcreteType>(); }
90 
91  /// This is a special trait that registers a given interface with an object.
92  template <typename ConcreteT>
93  struct Trait : public BaseTrait<ConcreteT, Trait> {
94  /// Define an accessor for the ID of this interface.
95  static TypeID getInterfaceID() { return TypeID::get<ConcreteType>(); }
96 
97  /// Provide an accessor to a static instance of the interface model for the
98  /// concrete T type.
99  /// The implementation is inspired from Sean Parent's concept-based
100  /// polymorphism. A key difference is that the set of classes erased is
101  /// statically known, which alleviates the need for using dynamic memory
102  /// allocation.
103  /// We use a zero-sized templated class `Model<ConcreteT>` to emit the
104  /// virtual table and generate a singleton object for each instantiation of
105  /// this class.
106  static Concept &instance() {
107  static Model<ConcreteT> singleton;
108  return singleton;
109  }
110  };
111 
112 protected:
113  /// Get the raw concept in the correct derived concept type.
114  const Concept *getImpl() const { return impl; }
115  Concept *getImpl() { return impl; }
116 
117 private:
118  /// A pointer to the impl concept object.
119  Concept *impl;
120 };

144 /// This class provides an efficient mapping between a given `Interface` type,
145 /// and a particular implementation of its concept.
146 class InterfaceMap {
147 public:
148  /// Construct an InterfaceMap with the given set of template types. For
149  /// convenience given that object trait lists may contain other non-interface
150  /// types, not all of the types need to be interfaces. The provided types that
151  /// do not represent interfaces are not added to the interface map.
152  template <typename... Types> static InterfaceMap get() {
153  return InterfaceMap(MapBuilder::create<Types...>());
154  }
155 
156  /// Returns an instance of the concept object for the given interface if it
157  /// was registered to this map, null otherwise.
158  template <typename T> typename T::Concept *lookup() const {
159  if (!interfaces)
160  return nullptr;
161  return reinterpret_cast<typename T::Concept *>(
162  interfaces->lookup(T::getInterfaceID()));
163  }
164 
165 private:
166  /// This struct provides support for building a map of interfaces.
167  class MapBuilder {
168  public:
169  template <typename... Types>
170  static std::unique_ptr<llvm::SmallDenseMap<TypeID, void *>> create() {
171  // Filter the provided types for those that are interfaces. This reduces
172  // the amount of maps that are generated.
173  return createImpl((typename FilterTypes<detect_get_interface_id,
174  Types...>::type *)nullptr);
175  }
176 
177  private:
178  /// Trait to check if T provides a static 'getInterfaceID' method.
179  template <typename T, typename... Args>
180  using has_get_interface_id = decltype(T::getInterfaceID());
181  template <typename T>
182  using detect_get_interface_id = llvm::is_detected<has_get_interface_id, T>;
183 
184  template <typename... Ts>
185  static std::unique_ptr<llvm::SmallDenseMap<TypeID, void *>>
186  createImpl(std::tuple<Ts...> *) {
187  // Only create an instance of the map if there are any interface types.
188  if (sizeof...(Ts) == 0)
189  return std::unique_ptr<llvm::SmallDenseMap<TypeID, void *>>();
190 
191  auto map = std::make_unique<llvm::SmallDenseMap<TypeID, void *>>();
192  (void)std::initializer_list<int>{
193  0, (map->try_emplace(Ts::getInterfaceID(), &Ts::instance()), 0)...};
194  return map;
195  }
196  };
197 
198 private:
199  InterfaceMap(std::unique_ptr<llvm::SmallDenseMap<TypeID, void *>> interfaces)
200  : interfaces(std::move(interfaces)) {}
201 
202  /// The internal map of interfaces. This is constructed statically for each
203  /// set of interfaces.
204  std::unique_ptr<llvm::SmallDenseMap<TypeID, void *>> interfaces;
205 };
