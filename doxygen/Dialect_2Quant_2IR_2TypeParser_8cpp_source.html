25 static IntegerType parseStorageType(DialectAsmParser &parser, bool &isSigned) {
26  auto typeLoc = parser.getCurrentLocation();
27  IntegerType type;
28 
29  // Parse storage type (alpha_ident, integer_literal).
30  StringRef identifier;
31  unsigned storageTypeWidth = 0;
32  if (failed(parser.parseOptionalKeyword(&identifier))) {
33  // If we didn't parse a keyword, this must be a signed type.
34  if (parser.parseType(type))
35  return nullptr;
36  isSigned = true;
37  storageTypeWidth = type.getWidth();
38 
39  // Otherwise, this must be an unsigned integer (`u` integer-literal).
40  } else {
41  if (!identifier.consume_front("u")) {
42  parser.emitError(typeLoc, "illegal storage type prefix");
43  return nullptr;
44  }
45  if (identifier.getAsInteger(10, storageTypeWidth)) {
46  parser.emitError(typeLoc, "expected storage type width");
47  return nullptr;
48  }
49  isSigned = false;
50  type = parser.getBuilder().getIntegerType(storageTypeWidth);
51  }
52 
53  if (storageTypeWidth == 0 ||
54  storageTypeWidth > QuantizedType::MaxStorageBits) {
55  parser.emitError(typeLoc, "illegal storage type size: ")
56  << storageTypeWidth;
57  return nullptr;
58  }
59 
60  return type;
61 }

63 static ParseResult parseStorageRange(DialectAsmParser &parser,
64  IntegerType storageType, bool isSigned,
65  int64_t &storageTypeMin,
66  int64_t &storageTypeMax) {
67  int64_t defaultIntegerMin = QuantizedType::getDefaultMinimumForInteger(
68  isSigned, storageType.getWidth());
69  int64_t defaultIntegerMax = QuantizedType::getDefaultMaximumForInteger(
70  isSigned, storageType.getWidth());
71  if (failed(parser.parseOptionalLess())) {
72  storageTypeMin = defaultIntegerMin;
73  storageTypeMax = defaultIntegerMax;
74  return success();
75  }
76 
77  // Explicit storage min and storage max.
78  llvm::SMLoc minLoc = parser.getCurrentLocation(), maxLoc;
79  if (parser.parseInteger(storageTypeMin) || parser.parseColon() ||
80  parser.getCurrentLocation(&maxLoc) ||
81  parser.parseInteger(storageTypeMax) || parser.parseGreater())
82  return failure();
83  if (storageTypeMin < defaultIntegerMin) {
84  return parser.emitError(minLoc, "illegal storage type minimum: ")
85  << storageTypeMin;
86  }
87  if (storageTypeMax > defaultIntegerMax) {
88  return parser.emitError(maxLoc, "illegal storage type maximum: ")
89  << storageTypeMax;
90  }
91  return success();
92 }

94 /// Parses a UniformQuantizedType.
95 ///
96 /// uniform_per_layer ::= `any<` storage-spec (expressed-type-spec)?`>`
97 /// storage-spec ::= storage-type (`<` storage-range `>`)?
98 /// storage-range ::= integer-literal `:` integer-literal
99 /// storage-type ::= (`i` | `u`) integer-literal
100 /// expressed-type-spec ::= `:` `f` integer-literal
101 static Type parseAnyType(DialectAsmParser &parser, Location loc) {
102  IntegerType storageType;
103  FloatType expressedType;
104  unsigned typeFlags = 0;
105  int64_t storageTypeMin;
106  int64_t storageTypeMax;
107 
108  // Type specification.
109  if (parser.parseLess())
110  return nullptr;
111 
112  // Storage type.
113  bool isSigned = false;
114  storageType = parseStorageType(parser, isSigned);
115  if (!storageType) {
116  return nullptr;
117  }
118  if (isSigned) {
119  typeFlags |= QuantizationFlags::Signed;
120  }
121 
122  // Storage type range.
123  if (parseStorageRange(parser, storageType, isSigned, storageTypeMin,
124  storageTypeMax)) {
125  return nullptr;
126  }
127 
128  // Optional expressed type.
129  if (succeeded(parser.parseOptionalColon())) {
130  if (parser.parseType(expressedType)) {
131  return nullptr;
132  }
133  }
134 
135  if (parser.parseGreater()) {
136  return nullptr;
137  }
138 
139  return AnyQuantizedType::getChecked(typeFlags, storageType, expressedType,
140  storageTypeMin, storageTypeMax, loc);
141 }

143 static ParseResult parseQuantParams(DialectAsmParser &parser, double &scale,
144  int64_t &zeroPoint) {
145  // scale[:zeroPoint]?
146  // scale.
147  if (parser.parseFloat(scale))
148  return failure();
149 
150  // zero point.
151  zeroPoint = 0;
152  if (failed(parser.parseOptionalColon())) {
153  // Default zero point.
154  return success();
155  }
156 
157  return parser.parseInteger(zeroPoint);
158 }

160 /// Parses a UniformQuantizedType.
161 ///
162 /// uniform_type ::= uniform_per_layer
163 /// | uniform_per_axis
164 /// uniform_per_layer ::= `uniform<` storage-spec expressed-type-spec
165 /// `,` scale-zero `>`
166 /// uniform_per_axis ::= `uniform<` storage-spec expressed-type-spec
167 /// axis-spec `,` scale-zero-list `>`
168 /// storage-spec ::= storage-type (`<` storage-range `>`)?
169 /// storage-range ::= integer-literal `:` integer-literal
170 /// storage-type ::= (`i` | `u`) integer-literal
171 /// expressed-type-spec ::= `:` `f` integer-literal
172 /// axis-spec ::= `:` integer-literal
173 /// scale-zero ::= float-literal `:` integer-literal
174 /// scale-zero-list ::= `{` scale-zero (`,` scale-zero)* `}`
175 static Type parseUniformType(DialectAsmParser &parser, Location loc) {
176  IntegerType storageType;
177  FloatType expressedType;
178  unsigned typeFlags = 0;
179  int64_t storageTypeMin;
180  int64_t storageTypeMax;
181  bool isPerAxis = false;
182  int32_t quantizedDimension;
183  SmallVector<double, 1> scales;
184  SmallVector<int64_t, 1> zeroPoints;
185 
186  // Type specification.
187  if (parser.parseLess()) {
188  return nullptr;
189  }
190 
191  // Storage type.
192  bool isSigned = false;
193  storageType = parseStorageType(parser, isSigned);
194  if (!storageType) {
195  return nullptr;
196  }
197  if (isSigned) {
198  typeFlags |= QuantizationFlags::Signed;
199  }
200 
201  // Storage type range.
202  if (parseStorageRange(parser, storageType, isSigned, storageTypeMin,
203  storageTypeMax)) {
204  return nullptr;
205  }
206 
207  // Expressed type.
208  if (parser.parseColon() || parser.parseType(expressedType)) {
209  return nullptr;
210  }
211 
212  // Optionally parse quantized dimension for per-axis quantization.
213  if (succeeded(parser.parseOptionalColon())) {
214  if (parser.parseInteger(quantizedDimension))
215  return nullptr;
216  isPerAxis = true;
217  }
218 
219  // Comma leading into range_spec.
220  if (parser.parseComma()) {
221  return nullptr;
222  }
223 
224  // Parameter specification.
225  // For per-axis, ranges are in a {} delimitted list.
226  if (isPerAxis) {
227  if (parser.parseLBrace()) {
228  return nullptr;
229  }
230  }
231 
232  // Parse scales/zeroPoints.
233  llvm::SMLoc scaleZPLoc = parser.getCurrentLocation();
234  do {
235  scales.resize(scales.size() + 1);
236  zeroPoints.resize(zeroPoints.size() + 1);
237  if (parseQuantParams(parser, scales.back(), zeroPoints.back())) {
238  return nullptr;
239  }
240  } while (isPerAxis && succeeded(parser.parseOptionalComma()));
241 
242  if (isPerAxis) {
243  if (parser.parseRBrace()) {
244  return nullptr;
245  }
246  }
247 
248  if (parser.parseGreater()) {
249  return nullptr;
250  }
251 
252  if (!isPerAxis && scales.size() > 1) {
253  return (parser.emitError(scaleZPLoc,
254  "multiple scales/zeroPoints provided, but "
255  "quantizedDimension wasn't specified"),
256  nullptr);
257  }
258 
259  if (isPerAxis) {
260  ArrayRef<double> scalesRef(scales.begin(), scales.end());
261  ArrayRef<int64_t> zeroPointsRef(zeroPoints.begin(), zeroPoints.end());
262  return UniformQuantizedPerAxisType::getChecked(
263  typeFlags, storageType, expressedType, scalesRef, zeroPointsRef,
264  quantizedDimension, storageTypeMin, storageTypeMax, loc);
265  }
266 
267  return UniformQuantizedType::getChecked(typeFlags, storageType, expressedType,
268  scales.front(), zeroPoints.front(),
269  storageTypeMin, storageTypeMax, loc);
270 }

272 /// Parse a type registered to this dialect.
273 Type QuantizationDialect::parseType(DialectAsmParser &parser) const {
274  Location loc = parser.getEncodedSourceLoc(parser.getNameLoc());
275 
276  // All types start with an identifier that we switch on.
277  StringRef typeNameSpelling;
278  if (failed(parser.parseKeyword(&typeNameSpelling)))
279  return nullptr;
280 
281  if (typeNameSpelling == "uniform")
282  return parseUniformType(parser, loc);
283  if (typeNameSpelling == "any")
284  return parseAnyType(parser, loc);
285 
286  parser.emitError(parser.getNameLoc(),
287  "unknown quantized type " + typeNameSpelling);
288  return nullptr;
289 }

291 static void printStorageType(QuantizedType type, DialectAsmPrinter &out) {
292  // storage type
293  unsigned storageWidth = type.getStorageTypeIntegralWidth();
294  bool isSigned = type.isSigned();
295  if (isSigned) {
296  out << "i" << storageWidth;
297  } else {
298  out << "u" << storageWidth;
299  }
300 
301  // storageTypeMin and storageTypeMax if not default.
302  int64_t defaultIntegerMin =
303  QuantizedType::getDefaultMinimumForInteger(isSigned, storageWidth);
304  int64_t defaultIntegerMax =
305  QuantizedType::getDefaultMaximumForInteger(isSigned, storageWidth);
306  if (defaultIntegerMin != type.getStorageTypeMin() ||
307  defaultIntegerMax != type.getStorageTypeMax()) {
308  out << "<" << type.getStorageTypeMin() << ":" << type.getStorageTypeMax()
309  << ">";
310  }
311 }

366 /// Print a type registered to this dialect.
367 void QuantizationDialect::printType(Type type, DialectAsmPrinter &os) const {
368  if (auto anyType = type.dyn_cast<AnyQuantizedType>())
369  printAnyQuantizedType(anyType, os);
370  else if (auto uniformType = type.dyn_cast<UniformQuantizedType>())
371  printUniformQuantizedType(uniformType, os);
372  else if (auto perAxisType = type.dyn_cast<UniformQuantizedPerAxisType>())
373  printUniformQuantizedPerAxisType(perAxisType, os);
374  else
375  llvm_unreachable("Unhandled quantized type");
376 }
