497  /// Walk the operation in postorder, calling the callback for each nested
498  /// operation(including this one). The callback method can take any of the
499  /// following forms:
500  /// void(Operation*) : Walk all operations opaquely.
501  /// * op->walk([](Operation *nestedOp) { ...});
502  /// void(OpT) : Walk all operations of the given derived type.
503  /// * op->walk([](ReturnOp returnOp) { ...});
504  /// WalkResult(Operation*|OpT) : Walk operations, but allow for
505  /// interruption/cancellation.
506  /// * op->walk([](... op) {
507  /// // Interrupt, i.e cancel, the walk based on some invariant.
508  /// if (some_invariant)
509  /// return WalkResult::interrupt();
510  /// return WalkResult::advance();
511  /// });
512  template <typename FnT, typename RetT = detail::walkResultType<FnT>>
513  RetT walk(FnT &&callback) {
514  return detail::walkOperations(this, std::forward<FnT>(callback));
515  }
