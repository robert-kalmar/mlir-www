25 /// A pattern to convert Generic Linalg operations which work on tensors to
26 /// use buffers. A buffer is allocated using BufferAssignmentPlacer for
27 /// each operation result. BufferPlacement pass should be later used to move
28 /// Alloc operations to the correct positions and insert the missing Dealloc
29 /// operations in the correct places.
30 class GenericOpConverter
31  : public BufferAssignmentOpConversionPattern<linalg::GenericOp> {
32 public:
33  using BufferAssignmentOpConversionPattern<
34  linalg::GenericOp>::BufferAssignmentOpConversionPattern;
35 
36  LogicalResult
37  matchAndRewrite(linalg::GenericOp op, ArrayRef<Value> operands,
38  ConversionPatternRewriter &rewriter) const final {
39  Location loc = op.getLoc();
40  ResultRange results = op.getOperation()->getResults();
41  SmallVector<Value, 2> newArgs, newResults;
42  newArgs.reserve(operands.size() + results.size());
43  newArgs.append(operands.begin(), operands.end());
44  newResults.reserve(results.size());
45 
46  // Update all types to memref types.
47  for (auto result : results) {
48  auto type = result.getType().cast<ShapedType>();
49  assert(type && "tensor to buffer conversion expects ranked results");
50  if (!type.hasStaticShape())
51  return rewriter.notifyMatchFailure(
52  op, "dynamic shapes not currently supported");
53  auto memrefType = MemRefType::get(type.getShape(), type.getElementType());
54 
55  // Compute alloc position and insert a custom allocation node.
56  OpBuilder::InsertionGuard guard(rewriter);
57  rewriter.restoreInsertionPoint(
58  bufferAssignment->computeAllocPosition(result));
59  auto alloc = rewriter.create<AllocOp>(loc, memrefType);
60  newArgs.push_back(alloc);
61  newResults.push_back(alloc);
62  }
63 
64  // Generate a new linalg operation that works on buffers.
65  auto linalgOp = rewriter.create<linalg::GenericOp>(
66  loc, llvm::None, newArgs, rewriter.getI64IntegerAttr(operands.size()),
67  rewriter.getI64IntegerAttr(results.size()), op.indexing_maps(),
68  op.iterator_types(), op.docAttr(), op.library_callAttr(),
69  op.symbol_sourceAttr());
70 
71  // Create a new block in the region of the new Generic Op.
72  Block &oldBlock = op.getRegion().front();
73  Region &newRegion = linalgOp.region();
74  Block *newBlock = rewriter.createBlock(&newRegion, newRegion.begin(),
75  oldBlock.getArgumentTypes());
76 
77  // Add the result arguments to the new block.
78  for (auto result : newResults)
79  newBlock->addArgument(
80  result.getType().cast<ShapedType>().getElementType());
81 
82  // Clone the body of the old block to the new block.
83  BlockAndValueMapping mapping;
84  for (unsigned i = 0; i < oldBlock.getNumArguments(); i++)
85  mapping.map(oldBlock.getArgument(i), newBlock->getArgument(i));
86  rewriter.setInsertionPointToEnd(newBlock);
87  for (auto &op : oldBlock.getOperations()) {
88  Operation *clonedOp = rewriter.clone(op, mapping);
89  mapping.map(op.getResults(), clonedOp->getResults());
90  }
91 
92  // Replace the results of the old Generic Op with the results of the new
93  // one.
94  rewriter.replaceOp(op, newResults);
95  return success();
96  }
97 };

99 /// Populate the given list with patterns to convert Linalg operations on
100 /// tensors to buffers.
101 static void populateConvertLinalgOnTensorsToBuffersPattern(
102  MLIRContext *context, BufferAssignmentPlacer *placer,
103  TypeConverter *converter, OwningRewritePatternList *patterns) {
104  populateWithBufferAssignmentOpConversionPatterns<
105  mlir::ReturnOp, mlir::ReturnOp, linalg::CopyOp,
106  /*allowMemrefFunctionResults=*/false>(context, placer, converter,
107  patterns);
108  patterns->insert<GenericOpConverter>(context, placer, converter);
109 }

111 /// Converts Linalg operations that work on tensor-type operands or results to
112 /// work on buffers.
113 struct ConvertLinalgOnTensorsToBuffers
114  : public LinalgOnTensorsToBuffersBase<ConvertLinalgOnTensorsToBuffers> {
115  void runOnOperation() override {
116  MLIRContext &context = getContext();
117  ConversionTarget target(context);
118  BufferAssignmentTypeConverter converter;
119 
120  // Mark all Standard operations legal.
121  target.addLegalDialect<StandardOpsDialect>();
122 
123  // Mark all Linalg operations illegal as long as they work on tensors.
124  auto isLegalOperation = [&](Operation *op) {
125  return converter.isLegal(op);
126  };
127  target.addDynamicallyLegalDialect<linalg::LinalgDialect>(
128  Optional<ConversionTarget::DynamicLegalityCallbackFn>(
129  isLegalOperation));
130 
131  // Mark Standard Return operations illegal as long as one operand is tensor.
132  target.addDynamicallyLegalOp<mlir::ReturnOp>([&](mlir::ReturnOp returnOp) {
133  return converter.isLegal(returnOp.getOperandTypes());
134  });
135 
136  // Mark the function operation illegal as long as an argument is tensor.
137  target.addDynamicallyLegalOp<FuncOp>([&](FuncOp funcOp) {
138  return converter.isSignatureLegal(funcOp.getType()) &&
139  llvm::none_of(funcOp.getType().getResults(),
140  [&](Type type) { return type.isa<MemRefType>(); }) &&
141  converter.isLegal(&funcOp.getBody());
142  });
143 
144  // Walk over all the functions to apply buffer assignment.
145  getOperation().walk([&](FuncOp function) -> WalkResult {
146  OwningRewritePatternList patterns;
147  BufferAssignmentPlacer placer(function);
148  populateConvertLinalgOnTensorsToBuffersPattern(&context, &placer,
149  &converter, &patterns);
150 
151  // Applying full conversion
152  return applyFullConversion(function, target, patterns);
153  });
154  }
155 };
156 } // end anonymous namespace
