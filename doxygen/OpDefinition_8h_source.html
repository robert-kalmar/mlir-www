269 /// This template defines the foldHook as used by AbstractOperation.
270 ///
271 /// The default implementation uses a general fold method that can be defined on
272 /// custom ops which can return multiple results.
273 template <typename ConcreteType, bool isSingleResult, typename = void>
274 class FoldingHook {
275 public:
276  /// This is an implementation detail of the constant folder hook for
277  /// AbstractOperation.
278  static LogicalResult foldHook(Operation *op, ArrayRef<Attribute> operands,
279  SmallVectorImpl<OpFoldResult> &results) {
280  return cast<ConcreteType>(op).fold(operands, results);
281  }
282 
283  /// This hook implements a generalized folder for this operation. Operations
284  /// can implement this to provide simplifications rules that are applied by
285  /// the Builder::createOrFold API and the canonicalization pass.
286  ///
287  /// This is an intentionally limited interface - implementations of this hook
288  /// can only perform the following changes to the operation:
289  ///
290  /// 1. They can leave the operation alone and without changing the IR, and
291  /// return failure.
292  /// 2. They can mutate the operation in place, without changing anything else
293  /// in the IR. In this case, return success.
294  /// 3. They can return a list of existing values that can be used instead of
295  /// the operation. In this case, fill in the results list and return
296  /// success. The caller will remove the operation and use those results
297  /// instead.
298  ///
299  /// This allows expression of some simple in-place canonicalizations (e.g.
300  /// "x+0 -> x", "min(x,y,x,z) -> min(x,y,z)", "x+y-x -> y", etc), as well as
301  /// generalized constant folding.
302  ///
303  /// If not overridden, this fallback implementation always fails to fold.
304  ///
305  LogicalResult fold(ArrayRef<Attribute> operands,
306  SmallVectorImpl<OpFoldResult> &results) {
307  return failure();
308  }
309 };

311 /// This template specialization defines the foldHook as used by
312 /// AbstractOperation for single-result operations. This gives the hook a nicer
313 /// signature that is easier to implement.
314 template <typename ConcreteType, bool isSingleResult>
315 class FoldingHook<ConcreteType, isSingleResult,
316  typename std::enable_if<isSingleResult>::type> {
317 public:
318  /// If the operation returns a single value, then the Op can be implicitly
319  /// converted to an Value. This yields the value of the only result.
320  operator Value() {
321  return static_cast<ConcreteType *>(this)->getOperation()->getResult(0);
322  }
323 
324  /// This is an implementation detail of the constant folder hook for
325  /// AbstractOperation.
326  static LogicalResult foldHook(Operation *op, ArrayRef<Attribute> operands,
327  SmallVectorImpl<OpFoldResult> &results) {
328  auto result = cast<ConcreteType>(op).fold(operands);
329  if (!result)
330  return failure();
331 
332  // Check if the operation was folded in place. In this case, the operation
333  // returns itself.
334  if (result.template dyn_cast<Value>() != op->getResult(0))
335  results.push_back(result);
336  return success();
337  }
338 
339  /// This hook implements a generalized folder for this operation. Operations
340  /// can implement this to provide simplifications rules that are applied by
341  /// the Builder::createOrFold API and the canonicalization pass.
342  ///
343  /// This is an intentionally limited interface - implementations of this hook
344  /// can only perform the following changes to the operation:
345  ///
346  /// 1. They can leave the operation alone and without changing the IR, and
347  /// return nullptr.
348  /// 2. They can mutate the operation in place, without changing anything else
349  /// in the IR. In this case, return the operation itself.
350  /// 3. They can return an existing SSA value that can be used instead of
351  /// the operation. In this case, return that value. The caller will
352  /// remove the operation and use that result instead.
353  ///
354  /// This allows expression of some simple in-place canonicalizations (e.g.
355  /// "x+0 -> x", "min(x,y,x,z) -> min(x,y,z)", "x+y-x -> y", etc), as well as
356  /// generalized constant folding.
357  ///
358  /// If not overridden, this fallback implementation always fails to fold.
359  ///
360  OpFoldResult fold(ArrayRef<Attribute> operands) { return {}; }
361 };

406 /// Helper class for implementing traits. Clients are not expected to interact
407 /// with this directly, so its members are all protected.
408 template <typename ConcreteType, template <typename> class TraitType>
409 class TraitBase {
410 protected:
411  /// Return the ultimate Operation being worked on.
412  Operation *getOperation() {
413  // We have to cast up to the trait type, then to the concrete type, then to
414  // the BaseState class in explicit hops because the concrete type will
415  // multiply derive from the (content free) TraitBase class, and we need to
416  // be able to disambiguate the path for the C++ compiler.
417  auto *trait = static_cast<TraitType<ConcreteType> *>(this);
418  auto *concrete = static_cast<ConcreteType *>(trait);
419  auto *base = static_cast<OpState *>(concrete);
420  return base->getOperation();
421  }
422 
423  /// Provide default implementations of trait hooks. This allows traits to
424  /// provide exactly the overrides they care about.
425  static LogicalResult verifyTrait(Operation *op) { return success(); }
426  static AbstractOperation::OperationProperties getTraitProperties() {
427  return 0;
428  }
429 };
