27 /// ModuleOp represents a module, or an operation containing one region with a
28 /// single block containing opaque operations. The region of a module is not
29 /// allowed to implicitly capture global values, and all external references
30 /// must use symbolic references via attributes(e.g. via a string name).

72  /// Iteration over the operations in the module.

75  iterator begin() { return getBody()->begin(); }

79  /// This returns a range of operations of the given type 'T' held within the

85  /// Insert the operation into the back of the body, before the terminator.
86  void push_back(Operation *op) {
87  insert(Block::iterator(getBody()->getTerminator()), op);
88  }
89 
90  /// Insert the operation at the given insertion point. Note: The operation is
91  /// never inserted after the terminator, even if the insertion point is end().
92  void insert(Operation *insertPt, Operation *op) {
93  insert(Block::iterator(insertPt), op);
94  }
95  void insert(Block::iterator insertPt, Operation *op) {
96  auto *body = getBody();
97  if (insertPt == body->end())
98  insertPt = Block::iterator(body->getTerminator());
99  body->getOperations().insert(insertPt, op);
100  }

110 /// The ModuleTerminatorOp is a special terminator operation for the body of a
111 /// ModuleOp, it has no semantic meaning beyond keeping the body of a ModuleOp
112 /// well-formed.
113 ///
114 /// This operation does _not_ have a custom syntax. However, ModuleOp will omit
115 /// the terminator in their custom syntax for brevity.
116 class ModuleTerminatorOp
117  : public Op<ModuleTerminatorOp, OpTrait::ZeroOperands, OpTrait::ZeroResult,
118  OpTrait::HasParent<ModuleOp>::Impl, OpTrait::IsTerminator> {
119 public:
120  using Op::Op;
121  static StringRef getOperationName() { return "module_terminator"; }
122  static void build(OpBuilder &, OperationState &) {}
123 };
