411  // Given subscript indices and array sizes in row-major order,
412  // i_n, i_{n-1}, ..., i_1
413  // s_n, s_{n-1}, ..., s_1

481 /// Generic implementation of one-to-one conversion from "SourceOp" to
482 /// "TargetOp" where the latter belongs to the LLVM dialect or an equivalent.
483 /// Upholds a convention that multi-result operations get converted into an
484 /// operation returning the LLVM IR structure type, in which case individual
485 /// values must be extracted from using LLVM::ExtractValueOp before being used.
486 template <typename SourceOp, typename TargetOp>
487 class OneToOneConvertToLLVMPattern : public ConvertOpToLLVMPattern<SourceOp> {
488 public:
489  using ConvertOpToLLVMPattern<SourceOp>::ConvertOpToLLVMPattern;
490  using Super = OneToOneConvertToLLVMPattern<SourceOp, TargetOp>;
491 
492  /// Converts the type of the result to an LLVM type, pass operands as is,
493  /// preserve attributes.
494  LogicalResult
495  matchAndRewrite(Operation *op, ArrayRef<Value> operands,
496  ConversionPatternRewriter &rewriter) const override {
497  return LLVM::detail::oneToOneRewrite(op, TargetOp::getOperationName(),
498  operands, this->typeConverter,
499  rewriter);
500  }
