65 static LogicalResult verify(ReductionOp op) {
66  // Verify for 1-D vector.
67  int64_t rank = op.getVectorType().getRank();
68  if (rank != 1)
69  return op.emitOpError("unsupported reduction rank: ") << rank;
70 
71  // Verify supported reduction kind.
72  auto kind = op.kind();
73  Type eltType = op.dest().getType();
74  if (kind == "add" || kind == "mul" || kind == "min" || kind == "max") {
75  if (!eltType.isF32() && !eltType.isF64() &&
76  !eltType.isSignlessInteger(32) && !eltType.isSignlessInteger(64))
77  return op.emitOpError("unsupported reduction type");
78  } else if (kind == "and" || kind == "or" || kind == "xor") {
79  if (!eltType.isSignlessInteger(32) && !eltType.isSignlessInteger(64))
80  return op.emitOpError("unsupported reduction type");
81  } else {
82  return op.emitOpError("unknown reduction kind: ") << kind;
83  }
84 
85  // Verify optional accumulator.
86  if (!op.acc().empty()) {
87  if (kind != "add" && kind != "mul")
88  return op.emitOpError("no accumulator for reduction kind: ") << kind;
89  if (!eltType.isF32() && !eltType.isF64())
90  return op.emitOpError("no accumulator for type: ") << eltType;
91  }
92 
93  return success();
94 }

226 static LogicalResult verifyOutputShape(
227  ContractionOp op, VectorType lhsType, VectorType rhsType, Type accType,
228  Type resType,
229  const std::vector<std::pair<int64_t, int64_t>> &contractingDimMap,
230  const std::vector<std::pair<int64_t, int64_t>> &batchDimMap) {
231  DenseSet<int64_t> lhsContractingDimSet;
232  DenseSet<int64_t> rhsContractingDimSet;
233  for (auto &dimPair : contractingDimMap) {
234  lhsContractingDimSet.insert(dimPair.first);
235  rhsContractingDimSet.insert(dimPair.second);
236  }
237  DenseSet<int64_t> rhsBatchDimSet;
238  for (auto &dimPair : batchDimMap)
239  rhsBatchDimSet.insert(dimPair.second);
240 
241  // Add free and batch dimensions from 'lhsType' to 'expectedResultDims'.
242  SmallVector<int64_t, 4> expectedResultDims;
243  for (int64_t i = 0, e = lhsType.getRank(); i < e; ++i) {
244  if (lhsContractingDimSet.count(i) > 0)
245  continue;
246  expectedResultDims.push_back(lhsType.getDimSize(i));
247  }
248 
249  // Add free dimensions from 'rhsType' to 'expectedResultDims'.
250  for (int64_t i = 0, e = rhsType.getRank(); i < e; ++i) {
251  if (rhsContractingDimSet.count(i) > 0 || rhsBatchDimSet.count(i) > 0)
252  continue;
253  expectedResultDims.push_back(rhsType.getDimSize(i));
254  }
255 
256  // Verify 'expectedResultDims'.
257  if (expectedResultDims.size() == 0) {
258  // No batch or free dimension implies a scalar result.
259  if (resType.isa<VectorType>() || accType.isa<VectorType>())
260  return op.emitOpError("invalid accumulator/result vector shape");
261  } else {
262  // At least one batch or free dimension implies a vector result.
263  auto resVectorType = resType.dyn_cast<VectorType>();
264  auto accVectorType = accType.dyn_cast<VectorType>();
265  if (!resVectorType || !accVectorType)
266  return op.emitOpError("invalid accumulator/result vector shape");
267 
268  // Infer expected result vector type. Lhs + rhs map and lhs + rhs vector
269  // types fully define the result vector type. This assumes the affine maps
270  // are well-formed, which must have been verified already.
271  MLIRContext *ctx = op.getContext();
272  AffineMap lhsMap = op.getIndexingMaps()[0];
273  AffineMap rhsMap = op.getIndexingMaps()[1];
274  SmallVector<AffineExpr, 4> extents(lhsMap.getNumInputs());
275  for (auto pair :
276  {std::make_pair(lhsType, lhsMap), std::make_pair(rhsType, rhsMap)}) {
277  VectorType v = pair.first;
278  auto map = pair.second;
279  for (unsigned idx = 0, e = v.getRank(); idx < e; ++idx) {
280  unsigned pos = map.getResult(idx).cast<AffineDimExpr>().getPosition();
281  if (!extents[pos])
282  extents[pos] = getAffineConstantExpr(v.getShape()[idx], ctx);
283  }
284  }
285  assert(llvm::all_of(extents, [](AffineExpr e) { return e; }) &&
286  "expected extent along all dimensions.");
287 
288  AffineMap resMap = op.getIndexingMaps()[2];
289  auto extentsMap = AffineMap::get(/*dimCount=*/extents.size(),
290  /*symCount=*/0, extents, ctx);
291  // Compose the resMap with the extentsMap, which is a constant map.
292  AffineMap expectedMap = simplifyAffineMap(resMap.compose(extentsMap));
293  assert(llvm::all_of(
294  expectedMap.getResults(),
295  [](AffineExpr e) { return e.isa<AffineConstantExpr>(); }) &&
296  "expected constant extent along all dimensions.");
297  // Extract the expected shape and build the type.
298  auto expectedShape = llvm::to_vector<4>(
299  llvm::map_range(expectedMap.getResults(), [](AffineExpr e) {
300  return e.cast<AffineConstantExpr>().getValue();
301  }));
302  auto expected =
303  VectorType::get(expectedShape, resVectorType.getElementType());
304  if (resVectorType != expected || accVectorType != expected)
305  return op.emitOpError(
306  "invalid accumulator/result vector shape, expected: ")
307  << expected;
308  }
309  return success();
310 }

312 static LogicalResult verify(ContractionOp op) {
313  auto lhsType = op.getLhsType();
314  auto rhsType = op.getRhsType();
315  auto accType = op.getAccType();
316  auto resType = op.getResultType();
317 
318  // Verify that an indexing map was specified for each vector operand.
319  if (op.indexing_maps().size() != 3)
320  return op.emitOpError("expected an indexing map for each vector operand");
321 
322  // Verify that each index map has 'numIterators' inputs, no symbols, and
323  // that the number of map outputs equals the rank of its associated
324  // vector operand.
325  unsigned numIterators = op.iterator_types().getValue().size();
326  for (auto it : llvm::enumerate(op.indexing_maps())) {
327  auto index = it.index();
328  auto map = it.value().cast<AffineMapAttr>().getValue();
329  if (map.getNumSymbols() != 0)
330  return op.emitOpError("expected indexing map ")
331  << index << " to have no symbols";
332  auto vectorType = op.getOperand(index).getType().dyn_cast<VectorType>();
333  unsigned rank = vectorType ? vectorType.getShape().size() : 0;
334  // Verify that the map has the right number of inputs, outputs, and indices.
335  // This also correctly accounts for (..) -> () for rank-0 results.
336  if (map.getNumDims() != numIterators)
337  return op.emitOpError("expected indexing map ")
338  << index << " to have " << numIterators << " number of inputs";
339  if (map.getNumResults() != rank)
340  return op.emitOpError("expected indexing map ")
341  << index << " to have " << rank << " number of outputs";
342  if (!map.isProjectedPermutation())
343  return op.emitOpError("expected indexing map ")
344  << index << " to be a projected permutation of its inputs";
345  }
346 
347  auto contractingDimMap = op.getContractingDimMap();
348  auto batchDimMap = op.getBatchDimMap();
349 
350  // Verify at least one contracting dimension pair was specified.
351  if (contractingDimMap.empty())
352  return op.emitOpError("expected at least one contracting dimension pair");
353 
354  // Verify contracting dimension map was properly constructed.
355  if (!verifyDimMap(lhsType, rhsType, contractingDimMap))
356  return op.emitOpError("invalid contracting dimension map");
357 
358  // Verify batch dimension map was properly constructed.
359  if (!verifyDimMap(lhsType, rhsType, batchDimMap))
360  return op.emitOpError("invalid batch dimension map");
361 
362  // Verify 'accType' and 'resType' shape.
363  if (failed(verifyOutputShape(op, lhsType, rhsType, accType, resType,
364  contractingDimMap, batchDimMap)))
365  return failure();
366 
367  // Verify that either two vector masks are set or none are set.
368  auto lhsMaskType = op.getLHSVectorMaskType();
369  auto rhsMaskType = op.getRHSVectorMaskType();
370  if ((lhsMaskType && !rhsMaskType) || (!lhsMaskType && rhsMaskType))
371  return op.emitOpError("invalid number of vector masks specified");
372  if (lhsMaskType && rhsMaskType) {
373  // Verify mask rank == argument rank.
374  if (lhsMaskType.getShape().size() != lhsType.getShape().size() ||
375  rhsMaskType.getShape().size() != rhsType.getShape().size())
376  return op.emitOpError("invalid vector mask rank");
377  }
378  return success();
379 }

394 static std::vector<std::pair<int64_t, int64_t>>
395 getDimMap(ArrayRef<AffineMap> indexingMaps, ArrayAttr iteratorTypes,
396  StringRef targetIteratorTypeName, MLIRContext *context) {
397  std::vector<std::pair<int64_t, int64_t>> dimMap;
398  for (auto it : llvm::enumerate(iteratorTypes)) {
399  auto iteratorTypeName = it.value().cast<StringAttr>().getValue();
400  if (iteratorTypeName != targetIteratorTypeName)
401  continue;
402  // Search lhs/rhs map results for 'targetExpr'.
403  auto targetExpr = getAffineDimExpr(it.index(), context);
404  int64_t lhsDim = getResultIndex(indexingMaps[0], targetExpr);
405  int64_t rhsDim = getResultIndex(indexingMaps[1], targetExpr);
406  if (lhsDim >= 0 && rhsDim >= 0)
407  dimMap.push_back({lhsDim, rhsDim});
408  }
409  return dimMap;
410 }

412 void ContractionOp::getIterationBounds(
413  SmallVectorImpl<int64_t> &iterationBounds) {
414  auto lhsShape = getLhsType().getShape();
415  auto resVectorType = getResultType().dyn_cast<VectorType>();
416  SmallVector<AffineMap, 4> indexingMaps(getIndexingMaps());
417  SmallVector<int64_t, 2> iterationShape;
418  for (auto it : llvm::enumerate(iterator_types())) {
419  // Search lhs/rhs map results for 'targetExpr'.
420  auto targetExpr = getAffineDimExpr(it.index(), getContext());
421  auto iteratorTypeName = it.value().cast<StringAttr>().getValue();
422  if (iteratorTypeName == getReductionIteratorTypeName()) {
423  // Get reduction dim size from lhs shape (same size in rhsShape).
424  int64_t lhsDimIndex = getResultIndex(indexingMaps[0], targetExpr);
425  assert(lhsDimIndex >= 0);
426  iterationBounds.push_back(lhsShape[lhsDimIndex]);
427  continue;
428  }
429  // Get parallel dimension size from result shape.
430  int64_t resDimIndex = getResultIndex(indexingMaps[2], targetExpr);
431  assert(resDimIndex >= 0);
432  assert(resVectorType != nullptr);
433  iterationBounds.push_back(resVectorType.getShape()[resDimIndex]);
434  }

603 /// Fold the result of an ExtractOp in place when it comes from a TransposeOp.
604 static LogicalResult foldExtractOpFromTranspose(ExtractOp extractOp) {
605  auto transposeOp = extractOp.vector().getDefiningOp<TransposeOp>();
606  if (!transposeOp)
607  return failure();
608 
609  auto permutation = extractVector<unsigned>(transposeOp.transp());
610  auto extractedPos = extractVector<int64_t>(extractOp.position());
611 
612  // If transposition permutation is larger than the ExtractOp, all minor
613  // dimensions must be an identity for folding to occur. If not, individual
614  // elements within the extracted value are transposed and this is not just a
615  // simple folding.
616  unsigned minorRank = permutation.size() - extractedPos.size();
617  MLIRContext *ctx = extractOp.getContext();
618  AffineMap permutationMap = AffineMap::getPermutationMap(permutation, ctx);
619  AffineMap minorMap = permutationMap.getMinorSubMap(minorRank);
620  if (minorMap && !minorMap.isMinorIdentity())
621  return failure();
622 
623  // %1 = transpose %0[x, y, z] : vector<axbxcxf32>
624  // %2 = extract %1[u, v] : vector<..xf32>
625  // may turn into:
626  // %2 = extract %0[w, x] : vector<..xf32>
627  // iff z == 2 and [w, x] = [x, y]^-1 o [u, v] here o denotes composition and
628  // -1 denotes the inverse.
629  permutationMap = permutationMap.getMajorSubMap(extractedPos.size());
630  // The major submap has fewer results but the same number of dims. To compose
631  // cleanly, we need to drop dims to form a "square matrix". This is possible
632  // because:
633  // (a) this is a permutation map and
634  // (b) the minor map has already been checked to be identity.
635  // Therefore, the major map cannot contain dims of position greater or equal
636  // than the number of results.
637  assert(llvm::all_of(permutationMap.getResults(),
638  [&](AffineExpr e) {
639  auto dim = e.dyn_cast<AffineDimExpr>();
640  return dim && dim.getPosition() <
641  permutationMap.getNumResults();
642  }) &&
643  "Unexpected map results depend on higher rank positions");
644  // Project on the first domain dimensions to allow composition.
645  permutationMap = AffineMap::get(permutationMap.getNumResults(), 0,
646  permutationMap.getResults(), ctx);
647 
648  extractOp.setOperand(transposeOp.vector());
649  // Compose the inverse permutation map with the extractedPos.
650  auto newExtractedPos =
651  inversePermutation(permutationMap).compose(extractedPos);
652  // OpBuilder is only used as a helper to build an I64ArrayAttr.
653  OpBuilder b(extractOp.getContext());
654  extractOp.setAttr(ExtractOp::getPositionAttrName(),
655  b.getI64ArrayAttr(newExtractedPos));
656 
657  return success();
658 }

660 /// Fold an ExtractOp that is fed by a chain of InsertOps and TransposeOps. The
661 /// result is always the input to some InsertOp.
662 static Value foldExtractOpFromInsertChainAndTranspose(ExtractOp extractOp) {
663  MLIRContext *context = extractOp.getContext();
664  AffineMap permutationMap;
665  auto extractedPos = extractVector<unsigned>(extractOp.position());
666  // Walk back a chain of InsertOp/TransposeOp until we hit a match.
667  // Compose TransposeOp permutations as we walk back.
668  auto insertOp = extractOp.vector().getDefiningOp<vector::InsertOp>();
669  auto transposeOp = extractOp.vector().getDefiningOp<vector::TransposeOp>();
670  while (insertOp || transposeOp) {
671  if (transposeOp) {
672  // If it is transposed, compose the map and iterate.
673  auto permutation = extractVector<unsigned>(transposeOp.transp());
674  AffineMap newMap = AffineMap::getPermutationMap(permutation, context);
675  if (!permutationMap)
676  permutationMap = newMap;
677  else if (newMap.getNumInputs() != permutationMap.getNumResults())
678  return Value();
679  else
680  permutationMap = newMap.compose(permutationMap);
681  // Compute insert/transpose for the next iteration.
682  Value transposed = transposeOp.vector();
683  insertOp = transposed.getDefiningOp<vector::InsertOp>();
684  transposeOp = transposed.getDefiningOp<vector::TransposeOp>();
685  continue;
686  }
687 
688  assert(insertOp);
689  Value insertionDest = insertOp.dest();
690  // If it is inserted into, either the position matches and we have a
691  // successful folding; or we iterate until we run out of
692  // InsertOp/TransposeOp. This is because `vector.insert %scalar, %vector`
693  // produces a new vector with 1 modified value/slice in exactly the static
694  // position we need to match.
695  auto insertedPos = extractVector<unsigned>(insertOp.position());
696  // Trivial permutations are solved with position equality checks.
697  if (!permutationMap || permutationMap.isIdentity()) {
698  if (extractedPos == insertedPos)
699  return insertOp.source();
700  // Fallthrough: if the position does not match, just skip to the next
701  // producing `vector.insert` / `vector.transpose`.
702  // Compute insert/transpose for the next iteration.
703  insertOp = insertionDest.getDefiningOp<vector::InsertOp>();
704  transposeOp = insertionDest.getDefiningOp<vector::TransposeOp>();
705  continue;
706  }
707 
708  // More advanced permutations require application of the permutation.
709  // However, the rank of `insertedPos` may be different from that of the
710  // `permutationMap`. To support such case, we need to:
711  // 1. apply on the `insertedPos.size()` major dimensions
712  // 2. check the other dimensions of the permutation form a minor identity.
713  assert(permutationMap.isPermutation() && "expected a permutation");
714  if (insertedPos.size() == extractedPos.size()) {
715  bool fold = true;
716  for (unsigned idx = 0, sz = extractedPos.size(); idx < sz; ++idx) {
717  auto pos =
718  permutationMap.getResult(idx).cast<AffineDimExpr>().getPosition();
719  if (pos >= sz || insertedPos[pos] != extractedPos[idx]) {
720  fold = false;
721  break;
722  }
723  }
724  if (fold) {
725  assert(permutationMap.getNumResults() >= insertedPos.size() &&
726  "expected map of rank larger than insert indexing");
727  unsigned minorRank =
728  permutationMap.getNumResults() - insertedPos.size();
729  AffineMap minorMap = permutationMap.getMinorSubMap(minorRank);
730  if (!minorMap || minorMap.isMinorIdentity())
731  return insertOp.source();
732  }
733  }
734 
735  // If we haven't found a match, just continue to the next producing
736  // `vector.insert` / `vector.transpose`.
737  // Compute insert/transpose for the next iteration.
738  insertOp = insertionDest.getDefiningOp<vector::InsertOp>();
739  transposeOp = insertionDest.getDefiningOp<vector::TransposeOp>();
740  }
741  return Value();
742 }

770 static LogicalResult
771 isValidExtractOrInsertSlicesType(Operation *op, VectorType vectorType,
772  TupleType tupleType, ArrayRef<int64_t> sizes,
773  ArrayRef<int64_t> strides) {
774  // Check for non-unit strides.
775  // TODO: Support non-1 strides.
776  if (llvm::any_of(strides, [](int64_t s) { return s != 1; }))
777  return op->emitError("requires unit strides");
778  // Check that 'vectorType' rank matches rank of tuple element vectors.
779  unsigned rank = vectorType.getRank();
780  auto is_vector_type_of_rank = [&](Type t) {
781  return t.isa<VectorType>() && t.cast<VectorType>().getRank() == rank;
782  };
783  if (!llvm::all_of(tupleType.getTypes(), is_vector_type_of_rank))
784  return op->emitError("requires vector tuple elements of rank ") << rank;
785  // Check that 'sizes' and 'strides' are of size == 'rank'.
786  if (sizes.size() != rank || strides.size() != rank)
787  return op->emitError("requires sizes and strides of rank ") << rank;
788 
789  // Generate each slice shape based on 'sizes', 'strides' and 'vectorType',
790  // and verify that the same matches the corresponding tuple element 'i'.
791  auto shape = vectorType.getShape();
792  auto sliceStrides = computeStrides(shape, sizes);
793  for (int64_t i = 0, e = tupleType.size(); i < e; ++i) {
794  auto vectorOffsets = delinearize(sliceStrides, i);
795  auto elementOffsets =
796  computeElementOffsetsFromVectorSliceOffsets(sizes, vectorOffsets);
797  auto sliceSizes = computeSliceSizes(shape, sizes, elementOffsets);
798  // Create slice VectorType type.
799  auto sliceVectorType =
800  VectorType::get(sliceSizes, vectorType.getElementType());
801  // Verify that 'sliceVectorType' matches tupleType.getTypes(i)
802  if (sliceVectorType != tupleType.getType(i))
803  return op->emitError("invalid tuple element type ") << sliceVectorType;
804  }
805  return success();
806 }

1399 // Pattern to rewrite a ExtractStridedSliceOp(ConstantMaskOp) -> ConstantMaskOp.
1400 class StridedSliceConstantMaskFolder final
1401  : public OpRewritePattern<ExtractStridedSliceOp> {
1402 public:
1403  using OpRewritePattern<ExtractStridedSliceOp>::OpRewritePattern;
1404 
1405  LogicalResult matchAndRewrite(ExtractStridedSliceOp extractStridedSliceOp,
1406  PatternRewriter &rewriter) const override {
1407  // Return if 'extractStridedSliceOp' operand is not defined by a
1408  // ConstantMaskOp.
1409  auto defOp = extractStridedSliceOp.vector().getDefiningOp();
1410  auto constantMaskOp = dyn_cast_or_null<ConstantMaskOp>(defOp);
1411  if (!constantMaskOp)
1412  return failure();
1413  // Return if 'extractStridedSliceOp' has non-unit strides.
1414  if (llvm::any_of(extractStridedSliceOp.strides(), [](Attribute attr) {
1415  return attr.cast<IntegerAttr>().getInt() != 1;
1416  }))
1417  return failure();
1418  // Gather constant mask dimension sizes.
1419  SmallVector<int64_t, 4> maskDimSizes;
1420  populateFromInt64AttrArray(constantMaskOp.mask_dim_sizes(), maskDimSizes);
1421  // Gather strided slice offsets and sizes.
1422  SmallVector<int64_t, 4> sliceOffsets;
1423  populateFromInt64AttrArray(extractStridedSliceOp.offsets(), sliceOffsets);
1424  SmallVector<int64_t, 4> sliceSizes;
1425  populateFromInt64AttrArray(extractStridedSliceOp.sizes(), sliceSizes);
1426 
1427  // Compute slice of vector mask region.
1428  SmallVector<int64_t, 4> sliceMaskDimSizes;
1429  assert(sliceOffsets.size() == maskDimSizes.size());
1430  for (auto it : llvm::zip(maskDimSizes, sliceOffsets, sliceSizes)) {
1431  int64_t maskDimSize = std::get<0>(it);
1432  int64_t sliceOffset = std::get<1>(it);
1433  int64_t sliceSize = std::get<2>(it);
1434  int64_t sliceMaskDimSize = std::max(
1435  static_cast<int64_t>(0),
1436  std::min(sliceOffset + sliceSize, maskDimSize) - sliceOffset);
1437  sliceMaskDimSizes.push_back(sliceMaskDimSize);
1438  }
1439  // If any of 'sliceMaskDimSizes' are zero, then set all to zero (masked
1440  // region is a conjunction of mask dim intervals).
1441  if (llvm::any_of(sliceMaskDimSizes, [](int64_t sz) { return sz == 0; }))
1442  sliceMaskDimSizes.assign(maskDimSizes.size(), 0);
1443 
1444  // Replace 'extractStridedSliceOp' with ConstantMaskOp with sliced mask
1445  // region.
1446  rewriter.replaceOpWithNewOp<ConstantMaskOp>(
1447  extractStridedSliceOp, extractStridedSliceOp.getResult().getType(),
1448  vector::getVectorSubscriptAttr(rewriter, sliceMaskDimSizes));
1449  return success();
1450  }

1974 /// Returns true if each element of 'a' is equal to the product of a contiguous
1975 /// sequence of the elements of 'b'. Returns false otherwise.
1976 static bool isValidShapeCast(ArrayRef<int64_t> a, ArrayRef<int64_t> b) {
1977  unsigned rankA = a.size();
1978  unsigned rankB = b.size();
1979  assert(rankA < rankB);
1980 
1981  unsigned i = 0;
1982  unsigned j = 0;
1983  while (i < rankA && j < rankB) {
1984  int64_t dimA = a[i];
1985  int64_t dimB = 1;
1986  while (dimB < dimA && j < rankB)
1987  dimB *= b[j++];
1988  if (dimA != dimB)
1989  break;
1990  ++i;
1991 
1992  // Handle the case when trailing dimensions are of size 1.
1993  // Include them into the contiguous sequence.
1994  auto isOne = [](int64_t v) { return v == 1; };
1995  if (i < rankA && llvm::all_of(a.slice(i), isOne))
1996  i = rankA;
1997  if (j < rankB && llvm::all_of(b.slice(j), isOne))
1998  j = rankB;
1999  }
2000 
2001  return i == rankA && j == rankB;
2002 }
