78 /// This trait provides APIs for Ops that behave like functions. In particular:
79 /// - Ops must be symbols, i.e. also have the `Symbol` trait;
80 /// - Ops have a single region with multiple blocks that corresponds to the body
81 /// of the function;
82 /// - the absence of a region corresponds to an external function;
83 /// - leading arguments of the first block of the region are treated as function
84 /// arguments;
85 /// - they can have argument attributes that are stored in a dictionary
86 /// attribute on the Op itself.
87 /// This trait does *NOT* provide type support for the functions, meaning that
88 /// concrete Ops must handle the type of the declared or defined function.
89 /// `getTypeAttrName()` is a convenience function that returns the name of the
90 /// attribute that can be used to store the function type, but the trait makes
91 /// no assumption based on it.
92 ///
93 /// - Concrete ops *must* define a member function `getNumFuncArguments()` that
94 /// returns the number of function arguments based exclusively on type (so
95 /// that it can be called on function declarations).
96 /// - Concrete ops *must* define a member function `getNumFuncResults()` that
97 /// returns the number of function results based exclusively on type (so that
98 /// it can be called on function declarations).
99 /// - To verify that the type respects op-specific invariants, concrete ops may
100 /// redefine the `verifyType()` hook that will be called after verifying the
101 /// presence of the `type` attribute and before any call to
102 /// `getNumFuncArguments`/`getNumFuncResults` from the verifier.
103 /// - To verify that the body respects op-specific invariants, concrete ops may
104 /// redefine the `verifyBody()` hook that will be called after verifying the
105 /// function type and the presence of the (potentially empty) body region.
106 template <typename ConcreteType>

394 /// Default verifier checks that if the entry block exists, it has the same
395 /// number of arguments as the function-like operation.
396 template <typename ConcreteType>
397 LogicalResult FunctionLike<ConcreteType>::verifyBody() {
398  auto funcOp = cast<ConcreteType>(this->getOperation());
399 
400  if (funcOp.isExternal())
401  return success();
402 
403  unsigned numArguments = funcOp.getNumArguments();
404  if (funcOp.front().getNumArguments() != numArguments)
405  return funcOp.emitOpError("entry block must have ")
406  << numArguments << " arguments to match function signature";
407 
408  return success();
409 }

411 template <typename ConcreteType>
412 LogicalResult FunctionLike<ConcreteType>::verifyTrait(Operation *op) {
413  MLIRContext *ctx = op->getContext();
414  auto funcOp = cast<ConcreteType>(op);
415 
416  if (!funcOp.isTypeAttrValid())
417  return funcOp.emitOpError("requires a type attribute '")
418  << getTypeAttrName() << '\'';
419 
420  if (failed(funcOp.verifyType()))
421  return failure();
422 
423  for (unsigned i = 0, e = funcOp.getNumArguments(); i != e; ++i) {
424  // Verify that all of the argument attributes are dialect attributes, i.e.
425  // that they contain a dialect prefix in their name. Call the dialect, if
426  // registered, to verify the attributes themselves.
427  for (auto attr : funcOp.getArgAttrs(i)) {
428  if (!attr.first.strref().contains('.'))
429  return funcOp.emitOpError("arguments may only have dialect attributes");
430  auto dialectNamePair = attr.first.strref().split('.');
431  if (auto *dialect = ctx->getRegisteredDialect(dialectNamePair.first)) {
432  if (failed(dialect->verifyRegionArgAttribute(op, /*regionIndex=*/0,
433  /*argIndex=*/i, attr)))
434  return failure();
435  }
436  }
437  }
438 
439  for (unsigned i = 0, e = funcOp.getNumResults(); i != e; ++i) {
440  // Verify that all of the result attributes are dialect attributes, i.e.
441  // that they contain a dialect prefix in their name. Call the dialect, if
442  // registered, to verify the attributes themselves.
443  for (auto attr : funcOp.getResultAttrs(i)) {
444  if (!attr.first.strref().contains('.'))
445  return funcOp.emitOpError("results may only have dialect attributes");
446  auto dialectNamePair = attr.first.strref().split('.');
447  if (auto *dialect = ctx->getRegisteredDialect(dialectNamePair.first)) {
448  if (failed(dialect->verifyRegionResultAttribute(op, /*regionIndex=*/0,
449  /*resultIndex=*/i,
450  attr)))
451  return failure();
452  }
453  }
454  }
455 
456  // Check that the op has exactly one region for the body.
457  if (op->getNumRegions() != 1)
458  return funcOp.emitOpError("expects one region");
459 
460  return funcOp.verifyBody();
461 }

488 void FunctionLike<ConcreteType>::setType(FunctionType newType) {
489  SmallVector<char, 16> nameBuf;
490  auto oldType = getType();
491  auto *concreteOp = static_cast<ConcreteType *>(this);
492 
493  for (int i = newType.getNumInputs(), e = oldType.getNumInputs(); i < e; i++)
494  concreteOp->removeAttr(getArgAttrName(i, nameBuf));
495  for (int i = newType.getNumResults(), e = oldType.getNumResults(); i < e; i++)
496  concreteOp->removeAttr(getResultAttrName(i, nameBuf));
