22 /// This class represents the benefit of a pattern match in a unitless scheme
23 /// that ranges from 0 (very little benefit) to 65K. The most common unit to
24 /// use here is the "number of operations matched" by the pattern.
25 ///
26 /// This also has a sentinel representation that can be used for patterns that
27 /// fail to match.
28 ///
29 class PatternBenefit {
30  enum { ImpossibleToMatchSentinel = 65535 };
31 
32 public:
33  PatternBenefit() : representation(ImpossibleToMatchSentinel) {}
34  PatternBenefit(unsigned benefit);
35  PatternBenefit(const PatternBenefit &) = default;
36  PatternBenefit &operator=(const PatternBenefit &) = default;
37 
38  static PatternBenefit impossibleToMatch() { return PatternBenefit(); }
39  bool isImpossibleToMatch() const { return *this == impossibleToMatch(); }
40 
41  /// If the corresponding pattern can match, return its benefit. If the
42  // corresponding pattern isImpossibleToMatch() then this aborts.
43  unsigned short getBenefit() const;
44 
45  bool operator==(const PatternBenefit &rhs) const {
46  return representation == rhs.representation;
47  }
48  bool operator!=(const PatternBenefit &rhs) const { return !(*this == rhs); }
49  bool operator<(const PatternBenefit &rhs) const {
50  return representation < rhs.representation;
51  }
52  bool operator>(const PatternBenefit &rhs) const { return rhs < *this; }
53  bool operator<=(const PatternBenefit &rhs) const { return !(*this > rhs); }
54  bool operator>=(const PatternBenefit &rhs) const { return !(*this < rhs); }
55 
56 private:
57  unsigned short representation;
58 };

64 /// Instances of Pattern can be matched against SSA IR. These matches get used
65 /// in ways dependent on their subclasses and the driver doing the matching.
66 /// For example, RewritePatterns implement a rewrite from one matched pattern
67 /// to a replacement DAG tile.
68 class Pattern {
69 public:
70  /// Return the benefit (the inverse of "cost") of matching this pattern. The
71  /// benefit of a Pattern is always static - rewrites that may have dynamic
72  /// benefit can be instantiated multiple times (different Pattern instances)
73  /// for each benefit that they may return, and be guarded by different match
74  /// condition predicates.
75  PatternBenefit getBenefit() const { return benefit; }
76 
77  /// Return the root node that this pattern matches. Patterns that can match
78  /// multiple root types return None.
79  Optional<OperationName> getRootKind() const { return rootKind; }
80 
81  //===--------------------------------------------------------------------===//
82  // Implementation hooks for patterns to implement.
83  //===--------------------------------------------------------------------===//
84 
85  /// Attempt to match against code rooted at the specified operation,
86  /// which is the same operation code as getRootKind().
87  virtual LogicalResult match(Operation *op) const = 0;
88 
89  virtual ~Pattern() {}
90 
91 protected:
92  /// This class acts as a special tag that makes the desire to match "any"
93  /// operation type explicit. This helps to avoid unnecessary usages of this
94  /// feature, and ensures that the user is making a conscious decision.
95  struct MatchAnyOpTypeTag {};
96 
97  /// This constructor is used for patterns that match against a specific
98  /// operation type. The `benefit` is the expected benefit of matching this
99  /// pattern.
100  Pattern(StringRef rootName, PatternBenefit benefit, MLIRContext *context);
101 
102  /// This constructor is used when a pattern may match against multiple
103  /// different types of operations. The `benefit` is the expected benefit of
104  /// matching this pattern. `MatchAnyOpTypeTag` is just a tag to ensure that
105  /// the "match any" behavior is what the user actually desired,
106  /// `MatchAnyOpTypeTag()` should always be supplied here.
107  Pattern(PatternBenefit benefit, MatchAnyOpTypeTag);
108 
109 private:
110  /// The root operation of the pattern. If the pattern matches a specific
111  /// operation, this contains the name of that operation. Contains None
112  /// otherwise.
113  Optional<OperationName> rootKind;
114 
115  /// The expected benefit of matching this pattern.
116  const PatternBenefit benefit;
117 
118  virtual void anchor();
119 };

121 /// RewritePattern is the common base class for all DAG to DAG replacements.
122 /// There are two possible usages of this class:
123 /// * Multi-step RewritePattern with "match" and "rewrite"
124 /// - By overloading the "match" and "rewrite" functions, the user can
125 /// separate the concerns of matching and rewriting.
126 /// * Single-step RewritePattern with "matchAndRewrite"
127 /// - By overloading the "matchAndRewrite" function, the user can perform
128 /// the rewrite in the same call as the match.
129 ///
130 class RewritePattern : public Pattern {
131 public:
132  /// Rewrite the IR rooted at the specified operation with the result of
133  /// this pattern, generating any new operations with the specified
134  /// builder. If an unexpected error is encountered (an internal
135  /// compiler error), it is emitted through the normal MLIR diagnostic
136  /// hooks and the IR is left in a valid state.
137  virtual void rewrite(Operation *op, PatternRewriter &rewriter) const;
138 
139  /// Attempt to match against code rooted at the specified operation,
140  /// which is the same operation code as getRootKind().
141  LogicalResult match(Operation *op) const override;
142 
143  /// Attempt to match against code rooted at the specified operation,
144  /// which is the same operation code as getRootKind(). If successful, this
145  /// function will automatically perform the rewrite.
146  virtual LogicalResult matchAndRewrite(Operation *op,
147  PatternRewriter &rewriter) const {
148  if (succeeded(match(op))) {
149  rewrite(op, rewriter);
150  return success();
151  }
152  return failure();
153  }
154 
155  /// Returns true if this pattern is known to result in recursive application,
156  /// i.e. this pattern may generate IR that also matches this pattern, but is
157  /// known to bound the recursion. This signals to a rewriter that it is safe
158  /// to apply this pattern recursively to generated IR.
159  virtual bool hasBoundedRewriteRecursion() const { return false; }
160 
161  /// Return a list of operations that may be generated when rewriting an
162  /// operation instance with this pattern.
163  ArrayRef<OperationName> getGeneratedOps() const { return generatedOps; }
164 
165 protected:
166  /// Construct a rewrite pattern with a certain benefit that matches the
167  /// operation with the given root name.
168  RewritePattern(StringRef rootName, PatternBenefit benefit,
169  MLIRContext *context)
170  : Pattern(rootName, benefit, context) {}
171  /// Construct a rewrite pattern with a certain benefit that matches any
172  /// operation type. `MatchAnyOpTypeTag` is just a tag to ensure that the
173  /// "match any" behavior is what the user actually desired,
174  /// `MatchAnyOpTypeTag()` should always be supplied here.
175  RewritePattern(PatternBenefit benefit, MatchAnyOpTypeTag tag)
176  : Pattern(benefit, tag) {}
177  /// Construct a rewrite pattern with a certain benefit that matches the
178  /// operation with the given root name. `generatedNames` contains the names of
179  /// operations that may be generated during a successful rewrite.
180  RewritePattern(StringRef rootName, ArrayRef<StringRef> generatedNames,
181  PatternBenefit benefit, MLIRContext *context);
182  /// Construct a rewrite pattern that may match any operation type.
183  /// `generatedNames` contains the names of operations that may be generated
184  /// during a successful rewrite. `MatchAnyOpTypeTag` is just a tag to ensure
185  /// that the "match any" behavior is what the user actually desired,
186  /// `MatchAnyOpTypeTag()` should always be supplied here.
187  RewritePattern(ArrayRef<StringRef> generatedNames, PatternBenefit benefit,
188  MLIRContext *context, MatchAnyOpTypeTag tag);
189 
190  /// A list of the potential operations that may be generated when rewriting
191  /// an op with this pattern.
192  SmallVector<OperationName, 2> generatedOps;
193 };

195 /// OpRewritePattern is a wrapper around RewritePattern that allows for
196 /// matching and rewriting against an instance of a derived operation class as
197 /// opposed to a raw Operation.
198 template <typename SourceOp> struct OpRewritePattern : public RewritePattern {
199  /// Patterns must specify the root operation name they match against, and can
200  /// also specify the benefit of the pattern matching.
201  OpRewritePattern(MLIRContext *context, PatternBenefit benefit = 1)
202  : RewritePattern(SourceOp::getOperationName(), benefit, context) {}
203 
204  /// Wrappers around the RewritePattern methods that pass the derived op type.
205  void rewrite(Operation *op, PatternRewriter &rewriter) const final {
206  rewrite(cast<SourceOp>(op), rewriter);
207  }
208  LogicalResult match(Operation *op) const final {
209  return match(cast<SourceOp>(op));
210  }
211  LogicalResult matchAndRewrite(Operation *op,
212  PatternRewriter &rewriter) const final {
213  return matchAndRewrite(cast<SourceOp>(op), rewriter);
214  }
215 
216  /// Rewrite and Match methods that operate on the SourceOp type. These must be
217  /// overridden by the derived pattern class.
218  virtual void rewrite(SourceOp op, PatternRewriter &rewriter) const {
219  llvm_unreachable("must override rewrite or matchAndRewrite");
220  }
221  virtual LogicalResult match(SourceOp op) const {
222  llvm_unreachable("must override match or matchAndRewrite");
223  }
224  virtual LogicalResult matchAndRewrite(SourceOp op,
225  PatternRewriter &rewriter) const {
226  if (succeeded(match(op))) {
227  rewrite(op, rewriter);
228  return success();
229  }
230  return failure();
231  }

238 /// This class coordinates the application of a pattern to the current function,
239 /// providing a way to create operations and keep track of what gets deleted.
240 ///
241 /// These class serves two purposes:
242 /// 1) it is the interface that patterns interact with to make mutations to the
243 /// IR they are being applied to.
244 /// 2) It is a base class that clients of the PatternMatcher use when they want
245 /// to apply patterns and observe their effects (e.g. to keep worklists or
246 /// other data structures up to date).
247 ///
248 class PatternRewriter : public OpBuilder, public OpBuilder::Listener {
249 public:
250  /// Create operation of specific op type at the current insertion point
251  /// without verifying to see if it is valid.
252  template <typename OpTy, typename... Args>
253  OpTy create(Location location, Args... args) {
254  OperationState state(location, OpTy::getOperationName());
255  OpTy::build(*this, state, args...);
256  auto *op = createOperation(state);
257  auto result = dyn_cast<OpTy>(op);
258  assert(result && "Builder didn't return the right type");
259  return result;
260  }
261 
262  /// Creates an operation of specific op type at the current insertion point.
263  /// If the result is an invalid op (the verifier hook fails), emit an error
264  /// and return null.
265  template <typename OpTy, typename... Args>
266  OpTy createChecked(Location location, Args... args) {
267  OperationState state(location, OpTy::getOperationName());
268  OpTy::build(*this, state, args...);
269  auto *op = createOperation(state);
270 
271  // If the Operation we produce is valid, return it.
272  if (!OpTy::verifyInvariants(op)) {
273  auto result = dyn_cast<OpTy>(op);
274  assert(result && "Builder didn't return the right type");
275  return result;
276  }
277 
278  // Otherwise, the error message got emitted. Just remove the operation
279  // we made.
280  op->erase();
281  return OpTy();
282  }

333  /// This method is used to notify the rewriter that an in-place operation

335  /// followed by a call to either `finalizeRootUpdate` or `cancelRootUpdate`.
336  /// This is a minor efficiency win (it avoids creating a new operation and

339

448  /// Add an instance of each of the pattern types 'Ts' to the pattern list with
449  /// the given arguments. Return a reference to `this` for chaining insertions.
450  /// Note: ConstructorArg is necessary here to separate the two variadic lists.
451  template <typename... Ts, typename ConstructorArg,
452  typename... ConstructorArgs,
453  typename = std::enable_if_t<sizeof...(Ts) != 0>>
454  OwningRewritePatternList &insert(ConstructorArg &&arg,
455  ConstructorArgs &&... args) {
456  // The following expands a call to emplace_back for each of the pattern
457  // types 'Ts'. This magic is necessary due to a limitation in the places
458  // that a parameter pack can be expanded in c++11.
459  // FIXME: In c++17 this can be simplified by using 'fold expressions'.
460  (void)std::initializer_list<int>{
461  0, (patterns.emplace_back(std::make_unique<Ts>(arg, args...)), 0)...};
462  return *this;
463  }
464 
465  /// Add an instance of each of the pattern types 'Ts'. Return a reference to
466  /// `this` for chaining insertions.
467  template <typename... Ts> OwningRewritePatternList &insert() {
468  (void)std::initializer_list<int>{
469  0, (patterns.emplace_back(std::make_unique<Ts>()), 0)...};
470  return *this;
471  }
472 
473  /// Add the given pattern to the pattern list.
474  void insert(std::unique_ptr<RewritePattern> pattern) {
475  patterns.emplace_back(std::move(pattern));
476  }
477 
478 private:
479  PatternListT patterns;
480 };
