36 /// Basic lowering implementation for one-to-one rewriting from AVX512 Ops to
37 /// LLVM Dialect Ops. Convert the type of the result to an LLVM type, pass
38 /// operands as is, preserve attributes.
39 template <typename SourceOp, typename TargetOp>
40 static LogicalResult
41 matchAndRewriteOneToOne(const ConvertToLLVMPattern &lowering,
42  LLVMTypeConverter &typeConverter, Operation *op,
43  ArrayRef<Value> operands,
44  ConversionPatternRewriter &rewriter) {
45  unsigned numResults = op->getNumResults();
46 
47  Type packedType;
48  if (numResults != 0) {
49  packedType = typeConverter.packFunctionResults(op->getResultTypes());
50  if (!packedType)
51  return failure();
52  }
53 
54  auto newOp = rewriter.create<TargetOp>(op->getLoc(), packedType, operands,
55  op->getAttrs());
56 
57  // If the operation produced 0 or 1 result, return them immediately.
58  if (numResults == 0)
59  return rewriter.eraseOp(op), success();
60  if (numResults == 1)
61  return rewriter.replaceOp(op, newOp.getOperation()->getResult(0)),
62  success();
63 
64  // Otherwise, it had been converted to an operation producing a structure.
65  // Extract individual results from the structure and return them as list.
66  SmallVector<Value, 4> results;
67  results.reserve(numResults);
68  for (unsigned i = 0; i < numResults; ++i) {
69  auto type = typeConverter.convertType(op->getResult(i).getType());
70  results.push_back(rewriter.create<LLVM::ExtractValueOp>(
71  op->getLoc(), type, newOp.getOperation()->getResult(0),
72  rewriter.getI64ArrayAttr(i)));
73  }
74  rewriter.replaceOp(op, results);
75  return success();
76 }

170 void ConvertAVX512ToLLVMPass::runOnOperation() {
171  // Convert to the LLVM IR dialect.
172  OwningRewritePatternList patterns;
173  LLVMTypeConverter converter(&getContext());
174  populateAVX512ToLLVMConversionPatterns(converter, patterns);
175  populateVectorToLLVMConversionPatterns(converter, patterns);
176  populateStdToLLVMConversionPatterns(converter, patterns);
177 
178  ConversionTarget target(getContext());
179  target.addLegalDialect<LLVM::LLVMDialect>();
180  target.addLegalDialect<LLVM::LLVMAVX512Dialect>();
181  target.addIllegalDialect<avx512::AVX512Dialect>();
182  if (failed(applyPartialConversion(getOperation(), target, patterns))) {
183  signalPassFailure();
184  }
185 }
