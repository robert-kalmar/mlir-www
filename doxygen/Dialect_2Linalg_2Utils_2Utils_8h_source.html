97 /// Returns the linearized list of all view dimensions in a linalgOp. Applying
98 /// the inverse, concatenated loopToOperandRangeMaps to this list allows the
99 /// derivation of loop ranges for any linalgOp.
100 template <typename ConcreteOp>
101 SmallVector<Value, 8> getViewSizes(OpBuilder &builder, ConcreteOp linalgOp) {
102  auto loc = linalgOp.getLoc();
103  SmallVector<Value, 8> res;
104  SmallVector<unsigned, 4> ranks;
105  for (auto v : linalgOp.getInputsAndOutputBuffers()) {
106  MemRefType t = v.getType().template cast<MemRefType>();
107  ranks.push_back(t.getRank());
108  for (unsigned i = 0; i < t.getRank(); ++i)
109  res.push_back(builder.create<DimOp>(loc, v, i));
110  }
111 
112  auto attr = linalgOp.template getAttrOfType<IntegerAttr>("symbol_source");
113  if (attr) {
114  // Find the correct position for inserting values for symbols.
115  unsigned numSymb = ranks[attr.getInt()], symbolsPos = 0;
116  for (unsigned idx = 0; idx < attr.getInt(); idx++)
117  symbolsPos += ranks[idx];
118 
119  // Append the end of the value list that corresponds to the
120  // values mapping to symbols. Since inside concatinated map symbols are
121  // repeated we have to repeat the sizes as well.
122 
123  // Reserve is mandatory to avoid a potential undefined behavior with
124  // pushing back to smallvector from itself.
125  res.reserve(res.size() + ranks.size() * numSymb);
126  for (unsigned idx = 0, s = ranks.size(); idx < s; ++idx)
127  for (unsigned idx2 = 0; idx2 < numSymb; ++idx2)
128  res.push_back(res[symbolsPos + idx2]);
129  }
130  return res;
131 }

146 /// Apply the permutation defined by `permutation` to `inVec`.
147 /// Element `i` in `inVec` is mapped to location `j = permutation[i]`.
148 /// E.g.: for an input vector `inVec = ['a', 'b', 'c']` and a permutation vector
149 /// `permutation = [2, 0, 1]`, this function leaves `inVec = ['c', 'a', 'b']`.
150 template <typename T, unsigned N>
151 void applyPermutationToVector(SmallVector<T, N> &inVec,
152  ArrayRef<unsigned> permutation) {
153  SmallVector<T, N> auxVec(inVec.size());
154  for (unsigned i = 0; i < permutation.size(); ++i)
155  auxVec[i] = inVec[permutation[i]];
156  inVec = auxVec;
157 }
