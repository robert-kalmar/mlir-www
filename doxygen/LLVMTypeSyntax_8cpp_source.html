27  case LLVMType::VoidType:

29  case LLVMType::HalfType:
30  return "half";
31  case LLVMType::BFloatType:
32  return "bfloat";
33  case LLVMType::FloatType:
34  return "float";
35  case LLVMType::DoubleType:
36  return "double";
37  case LLVMType::FP128Type:
38  return "fp128";
39  case LLVMType::X86FP80Type:
40  return "x86_fp80";
41  case LLVMType::PPCFP128Type:
42  return "ppc_fp128";
43  case LLVMType::X86MMXType:
44  return "x86_mmx";
45  case LLVMType::TokenType:
46  return "token";
47  case LLVMType::LabelType:
48  return "label";
49  case LLVMType::MetadataType:
50  return "metadata";
51  case LLVMType::FunctionType:
52  return "func";

54  return "i";

68 /// Prints the body of a structure type. Uses `stack` to avoid printing
69 /// recursive structs indefinitely.
70 static void printStructTypeBody(llvm::raw_ostream &os, LLVMStructType type,
71  llvm::SetVector<StringRef> &stack) {
72  if (type.isIdentified() && type.isOpaque()) {
73  os << "opaque";
74  return;
75  }
76 
77  if (type.isPacked())
78  os << "packed ";
79 
80  // Put the current type on stack to avoid infinite recursion.
81  os << '(';
82  if (type.isIdentified())
83  stack.insert(type.getName());
84  llvm::interleaveComma(type.getBody(), os, [&](LLVMType subtype) {
85  printTypeImpl(os, subtype, stack);
86  });
87  if (type.isIdentified())
88  stack.pop_back();
89  os << ')';
90 }

92 /// Prints a structure type. Uses `stack` to keep track of the identifiers of
93 /// the structs being printed. Checks if the identifier of a struct is contained
94 /// in `stack`, i.e. whether a self-reference to a recursive stack is being
95 /// printed, and only prints the name to avoid infinite recursion.
96 static void printStructType(llvm::raw_ostream &os, LLVMStructType type,
97  llvm::SetVector<StringRef> &stack) {
98  os << "<";
99  if (type.isIdentified()) {
100  os << '"' << type.getName() << '"';
101  // If we are printing a reference to one of the enclosing structs, just
102  // print the name and stop to avoid infinitely long output.
103  if (stack.count(type.getName())) {
104  os << '>';
105  return;
106  }
107  os << ", ";
108  }
109 
110  printStructTypeBody(os, type, stack);
111  os << '>';
112 }

140 /// Prints the given LLVM dialect type recursively. This leverages closedness of
141 /// the LLVM dialect type system to avoid printing the dialect prefix
142 /// repeatedly. For recursive structures, only prints the name of the structure
143 /// when printing a self-reference. Note that this does not apply to sibling
144 /// references. For example,
145 /// struct<"a", (ptr<struct<"a">>)>
146 /// struct<"c", (ptr<struct<"b", (ptr<struct<"c">>)>>,
147 /// ptr<struct<"b", (ptr<struct<"c">>)>>)>
148 /// note that "b" is printed twice.
149 static void printTypeImpl(llvm::raw_ostream &os, LLVMType type,
150  llvm::SetVector<StringRef> &stack) {
151  if (!type) {
152  os << "<<NULL-TYPE>>";
153  return;
154  }
155 
156  unsigned kind = type.getKind();
157  os << getTypeKeyword(type);
158 
159  // Trivial types only consist of their keyword.
160  if (LLVMType::FIRST_TRIVIAL_TYPE <= kind &&
161  kind <= LLVMType::LAST_TRIVIAL_TYPE)
162  return;
163 
164  if (auto intType = type.dyn_cast<LLVMIntegerType>()) {
165  os << intType.getBitWidth();
166  return;
167  }
168 
169  if (auto ptrType = type.dyn_cast<LLVMPointerType>()) {
170  os << '<';
171  printTypeImpl(os, ptrType.getElementType(), stack);
172  if (ptrType.getAddressSpace() != 0)
173  os << ", " << ptrType.getAddressSpace();
174  os << '>';
175  return;
176  }
177 
178  if (auto arrayType = type.dyn_cast<LLVMArrayType>())
179  return printArrayOrVectorType(os, arrayType, stack);
180  if (auto vectorType = type.dyn_cast<LLVMFixedVectorType>())
181  return printArrayOrVectorType(os, vectorType, stack);
182 
183  if (auto vectorType = type.dyn_cast<LLVMScalableVectorType>()) {
184  os << "<? x " << vectorType.getMinNumElements() << " x ";
185  printTypeImpl(os, vectorType.getElementType(), stack);
186  os << '>';
187  return;
188  }
189 
190  if (auto structType = type.dyn_cast<LLVMStructType>())
191  return printStructType(os, structType, stack);
192 
193  printFunctionType(os, type.cast<LLVMFunctionType>(), stack);
194 }

216 /// Parses an LLVM dialect function type.
217 /// llvm-type :: = `func<` llvm-type `(` llvm-type-list `...`? `)>`
218 static LLVMFunctionType parseFunctionType(DialectAsmParser &parser,
219  llvm::SetVector<StringRef> &stack) {
220  LLVMType returnType;
221  if (parser.parseLess() || parseTypeImpl(parser, stack, returnType) ||
222  parser.parseLParen())
223  return LLVMFunctionType();
224 
225  // Function type without arguments.
226  if (succeeded(parser.parseOptionalRParen())) {
227  if (succeeded(parser.parseGreater()))
228  return LLVMFunctionType::get(returnType, {}, /*isVarArg=*/false);
229  return LLVMFunctionType();
230  }
231 
232  // Parse arguments.
233  SmallVector<LLVMType, 8> argTypes;
234  do {
235  if (succeeded(parser.parseOptionalEllipsis())) {
236  if (parser.parseOptionalRParen() || parser.parseOptionalGreater())
237  return LLVMFunctionType();
238  return LLVMFunctionType::get(returnType, argTypes, /*isVarArg=*/true);
239  }
240 
241  argTypes.push_back(parseTypeImpl(parser, stack));
242  if (!argTypes.back())
243  return LLVMFunctionType();
244  } while (succeeded(parser.parseOptionalComma()));
245 
246  if (parser.parseOptionalRParen() || parser.parseOptionalGreater())
247  return LLVMFunctionType();
248  return LLVMFunctionType::get(returnType, argTypes, /*isVarArg=*/false);
249 }

251 /// Parses an LLVM dialect pointer type.
252 /// llvm-type ::= `ptr<` llvm-type (`,` integer)? `>`
253 static LLVMPointerType parsePointerType(DialectAsmParser &parser,
254  llvm::SetVector<StringRef> &stack) {
255  LLVMType elementType;
256  if (parser.parseLess() || parseTypeImpl(parser, stack, elementType))
257  return LLVMPointerType();
258 
259  unsigned addressSpace = 0;
260  if (succeeded(parser.parseOptionalComma()) &&
261  failed(parser.parseInteger(addressSpace)))
262  return LLVMPointerType();
263  if (failed(parser.parseGreater()))
264  return LLVMPointerType();
265  return LLVMPointerType::get(elementType, addressSpace);
266 }

268 /// Parses an LLVM dialect vector type.
269 /// llvm-type ::= `vec<` `? x`? integer `x` llvm-type `>`
270 /// Supports both fixed and scalable vectors.
271 static LLVMVectorType parseVectorType(DialectAsmParser &parser,
272  llvm::SetVector<StringRef> &stack) {
273  SmallVector<int64_t, 2> dims;
274  llvm::SMLoc dimPos;
275  LLVMType elementType;
276  if (parser.parseLess() || parser.getCurrentLocation(&dimPos) ||
277  parser.parseDimensionList(dims, /*allowDynamic=*/true) ||
278  parseTypeImpl(parser, stack, elementType) || parser.parseGreater())
279  return LLVMVectorType();
280 
281  // We parsed a generic dimension list, but vectors only support two forms:
282  // - single non-dynamic entry in the list (fixed vector);
283  // - two elements, the first dynamic (indicated by -1) and the second
284  // non-dynamic (scalable vector).
285  if (dims.empty() || dims.size() > 2 ||
286  ((dims.size() == 2) ^ (dims[0] == -1)) ||
287  (dims.size() == 2 && dims[1] == -1)) {
288  parser.emitError(dimPos)
289  << "expected '? x <integer> x <type>' or '<integer> x <type>'";
290  return LLVMVectorType();
291  }
292 
293  bool isScalable = dims.size() == 2;
294  return isScalable ? static_cast<LLVMVectorType>(
295  LLVMScalableVectorType::get(elementType, dims[1]))
296  : LLVMFixedVectorType::get(elementType, dims[0]);
297 }

299 /// Parses an LLVM dialect array type.
300 /// llvm-type ::= `array<` integer `x` llvm-type `>`
301 static LLVMArrayType parseArrayType(DialectAsmParser &parser,
302  llvm::SetVector<StringRef> &stack) {
303  SmallVector<int64_t, 1> dims;
304  llvm::SMLoc sizePos;
305  LLVMType elementType;
306  if (parser.parseLess() || parser.getCurrentLocation(&sizePos) ||
307  parser.parseDimensionList(dims, /*allowDynamic=*/false) ||
308  parseTypeImpl(parser, stack, elementType) || parser.parseGreater())
309  return LLVMArrayType();
310 
311  if (dims.size() != 1) {
312  parser.emitError(sizePos) << "expected ? x <type>";
313  return LLVMArrayType();
314  }
315 
316  return LLVMArrayType::get(elementType, dims[0]);
317 }

319 /// Attempts to set the body of an identified structure type. Reports a parsing
320 /// error at `subtypesLoc` in case of failure, uses `stack` to make sure the
321 /// types printed in the error message look like they did when parsed.
322 static LLVMStructType trySetStructBody(LLVMStructType type,
323  ArrayRef<LLVMType> subtypes,
324  bool isPacked, DialectAsmParser &parser,
325  llvm::SMLoc subtypesLoc,
326  llvm::SetVector<StringRef> &stack) {
327  if (succeeded(type.setBody(subtypes, isPacked)))
328  return type;
329 
330  std::string currentBody;
331  llvm::raw_string_ostream currentBodyStream(currentBody);
332  printStructTypeBody(currentBodyStream, type, stack);
333  auto diag = parser.emitError(subtypesLoc)
334  << "identified type already used with a different body";
335  diag.attachNote() << "existing body: " << currentBodyStream.str();
336  return LLVMStructType();
337 }

339 /// Parses an LLVM dialect structure type.
340 /// llvm-type ::= `struct<` (string-literal `,`)? `packed`?
341 /// `(` llvm-type-list `)` `>`
342 /// | `struct<` string-literal `>`
343 /// | `struct<` string-literal `, opaque>`
344 static LLVMStructType parseStructType(DialectAsmParser &parser,
345  llvm::SetVector<StringRef> &stack) {
346  MLIRContext *ctx = parser.getBuilder().getContext();
347 
348  if (failed(parser.parseLess()))
349  return LLVMStructType();
350 
351  // If we are parsing a self-reference to a recursive struct, i.e. the parsing
352  // stack already contains a struct with the same identifier, bail out after
353  // the name.
354  StringRef name;
355  bool isIdentified = succeeded(parser.parseOptionalString(&name));
356  if (isIdentified) {
357  if (stack.count(name)) {
358  if (failed(parser.parseGreater()))
359  return LLVMStructType();
360  return LLVMStructType::getIdentified(ctx, name);
361  }
362  if (failed(parser.parseComma()))
363  return LLVMStructType();
364  }
365 
366  // Handle intentionally opaque structs.
367  llvm::SMLoc kwLoc = parser.getCurrentLocation();
368  if (succeeded(parser.parseOptionalKeyword("opaque"))) {
369  if (!isIdentified)
370  return parser.emitError(kwLoc, "only identified structs can be opaque"),
371  LLVMStructType();
372  if (failed(parser.parseGreater()))
373  return LLVMStructType();
374  auto type = LLVMStructType::getOpaque(name, ctx);
375  if (!type.isOpaque()) {
376  parser.emitError(kwLoc, "redeclaring defined struct as opaque");
377  return LLVMStructType();
378  }
379  return type;
380  }
381 
382  // Check for packedness.
383  bool isPacked = succeeded(parser.parseOptionalKeyword("packed"));
384  if (failed(parser.parseLParen()))
385  return LLVMStructType();
386 
387  // Fast pass for structs with zero subtypes.
388  if (succeeded(parser.parseOptionalRParen())) {
389  if (failed(parser.parseGreater()))
390  return LLVMStructType();
391  if (!isIdentified)
392  return LLVMStructType::getLiteral(ctx, {}, isPacked);
393  auto type = LLVMStructType::getIdentified(ctx, name);
394  return trySetStructBody(type, {}, isPacked, parser, kwLoc, stack);
395  }
396 
397  // Parse subtypes. For identified structs, put the identifier of the struct on
398  // the stack to support self-references in the recursive calls.
399  SmallVector<LLVMType, 4> subtypes;
400  llvm::SMLoc subtypesLoc = parser.getCurrentLocation();
401  do {
402  if (isIdentified)
403  stack.insert(name);
404  LLVMType type = parseTypeImpl(parser, stack);
405  if (!type)
406  return LLVMStructType();
407  subtypes.push_back(type);
408  if (isIdentified)
409  stack.pop_back();
410  } while (succeeded(parser.parseOptionalComma()));
411 
412  if (parser.parseRParen() || parser.parseGreater())
413  return LLVMStructType();
414 
415  // Construct the struct with body.
416  if (!isIdentified)
417  return LLVMStructType::getLiteral(ctx, subtypes, isPacked);
418  auto type = LLVMStructType::getIdentified(ctx, name);
419  return trySetStructBody(type, subtypes, isPacked, parser, subtypesLoc, stack);
420 }

422 /// Parses one of the LLVM dialect types.
423 static LLVMType parseTypeImpl(DialectAsmParser &parser,
424  llvm::SetVector<StringRef> &stack) {
425  // Special case for integers (i[1-9][0-9]*) that are literals rather than
426  // keywords for the parser, so they are not caught by the main dispatch below.
427  // Try parsing it a built-in integer type instead.
428  Type maybeIntegerType;
429  MLIRContext *ctx = parser.getBuilder().getContext();
430  llvm::SMLoc keyLoc = parser.getCurrentLocation();
431  OptionalParseResult result = parser.parseOptionalType(maybeIntegerType);
432  if (result.hasValue()) {
433  if (failed(*result))
434  return LLVMType();
435 
436  if (!maybeIntegerType.isSignlessInteger()) {
437  parser.emitError(keyLoc) << "unexpected type, expected i* or keyword";
438  return LLVMType();
439  }
440  return LLVMIntegerType::get(ctx, maybeIntegerType.getIntOrFloatBitWidth());
441  }
442 
443  // Dispatch to concrete functions.
444  StringRef key;
445  if (failed(parser.parseKeyword(&key)))
446  return LLVMType();
447 
448  return llvm::StringSwitch<function_ref<LLVMType()>>(key)
449  .Case("void", [&] { return LLVMVoidType::get(ctx); })
450  .Case("half", [&] { return LLVMHalfType::get(ctx); })
451  .Case("bfloat", [&] { return LLVMBFloatType::get(ctx); })
452  .Case("float", [&] { return LLVMFloatType::get(ctx); })
453  .Case("double", [&] { return LLVMDoubleType::get(ctx); })
454  .Case("fp128", [&] { return LLVMFP128Type::get(ctx); })
455  .Case("x86_fp80", [&] { return LLVMX86FP80Type::get(ctx); })
456  .Case("ppc_fp128", [&] { return LLVMPPCFP128Type::get(ctx); })
457  .Case("x86_mmx", [&] { return LLVMX86MMXType::get(ctx); })
458  .Case("token", [&] { return LLVMTokenType::get(ctx); })
459  .Case("label", [&] { return LLVMLabelType::get(ctx); })
460  .Case("metadata", [&] { return LLVMMetadataType::get(ctx); })
461  .Case("func", [&] { return parseFunctionType(parser, stack); })
462  .Case("ptr", [&] { return parsePointerType(parser, stack); })
463  .Case("vec", [&] { return parseVectorType(parser, stack); })
464  .Case("array", [&] { return parseArrayType(parser, stack); })
465  .Case("struct", [&] { return parseStructType(parser, stack); })
466  .Default([&] {
467  parser.emitError(keyLoc) << "unknown LLVM type: " << key;
468  return LLVMType();
469  })();
470 }
