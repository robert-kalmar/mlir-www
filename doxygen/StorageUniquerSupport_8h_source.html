35 /// Helper class for implementing traits for storage classes. Clients are not
36 /// expected to interact with this directly, so its members are all protected.
37 template <typename ConcreteType, template <typename> class TraitType>
38 class StorageUserTraitBase {
39 protected:
40  /// Return the derived instance.
41  ConcreteType getInstance() const {
42  // We have to cast up to the trait type, then to the concrete type because
43  // the concrete type will multiply derive from the (content free) TraitBase
44  // class, and we need to be able to disambiguate the path for the C++
45  // compiler.
46  auto *trait = static_cast<const TraitType<ConcreteType> *>(this);
47  return *static_cast<const ConcreteType *>(trait);
48  }
49 };

55 /// Utility class for implementing users of storage classes uniqued by a
56 /// StorageUniquer. Clients are not expected to interact with this class
57 /// directly.
58 template <typename ConcreteT, typename BaseT, typename StorageT,
59  typename UniquerT, template <typename T> class... Traits>
60 class StorageUserBase : public BaseT, public Traits<ConcreteT>... {
61 public:
62  using BaseT::BaseT;
63 
64  /// Utility declarations for the concrete attribute class.
65  using Base = StorageUserBase<ConcreteT, BaseT, StorageT, UniquerT, Traits...>;
66  using ImplType = StorageT;
67 
68  /// Return a unique identifier for the concrete type.
69  static TypeID getTypeID() { return TypeID::get<ConcreteT>(); }
70 
71  /// Provide an implementation of 'classof' that compares the type id of the
72  /// provided value with that of the concerete type.
73  template <typename T> static bool classof(T val) {
74  static_assert(std::is_convertible<ConcreteT, T>::value,
75  "casting from a non-convertible type");
76  return val.getTypeID() == getTypeID();
77  }
78 
79  /// Returns an interface map for the interfaces registered to this storage
80  /// user. This should not be used directly.
81  static detail::InterfaceMap getInterfaceMap() {
82  return detail::InterfaceMap::template get<Traits<ConcreteT>...>();
83  }
84 
85 protected:
86  /// Get or create a new ConcreteT instance within the ctx. This
87  /// function is guaranteed to return a non null object and will assert if
88  /// the arguments provided are invalid.
89  template <typename... Args>
90  static ConcreteT get(MLIRContext *ctx, unsigned kind, Args... args) {
91  // Ensure that the invariants are correct for construction.
92  assert(succeeded(ConcreteT::verifyConstructionInvariants(
93  generateUnknownStorageLocation(ctx), args...)));
94  return UniquerT::template get<ConcreteT>(ctx, kind, args...);
95  }
96 
97  /// Get or create a new ConcreteT instance within the ctx, defined at
98  /// the given, potentially unknown, location. If the arguments provided are
99  /// invalid then emit errors and return a null object.
100  template <typename LocationT, typename... Args>
101  static ConcreteT getChecked(LocationT loc, unsigned kind, Args... args) {
102  // If the construction invariants fail then we return a null attribute.
103  if (failed(ConcreteT::verifyConstructionInvariants(loc, args...)))
104  return ConcreteT();
105  return UniquerT::template get<ConcreteT>(loc.getContext(), kind, args...);
106  }
107 
108  /// Mutate the current storage instance. This will not change the unique key.
109  /// The arguments are forwarded to 'ConcreteT::mutate'.
110  template <typename... Args> LogicalResult mutate(Args &&...args) {
111  return UniquerT::template mutate<ConcreteT>(this->getContext(), getImpl(),
112  std::forward<Args>(args)...);
113  }
114 
115  /// Default implementation that just returns success.
116  template <typename... Args>
117  static LogicalResult verifyConstructionInvariants(Args... args) {
118  return success();
119  }
120 
121  /// Utility for easy access to the storage instance.
122  ImplType *getImpl() const { return static_cast<ImplType *>(this->impl); }
123 };
