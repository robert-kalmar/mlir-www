268 static ParseResult parseAllocLikeOp(OpAsmParser &parser,
269  OperationState &result) {
270  MemRefType type;
271 
272  // Parse the dimension operands and optional symbol operands, followed by a
273  // memref type.
274  unsigned numDimOperands;
275  if (parseDimAndSymbolList(parser, result.operands, numDimOperands) ||
276  parser.parseOptionalAttrDict(result.attributes) ||
277  parser.parseColonType(type))
278  return failure();
279 
280  // Check numDynamicDims against number of question marks in memref type.
281  // Note: this check remains here (instead of in verify()), because the
282  // partition between dim operands and symbol operands is lost after parsing.
283  // Verification still checks that the total number of operands matches
284  // the number of symbols in the affine map, plus the number of dynamic
285  // dimensions in the memref.
286  if (numDimOperands != type.getNumDynamicDims())
287  return parser.emitError(parser.getNameLoc())
288  << "dimension operand count does not equal memref dynamic dimension "
289  "count";
290  result.types.push_back(type);
291  return success();

294 template <typename AllocLikeOp>
295 static LogicalResult verify(AllocLikeOp op) {
296  static_assert(llvm::is_one_of<AllocLikeOp, AllocOp, AllocaOp>::value,
297  "applies to only alloc or alloca");
298  auto memRefType = op.getResult().getType().template dyn_cast<MemRefType>();
299  if (!memRefType)
300  return op.emitOpError("result must be a memref");
301 
302  unsigned numSymbols = 0;
303  if (!memRefType.getAffineMaps().empty()) {
304  // Store number of symbols used in affine map (used in subsequent check).
305  AffineMap affineMap = memRefType.getAffineMaps()[0];
306  numSymbols = affineMap.getNumSymbols();
307  }
308 
309  // Check that the total number of operands matches the number of symbols in
310  // the affine map, plus the number of dynamic dimensions specified in the
311  // memref type.
312  unsigned numDynamicDims = memRefType.getNumDynamicDims();
313  if (op.getNumOperands() != numDynamicDims + numSymbols)
314  return op.emitOpError(
315  "operand count does not equal dimension plus symbol operand count");
316 
317  // Verify that all operands are of type Index.
318  for (auto operandType : op.getOperandTypes())
319  if (!operandType.isIndex())
320  return op.emitOpError("requires operands to be of type Index");
321 
322  if (std::is_same<AllocLikeOp, AllocOp>::value)
323  return success();
324 
325  // An alloca op needs to have an ancestor with an allocation scope trait.
326  if (!op.template getParentWithTrait<OpTrait::AutomaticAllocationScope>())
327  return op.emitOpError(
328  "requires an ancestor op with AutomaticAllocationScope trait");
329 
330  return success();
331 }

334 /// Fold constant dimensions into an alloc like operation.
335 template <typename AllocLikeOp>
336 struct SimplifyAllocConst : public OpRewritePattern<AllocLikeOp> {
337  using OpRewritePattern<AllocLikeOp>::OpRewritePattern;
338 
339  LogicalResult matchAndRewrite(AllocLikeOp alloc,
340  PatternRewriter &rewriter) const override {
341  // Check to see if any dimensions operands are constants. If so, we can
342  // substitute and drop them.
343  if (llvm::none_of(alloc.getOperands(), [](Value operand) {
344  return matchPattern(operand, m_ConstantIndex());
345  }))
346  return failure();
347 
348  auto memrefType = alloc.getType();
349 
350  // Ok, we have one or more constant operands. Collect the non-constant ones
351  // and keep track of the resultant memref type to build.
352  SmallVector<int64_t, 4> newShapeConstants;
353  newShapeConstants.reserve(memrefType.getRank());
354  SmallVector<Value, 4> newOperands;
355 
356  unsigned dynamicDimPos = 0;
357  for (unsigned dim = 0, e = memrefType.getRank(); dim < e; ++dim) {
358  int64_t dimSize = memrefType.getDimSize(dim);
359  // If this is already static dimension, keep it.
360  if (dimSize != -1) {
361  newShapeConstants.push_back(dimSize);
362  continue;
363  }
364  auto *defOp = alloc.getOperand(dynamicDimPos).getDefiningOp();
365  if (auto constantIndexOp = dyn_cast_or_null<ConstantIndexOp>(defOp)) {
366  // Dynamic shape dimension will be folded.
367  newShapeConstants.push_back(constantIndexOp.getValue());
368  } else {
369  // Dynamic shape dimension not folded; copy operand from old memref.
370  newShapeConstants.push_back(-1);
371  newOperands.push_back(alloc.getOperand(dynamicDimPos));
372  }
373  dynamicDimPos++;
374  }
375 
376  // Create new memref type (which will have fewer dynamic dimensions).
377  MemRefType newMemRefType =
378  MemRefType::Builder(memrefType).setShape(newShapeConstants);
379  assert(static_cast<int64_t>(newOperands.size()) ==
380  newMemRefType.getNumDynamicDims());
381 
382  // Create and insert the alloc op for the new memref.
383  auto newAlloc = rewriter.create<AllocLikeOp>(alloc.getLoc(), newMemRefType,
384  newOperands, IntegerAttr());
385  // Insert a cast so we have the same type as the old alloc.
386  auto resultCast = rewriter.create<MemRefCastOp>(alloc.getLoc(), newAlloc,
387  alloc.getType());
388 
389  rewriter.replaceOp(alloc, {resultCast});
390  return success();
391  }
392 };

597 /// Given a successor, try to collapse it to a new destination if it only
598 /// contains a passthrough unconditional branch. If the successor is
599 /// collapsable, `successor` and `successorOperands` are updated to reference
600 /// the new destination and values. `argStorage` is an optional storage to use
601 /// if operands to the collapsed successor need to be remapped.
602 static LogicalResult collapseBranch(Block *&successor,
603  ValueRange &successorOperands,
604  SmallVectorImpl<Value> &argStorage) {
605  // Check that the successor only contains a unconditional branch.
606  if (std::next(successor->begin()) != successor->end())
607  return failure();
608  // Check that the terminator is an unconditional branch.
609  BranchOp successorBranch = dyn_cast<BranchOp>(successor->getTerminator());
610  if (!successorBranch)
611  return failure();
612  // Check that the arguments are only used within the terminator.
613  for (BlockArgument arg : successor->getArguments()) {
614  for (Operation *user : arg.getUsers())
615  if (user != successorBranch)
616  return failure();
617  }
618  // Don't try to collapse branches to infinite loops.
619  Block *successorDest = successorBranch.getDest();
620  if (successorDest == successor)
621  return failure();
622 
623  // Update the operands to the successor. If the branch parent has no
624  // arguments, we can use the branch operands directly.
625  OperandRange operands = successorBranch.getOperands();
626  if (successor->args_empty()) {
627  successor = successorDest;
628  successorOperands = operands;
629  return success();
630  }
631 
632  // Otherwise, we need to remap any argument operands.
633  for (Value operand : operands) {
634  BlockArgument argOperand = operand.dyn_cast<BlockArgument>();
635  if (argOperand && argOperand.getOwner() == successor)

866 /// Compute `lhs` `pred` `rhs`, where `pred` is one of the known floating point
867 /// comparison predicates.
868 bool mlir::applyCmpPredicate(CmpFPredicate predicate, const APFloat &lhs,
869  const APFloat &rhs) {
870  auto cmpResult = lhs.compare(rhs);
871  switch (predicate) {
872  case CmpFPredicate::AlwaysFalse:
873  return false;
874  case CmpFPredicate::OEQ:
875  return cmpResult == APFloat::cmpEqual;
876  case CmpFPredicate::OGT:
877  return cmpResult == APFloat::cmpGreaterThan;
878  case CmpFPredicate::OGE:
879  return cmpResult == APFloat::cmpGreaterThan ||
880  cmpResult == APFloat::cmpEqual;
881  case CmpFPredicate::OLT:
882  return cmpResult == APFloat::cmpLessThan;
883  case CmpFPredicate::OLE:
884  return cmpResult == APFloat::cmpLessThan || cmpResult == APFloat::cmpEqual;
885  case CmpFPredicate::ONE:
886  return cmpResult != APFloat::cmpUnordered && cmpResult != APFloat::cmpEqual;
887  case CmpFPredicate::ORD:
888  return cmpResult != APFloat::cmpUnordered;
889  case CmpFPredicate::UEQ:
890  return cmpResult == APFloat::cmpUnordered || cmpResult == APFloat::cmpEqual;
891  case CmpFPredicate::UGT:
892  return cmpResult == APFloat::cmpUnordered ||
893  cmpResult == APFloat::cmpGreaterThan;
894  case CmpFPredicate::UGE:
895  return cmpResult == APFloat::cmpUnordered ||
896  cmpResult == APFloat::cmpGreaterThan ||
897  cmpResult == APFloat::cmpEqual;
898  case CmpFPredicate::ULT:
899  return cmpResult == APFloat::cmpUnordered ||
900  cmpResult == APFloat::cmpLessThan;
901  case CmpFPredicate::ULE:
902  return cmpResult == APFloat::cmpUnordered ||
903  cmpResult == APFloat::cmpLessThan || cmpResult == APFloat::cmpEqual;
904  case CmpFPredicate::UNE:
905  return cmpResult != APFloat::cmpEqual;
906  case CmpFPredicate::UNO:
907  return cmpResult == APFloat::cmpUnordered;
908  case CmpFPredicate::AlwaysTrue:
909  return true;
910  }
911  llvm_unreachable("unknown comparison predicate");
912 }

974  ValueRange trueDestOperands = condbr.getTrueOperands();

987  rewriter.replaceOpWithNewOp<CondBranchOp>(condbr, condbr.getCondition(),

994 /// cond_br %cond, ^bb1(A, ..., N), ^bb1(A, ..., N)
995 /// -> br ^bb1(A, ..., N)
996 ///
997 /// cond_br %cond, ^bb1(A), ^bb1(B)
998 /// -> %select = select %cond, A, B
999 /// br ^bb1(%select)
1000 ///
1001 struct SimplifyCondBranchIdenticalSuccessors
1002  : public OpRewritePattern<CondBranchOp> {
1003  using OpRewritePattern<CondBranchOp>::OpRewritePattern;
1004 
1005  LogicalResult matchAndRewrite(CondBranchOp condbr,
1006  PatternRewriter &rewriter) const override {
1007  // Check that the true and false destinations are the same and have the same
1008  // operands.
1009  Block *trueDest = condbr.trueDest();
1010  if (trueDest != condbr.falseDest())
1011  return failure();
1012 
1013  // If all of the operands match, no selects need to be generated.
1014  OperandRange trueOperands = condbr.getTrueOperands();
1015  OperandRange falseOperands = condbr.getFalseOperands();
1016  if (trueOperands == falseOperands) {
1017  rewriter.replaceOpWithNewOp<BranchOp>(condbr, trueDest, trueOperands);
1018  return success();
1019  }
1020 
1021  // Otherwise, if the current block is the only predecessor insert selects
1022  // for any mismatched branch operands.
1023  if (trueDest->getUniquePredecessor() != condbr.getOperation()->getBlock())
1024  return failure();
1025 
1026  // Generate a select for any operands that differ between the two.
1027  SmallVector<Value, 8> mergedOperands;
1028  mergedOperands.reserve(trueOperands.size());
1029  Value condition = condbr.getCondition();
1030  for (auto it : llvm::zip(trueOperands, falseOperands)) {
1031  if (std::get<0>(it) == std::get<1>(it))
1032  mergedOperands.push_back(std::get<0>(it));
1033  else
1034  mergedOperands.push_back(rewriter.create<SelectOp>(
1035  condbr.getLoc(), condition, std::get<0>(it), std::get<1>(it)));
1036  }
1037 
1038  rewriter.replaceOpWithNewOp<BranchOp>(condbr, trueDest, mergedOperands);
1039  return success();
1040  }
1041 };

1425 ParseResult DmaStartOp::parse(OpAsmParser &parser, OperationState &result) {
1426  OpAsmParser::OperandType srcMemRefInfo;
1427  SmallVector<OpAsmParser::OperandType, 4> srcIndexInfos;
1428  OpAsmParser::OperandType dstMemRefInfo;
1429  SmallVector<OpAsmParser::OperandType, 4> dstIndexInfos;
1430  OpAsmParser::OperandType numElementsInfo;
1431  OpAsmParser::OperandType tagMemrefInfo;
1432  SmallVector<OpAsmParser::OperandType, 4> tagIndexInfos;
1433  SmallVector<OpAsmParser::OperandType, 2> strideInfo;
1434 
1435  SmallVector<Type, 3> types;
1436  auto indexType = parser.getBuilder().getIndexType();
1437 
1438  // Parse and resolve the following list of operands:
1439  // *) source memref followed by its indices (in square brackets).
1440  // *) destination memref followed by its indices (in square brackets).
1441  // *) dma size in KiB.
1442  if (parser.parseOperand(srcMemRefInfo) ||
1443  parser.parseOperandList(srcIndexInfos, OpAsmParser::Delimiter::Square) ||
1444  parser.parseComma() || parser.parseOperand(dstMemRefInfo) ||

1447  parser.parseComma() || parser.parseOperand(tagMemrefInfo) ||
1448  parser.parseOperandList(tagIndexInfos, OpAsmParser::Delimiter::Square))
1449  return failure();
1450 
1451  // Parse optional stride and elements per stride.
1452  if (parser.parseTrailingOperandList(strideInfo))
1453  return failure();
1454 
1455  bool isStrided = strideInfo.size() == 2;
1456  if (!strideInfo.empty() && !isStrided) {
1457  return parser.emitError(parser.getNameLoc(),
1458  "expected two stride related operands");
1459  }
1460 
1461  if (parser.parseColonTypeList(types))
1462  return failure();
1463  if (types.size() != 3)
1464  return parser.emitError(parser.getNameLoc(), "fewer/more types expected");
1465

1467  parser.resolveOperands(srcIndexInfos, indexType, result.operands) ||
1468  parser.resolveOperand(dstMemRefInfo, types[1], result.operands) ||
1469  parser.resolveOperands(dstIndexInfos, indexType, result.operands) ||

1471  parser.resolveOperand(numElementsInfo, indexType, result.operands) ||

1473  // tag indices should be index.
1474  parser.resolveOperands(tagIndexInfos, indexType, result.operands))
1475  return failure();

1478  if (parser.resolveOperands(strideInfo, indexType, result.operands))
1479  return failure();
1480  }
1481 
1482  return success();

1485 LogicalResult DmaStartOp::verify() {

1487 
1488  // Mandatory non-variadic operands are: src memref, dst memref, tag memref and
1489  // the number of elements.
1490  if (numOperands < 4)
1491  return emitOpError("expected at least 4 operands");

1494  // calls rely on some type properties to compute the operand position.

1497  return emitOpError("expected source to be of memref type");

1499  return emitOpError() << "expected at least " << getSrcMemRefRank() + 4

1502  !llvm::all_of(getSrcIndices().getTypes(),
1503  [](Type t) { return t.isIndex(); }))
1504  return emitOpError("expected source indices to be of index type");
1505 
1506  // 2. Destination memref.
1507  if (!getDstMemRef().getType().isa<MemRefType>())
1508  return emitOpError("expected destination to be of memref type");
1509  unsigned numExpectedOperands = getSrcMemRefRank() + getDstMemRefRank() + 4;
1510  if (numOperands < numExpectedOperands)
1511  return emitOpError() << "expected at least " << numExpectedOperands
1512  << " operands";
1513  if (!getDstIndices().empty() &&
1514  !llvm::all_of(getDstIndices().getTypes(),
1515  [](Type t) { return t.isIndex(); }))
1516  return emitOpError("expected destination indices to be of index type");

1518  // 3. Number of elements.
1519  if (!getNumElements().getType().isIndex())
1520  return emitOpError("expected num elements to be of index type");
1521 
1522  // 4. Tag memref.
1523  if (!getTagMemRef().getType().isa<MemRefType>())
1524  return emitOpError("expected tag to be of memref type");
1525  numExpectedOperands += getTagMemRefRank();
1526  if (numOperands < numExpectedOperands)
1527  return emitOpError() << "expected at least " << numExpectedOperands
1528  << " operands";
1529  if (!getTagIndices().empty() &&
1530  !llvm::all_of(getTagIndices().getTypes(),
1531  [](Type t) { return t.isIndex(); }))
1532  return emitOpError("expected tag indices to be of index type");

1535  if (getSrcMemorySpace() == getDstMemorySpace())
1536  return emitOpError("DMA should be between different memory spaces");
1537

1548  return emitOpError(

1853 bool MemRefCastOp::areCastCompatible(Type a, Type b) {
1854  auto aT = a.dyn_cast<MemRefType>();

1856 
1857  auto uaT = a.dyn_cast<UnrankedMemRefType>();
1858  auto ubT = b.dyn_cast<UnrankedMemRefType>();
1859 
1860  if (aT && bT) {
1861  if (aT.getElementType() != bT.getElementType())
1862  return false;
1863  if (aT.getAffineMaps() != bT.getAffineMaps()) {
1864  int64_t aOffset, bOffset;
1865  SmallVector<int64_t, 4> aStrides, bStrides;
1866  if (failed(getStridesAndOffset(aT, aStrides, aOffset)) ||
1867  failed(getStridesAndOffset(bT, bStrides, bOffset)) ||
1868  aStrides.size() != bStrides.size())
1869  return false;
1870 
1871  // Strides along a dimension/offset are compatible if the value in the
1872  // source memref is static and the value in the target memref is the
1873  // same. They are also compatible if either one is dynamic (see
1874  // description of MemRefCastOp for details).
1875  auto checkCompatible = [](int64_t a, int64_t b) {
1876  return (a == MemRefType::getDynamicStrideOrOffset() ||
1877  b == MemRefType::getDynamicStrideOrOffset() || a == b);
1878  };
1879  if (!checkCompatible(aOffset, bOffset))
1880  return false;
1881  for (auto aStride : enumerate(aStrides))
1882  if (!checkCompatible(aStride.value(), bStrides[aStride.index()]))
1883  return false;
1884  }
1885  if (aT.getMemorySpace() != bT.getMemorySpace())
1886  return false;
1887 
1888  // They must have the same rank, and any specified dimensions must match.
1889  if (aT.getRank() != bT.getRank())
1890  return false;
1891 
1892  for (unsigned i = 0, e = aT.getRank(); i != e; ++i) {
1893  int64_t aDim = aT.getDimSize(i), bDim = bT.getDimSize(i);
1894  if (aDim != -1 && bDim != -1 && aDim != bDim)
1895  return false;
1896  }
1897  return true;
1898  } else {
1899  if (!aT && !uaT)
1900  return false;
1901  if (!bT && !ubT)
1902  return false;
1903  // Unranked to unranked casting is unsupported
1904  if (uaT && ubT)
1905  return false;
1906 
1907  auto aEltType = (aT) ? aT.getElementType() : uaT.getElementType();
1908  auto bEltType = (bT) ? bT.getElementType() : ubT.getElementType();
1909  if (aEltType != bEltType)
1910  return false;
1911 
1912  auto aMemSpace = (aT) ? aT.getMemorySpace() : uaT.getMemorySpace();
1913  auto bMemSpace = (bT) ? bT.getMemorySpace() : ubT.getMemorySpace();
1914  if (aMemSpace != bMemSpace)
1915  return false;
1916 
1917  return true;
1918  }
1919 
1920  return false;
1921 }

2426  for (auto it : llvm::zip(staticOffsets, sourceStrides)) {
2427  auto staticOffset = std::get<0>(it), targetStride = std::get<1>(it);
2428  using namespace saturated_arith;
2429  targetOffset = Wrapper(targetOffset) + Wrapper(staticOffset) * targetStride;
2430  }

2433  // `sourceStrides_i * staticStrides_i`.

2435  targetStrides.reserve(staticOffsets.size());
2436  for (auto it : llvm::zip(sourceStrides, staticStrides)) {
2437  auto sourceStride = std::get<0>(it), staticStride = std::get<1>(it);
2438  using namespace saturated_arith;
2439  targetStrides.push_back(Wrapper(sourceStride) * staticStride);
2440  }
2441 
2442  // The type is now known.
2443  return MemRefType::get(
2444  staticSizes, sourceMemRefType.getElementType(),
2445  makeStridedLinearLayoutMap(targetStrides, targetOffset,
2446  sourceMemRefType.getContext()),
2447  sourceMemRefType.getMemorySpace());

2813 /// Determines whether MemRefCastOp casts to a more dynamic version of the
2814 /// source memref. This is useful to to fold a memref_cast into a consuming op
2815 /// and implement canonicalization patterns for ops in different dialects that
2816 /// may consume the results of memref_cast operations. Such foldable memref_cast
2817 /// operations are typically inserted as `view` and `subview` ops are
2818 /// canonicalized, to preserve the type compatibility of their uses.
2819 ///
2820 /// Returns true when all conditions are met:
2821 /// 1. source and result are ranked memrefs with strided semantics and same
2822 /// element type and rank.
2823 /// 2. each of the source's size, offset or stride has more static information
2824 /// than the corresponding result's size, offset or stride.
2825 ///
2826 /// Example 1:
2827 /// ```mlir
2828 /// %1 = memref_cast %0 : memref<8x16xf32> to memref<?x?xf32>
2829 /// %2 = consumer %1 ... : memref<?x?xf32> ...
2830 /// ```
2831 ///
2832 /// may fold into:
2833 ///
2834 /// ```mlir
2835 /// %2 = consumer %0 ... : memref<8x16xf32> ...
2836 /// ```
2837 ///
2838 /// Example 2:
2839 /// ```
2840 /// %1 = memref_cast %0 : memref<?x16xf32, affine_map<(i, j)->(16 * i + j)>>
2841 /// to memref<?x?xf32>
2842 /// consumer %1 : memref<?x?xf32> ...
2843 /// ```
2844 ///
2845 /// may fold into:
2846 ///
2847 /// ```
2848 /// consumer %0 ... : memref<?x16xf32, affine_map<(i, j)->(16 * i + j)>>
2849 /// ```
2850 bool mlir::canFoldIntoConsumerOp(MemRefCastOp castOp) {
2851  MemRefType sourceType = castOp.source().getType().dyn_cast<MemRefType>();
2852  MemRefType resultType = castOp.getType().dyn_cast<MemRefType>();
2853 
2854  // Requires ranked MemRefType.
2855  if (!sourceType || !resultType)
2856  return false;
2857 
2858  // Requires same elemental type.
2859  if (sourceType.getElementType() != resultType.getElementType())
2860  return false;
2861 
2862  // Requires same rank.
2863  if (sourceType.getRank() != resultType.getRank())
2864  return false;
2865 
2866  // Only fold casts between strided memref forms.
2867  int64_t sourceOffset, resultOffset;
2868  SmallVector<int64_t, 4> sourceStrides, resultStrides;
2869  if (failed(getStridesAndOffset(sourceType, sourceStrides, sourceOffset)) ||
2870  failed(getStridesAndOffset(resultType, resultStrides, resultOffset)))
2871  return false;
2872 
2873  // If cast is towards more static sizes along any dimension, don't fold.
2874  for (auto it : llvm::zip(sourceType.getShape(), resultType.getShape())) {
2875  auto ss = std::get<0>(it), st = std::get<1>(it);
2876  if (ss != st)
2877  if (MemRefType::isDynamic(ss) && !MemRefType::isDynamic(st))
2878  return false;
2879  }
2880 
2881  // If cast is towards more static offset along any dimension, don't fold.
2882  if (sourceOffset != resultOffset)
2883  if (MemRefType::isDynamicStrideOrOffset(sourceOffset) &&
2884  !MemRefType::isDynamicStrideOrOffset(resultOffset))
2885  return false;
2886 
2887  // If cast is towards more static strides along any dimension, don't fold.
2888  for (auto it : llvm::zip(sourceStrides, resultStrides)) {
2889  auto ss = std::get<0>(it), st = std::get<1>(it);
2890  if (ss != st)
2891  if (MemRefType::isDynamicStrideOrOffset(ss) &&
2892  !MemRefType::isDynamicStrideOrOffset(st))
2893  return false;
2894  }
2895 
2896  return true;
2897 }
