45 /// Returns true if the given region contains spv.Return or spv.ReturnValue ops.
46 static inline bool containsReturn(Region &region) {
47  return llvm::any_of(region, [](Block &block) {
48  Operation *terminator = block.getTerminator();
49  return isa<spirv::ReturnOp, spirv::ReturnValueOp>(terminator);
50  });
51 }

53 namespace {
54 /// This class defines the interface for inlining within the SPIR-V dialect.
55 struct SPIRVInlinerInterface : public DialectInlinerInterface {
56  using DialectInlinerInterface::DialectInlinerInterface;
57 
58  /// Returns true if the given region 'src' can be inlined into the region
59  /// 'dest' that is attached to an operation registered to the current dialect.
60  bool isLegalToInline(Region *dest, Region *src,
61  BlockAndValueMapping &) const final {
62  // Return true here when inlining into spv.func, spv.selection, and
63  // spv.loop operations.
64  auto *op = dest->getParentOp();
65  return isa<spirv::FuncOp, spirv::SelectionOp, spirv::LoopOp>(op);
66  }
67 
68  /// Returns true if the given operation 'op', that is registered to this
69  /// dialect, can be inlined into the region 'dest' that is attached to an
70  /// operation registered to the current dialect.
71  bool isLegalToInline(Operation *op, Region *dest,
72  BlockAndValueMapping &) const final {
73  // TODO: Enable inlining structured control flows with return.
74  if ((isa<spirv::SelectionOp, spirv::LoopOp>(op)) &&
75  containsReturn(op->getRegion(0)))
76  return false;
77  // TODO: we need to filter OpKill here to avoid inlining it to
78  // a loop continue construct:
79  // https://github.com/KhronosGroup/SPIRV-Headers/issues/86
80  // However OpKill is fragment shader specific and we don't support it yet.
81  return true;
82  }
83 
84  /// Handle the given inlined terminator by replacing it with a new operation
85  /// as necessary.
86  void handleTerminator(Operation *op, Block *newDest) const final {
87  if (auto returnOp = dyn_cast<spirv::ReturnOp>(op)) {
88  OpBuilder(op).create<spirv::BranchOp>(op->getLoc(), newDest);
89  op->erase();
90  } else if (auto retValOp = dyn_cast<spirv::ReturnValueOp>(op)) {
91  llvm_unreachable("unimplemented spv.ReturnValue in inliner");
92  }
93  }
94 
95  /// Handle the given inlined terminator by replacing it with a new operation
96  /// as necessary.
97  void handleTerminator(Operation *op,
98  ArrayRef<Value> valuesToRepl) const final {
99  // Only spv.ReturnValue needs to be handled here.
100  auto retValOp = dyn_cast<spirv::ReturnValueOp>(op);
101  if (!retValOp)
102  return;
103 
104  // Replace the values directly with the return operands.
105  assert(valuesToRepl.size() == 1 &&
106  "spv.ReturnValue expected to only handle one result");
107  valuesToRepl.front().replaceAllUsesWith(retValOp.value());
108  }
109 };

154 static Type parseAndVerifyType(SPIRVDialect const &dialect,
155  DialectAsmParser &parser) {
156  Type type;
157  llvm::SMLoc typeLoc = parser.getCurrentLocation();
158  if (parser.parseType(type))
159  return Type();
160 
161  // Allow SPIR-V dialect types
162  if (&type.getDialect() == &dialect)
163  return type;
164 
165  // Check other allowed types
166  if (auto t = type.dyn_cast<FloatType>()) {
167  if (type.isBF16()) {
168  parser.emitError(typeLoc, "cannot use 'bf16' to compose SPIR-V types");
169  return Type();
170  }
171  } else if (auto t = type.dyn_cast<IntegerType>()) {
172  if (!ScalarType::isValid(t)) {
173  parser.emitError(typeLoc,
174  "only 1/8/16/32/64-bit integer type allowed but found ")
175  << type;
176  return Type();
177  }
178  } else if (auto t = type.dyn_cast<VectorType>()) {
179  if (t.getRank() != 1) {
180  parser.emitError(typeLoc, "only 1-D vector allowed but found ") << t;
181  return Type();
182  }
183  if (t.getNumElements() > 4) {
184  parser.emitError(
185  typeLoc, "vector length has to be less than or equal to 4 but found ")
186  << t.getNumElements();
187  return Type();
188  }
189  } else {
190  parser.emitError(typeLoc, "cannot use ")
191  << type << " to compose SPIR-V types";
192  return Type();
193  }
194 
195  return type;
196 }

198 static Type parseAndVerifyMatrixType(SPIRVDialect const &dialect,
199  DialectAsmParser &parser) {
200  Type type;
201  llvm::SMLoc typeLoc = parser.getCurrentLocation();
202  if (parser.parseType(type))
203  return Type();
204 
205  if (auto t = type.dyn_cast<VectorType>()) {
206  if (t.getRank() != 1) {
207  parser.emitError(typeLoc, "only 1-D vector allowed but found ") << t;
208  return Type();
209  }
210  if (t.getNumElements() > 4 || t.getNumElements() < 2) {
211  parser.emitError(typeLoc,
212  "matrix columns size has to be less than or equal "
213  "to 4 and greater than or equal 2, but found ")
214  << t.getNumElements();
215  return Type();
216  }
217 
218  if (!t.getElementType().isa<FloatType>()) {
219  parser.emitError(typeLoc, "matrix columns' elements must be of "
220  "Float type, got ")
221  << t.getElementType();
222  return Type();
223  }
224  } else {
225  parser.emitError(typeLoc, "matrix must be composed using vector "
226  "type, got ")
227  << type;
228  return Type();
229  }
230 
231  return type;
232 }

234 /// Parses an optional `, stride = N` assembly segment. If no parsing failure
235 /// occurs, writes `N` to `stride` if existing and writes 0 to `stride` if
236 /// missing.
237 static LogicalResult parseOptionalArrayStride(const SPIRVDialect &dialect,
238  DialectAsmParser &parser,
239  unsigned &stride) {
240  if (failed(parser.parseOptionalComma())) {
241  stride = 0;
242  return success();
243  }
244 
245  if (parser.parseKeyword("stride") || parser.parseEqual())
246  return failure();
247 
248  llvm::SMLoc strideLoc = parser.getCurrentLocation();
249  Optional<unsigned> optStride = parseAndVerify<unsigned>(dialect, parser);
250  if (!optStride)
251  return failure();
252 
253  if (!(stride = optStride.getValue())) {
254  parser.emitError(strideLoc, "ArrayStride must be greater than zero");
255  return failure();
256  }
257  return success();
258 }

260 // element-type ::= integer-type
261 // | floating-point-type
262 // | vector-type
263 // | spirv-type
264 //
265 // array-type ::= `!spv.array` `<` integer-literal `x` element-type
266 // (`,` `stride` `=` integer-literal)? `>`
267 static Type parseArrayType(SPIRVDialect const &dialect,
268  DialectAsmParser &parser) {
269  if (parser.parseLess())
270  return Type();
271 
272  SmallVector<int64_t, 1> countDims;
273  llvm::SMLoc countLoc = parser.getCurrentLocation();
274  if (parser.parseDimensionList(countDims, /*allowDynamic=*/false))
275  return Type();
276  if (countDims.size() != 1) {
277  parser.emitError(countLoc,
278  "expected single integer for array element count");
279  return Type();
280  }
281 
282  // According to the SPIR-V spec:
283  // "Length is the number of elements in the array. It must be at least 1."
284  int64_t count = countDims[0];
285  if (count == 0) {
286  parser.emitError(countLoc, "expected array length greater than 0");
287  return Type();
288  }
289 
290  Type elementType = parseAndVerifyType(dialect, parser);
291  if (!elementType)
292  return Type();
293 
294  unsigned stride = 0;
295  if (failed(parseOptionalArrayStride(dialect, parser, stride)))
296  return Type();
297 
298  if (parser.parseGreater())
299  return Type();
300  return ArrayType::get(elementType, count, stride);
301 }

303 // cooperative-matrix-type ::= `!spv.coopmatrix` `<` element-type ',' scope ','
304 // rows ',' coloumns>`
305 static Type parseCooperativeMatrixType(SPIRVDialect const &dialect,
306  DialectAsmParser &parser) {
307  if (parser.parseLess())
308  return Type();
309 
310  SmallVector<int64_t, 2> dims;
311  llvm::SMLoc countLoc = parser.getCurrentLocation();
312  if (parser.parseDimensionList(dims, /*allowDynamic=*/false))
313  return Type();
314 
315  if (dims.size() != 2) {
316  parser.emitError(countLoc, "expected rows and columns size");
317  return Type();
318  }
319 
320  auto elementTy = parseAndVerifyType(dialect, parser);
321  if (!elementTy)
322  return Type();
323 
324  Scope scope;
325  if (parser.parseComma() || parseEnumKeywordAttr(scope, parser, "scope <id>"))
326  return Type();
327 
328  if (parser.parseGreater())
329  return Type();
330  return CooperativeMatrixNVType::get(elementTy, scope, dims[0], dims[1]);
331 }

387 // matrix-type ::= `!spv.matrix` `<` integer-literal `x` element-type `>`
388 static Type parseMatrixType(SPIRVDialect const &dialect,
389  DialectAsmParser &parser) {
390  if (parser.parseLess())
391  return Type();
392 
393  SmallVector<int64_t, 1> countDims;
394  llvm::SMLoc countLoc = parser.getCurrentLocation();
395  if (parser.parseDimensionList(countDims, /*allowDynamic=*/false))
396  return Type();
397  if (countDims.size() != 1) {
398  parser.emitError(countLoc, "expected single unsigned "
399  "integer for number of columns");
400  return Type();
401  }
402 
403  int64_t columnCount = countDims[0];
404  // According to the specification, Matrices can have 2, 3, or 4 columns
405  if (columnCount < 2 || columnCount > 4) {
406  parser.emitError(countLoc, "matrix is expected to have 2, 3, or 4 "
407  "columns");
408  return Type();
409  }
410 
411  Type columnType = parseAndVerifyMatrixType(dialect, parser);
412  if (!columnType)
413  return Type();
414 
415  if (parser.parseGreater())
416  return Type();
417 
418  return MatrixType::get(columnType, columnCount);
419 }

464 // Functor object to parse a comma separated list of specs. The function
465 // parseAndVerify does the actual parsing and verification of individual
466 // elements. This is a functor since parsing the last element of the list
467 // (termination condition) needs partial specialization.
468 template <typename ParseType, typename... Args>
469 struct ParseCommaSeparatedList {
470  Optional<std::tuple<ParseType, Args...>>
471  operator()(SPIRVDialect const &dialect, DialectAsmParser &parser) const {
472  auto parseVal = parseAndVerify<ParseType>(dialect, parser);
473  if (!parseVal)
474  return llvm::None;
475 
476  auto numArgs = std::tuple_size<std::tuple<Args...>>::value;
477  if (numArgs != 0 && failed(parser.parseComma()))
478  return llvm::None;
479  auto remainingValues = ParseCommaSeparatedList<Args...>{}(dialect, parser);
480  if (!remainingValues)
481  return llvm::None;
482  return std::tuple_cat(std::tuple<ParseType>(parseVal.getValue()),
483  remainingValues.getValue());
484  }
485 };
486 
487 // Partial specialization of the function to parse a comma separated list of
488 // specs to parse the last element of the list.
489 template <typename ParseType>
490 struct ParseCommaSeparatedList<ParseType> {
491  Optional<std::tuple<ParseType>> operator()(SPIRVDialect const &dialect,
492  DialectAsmParser &parser) const {
493  if (auto value = parseAndVerify<ParseType>(dialect, parser))
494  return std::tuple<ParseType>(value.getValue());
495  return llvm::None;
496  }

516  DialectAsmParser &parser) {

518  return Type();

522  ImageSamplingInfo, ImageSamplerUseInfo,
523  ImageFormat>{}(dialect, parser);

532 // Parse decorations associated with a member.
533 static ParseResult parseStructMemberDecorations(
534  SPIRVDialect const &dialect, DialectAsmParser &parser,
535  ArrayRef<Type> memberTypes,
536  SmallVectorImpl<StructType::OffsetInfo> &offsetInfo,
537  SmallVectorImpl<StructType::MemberDecorationInfo> &memberDecorationInfo) {
538 
539  // Check if the first element is offset.
540  llvm::SMLoc offsetLoc = parser.getCurrentLocation();
541  StructType::OffsetInfo offset = 0;
542  OptionalParseResult offsetParseResult = parser.parseOptionalInteger(offset);
543  if (offsetParseResult.hasValue()) {
544  if (failed(*offsetParseResult))
545  return failure();
546 
547  if (offsetInfo.size() != memberTypes.size() - 1) {
548  return parser.emitError(offsetLoc,
549  "offset specification must be given for "
550  "all members");
551  }
552  offsetInfo.push_back(offset);
553  }
554 
555  // Check for no spirv::Decorations.
556  if (succeeded(parser.parseOptionalRSquare()))
557  return success();
558 
559  // If there was an offset, make sure to parse the comma.
560  if (offsetParseResult.hasValue() && parser.parseComma())
561  return failure();
562 
563  // Check for spirv::Decorations.
564  do {
565  auto memberDecoration = parseAndVerify<spirv::Decoration>(dialect, parser);
566  if (!memberDecoration)
567  return failure();
568 
569  // Parse member decoration value if it exists.
570  if (succeeded(parser.parseOptionalEqual())) {
571  auto memberDecorationValue =
572  parseAndVerifyInteger<uint32_t>(dialect, parser);
573 
574  if (!memberDecorationValue)
575  return failure();
576 
577  memberDecorationInfo.emplace_back(
578  static_cast<uint32_t>(memberTypes.size() - 1), 1,
579  memberDecoration.getValue(), memberDecorationValue.getValue());
580  } else {
581  memberDecorationInfo.emplace_back(
582  static_cast<uint32_t>(memberTypes.size() - 1), 0,
583  memberDecoration.getValue(), 0);
584  }
585 
586  } while (succeeded(parser.parseOptionalComma()));
587 
588  return parser.parseRSquare();
589 }

591 // struct-member-decoration ::= integer-literal? spirv-decoration*
592 // struct-type ::= `!spv.struct<` spirv-type (`[` struct-member-decoration `]`)?
593 // (`, ` spirv-type (`[` struct-member-decoration `]`)? `>`
594 static Type parseStructType(SPIRVDialect const &dialect,
595  DialectAsmParser &parser) {
596  if (parser.parseLess())
597  return Type();
598 
599  if (succeeded(parser.parseOptionalGreater()))
600  return StructType::getEmpty(dialect.getContext());
601 
602  SmallVector<Type, 4> memberTypes;
603  SmallVector<StructType::OffsetInfo, 4> offsetInfo;
604  SmallVector<StructType::MemberDecorationInfo, 4> memberDecorationInfo;
605 
606  do {
607  Type memberType;
608  if (parser.parseType(memberType))
609  return Type();
610  memberTypes.push_back(memberType);
611 
612  if (succeeded(parser.parseOptionalLSquare())) {
613  if (parseStructMemberDecorations(dialect, parser, memberTypes, offsetInfo,
614  memberDecorationInfo)) {
615  return Type();
616  }
617  }
618  } while (succeeded(parser.parseOptionalComma()));
619 
620  if (!offsetInfo.empty() && memberTypes.size() != offsetInfo.size()) {
621  parser.emitError(parser.getNameLoc(),
622  "offset specification must be given for all members");
623  return Type();
624  }
625  if (parser.parseGreater())
626  return Type();
627  return StructType::get(memberTypes, offsetInfo, memberDecorationInfo);
628 }

630 // spirv-type ::= array-type
631 // | element-type
632 // | image-type
633 // | pointer-type
634 // | runtime-array-type
635 // | struct-type
636 Type SPIRVDialect::parseType(DialectAsmParser &parser) const {
637  StringRef keyword;
638  if (parser.parseKeyword(&keyword))
639  return Type();
640 
641  if (keyword == "array")
642  return parseArrayType(*this, parser);
643  if (keyword == "coopmatrix")
644  return parseCooperativeMatrixType(*this, parser);
645  if (keyword == "image")
646  return parseImageType(*this, parser);
647  if (keyword == "ptr")
648  return parsePointerType(*this, parser);
649  if (keyword == "rtarray")
650  return parseRuntimeArrayType(*this, parser);
651  if (keyword == "struct")
652  return parseStructType(*this, parser);
653  if (keyword == "matrix")
654  return parseMatrixType(*this, parser);
655  parser.emitError(parser.getNameLoc(), "unknown SPIR-V type: ") << keyword;
656  return Type();
657 }

742 /// Parses a comma-separated list of keywords, invokes `processKeyword` on each
743 /// of the parsed keyword, and returns failure if any error occurs.
744 static ParseResult parseKeywordList(
745  DialectAsmParser &parser,
746  function_ref<LogicalResult(llvm::SMLoc, StringRef)> processKeyword) {
747  if (parser.parseLSquare())
748  return failure();
749 
750  // Special case for empty list.
751  if (succeeded(parser.parseOptionalRSquare()))
752  return success();
753 
754  // Keep parsing the keyword and an optional comma following it. If the comma
755  // is successfully parsed, then we have more keywords to parse.
756  do {
757  auto loc = parser.getCurrentLocation();
758  StringRef keyword;
759  if (parser.parseKeyword(&keyword) || failed(processKeyword(loc, keyword)))
760  return failure();
761  } while (succeeded(parser.parseOptionalComma()));
762 
763  if (parser.parseRSquare())
764  return failure();
765 
766  return success();
767 }
768 
769 /// Parses a spirv::InterfaceVarABIAttr.
770 static Attribute parseInterfaceVarABIAttr(DialectAsmParser &parser) {
771  if (parser.parseLess())
772  return {};
773 
774  Builder &builder = parser.getBuilder();
775 
776  if (parser.parseLParen())
777  return {};
778 
779  IntegerAttr descriptorSetAttr;
780  {
781  auto loc = parser.getCurrentLocation();
782  uint32_t descriptorSet = 0;
783  auto descriptorSetParseResult = parser.parseOptionalInteger(descriptorSet);
784 
785  if (!descriptorSetParseResult.hasValue() ||
786  failed(*descriptorSetParseResult)) {
787  parser.emitError(loc, "missing descriptor set");
788  return {};
789  }
790  descriptorSetAttr = builder.getI32IntegerAttr(descriptorSet);
791  }
792 
793  if (parser.parseComma())
794  return {};
795 
796  IntegerAttr bindingAttr;
797  {
798  auto loc = parser.getCurrentLocation();
799  uint32_t binding = 0;
800  auto bindingParseResult = parser.parseOptionalInteger(binding);
801 
802  if (!bindingParseResult.hasValue() || failed(*bindingParseResult)) {
803  parser.emitError(loc, "missing binding");
804  return {};
805  }
806  bindingAttr = builder.getI32IntegerAttr(binding);
807  }
808 
809  if (parser.parseRParen())
810  return {};
811 
812  IntegerAttr storageClassAttr;
813  {
814  if (succeeded(parser.parseOptionalComma())) {
815  auto loc = parser.getCurrentLocation();
816  StringRef storageClass;
817  if (parser.parseKeyword(&storageClass))
818  return {};
819 
820  if (auto storageClassSymbol =
821  spirv::symbolizeStorageClass(storageClass)) {
822  storageClassAttr = builder.getI32IntegerAttr(
823  static_cast<uint32_t>(*storageClassSymbol));
824  } else {
825  parser.emitError(loc, "unknown storage class: ") << storageClass;
826  return {};
827  }
828  }
829  }
830 
831  if (parser.parseGreater())
832  return {};
833 
834  return spirv::InterfaceVarABIAttr::get(descriptorSetAttr, bindingAttr,
835  storageClassAttr);
836 }

838 static Attribute parseVerCapExtAttr(DialectAsmParser &parser) {
839  if (parser.parseLess())
840  return {};
841 
842  Builder &builder = parser.getBuilder();
843 
844  IntegerAttr versionAttr;
845  {
846  auto loc = parser.getCurrentLocation();
847  StringRef version;
848  if (parser.parseKeyword(&version) || parser.parseComma())
849  return {};
850 
851  if (auto versionSymbol = spirv::symbolizeVersion(version)) {
852  versionAttr =
853  builder.getI32IntegerAttr(static_cast<uint32_t>(*versionSymbol));
854  } else {
855  parser.emitError(loc, "unknown version: ") << version;
856  return {};
857  }
858  }
859 
860  ArrayAttr capabilitiesAttr;
861  {
862  SmallVector<Attribute, 4> capabilities;
863  llvm::SMLoc errorloc;
864  StringRef errorKeyword;
865 
866  auto processCapability = [&](llvm::SMLoc loc, StringRef capability) {
867  if (auto capSymbol = spirv::symbolizeCapability(capability)) {
868  capabilities.push_back(
869  builder.getI32IntegerAttr(static_cast<uint32_t>(*capSymbol)));
870  return success();
871  }
872  return errorloc = loc, errorKeyword = capability, failure();
873  };
874  if (parseKeywordList(parser, processCapability) || parser.parseComma()) {
875  if (!errorKeyword.empty())
876  parser.emitError(errorloc, "unknown capability: ") << errorKeyword;
877  return {};
878  }
879 
880  capabilitiesAttr = builder.getArrayAttr(capabilities);
881  }
882 
883  ArrayAttr extensionsAttr;
884  {
885  SmallVector<Attribute, 1> extensions;
886  llvm::SMLoc errorloc;
887  StringRef errorKeyword;
888 
889  auto processExtension = [&](llvm::SMLoc loc, StringRef extension) {
890  if (spirv::symbolizeExtension(extension)) {
891  extensions.push_back(builder.getStringAttr(extension));
892  return success();
893  }
894  return errorloc = loc, errorKeyword = extension, failure();
895  };
896  if (parseKeywordList(parser, processExtension)) {
897  if (!errorKeyword.empty())
898  parser.emitError(errorloc, "unknown extension: ") << errorKeyword;
899  return {};
900  }
901 
902  extensionsAttr = builder.getArrayAttr(extensions);
903  }
904 
905  if (parser.parseGreater())
906  return {};
907 
908  return spirv::VerCapExtAttr::get(versionAttr, capabilitiesAttr,
909  extensionsAttr);
910 }

912 /// Parses a spirv::TargetEnvAttr.
913 static Attribute parseTargetEnvAttr(DialectAsmParser &parser) {
914  if (parser.parseLess())
915  return {};
916 
917  spirv::VerCapExtAttr tripleAttr;
918  if (parser.parseAttribute(tripleAttr) || parser.parseComma())
919  return {};
920 
921  DictionaryAttr limitsAttr;
922  {
923  auto loc = parser.getCurrentLocation();
924  if (parser.parseAttribute(limitsAttr))
925  return {};
926 
927  if (!limitsAttr.isa<spirv::ResourceLimitsAttr>()) {
928  parser.emitError(
929  loc,
930  "limits must be a dictionary attribute containing two 32-bit integer "
931  "attributes 'max_compute_workgroup_invocations' and "
932  "'max_compute_workgroup_size'");
933  return {};
934  }
935  }
936 
937  if (parser.parseGreater())
938  return {};
939 
940  return spirv::TargetEnvAttr::get(tripleAttr, limitsAttr);
941 }

943 Attribute SPIRVDialect::parseAttribute(DialectAsmParser &parser,
944  Type type) const {
945  // SPIR-V attributes are dictionaries so they do not have type.
946  if (type) {
947  parser.emitError(parser.getNameLoc(), "unexpected type");
948  return {};
949  }
950 
951  // Parse the kind keyword first.
952  StringRef attrKind;
953  if (parser.parseKeyword(&attrKind))
954  return {};
955 
956  if (attrKind == spirv::TargetEnvAttr::getKindName())
957  return parseTargetEnvAttr(parser);
958  if (attrKind == spirv::VerCapExtAttr::getKindName())
959  return parseVerCapExtAttr(parser);
960  if (attrKind == spirv::InterfaceVarABIAttr::getKindName())
961  return parseInterfaceVarABIAttr(parser);
962 
963  parser.emitError(parser.getNameLoc(), "unknown SPIR-V attribute kind: ")
964  << attrKind;
965  return {};
966 }
