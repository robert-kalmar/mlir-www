33 /// Performs standalone tiling of a single LinalgOp by `tileSizes`.
34 /// and permute the loop nest according to `interchangeVector`
35 /// The permutation is expressed as a list of integers that specify
36 /// the new ordering of the loop nest. The length of `interchangeVector`
37 /// must be equal to the length of `tileSizes`.
38 /// An empty vector is interpreted as the identity permutation and the
39 /// transformation returns early.
40 ///
41 /// Returns a struct containing the tiled loops in the specified order
42 /// and the cloned op if successful, llvm::None otherwise.
43 ///
44 /// E.g. the permutation `(i,j,k) -> (j,k,i)` is expressed by
45 /// `interchangeVector = [1,2,0]`. All values in `interchangeVector` must be
46 /// integers, in the range 0..`tileSizes.size()` without duplications
47 /// (i.e. `[1,1,2]` is an invalid permutation).
48 Optional<TiledLinalgOp> tileLinalgOp(OpBuilder &b, LinalgOp op,
49  const LinalgTilingOptions &options);

83 struct LinalgPromotionOptions {
84  /// Indices of subViews to promote. If `None`, try to promote all operands.
85  Optional<DenseSet<unsigned>> operandsToPromote = None;
86  LinalgPromotionOptions &setOperandsToPromote(ArrayRef<int64_t> operands) {
87  operandsToPromote = DenseSet<unsigned>();
88  operandsToPromote->insert(operands.begin(), operands.end());
89  return *this;
90  }
91  /// If ith element of `useFullTiles` is true the full view should be used for
92  /// the promoted buffer of the ith operand in `operandsToPromote`. Otherwise
93  /// the partial view will be used.
94  /// The decision is defaulted to `useFullTileBuffersDefault` when
95  /// `useFullTileBuffers` is None and for operands missing from
96  /// `useFullTileBuffers`.
97  Optional<llvm::SmallBitVector> useFullTileBuffers = None;
98  LinalgPromotionOptions &setUseFullTileBuffers(ArrayRef<bool> useFullTiles) {
99  unsigned size = useFullTiles.size();
100  llvm::SmallBitVector tmp(size, false);
101  for (unsigned i = 0; i < size; ++i)
102  tmp[i] = useFullTiles[i];
103  useFullTileBuffers = tmp;
104  return *this;
105  }
106  /// If true all operands unspecified by `useFullTileBuffers` will use the full
107  /// view, otherwise the partial view.
108  bool useFullTileBuffersDefault = false;
109  LinalgPromotionOptions &setUseFullTileBuffersByDefault(bool use) {
110  useFullTileBuffersDefault = use;
111  return *this;
112  }
113  /// Allow the use of dynamicaly-sized buffers.
114  bool dynamicBuffers = false;
115  LinalgPromotionOptions &setDynamicBuffers(unsigned dynamic) {
116  dynamicBuffers = dynamic;
117  return *this;
118  }
119  /// Alignment of promoted buffer. If `None` do not specify alignment.
120  Optional<unsigned> alignment = None;
121  LinalgPromotionOptions &setAlignment(unsigned align) {
122  alignment = align;
123  return *this;
124  }
125  /// Use alloca with the default allocation scheme.

405 template <typename OpTy> struct LinalgLoweringPattern : public RewritePattern {
406  LinalgLoweringPattern(MLIRContext *context, LinalgLoweringType loweringType,
407  LinalgMarker marker = LinalgMarker(),
408  PatternBenefit benefit = 1)
409  : RewritePattern(OpTy::getOperationName(), {}, benefit, context),
410  marker(marker), loweringType(loweringType) {}
411  // TODO: Move implementation to .cpp once named ops are auto-generated.
412  LogicalResult matchAndRewrite(Operation *op,
413  PatternRewriter &rewriter) const override {
414  LinalgOp linalgOp = dyn_cast<LinalgOp>(op);
415  if (!linalgOp)
416  return failure();
417  if (failed(marker.checkAndNotify(rewriter, linalgOp)))
418  return failure();
419 
420  if (loweringType == LinalgLoweringType::LibraryCall) {
421  // TODO: Move lowering to library calls here.
422  return failure();
423  } else if (loweringType == LinalgLoweringType::Loops) {
424  if (failed(linalgOpToLoops(rewriter, op)))
425  return failure();
426  } else if (loweringType == LinalgLoweringType::AffineLoops) {
427  if (failed(linalgOpToAffineLoops(rewriter, op)))
428  return failure();
429  } else if (failed(linalgOpToParallelLoops(rewriter, op))) {
430  return failure();
431  }
432  rewriter.eraseOp(op);
433  return success();
434  }
435 
436 private:
437  /// LinalgTransformMarker handles special attribute manipulations.
438  LinalgMarker marker;
439  /// Controls whether the pattern lowers to library calls, scf.for, affine.for
440  /// or scf.parallel.
441  LinalgLoweringType loweringType;
442 };
