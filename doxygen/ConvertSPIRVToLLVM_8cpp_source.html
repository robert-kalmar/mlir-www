53 static unsigned getBitWidth(Type type) {
54  assert((type.isIntOrFloat() || type.isa<VectorType>()) &&

109 /// Utility function for bitfiled ops:
110 /// - `BitFieldInsert`
111 /// - `BitFieldSExtract`
112 /// - `BitFieldUExtract`
113 /// Truncates or extends the value. If the bitwidth of the value is the same as
114 /// `dstType` bitwidth, the value remains unchanged.
115 static Value optionallyTruncateOrExtend(Location loc, Value value, Type dstType,
116  PatternRewriter &rewriter) {
117  auto srcType = value.getType();
118  auto llvmType = dstType.cast<LLVM::LLVMType>();
119  unsigned targetBitWidth = getLLVMTypeBitWidth(llvmType);
120  unsigned valueBitWidth =
121  srcType.isa<LLVM::LLVMType>()
122  ? getLLVMTypeBitWidth(srcType.cast<LLVM::LLVMType>())
123  : getBitWidth(srcType);
124 
125  if (valueBitWidth < targetBitWidth)
126  return rewriter.create<LLVM::ZExtOp>(loc, llvmType, value);
127  // If the bit widths of `Count` and `Offset` are greater than the bit width
128  // of the target type, they are truncated. Truncation is safe since `Count`
129  // and `Offset` must be no more than 64 for op behaviour to be defined. Hence,
130  // both values can be expressed in 8 bits.
131  if (valueBitWidth > targetBitWidth)
132  return rewriter.create<LLVM::TruncOp>(loc, llvmType, value);
133  return value;
134 }

136 /// Broadcasts the value to vector with `numElements` number of elements.
137 static Value broadcast(Location loc, Value toBroadcast, unsigned numElements,
138  LLVMTypeConverter &typeConverter,
139  ConversionPatternRewriter &rewriter) {
140  auto vectorType = VectorType::get(numElements, toBroadcast.getType());
141  auto llvmVectorType = typeConverter.convertType(vectorType);
142  auto llvmI32Type = typeConverter.convertType(rewriter.getIntegerType(32));
143  Value broadcasted = rewriter.create<LLVM::UndefOp>(loc, llvmVectorType);
144  for (unsigned i = 0; i < numElements; ++i) {
145  auto index = rewriter.create<LLVM::ConstantOp>(
146  loc, llvmI32Type, rewriter.getI32IntegerAttr(i));
147  broadcasted = rewriter.create<LLVM::InsertElementOp>(
148  loc, llvmVectorType, broadcasted, toBroadcast, index);
149  }
150  return broadcasted;
151 }

153 /// Broadcasts the value. If `srcType` is a scalar, the value remains unchanged.

164 /// Utility function for bitfiled ops: `BitFieldInsert`, `BitFieldSExtract` and
165 /// `BitFieldUExtract`.
166 /// Broadcast `Offset` and `Count` to match the type of `Base`. If `Base` is of
167 /// a vector type, construct a vector that has:
168 /// - same number of elements as `Base`
169 /// - each element has the type that is the same as the type of `Offset` or
170 /// `Count`
171 /// - each element has the same value as `Offset` or `Count`
172 /// Then cast `Offset` and `Count` if their bit width is different
173 /// from `Base` bit width.
174 static Value processCountOrOffset(Location loc, Value value, Type srcType,
175  Type dstType, LLVMTypeConverter &converter,
176  ConversionPatternRewriter &rewriter) {
177  Value broadcasted =
178  optionallyBroadcast(loc, value, srcType, converter, rewriter);
179  return optionallyTruncateOrExtend(loc, broadcasted, dstType, rewriter);
180 }

276 class BitFieldInsertPattern
277  : public SPIRVToLLVMConversion<spirv::BitFieldInsertOp> {
278 public:
279  using SPIRVToLLVMConversion<spirv::BitFieldInsertOp>::SPIRVToLLVMConversion;
280 
281  LogicalResult
282  matchAndRewrite(spirv::BitFieldInsertOp op, ArrayRef<Value> operands,
283  ConversionPatternRewriter &rewriter) const override {
284  auto srcType = op.getType();
285  auto dstType = typeConverter.convertType(srcType);
286  if (!dstType)
287  return failure();
288  Location loc = op.getLoc();
289 
290  // Process `Offset` and `Count`: broadcast and extend/truncate if needed.
291  Value offset = processCountOrOffset(loc, op.offset(), srcType, dstType,
292  typeConverter, rewriter);
293  Value count = processCountOrOffset(loc, op.count(), srcType, dstType,
294  typeConverter, rewriter);
295 
296  // Create a mask with bits set outside [Offset, Offset + Count - 1].
297  Value minusOne = createConstantAllBitsSet(loc, srcType, dstType, rewriter);
298  Value maskShiftedByCount =
299  rewriter.create<LLVM::ShlOp>(loc, dstType, minusOne, count);
300  Value negated = rewriter.create<LLVM::XOrOp>(loc, dstType,
301  maskShiftedByCount, minusOne);
302  Value maskShiftedByCountAndOffset =
303  rewriter.create<LLVM::ShlOp>(loc, dstType, negated, offset);
304  Value mask = rewriter.create<LLVM::XOrOp>(
305  loc, dstType, maskShiftedByCountAndOffset, minusOne);
306 
307  // Extract unchanged bits from the `Base` that are outside of
308  // [Offset, Offset + Count - 1]. Then `or` with shifted `Insert`.
309  Value baseAndMask =
310  rewriter.create<LLVM::AndOp>(loc, dstType, op.base(), mask);
311  Value insertShiftedByOffset =
312  rewriter.create<LLVM::ShlOp>(loc, dstType, op.insert(), offset);
313  rewriter.replaceOpWithNewOp<LLVM::OrOp>(op, dstType, baseAndMask,
314  insertShiftedByOffset);
315  return success();
316  }

319 /// Converts SPIR-V ConstantOp with scalar or vector type.
320 class ConstantScalarAndVectorPattern
321  : public SPIRVToLLVMConversion<spirv::ConstantOp> {
322 public:
323  using SPIRVToLLVMConversion<spirv::ConstantOp>::SPIRVToLLVMConversion;
324 
325  LogicalResult
326  matchAndRewrite(spirv::ConstantOp constOp, ArrayRef<Value> operands,
327  ConversionPatternRewriter &rewriter) const override {
328  auto srcType = constOp.getType();
329  if (!srcType.isa<VectorType>() && !srcType.isIntOrFloat())
330  return failure();
331 
332  auto dstType = typeConverter.convertType(srcType);
333  if (!dstType)
334  return failure();
335 
336  // SPIR-V constant can be a signed/unsigned integer, which has to be
337  // casted to signless integer when converting to LLVM dialect. Removing the
338  // sign bit may have unexpected behaviour. However, it is better to handle
339  // it case-by-case, given that the purpose of the conversion is not to
340  // cover all possible corner cases.
341  if (isSignedIntegerOrVector(srcType) ||
342  isUnsignedIntegerOrVector(srcType)) {
343  auto *context = rewriter.getContext();
344  auto signlessType = IntegerType::get(getBitWidth(srcType), context);
345 
346  if (srcType.isa<VectorType>()) {
347  auto dstElementsAttr = constOp.value().cast<DenseIntElementsAttr>();
348  rewriter.replaceOpWithNewOp<LLVM::ConstantOp>(
349  constOp, dstType,
350  dstElementsAttr.mapValues(
351  signlessType, [&](const APInt &value) { return value; }));
352  return success();
353  }
354  auto srcAttr = constOp.value().cast<IntegerAttr>();
355  auto dstAttr = rewriter.getIntegerAttr(signlessType, srcAttr.getValue());
356  rewriter.replaceOpWithNewOp<LLVM::ConstantOp>(constOp, dstType, dstAttr);
357  return success();
358  }
359  rewriter.replaceOpWithNewOp<LLVM::ConstantOp>(constOp, dstType, operands,
360  constOp.getAttrs());
361  return success();
362  }

365 class BitFieldSExtractPattern
366  : public SPIRVToLLVMConversion<spirv::BitFieldSExtractOp> {
367 public:
368  using SPIRVToLLVMConversion<spirv::BitFieldSExtractOp>::SPIRVToLLVMConversion;
369 
370  LogicalResult
371  matchAndRewrite(spirv::BitFieldSExtractOp op, ArrayRef<Value> operands,
372  ConversionPatternRewriter &rewriter) const override {
373  auto srcType = op.getType();
374  auto dstType = typeConverter.convertType(srcType);
375  if (!dstType)
376  return failure();
377  Location loc = op.getLoc();
378 
379  // Process `Offset` and `Count`: broadcast and extend/truncate if needed.
380  Value offset = processCountOrOffset(loc, op.offset(), srcType, dstType,
381  typeConverter, rewriter);
382  Value count = processCountOrOffset(loc, op.count(), srcType, dstType,
383  typeConverter, rewriter);
384 
385  // Create a constant that holds the size of the `Base`.
386  IntegerType integerType;
387  if (auto vecType = srcType.dyn_cast<VectorType>())
388  integerType = vecType.getElementType().cast<IntegerType>();
389  else
390  integerType = srcType.cast<IntegerType>();
391 
392  auto baseSize = rewriter.getIntegerAttr(integerType, getBitWidth(srcType));
393  Value size =
394  srcType.isa<VectorType>()
395  ? rewriter.create<LLVM::ConstantOp>(
396  loc, dstType,
397  SplatElementsAttr::get(srcType.cast<ShapedType>(), baseSize))
398  : rewriter.create<LLVM::ConstantOp>(loc, dstType, baseSize);
399 
400  // Shift `Base` left by [sizeof(Base) - (Count + Offset)], so that the bit
401  // at Offset + Count - 1 is the most significant bit now.
402  Value countPlusOffset =
403  rewriter.create<LLVM::AddOp>(loc, dstType, count, offset);
404  Value amountToShiftLeft =
405  rewriter.create<LLVM::SubOp>(loc, dstType, size, countPlusOffset);
406  Value baseShiftedLeft = rewriter.create<LLVM::ShlOp>(
407  loc, dstType, op.base(), amountToShiftLeft);
408 
409  // Shift the result right, filling the bits with the sign bit.
410  Value amountToShiftRight =
411  rewriter.create<LLVM::AddOp>(loc, dstType, offset, amountToShiftLeft);
412  rewriter.replaceOpWithNewOp<LLVM::AShrOp>(op, dstType, baseShiftedLeft,
413  amountToShiftRight);
414  return success();
415  }

721 /// Converts `spv.loop` to LLVM dialect. All blocks within selection should be
722 /// reachable for conversion to succeed.
723 /// The structure of the loop in LLVM dialect will be the following:
724 ///
725 /// +------------------------------------+
726 /// | <code before spv.loop> |
727 /// | llvm.br ^header |
728 /// +------------------------------------+
729 /// |
730 /// +----------------+ |
731 /// | | |
732 /// | V V
733 /// | +------------------------------------+
734 /// | | ^header: |
735 /// | | <header code> |
736 /// | | llvm.cond_br %cond, ^body, ^exit |
737 /// | +------------------------------------+
738 /// | |
739 /// | |----------------------+
740 /// | | |
741 /// | V |
742 /// | +------------------------------------+ |
743 /// | | ^body: | |
744 /// | | <body code> | |
745 /// | | llvm.br ^continue | |
746 /// | +------------------------------------+ |
747 /// | | |
748 /// | V |
749 /// | +------------------------------------+ |
750 /// | | ^continue: | |
751 /// | | <continue code> | |
752 /// | | llvm.br ^header | |
753 /// | +------------------------------------+ |
754 /// | | |
755 /// +---------------+ +----------------------+
756 /// |
757 /// V
758 /// +------------------------------------+
759 /// | ^exit: |
760 /// | llvm.br ^remaining |
761 /// +------------------------------------+
762 /// |
763 /// V
764 /// +------------------------------------+
765 /// | ^remaining: |
766 /// | <code after spv.loop> |
767 /// +------------------------------------+
768 ///
769 class LoopPattern : public SPIRVToLLVMConversion<spirv::LoopOp> {
770 public:
771  using SPIRVToLLVMConversion<spirv::LoopOp>::SPIRVToLLVMConversion;
772 
773  LogicalResult
774  matchAndRewrite(spirv::LoopOp loopOp, ArrayRef<Value> operands,
775  ConversionPatternRewriter &rewriter) const override {
776  // There is no support of loop control at the moment.
777  if (loopOp.loop_control() != spirv::LoopControl::None)
778  return failure();
779 
780  Location loc = loopOp.getLoc();
781 
782  // Split the current block after `spv.loop`. The remaing ops will be used in
783  // `endBlock`.
784  Block *currentBlock = rewriter.getBlock();
785  auto position = Block::iterator(loopOp);
786  Block *endBlock = rewriter.splitBlock(currentBlock, position);
787 
788  // Remove entry block and create a branch in the current block going to the
789  // header block.
790  Block *entryBlock = loopOp.getEntryBlock();
791  assert(entryBlock->getOperations().size() == 1);
792  auto brOp = dyn_cast<spirv::BranchOp>(entryBlock->getOperations().front());
793  if (!brOp)
794  return failure();
795  Block *headerBlock = loopOp.getHeaderBlock();
796  rewriter.setInsertionPointToEnd(currentBlock);
797  rewriter.create<LLVM::BrOp>(loc, brOp.getBlockArguments(), headerBlock);
798  rewriter.eraseBlock(entryBlock);
799 
800  // Branch from merge block to end block.
801  Block *mergeBlock = loopOp.getMergeBlock();
802  Operation *terminator = mergeBlock->getTerminator();
803  ValueRange terminatorOperands = terminator->getOperands();
804  rewriter.setInsertionPointToEnd(mergeBlock);
805  rewriter.create<LLVM::BrOp>(loc, terminatorOperands, endBlock);
806 
807  rewriter.inlineRegionBefore(loopOp.body(), endBlock);
808  rewriter.replaceOp(loopOp, endBlock->getArguments());
809  return success();
810  }
811 };

825 /// Converts `spv.selection` with `spv.BranchConditional` in its header block.
826 /// All blocks within selection should be reachable for conversion to succeed.
827 class SelectionPattern : public SPIRVToLLVMConversion<spirv::SelectionOp> {
828 public:
829  using SPIRVToLLVMConversion<spirv::SelectionOp>::SPIRVToLLVMConversion;
830 
831  LogicalResult
832  matchAndRewrite(spirv::SelectionOp op, ArrayRef<Value> operands,
833  ConversionPatternRewriter &rewriter) const override {
834  // There is no support for `Flatten` or `DontFlatten` selection control at
835  // the moment. This are just compiler hints and can be performed during the
836  // optimization passes.
837  if (op.selection_control() != spirv::SelectionControl::None)
838  return failure();
839 
840  // `spv.selection` should have at least two blocks: one selection header
841  // block and one merge block. If no blocks are present, or control flow
842  // branches straight to merge block (two blocks are present), the op is
843  // redundant and it is erased.
844  if (op.body().getBlocks().size() <= 2) {
845  rewriter.eraseOp(op);
846  return success();
847  }
848 
849  Location loc = op.getLoc();
850 
851  // Split the current block after `spv.selection`. The remaing ops will be
852  // used in `continueBlock`.
853  auto *currentBlock = rewriter.getInsertionBlock();
854  rewriter.setInsertionPointAfter(op);
855  auto position = rewriter.getInsertionPoint();
856  auto *continueBlock = rewriter.splitBlock(currentBlock, position);
857 
858  // Extract conditional branch information from the header block. By SPIR-V
859  // dialect spec, it should contain `spv.BranchConditional` or `spv.Switch`
860  // op. Note that `spv.Switch op` is not supported at the moment in the
861  // SPIR-V dialect. Remove this block when finished.
862  auto *headerBlock = op.getHeaderBlock();
863  assert(headerBlock->getOperations().size() == 1);
864  auto condBrOp = dyn_cast<spirv::BranchConditionalOp>(
865  headerBlock->getOperations().front());
866  if (!condBrOp)
867  return failure();
868  rewriter.eraseBlock(headerBlock);
869 
870  // Branch from merge block to continue block.
871  auto *mergeBlock = op.getMergeBlock();
872  Operation *terminator = mergeBlock->getTerminator();
873  ValueRange terminatorOperands = terminator->getOperands();
874  rewriter.setInsertionPointToEnd(mergeBlock);
875  rewriter.create<LLVM::BrOp>(loc, terminatorOperands, continueBlock);
876 
877  // Link current block to `true` and `false` blocks within the selection.
878  Block *trueBlock = condBrOp.getTrueBlock();
879  Block *falseBlock = condBrOp.getFalseBlock();
880  rewriter.setInsertionPointToEnd(currentBlock);
881  rewriter.create<LLVM::CondBrOp>(loc, condBrOp.condition(), trueBlock,
882  condBrOp.trueTargetOperands(), falseBlock,
883  condBrOp.falseTargetOperands());
884 
885  rewriter.inlineRegionBefore(op.body(), continueBlock);
886  rewriter.replaceOp(op, continueBlock->getArguments());
887  return success();
888  }
889 };
