23  switch (getToken().getKind()) {

25  case Token::kw_memref:
26  case Token::kw_tensor:
27  case Token::kw_complex:

33  case Token::kw_f32:
34  case Token::kw_f64:
35  case Token::kw_index:

87 /// Parse a parenthesized list of types.
88 ///
89 /// type-list-parens ::= `(` `)`
90 /// | `(` type-list-no-parens `)`
91 ///
92 ParseResult Parser::parseTypeListParens(SmallVectorImpl<Type> &elements) {
93  if (parseToken(Token::l_paren, "expected '('"))
94  return failure();
95 
96  // Handle empty lists.
97  if (getToken().is(Token::r_paren))
98  return consumeToken(), success();
99 
100  if (parseTypeListNoParens(elements) ||
101  parseToken(Token::r_paren, "expected ')'"))
102  return failure();
103  return success();
104 }

106 /// Parse a complex type.
107 ///
108 /// complex-type ::= `complex` `<` type `>`
109 ///
110 Type Parser::parseComplexType() {
111  consumeToken(Token::kw_complex);
112 
113  // Parse the '<'.
114  if (parseToken(Token::less, "expected '<' in complex type"))
115  return nullptr;
116 
117  llvm::SMLoc elementTypeLoc = getToken().getLoc();
118  auto elementType = parseType();

120  parseToken(Token::greater, "expected '>' in complex type"))
121  return nullptr;

145 /// Parse the offset and strides from a strided layout specification.
146 ///
147 /// strided-layout ::= `offset:` dimension `,` `strides: ` stride-list
148 ///
149 ParseResult Parser::parseStridedLayout(int64_t &offset,
150  SmallVectorImpl<int64_t> &strides) {
151  // Parse offset.
152  consumeToken(Token::kw_offset);
153  if (!consumeIf(Token::colon))
154  return emitError("expected colon after `offset` keyword");
155  auto maybeOffset = getToken().getUnsignedIntegerValue();
156  bool question = getToken().is(Token::question);
157  if (!maybeOffset && !question)
158  return emitError("invalid offset");
159  offset = maybeOffset ? static_cast<int64_t>(maybeOffset.getValue())
160  : MemRefType::getDynamicStrideOrOffset();
161  consumeToken();
162 
163  if (!consumeIf(Token::comma))
164  return emitError("expected comma after offset value");
165 
166  // Parse stride list.
167  if (!consumeIf(Token::kw_strides))
168  return emitError("expected `strides` keyword after offset specification");
169  if (!consumeIf(Token::colon))
170  return emitError("expected colon after `strides` keyword");
171  if (failed(parseStrideList(strides)))
172  return emitError("invalid braces-enclosed stride list");
173  if (llvm::any_of(strides, [](int64_t st) { return st == 0; }))
174  return emitError("invalid memref stride");
175 
176  return success();
177 }

179 /// Parse a memref type.
180 ///
181 /// memref-type ::= ranked-memref-type | unranked-memref-type
182 ///
183 /// ranked-memref-type ::= `memref` `<` dimension-list-ranked type
184 /// (`,` semi-affine-map-composition)? (`,`
185 /// memory-space)? `>`
186 ///
187 /// unranked-memref-type ::= `memref` `<*x` type (`,` memory-space)? `>`
188 ///
189 /// semi-affine-map-composition ::= (semi-affine-map `,` )* semi-affine-map
190 /// memory-space ::= integer-literal /* | TODO: address-space-id */
191 ///
192 Type Parser::parseMemRefType() {
193  consumeToken(Token::kw_memref);
194 
195  if (parseToken(Token::less, "expected '<' in memref type"))
196  return nullptr;
197 
198  bool isUnranked;
199  SmallVector<int64_t, 4> dimensions;
200 
201  if (consumeIf(Token::star)) {
202  // This is an unranked memref type.
203  isUnranked = true;
204  if (parseXInDimensionList())
205  return nullptr;
206 
207  } else {
208  isUnranked = false;
209  if (parseDimensionListRanked(dimensions))
210  return nullptr;
211  }
212 
213  // Parse the element type.
214  auto typeLoc = getToken().getLoc();
215  auto elementType = parseType();
216  if (!elementType)
217  return nullptr;
218 
219  // Check that memref is formed from allowed types.
220  if (!elementType.isIntOrIndexOrFloat() &&
221  !elementType.isa<VectorType, ComplexType>())
222  return emitError(typeLoc, "invalid memref element type"), nullptr;
223 
224  // Parse semi-affine-map-composition.
225  SmallVector<AffineMap, 2> affineMapComposition;
226  Optional<unsigned> memorySpace;
227  unsigned numDims = dimensions.size();
228 
229  auto parseElt = [&]() -> ParseResult {
230  // Check for the memory space.
231  if (getToken().is(Token::integer)) {
232  if (memorySpace)
233  return emitError("multiple memory spaces specified in memref type");
234  memorySpace = getToken().getUnsignedIntegerValue();
235  if (!memorySpace.hasValue())
236  return emitError("invalid memory space in memref type");
237  consumeToken(Token::integer);
238  return success();
239  }
240  if (isUnranked)
241  return emitError("cannot have affine map for unranked memref type");
242  if (memorySpace)
243  return emitError("expected memory space to be last in memref type");
244 
245  AffineMap map;
246  llvm::SMLoc mapLoc = getToken().getLoc();
247  if (getToken().is(Token::kw_offset)) {
248  int64_t offset;
249  SmallVector<int64_t, 4> strides;
250  if (failed(parseStridedLayout(offset, strides)))
251  return failure();
252  // Construct strided affine map.
253  map = makeStridedLinearLayoutMap(strides, offset, state.context);
254  } else {
255  // Parse an affine map attribute.
256  auto affineMap = parseAttribute();
257  if (!affineMap)
258  return failure();
259  auto affineMapAttr = affineMap.dyn_cast<AffineMapAttr>();
260  if (!affineMapAttr)
261  return emitError("expected affine map in memref type");
262  map = affineMapAttr.getValue();
263  }
264 
265  if (map.getNumDims() != numDims) {
266  size_t i = affineMapComposition.size();
267  return emitError(mapLoc, "memref affine map dimension mismatch between ")
268  << (i == 0 ? Twine("memref rank") : "affine map " + Twine(i))
269  << " and affine map" << i + 1 << ": " << numDims
270  << " != " << map.getNumDims();
271  }
272  numDims = map.getNumResults();
273  affineMapComposition.push_back(map);
274  return success();
275  };
276 
277  // Parse a list of mappings and address space if present.
278  if (!consumeIf(Token::greater)) {
279  // Parse comma separated list of affine maps, followed by memory space.
280  if (parseToken(Token::comma, "expected ',' or '>' in memref type") ||
281  parseCommaSeparatedListUntil(Token::greater, parseElt,
282  /*allowEmptyList=*/false)) {
283  return nullptr;
284  }
285  }
286 
287  if (isUnranked)
288  return UnrankedMemRefType::get(elementType, memorySpace.getValueOr(0));
289 
290  return MemRefType::get(dimensions, elementType, affineMapComposition,
291  memorySpace.getValueOr(0));
292 }

294 /// Parse any type except the function type.
295 ///
296 /// non-function-type ::= integer-type
297 /// | index-type
298 /// | float-type
299 /// | extended-type
300 /// | vector-type
301 /// | tensor-type
302 /// | memref-type
303 /// | complex-type
304 /// | tuple-type
305 /// | none-type
306 ///
307 /// index-type ::= `index`
308 /// float-type ::= `f16` | `bf16` | `f32` | `f64`
309 /// none-type ::= `none`
310 ///
311 Type Parser::parseNonFunctionType() {
312  switch (getToken().getKind()) {
313  default:
314  return (emitError("expected non-function type"), nullptr);
315  case Token::kw_memref:
316  return parseMemRefType();
317  case Token::kw_tensor:
318  return parseTensorType();
319  case Token::kw_complex:
320  return parseComplexType();
321  case Token::kw_tuple:
322  return parseTupleType();
323  case Token::kw_vector:
324  return parseVectorType();
325  // integer-type
326  case Token::inttype: {
327  auto width = getToken().getIntTypeBitwidth();
328  if (!width.hasValue())
329  return (emitError("invalid integer width"), nullptr);
330  if (width.getValue() > IntegerType::kMaxWidth) {
331  emitError(getToken().getLoc(), "integer bitwidth is limited to ")
332  << IntegerType::kMaxWidth << " bits";
333  return nullptr;
334  }
335 
336  IntegerType::SignednessSemantics signSemantics = IntegerType::Signless;
337  if (Optional<bool> signedness = getToken().getIntTypeSignedness())
338  signSemantics = *signedness ? IntegerType::Signed : IntegerType::Unsigned;
339 
340  auto loc = getEncodedSourceLocation(getToken().getLoc());
341  consumeToken(Token::inttype);
342  return IntegerType::getChecked(width.getValue(), signSemantics, loc);
343  }
344 
345  // float-type
346  case Token::kw_bf16:
347  consumeToken(Token::kw_bf16);
348  return builder.getBF16Type();
349  case Token::kw_f16:
350  consumeToken(Token::kw_f16);
351  return builder.getF16Type();
352  case Token::kw_f32:
353  consumeToken(Token::kw_f32);
354  return builder.getF32Type();
355  case Token::kw_f64:
356  consumeToken(Token::kw_f64);
357  return builder.getF64Type();
358 
359  // index-type
360  case Token::kw_index:
361  consumeToken(Token::kw_index);
362  return builder.getIndexType();
363 
364  // none-type
365  case Token::kw_none:
366  consumeToken(Token::kw_none);
367  return builder.getNoneType();
368 
369  // extended type
370  case Token::exclamation_identifier:
371  return parseExtendedType();
372  }
373 }

375 /// Parse a tensor type.
376 ///
377 /// tensor-type ::= `tensor` `<` dimension-list type `>`
378 /// dimension-list ::= dimension-list-ranked | `*x`
379 ///
380 Type Parser::parseTensorType() {
381  consumeToken(Token::kw_tensor);
382 
383  if (parseToken(Token::less, "expected '<' in tensor type"))
384  return nullptr;
385 
386  bool isUnranked;
387  SmallVector<int64_t, 4> dimensions;
388 
389  if (consumeIf(Token::star)) {
390  // This is an unranked tensor type.
391  isUnranked = true;
392 
393  if (parseXInDimensionList())
394  return nullptr;
395 
396  } else {
397  isUnranked = false;
398  if (parseDimensionListRanked(dimensions))
399  return nullptr;
400  }
401 
402  // Parse the element type.
403  auto elementTypeLoc = getToken().getLoc();
404  auto elementType = parseType();
405  if (!elementType || parseToken(Token::greater, "expected '>' in tensor type"))
406  return nullptr;

408  return emitError(elementTypeLoc, "invalid tensor element type"), nullptr;
409 
410  if (isUnranked)

412  return RankedTensorType::get(dimensions, elementType);

474 /// Parse a dimension list of a tensor or memref type. This populates the
475 /// dimension list, using -1 for the `?` dimensions if `allowDynamic` is set and
476 /// errors out on `?` otherwise.
477 ///
478 /// dimension-list-ranked ::= (dimension `x`)*
479 /// dimension ::= `?` | decimal-literal
480 ///
481 /// When `allowDynamic` is not set, this is used to parse:
482 ///
483 /// static-dimension-list ::= (decimal-literal `x`)*
484 ParseResult
485 Parser::parseDimensionListRanked(SmallVectorImpl<int64_t> &dimensions,
486  bool allowDynamic) {
487  while (getToken().isAny(Token::integer, Token::question)) {
488  if (consumeIf(Token::question)) {
489  if (!allowDynamic)
490  return emitError("expected static shape");
491  dimensions.push_back(-1);
492  } else {
493  // Hexadecimal integer literals (starting with `0x`) are not allowed in
494  // aggregate type declarations. Therefore, `0xf32` should be processed as
495  // a sequence of separate elements `0`, `x`, `f32`.
496  if (getTokenSpelling().size() > 1 && getTokenSpelling()[1] == 'x') {
497  // We can get here only if the token is an integer literal. Hexadecimal
498  // integer literals can only start with `0x` (`1x` wouldn't lex as a
499  // literal, just `1` would, at which point we don't get into this
500  // branch).
501  assert(getTokenSpelling()[0] == '0' && "invalid integer literal");
502  dimensions.push_back(0);
503  state.lex.resetPointer(getTokenSpelling().data() + 1);
504  consumeToken();
505  } else {
506  // Make sure this integer value is in bound and valid.
507  auto dimension = getToken().getUnsignedIntegerValue();
508  if (!dimension.hasValue())
509  return emitError("invalid dimension");
510  dimensions.push_back((int64_t)dimension.getValue());
511  consumeToken(Token::integer);
512  }
513  }
514 
515  // Make sure we have an 'x' or something like 'xbf32'.
516  if (parseXInDimensionList())
517  return failure();
518  }
519 
520  return success();
521 }

523 /// Parse an 'x' token in a dimension list, handling the case where the x is
524 /// juxtaposed with an element type, as in "xf32", leaving the "f32" as the next
525 /// token.
526 ParseResult Parser::parseXInDimensionList() {
527  if (getToken().isNot(Token::bare_identifier) || getTokenSpelling()[0] != 'x')
528  return emitError("expected 'x' in dimension list");
529 
530  // If we had a prefix of 'x', lex the next token immediately after the 'x'.
531  if (getTokenSpelling().size() != 1)
532  state.lex.resetPointer(getTokenSpelling().data() + 1);
533 
534  // Consume the 'x'.
535  consumeToken(Token::bare_identifier);
536 
537  return success();
538 }

540 // Parse a comma-separated list of dimensions, possibly empty:
541 // stride-list ::= `[` (dimension (`,` dimension)*)? `]`
542 ParseResult Parser::parseStrideList(SmallVectorImpl<int64_t> &dimensions) {
543  if (!consumeIf(Token::l_square))
544  return failure();
545  // Empty list early exit.
546  if (consumeIf(Token::r_square))
547  return success();
548  while (true) {
549  if (consumeIf(Token::question)) {
550  dimensions.push_back(MemRefType::getDynamicStrideOrOffset());
551  } else {
552  // This must be an integer value.
553  int64_t val;
554  if (getToken().getSpelling().getAsInteger(10, val))
555  return emitError("invalid integer value: ") << getToken().getSpelling();
556  // Make sure it is not the one value for `?`.
557  if (ShapedType::isDynamic(val))
558  return emitError("invalid integer value: ")
559  << getToken().getSpelling()
560  << ", use `?` to specify a dynamic dimension";
561  dimensions.push_back(val);
562  consumeToken(Token::integer);
563  }
564  if (!consumeIf(Token::comma))
565  break;
566  }
567  if (!consumeIf(Token::r_square))
568  return failure();
569  return success();
570 }
