57 enum class OperationProperty {
58  /// This bit is set for an operation if it is a commutative
59  /// operation: that is an operator where order of operands does not
60  /// change the result of the operation. For example, in a binary
61  /// commutative operation, "a op b" and "b op a" produce the same
62  /// results.
63  Commutative = 0x1,
64 
65  /// This bit is set for an operation if it is a terminator: that means
66  /// an operation at the end of a block.
67  Terminator = 0x2,
68 
69  /// This bit is set for operations that are completely isolated from above.
70  /// This is used for operations whose regions are explicit capture only, i.e.
71  /// they are never allowed to implicitly reference values defined above the
72  /// parent operation.
73  IsolatedFromAbove = 0x4,
74 };

76 /// This is a "type erased" representation of a registered operation. This
77 /// should only be used by things like the AsmPrinter and other things that need
78 /// to be parameterized by generic operation hooks. Most user code should use
79 /// the concrete operation types.
80 class AbstractOperation {
81 public:
82  using OperationProperties = uint32_t;
83 
84  /// This is the name of the operation.
85  const StringRef name;
86 
87  /// This is the dialect that this operation belongs to.
88  Dialect &dialect;
89 
90  /// The unique identifier of the derived Op class.
91  TypeID typeID;
92 
93  /// Use the specified object to parse this ops custom assembly format.
94  ParseResult (&parseAssembly)(OpAsmParser &parser, OperationState &result);
95 
96  /// This hook implements the AsmPrinter for this operation.
97  void (&printAssembly)(Operation *op, OpAsmPrinter &p);
98 
99  /// This hook implements the verifier for this operation. It should emits an
100  /// error message and returns failure if a problem is detected, or returns
101  /// success if everything is ok.
102  LogicalResult (&verifyInvariants)(Operation *op);
103 
104  /// This hook implements a generalized folder for this operation. Operations
105  /// can implement this to provide simplifications rules that are applied by
106  /// the Builder::createOrFold API and the canonicalization pass.
107  ///
108  /// This is an intentionally limited interface - implementations of this hook
109  /// can only perform the following changes to the operation:
110  ///
111  /// 1. They can leave the operation alone and without changing the IR, and
112  /// return failure.
113  /// 2. They can mutate the operation in place, without changing anything else
114  /// in the IR. In this case, return success.
115  /// 3. They can return a list of existing values that can be used instead of
116  /// the operation. In this case, fill in the results list and return
117  /// success. The caller will remove the operation and use those results
118  /// instead.
119  ///
120  /// This allows expression of some simple in-place canonicalizations (e.g.
121  /// "x+0 -> x", "min(x,y,x,z) -> min(x,y,z)", "x+y-x -> y", etc), as well as
122  /// generalized constant folding.
123  LogicalResult (&foldHook)(Operation *op, ArrayRef<Attribute> operands,
124  SmallVectorImpl<OpFoldResult> &results);
125 
126  /// This hook returns any canonicalization pattern rewrites that the operation
127  /// supports, for use by the canonicalization pass.
128  void (&getCanonicalizationPatterns)(OwningRewritePatternList &results,
129  MLIRContext *context);
130 
131  /// Returns whether the operation has a particular property.
132  bool hasProperty(OperationProperty property) const {
133  return opProperties & static_cast<OperationProperties>(property);
134  }
135 
136  /// Returns an instance of the concept object for the given interface if it
137  /// was registered to this operation, null otherwise. This should not be used
138  /// directly.
139  template <typename T> typename T::Concept *getInterface() const {
140  return interfaceMap.lookup<T>();
141  }
142 
143  /// Returns if the operation has a particular trait.
144  template <template <typename T> class Trait> bool hasTrait() const {
145  return hasRawTrait(TypeID::get<Trait>());
146  }
147 
148  /// Look up the specified operation in the specified MLIRContext and return a
149  /// pointer to it if present. Otherwise, return a null pointer.
150  static const AbstractOperation *lookup(StringRef opName,
151  MLIRContext *context);
152 
153  /// This constructor is used by Dialect objects when they register the list of
154  /// operations they contain.
155  template <typename T> static AbstractOperation get(Dialect &dialect) {
156  return AbstractOperation(
157  T::getOperationName(), dialect, T::getOperationProperties(),
158  TypeID::get<T>(), T::parseAssembly, T::printAssembly,
159  T::verifyInvariants, T::foldHook, T::getCanonicalizationPatterns,
160  T::getInterfaceMap(), T::hasTrait);
161  }
162 
163 private:
164  AbstractOperation(
165  StringRef name, Dialect &dialect, OperationProperties opProperties,
166  TypeID typeID,
167  ParseResult (&parseAssembly)(OpAsmParser &parser, OperationState &result),
168  void (&printAssembly)(Operation *op, OpAsmPrinter &p),
169  LogicalResult (&verifyInvariants)(Operation *op),
170  LogicalResult (&foldHook)(Operation *op, ArrayRef<Attribute> operands,
171  SmallVectorImpl<OpFoldResult> &results),
172  void (&getCanonicalizationPatterns)(OwningRewritePatternList &results,
173  MLIRContext *context),
174  detail::InterfaceMap &&interfaceMap, bool (&hasTrait)(TypeID traitID))
175  : name(name), dialect(dialect), typeID(typeID),
176  parseAssembly(parseAssembly), printAssembly(printAssembly),
177  verifyInvariants(verifyInvariants), foldHook(foldHook),
178  getCanonicalizationPatterns(getCanonicalizationPatterns),
179  opProperties(opProperties), interfaceMap(std::move(interfaceMap)),
180  hasRawTrait(hasTrait) {}
181 
182  /// The properties of the operation.
183  const OperationProperties opProperties;
184 
185  /// A map of interfaces that were registered to this operation.
186  detail::InterfaceMap interfaceMap;
187 
188  /// This hook returns if the operation contains the trait corresponding
189  /// to the given TypeID.
190  bool (&hasRawTrait)(TypeID traitID);
191 };

348 /// This represents an operation in an abstracted form, suitable for use with
349 /// the builder APIs. This object is a large and heavy weight object meant to
350 /// be used as a temporary object on the stack. It is generally unwise to put
351 /// this in a collection.
352 struct OperationState {
353  Location location;
354  OperationName name;
355  SmallVector<Value, 4> operands;
356  /// Types of the results of this operation.
357  SmallVector<Type, 4> types;
358  NamedAttrList attributes;
359  /// Successors of this operation and their respective operands.
360  SmallVector<Block *, 1> successors;
361  /// Regions that the op will hold.
362  SmallVector<std::unique_ptr<Region>, 1> regions;
363 
364 public:
365  OperationState(Location location, StringRef name);
366 
367  OperationState(Location location, OperationName name);
368 
369  OperationState(Location location, StringRef name, ValueRange operands,
370  ArrayRef<Type> types, ArrayRef<NamedAttribute> attributes,
371  ArrayRef<Block *> successors = {},
372  MutableArrayRef<std::unique_ptr<Region>> regions = {});
373 
374  void addOperands(ValueRange newOperands);
375 
376  void addTypes(ArrayRef<Type> newTypes) {
377  types.append(newTypes.begin(), newTypes.end());
378  }
379  template <typename RangeT>
380  std::enable_if_t<!std::is_convertible<RangeT, ArrayRef<Type>>::value>
381  addTypes(RangeT &&newTypes) {
382  types.append(newTypes.begin(), newTypes.end());
383  }
384 
385  /// Add an attribute with the specified name.
386  void addAttribute(StringRef name, Attribute attr) {
387  addAttribute(Identifier::get(name, getContext()), attr);
388  }
389 
390  /// Add an attribute with the specified name.
391  void addAttribute(Identifier name, Attribute attr) {
392  attributes.append(name, attr);
393  }
394 
395  /// Add an array of named attributes.
396  void addAttributes(ArrayRef<NamedAttribute> newAttributes) {
397  attributes.append(newAttributes);
398  }
399 
400  /// Add an array of successors.
401  void addSuccessors(ArrayRef<Block *> newSuccessors) {
402  successors.append(newSuccessors.begin(), newSuccessors.end());
403  }
404  void addSuccessors(Block *successor) { successors.push_back(successor); }
405  void addSuccessors(SuccessorRange newSuccessors);
406 
407  /// Create a region that should be attached to the operation. These regions
408  /// can be filled in immediately without waiting for Operation to be
409  /// created. When it is, the region bodies will be transferred.
410  Region *addRegion();
411 
412  /// Take a region that should be attached to the Operation. The body of the
413  /// region will be transferred when the Operation is constructed. If the
414  /// region is null, a new empty region will be attached to the Operation.
415  void addRegion(std::unique_ptr<Region> &&region);
416 
417  /// Get the context held by this operation state.
418  MLIRContext *getContext() const { return location->getContext(); }
419 };

426 /// This class contains the information for a trailing operand storage.
427 struct TrailingOperandStorage final
428  : public llvm::TrailingObjects<TrailingOperandStorage, OpOperand> {
429  ~TrailingOperandStorage() {
430  for (auto &operand : getOperands())
431  operand.~OpOperand();
432  }
433 
434  /// Return the operands held by this storage.
435  MutableArrayRef<OpOperand> getOperands() {
436  return {getTrailingObjects<OpOperand>(), numOperands};
437  }
438 
439  /// The number of operands within the storage.
440  unsigned numOperands;
441  /// The total capacity number of operands that the storage can hold.
442  unsigned capacity : 31;
443  /// We reserve a range of bits for use by the operand storage.
444  unsigned reserved : 1;
445 };

447 /// This class handles the management of operation operands. Operands are
448 /// stored either in a trailing array, or a dynamically resizable vector.
449 class OperandStorage final
450  : private llvm::TrailingObjects<OperandStorage, OpOperand> {
451 public:
452  OperandStorage(Operation *owner, ValueRange values);
453  ~OperandStorage();
454 
455  /// Replace the operands contained in the storage with the ones provided in
456  /// 'values'.
457  void setOperands(Operation *owner, ValueRange values);
458 
459  /// Replace the operands beginning at 'start' and ending at 'start' + 'length'
460  /// with the ones provided in 'operands'. 'operands' may be smaller or larger
461  /// than the range pointed to by 'start'+'length'.
462  void setOperands(Operation *owner, unsigned start, unsigned length,
463  ValueRange operands);
464 
465  /// Erase the operands held by the storage within the given range.
466  void eraseOperands(unsigned start, unsigned length);
467 
468  /// Get the operation operands held by the storage.
469  MutableArrayRef<OpOperand> getOperands() {
470  return getStorage().getOperands();
471  }
472 
473  /// Return the number of operands held in the storage.
474  unsigned size() { return getStorage().numOperands; }
475 
476  /// Returns the additional size necessary for allocating this object.
477  static size_t additionalAllocSize(unsigned numOperands) {
478  return additionalSizeToAlloc<OpOperand>(numOperands);
479  }
480 
481 private:
482  enum : uint64_t {
483  /// The bit used to mark the storage as dynamic.
484  DynamicStorageBit = 1ull << 63ull
485  };
486 
487  /// Resize the storage to the given size. Returns the array containing the new
488  /// operands.
489  MutableArrayRef<OpOperand> resize(Operation *owner, unsigned newSize);
490 
491  /// Returns the current internal storage instance.
492  TrailingOperandStorage &getStorage() {
493  return LLVM_UNLIKELY(isDynamicStorage()) ? getDynamicStorage()
494  : getInlineStorage();
495  }
496 
497  /// Returns the storage container if the storage is inline.
498  TrailingOperandStorage &getInlineStorage() {
499  assert(!isDynamicStorage() && "expected storage to be inline");
500  static_assert(sizeof(TrailingOperandStorage) == sizeof(uint64_t),
501  "inline storage representation must match the opaque "
502  "representation");
503  return inlineStorage;
504  }
505 
506  /// Returns the storage container if this storage is dynamic.
507  TrailingOperandStorage &getDynamicStorage() {
508  assert(isDynamicStorage() && "expected dynamic storage");
509  uint64_t maskedRepresentation = representation & ~DynamicStorageBit;
510  return *reinterpret_cast<TrailingOperandStorage *>(maskedRepresentation);
511  }
512 
513  /// Returns true if the storage is currently dynamic.
514  bool isDynamicStorage() const { return representation & DynamicStorageBit; }
515 
516  /// The current representation of the storage. This is either a
517  /// InlineOperandStorage, or a pointer to a InlineOperandStorage.
518  union {
519  TrailingOperandStorage inlineStorage;
520  uint64_t representation;
521  };
522 
523  /// This stuff is used by the TrailingObjects template.
524  friend llvm::TrailingObjects<OperandStorage, OpOperand>;
525 };
526 } // end namespace detail
