32 LogicalResult BroadcastOpConverter::matchAndRewrite(
33  BroadcastOp op, ArrayRef<Value> operands,
34  ConversionPatternRewriter &rewriter) const {
35  // For now, this lowering is only defined on `tensor<?xindex>` operands, not
36  // on shapes.
37  if (op.getType().isa<ShapeType>())
38  return failure();
39 
40  assert(!op.lhs().getType().isa<ShapeType>() &&
41  !op.rhs().getType().isa<ShapeType>());
42  auto loc = op.getLoc();
43  BroadcastOp::Adaptor transformed(operands);
44  Value zero = rewriter.create<ConstantIndexOp>(loc, 0);
45  Value one = rewriter.create<ConstantIndexOp>(loc, 1);
46 
47  // Find smaller and greater rank and extent tensor.
48  Value lhsRank = rewriter.create<DimOp>(loc, transformed.lhs(), zero);
49  Value rhsRank = rewriter.create<DimOp>(loc, transformed.rhs(), zero);
50  Value lhsSmaller =
51  rewriter.create<CmpIOp>(loc, CmpIPredicate::ule, lhsRank, rhsRank);
52  Type indexTy = rewriter.getIndexType();
53  Type extentTensorTy = op.getType();
54  auto ifOp = rewriter.create<IfOp>(
55  loc, TypeRange{indexTy, extentTensorTy, indexTy, extentTensorTy},
56  lhsSmaller,
57  [&](OpBuilder &b, Location loc) {
58  b.create<scf::YieldOp>(loc, ValueRange{lhsRank, transformed.lhs(),
59  rhsRank, transformed.rhs()});
60  },
61  [&](OpBuilder &b, Location loc) {
62  b.create<scf::YieldOp>(loc, ValueRange{rhsRank, transformed.rhs(),
63  lhsRank, transformed.lhs()});
64  });
65  Value smallerRank = ifOp.getResult(0);
66  Value smallerOperand = ifOp.getResult(1);
67  Value greaterRank = ifOp.getResult(2);
68  Value greaterOperand = ifOp.getResult(3);
69 
70  // Allocate stack memory for the broadcasted extent tensor.
71  Type memTy = MemRefType::get({ShapedType::kDynamicSize}, indexTy);
72  Value mem = rewriter.create<AllocaOp>(loc, memTy, ValueRange{greaterRank});
73 
74  // Copy extents from greater operand that are not challenged.
75  Value rankDiff =
76  rewriter.create<SubIOp>(loc, indexTy, greaterRank, smallerRank);
77  rewriter.create<ForOp>(loc, zero, rankDiff, one, llvm::None,
78  [&](OpBuilder &b, Location loc, Value iv, ValueRange) {
79  Value extent = b.create<ExtractElementOp>(
80  loc, greaterOperand, ValueRange{iv});
81  b.create<StoreOp>(loc, extent, mem, ValueRange{iv});
82  b.create<scf::YieldOp>(loc);
83  });
84 
85  // Determine remaining broadcasted extents.
86  rewriter.create<ForOp>(
87  loc, rankDiff, greaterRank, one, llvm::None,
88  [&](OpBuilder &b, Location loc, Value iv, ValueRange) {
89  Value greaterOperandExtent =
90  b.create<ExtractElementOp>(loc, greaterOperand, ValueRange{iv});
91  Value greaterOperandExtentIsOne =
92  b.create<CmpIOp>(loc, CmpIPredicate::eq, greaterOperandExtent, one);
93  auto ifOp = b.create<IfOp>(
94  loc, TypeRange{indexTy}, greaterOperandExtentIsOne,
95  [&](OpBuilder &b, Location loc) {
96  Value ivShifted = b.create<SubIOp>(loc, indexTy, iv, rankDiff);
97  Value smallerOperandExtent = b.create<ExtractElementOp>(
98  loc, smallerOperand, ValueRange{ivShifted});
99  b.create<scf::YieldOp>(loc, smallerOperandExtent);
100  },
101  [&](OpBuilder &b, Location loc) {
102  b.create<scf::YieldOp>(loc, greaterOperandExtent);
103  });
104  Value extent = ifOp.getResult(0);
105  b.create<StoreOp>(loc, extent, mem, ValueRange{iv});
106  b.create<scf::YieldOp>(loc);
107  });
108 
109  // Load broadcasted shape as an extent tensor.
110  rewriter.replaceOpWithNewOp<TensorLoadOp>(op, mem);
111  return success();
112 }

115 /// Converts `shape.shape_eq` to an `scf.for` loop. For now, the lowering is
116 /// only defined on `tensor<?xindex>` operands. The test for equality first
117 /// compares their size and, if equal, checks every extent for equality.
118 ///
119 /// Example:
120 ///
121 /// %result = shape.shape_eq %a, %b : tensor<?xindex>, tensor<?xindex>
122 ///
123 /// becomes
124 ///
125 /// %c0 = constant 0 : index
126 /// %0 = dim %arg0, %c0 : tensor<?xindex>
127 /// %1 = dim %arg1, %c0 : tensor<?xindex>
128 /// %2 = cmpi "eq", %0, %1 : index
129 /// %result = scf.if %2 -> (i1) {
130 /// %c1 = constant 1 : index
131 /// %true = constant true
132 /// %4 = scf.for %arg2 = %c0 to %0 step %c1 iter_args(%arg3 = %true) -> (i1) {
133 /// %5 = extract_element %arg0[%arg2] : tensor<?xindex>
134 /// %6 = extract_element %arg1[%arg2] : tensor<?xindex>
135 /// %7 = cmpi "eq", %5, %6 : index
136 /// %8 = and %arg3, %7 : i1
137 /// scf.yield %8 : i1
138 /// }
139 /// scf.yield %4 : i1
140 /// } else {
141 /// %false = constant false
142 /// scf.yield %false : i1
143 /// }
144 ///
145 struct ShapeEqOpConverter : public OpConversionPattern<ShapeEqOp> {
146  using OpConversionPattern<ShapeEqOp>::OpConversionPattern;
147 
148  LogicalResult
149  matchAndRewrite(ShapeEqOp op, ArrayRef<Value> operands,
150  ConversionPatternRewriter &rewriter) const override;
151 };
152 } // namespace

154 LogicalResult
155 ShapeEqOpConverter::matchAndRewrite(ShapeEqOp op, ArrayRef<Value> operands,
156  ConversionPatternRewriter &rewriter) const {
157  // For now, this lowering is only defined on `tensor<?xindex>` operands, not
158  // on shapes.
159  if (op.lhs().getType().isa<ShapeType>() ||
160  op.rhs().getType().isa<ShapeType>()) {
161  return failure();
162  }
163 
164  ShapeEqOp::Adaptor transformed(operands);
165  auto loc = op.getLoc();
166  Type indexTy = rewriter.getIndexType();
167  Value zero = rewriter.create<ConstantIndexOp>(loc, 0);
168  Value lhsRank = rewriter.create<DimOp>(loc, indexTy, transformed.lhs(), zero);
169  Value rhsRank = rewriter.create<DimOp>(loc, indexTy, transformed.rhs(), zero);
170  Value eqRank =
171  rewriter.create<CmpIOp>(loc, CmpIPredicate::eq, lhsRank, rhsRank);
172  Type i1Ty = rewriter.getI1Type();
173  rewriter.replaceOpWithNewOp<IfOp>(
174  op, i1Ty, eqRank,
175  [&](OpBuilder &b, Location loc) {
176  Value one = b.create<ConstantIndexOp>(loc, 1);
177  Value init = b.create<ConstantOp>(loc, i1Ty, b.getBoolAttr(true));
178  auto loop = b.create<scf::ForOp>(
179  loc, zero, lhsRank, one, ValueRange{init},
180  [&](OpBuilder &b, Location nestedLoc, Value iv, ValueRange args) {
181  Value conj = args[0];
182  Value lhsExtent =
183  b.create<ExtractElementOp>(loc, transformed.lhs(), iv);
184  Value rhsExtent =
185  b.create<ExtractElementOp>(loc, transformed.rhs(), iv);
186  Value eqExtent = b.create<CmpIOp>(loc, CmpIPredicate::eq,
187  lhsExtent, rhsExtent);
188  Value conjNext = b.create<AndOp>(loc, conj, eqExtent);
189  b.create<scf::YieldOp>(loc, ValueRange({conjNext}));
190  });
191  b.create<scf::YieldOp>(loc, loop.getResults());
192  },
193  [&](OpBuilder &b, Location loc) {
194  Value result = b.create<ConstantOp>(loc, i1Ty, b.getBoolAttr(false));
195  b.create<scf::YieldOp>(loc, result);
196  });
197  return success();
198 }

212 LogicalResult
213 ReduceOpConverter::matchAndRewrite(shape::ReduceOp op, ArrayRef<Value> operands,
214  ConversionPatternRewriter &rewriter) const {
215  // For now, this lowering is only defined on `tensor<?xindex>` operands.
216  if (op.shape().getType().isa<ShapeType>())
217  return failure();
218 
219  auto loc = op.getLoc();
220  shape::ReduceOp::Adaptor transformed(operands);
221 
222  Value zero = rewriter.create<ConstantIndexOp>(loc, 0);
223  Value one = rewriter.create<ConstantIndexOp>(loc, 1);
224  Type indexTy = rewriter.getIndexType();
225  Value rank = rewriter.create<DimOp>(loc, indexTy, transformed.shape(), zero);
226 
227  auto loop = rewriter.create<scf::ForOp>(
228  loc, zero, rank, one, op.initVals(),
229  [&](OpBuilder &b, Location loc, Value iv, ValueRange args) {
230  Value extent = b.create<ExtractElementOp>(loc, transformed.shape(), iv);
231 
232  SmallVector<Value, 2> mappedValues{iv, extent};
233  mappedValues.append(args.begin(), args.end());
234 
235  BlockAndValueMapping mapping;
236  Block *reduceBody = op.getBody();
237  mapping.map(reduceBody->getArguments(), mappedValues);
238  for (auto &nested : reduceBody->without_terminator())
239  b.clone(nested, mapping);
240 
241  SmallVector<Value, 2> mappedResults;
242  for (auto result : reduceBody->getTerminator()->getOperands())
243  mappedResults.push_back(mapping.lookup(result));
244  b.create<scf::YieldOp>(loc, mappedResults);
245  });
246 
247  rewriter.replaceOp(op, loop.getResults());
248  return success();
249 }

263 LogicalResult
264 ShapeOfOpConverter::matchAndRewrite(ShapeOfOp op, ArrayRef<Value> operands,
265  ConversionPatternRewriter &rewriter) const {
266  // For now, this lowering supports only error-free arguments.
267  if (op.getType().isa<ShapeType>())
268  return failure();
269 
270  // For ranked tensors `shape_of` lowers to `std` and the pattern can be
271  // found in the corresponding pass.
272  ShapeOfOp::Adaptor transformed(operands);
273  Value arg = transformed.arg();
274  Type argTy = arg.getType();
275  if (argTy.isa<RankedTensorType>())
276  return failure();
277 
278  // Allocate stack memory.
279  auto loc = op.getLoc();
280  Value rank = rewriter.create<mlir::RankOp>(loc, arg);
281  Type indexTy = rewriter.getIndexType();
282  Type memTy = MemRefType::get({ShapedType::kDynamicSize}, indexTy);
283  Value mem = rewriter.create<AllocaOp>(loc, memTy, ValueRange{rank});
284 
285  // Copy shape extents to stack-allocated memory.
286  Value zero = rewriter.create<ConstantIndexOp>(loc, 0);
287  Value one = rewriter.create<ConstantIndexOp>(loc, 1);
288  rewriter.create<scf::ForOp>(
289  loc, zero, rank, one, llvm::None,
290  [&](OpBuilder &b, Location loc, Value iv, ValueRange args) {
291  Value dim = rewriter.create<DimOp>(loc, arg, iv);
292  rewriter.create<StoreOp>(loc, dim, mem, ValueRange{iv});
293  rewriter.create<scf::YieldOp>(loc);
294  });
295 
296  // Load extents to tensor value.
297  rewriter.replaceOpWithNewOp<TensorLoadOp>(op.getOperation(), mem);
298  return success();
299 }

311  // Populate conversion patterns.
