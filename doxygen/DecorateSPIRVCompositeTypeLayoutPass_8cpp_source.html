26 class SPIRVGlobalVariableOpLayoutInfoDecoration
27  : public OpRewritePattern<spirv::GlobalVariableOp> {
28 public:
29  using OpRewritePattern<spirv::GlobalVariableOp>::OpRewritePattern;
30 
31  LogicalResult matchAndRewrite(spirv::GlobalVariableOp op,
32  PatternRewriter &rewriter) const override {
33  SmallVector<NamedAttribute, 4> globalVarAttrs;
34 
35  auto ptrType = op.type().cast<spirv::PointerType>();
36  auto structType = VulkanLayoutUtils::decorateType(
37  ptrType.getPointeeType().cast<spirv::StructType>());
38  auto decoratedType =
39  spirv::PointerType::get(structType, ptrType.getStorageClass());
40 
41  // Save all named attributes except "type" attribute.
42  for (const auto &attr : op.getAttrs()) {
43  if (attr.first == "type") {
44  continue;
45  }
46  globalVarAttrs.push_back(attr);
47  }
48 
49  rewriter.replaceOpWithNewOp<spirv::GlobalVariableOp>(
50  op, TypeAttr::get(decoratedType), globalVarAttrs);
51  return success();
52  }
53 };

55 class SPIRVAddressOfOpLayoutInfoDecoration
56  : public OpRewritePattern<spirv::AddressOfOp> {
57 public:
58  using OpRewritePattern<spirv::AddressOfOp>::OpRewritePattern;
59 
60  LogicalResult matchAndRewrite(spirv::AddressOfOp op,
61  PatternRewriter &rewriter) const override {
62  auto spirvModule = op.getParentOfType<spirv::ModuleOp>();
63  auto varName = op.variable();
64  auto varOp = spirvModule.lookupSymbol<spirv::GlobalVariableOp>(varName);
65 
66  rewriter.replaceOpWithNewOp<spirv::AddressOfOp>(
67  op, varOp.type(), rewriter.getSymbolRefAttr(varName));
68  return success();
69  }

87 void DecorateSPIRVCompositeTypeLayoutPass::runOnOperation() {
88  auto module = getOperation();
89  OwningRewritePatternList patterns;
90  populateSPIRVLayoutInfoPatterns(patterns, module.getContext());
91  ConversionTarget target(*(module.getContext()));
92  target.addLegalDialect<spirv::SPIRVDialect>();
93  target.addLegalOp<FuncOp>();
94  target.addDynamicallyLegalOp<spirv::GlobalVariableOp>(
95  [](spirv::GlobalVariableOp op) {
96  return VulkanLayoutUtils::isLegalType(op.type());
97  });
98 
99  // Change the type for the direct users.
100  target.addDynamicallyLegalOp<spirv::AddressOfOp>([](spirv::AddressOfOp op) {
101  return VulkanLayoutUtils::isLegalType(op.pointer().getType());
102  });
103 
104  // TODO: Change the type for the indirect users such as spv.Load, spv.Store,
105  // spv.FunctionCall and so on.
106 
107  for (auto spirvModule : module.getOps<spirv::ModuleOp>()) {
108  if (failed(applyFullConversion(spirvModule, target, patterns))) {
109  signalPassFailure();
110  }
111  }
112 }
