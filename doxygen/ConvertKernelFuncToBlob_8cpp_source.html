42 /// A pass converting tagged kernel modules to a blob with target instructions.
43 ///
44 /// If tagged as a kernel module, each contained function is translated to
45 /// user-specified IR. A user provided BlobGenerator then compiles the IR to
46 /// GPU binary code, which is then attached as an attribute to the function.
47 /// The function body is erased.
48 class GpuKernelToBlobPass
49  : public PassWrapper<GpuKernelToBlobPass, OperationPass<gpu::GPUModuleOp>> {
50 public:
51  GpuKernelToBlobPass(LoweringCallback loweringCallback,
52  BlobGenerator blobGenerator, StringRef triple,
53  StringRef targetChip, StringRef features,
54  StringRef gpuBinaryAnnotation)
55  : loweringCallback(loweringCallback), blobGenerator(blobGenerator),
56  triple(triple), targetChip(targetChip), features(features),
57  blobAnnotation(gpuBinaryAnnotation) {}
58 
59  void runOnOperation() override {
60  gpu::GPUModuleOp module = getOperation();
61 
62  // Lower the module to an LLVM IR module using a separate context to enable
63  // multi-threaded processing.
64  llvm::LLVMContext llvmContext;
65  std::unique_ptr<llvm::Module> llvmModule =
66  loweringCallback(module, llvmContext, "LLVMDialectModule");
67  if (!llvmModule)
68  return signalPassFailure();
69 
70  // Translate the llvm module to a target blob and attach the result as
71  // attribute to the module.
72  if (auto blobAttr = translateGPUModuleToBinaryAnnotation(
73  *llvmModule, module.getLoc(), module.getName()))
74  module.setAttr(blobAnnotation, blobAttr);
75  else
76  signalPassFailure();
77  }
78 
79 private:
80  std::string translateModuleToISA(llvm::Module &module,
81  llvm::TargetMachine &targetMachine);
82 
83  /// Converts llvmModule to a blob with target instructions using the
84  /// user-provided generator. Location is used for error reporting and name is
85  /// forwarded to the blob generator to use in its logging mechanisms.
86  OwnedBlob convertModuleToBlob(llvm::Module &llvmModule, Location loc,
87  StringRef name);
88 
89  /// Translates llvmModule to a blob with target instructions and returns the
90  /// result as attribute.
91  StringAttr translateGPUModuleToBinaryAnnotation(llvm::Module &llvmModule,
92  Location loc, StringRef name);
93 
94  LoweringCallback loweringCallback;
95  BlobGenerator blobGenerator;
96  llvm::Triple triple;
97  StringRef targetChip;
98  StringRef features;
99  StringRef blobAnnotation;
100 };

104 std::string
105 GpuKernelToBlobPass::translateModuleToISA(llvm::Module &module,
106  llvm::TargetMachine &targetMachine) {
107  std::string targetISA;
108  {
109  llvm::raw_string_ostream stream(targetISA);
110  llvm::buffer_ostream pstream(stream);
111  llvm::legacy::PassManager codegenPasses;
112  targetMachine.addPassesToEmitFile(codegenPasses, pstream, nullptr,
113  llvm::CGFT_AssemblyFile);
114  codegenPasses.run(module);
115  }
116 
117  return targetISA;

120 OwnedBlob GpuKernelToBlobPass::convertModuleToBlob(llvm::Module &llvmModule,
121  Location loc,
122  StringRef name) {
123  std::unique_ptr<llvm::TargetMachine> targetMachine;
124  {
125  std::string error;
126  const llvm::Target *target =
127  llvm::TargetRegistry::lookupTarget("", triple, error);
128  if (target == nullptr) {
129  emitError(loc, "cannot initialize target triple");
130  return {};
131  }
132  targetMachine.reset(target->createTargetMachine(triple.str(), targetChip,
133  features, {}, {}));
134  }
135 
136  llvmModule.setDataLayout(targetMachine->createDataLayout());
137 
138  auto targetISA = translateModuleToISA(llvmModule, *targetMachine);
139 
140  return blobGenerator(targetISA, loc, name);
141 }
