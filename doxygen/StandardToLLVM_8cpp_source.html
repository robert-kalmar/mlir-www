44 static LLVM::LLVMType unwrap(Type type) {
45  if (!type)

52  return wrappedLLVMType;

55 /// Callback to convert function argument types. It converts a MemRef function
56 /// argument to a list of non-aggregate types containing descriptor
57 /// information, and an UnrankedmemRef function argument to a list containing
58 /// the rank and a pointer to a descriptor struct.
59 LogicalResult mlir::structFuncArgTypeConverter(LLVMTypeConverter &converter,
60  Type type,
61  SmallVectorImpl<Type> &result) {
62  if (auto memref = type.dyn_cast<MemRefType>()) {
63  auto converted = converter.convertMemRefSignature(memref);
64  if (converted.empty())
65  return failure();
66  result.append(converted.begin(), converted.end());
67  return success();
68  }
69  if (type.isa<UnrankedMemRefType>()) {
70  auto converted = converter.convertUnrankedMemRefSignature();
71  if (converted.empty())
72  return failure();
73  result.append(converted.begin(), converted.end());
74  return success();
75  }
76  auto converted = converter.convertType(type);
77  if (!converted)
78  return failure();
79  result.push_back(converted);
80  return success();
81 }

248 /// In signatures, MemRef descriptors are expanded into lists of non-aggregate
249 /// values.
250 SmallVector<Type, 5>
251 LLVMTypeConverter::convertMemRefSignature(MemRefType type) {
252  SmallVector<Type, 5> results;
253  assert(isStrided(type) &&
254  "Non-strided layout maps must have been normalized away");
255 
256  LLVM::LLVMType elementType = unwrap(convertType(type.getElementType()));
257  if (!elementType)
258  return {};
259  auto indexTy = getIndexType();
260 
261  results.insert(results.begin(), 2,
262  elementType.getPointerTo(type.getMemorySpace()));
263  results.push_back(indexTy);
264  auto rank = type.getRank();
265  results.insert(results.end(), 2 * rank, indexTy);
266  return results;
267 }

275 // Function types are converted to LLVM Function types by recursively converting
276 // argument and result types. If MLIR Function has zero results, the LLVM
277 // Function has one VoidType result. If MLIR Function has more than one result,
278 // they are into an LLVM StructType in their order of appearance.
279 LLVM::LLVMType LLVMTypeConverter::convertFunctionSignature(
280  FunctionType type, bool isVariadic,
281  LLVMTypeConverter::SignatureConversion &result) {
282  // Select the argument converter depending on the calling convetion.
283  auto funcArgConverter = options.useBarePtrCallConv
284  ? barePtrFuncArgTypeConverter
285  : structFuncArgTypeConverter;
286  // Convert argument types one by one and check for errors.
287  for (auto &en : llvm::enumerate(type.getInputs())) {
288  Type type = en.value();
289  SmallVector<Type, 8> converted;
290  if (failed(funcArgConverter(*this, type, converted)))
291  return {};
292  result.addInputs(en.index(), converted);
293  }
294 
295  SmallVector<LLVM::LLVMType, 8> argTypes;
296  argTypes.reserve(llvm::size(result.getConvertedTypes()));
297  for (Type type : result.getConvertedTypes())
298  argTypes.push_back(unwrap(type));
299 
300  // If function does not return anything, create the void result type,
301  // if it returns on element, convert it, otherwise pack the result types into
302  // a struct.
303  LLVM::LLVMType resultType =
304  type.getNumResults() == 0
305  ? LLVM::LLVMType::getVoidTy(&getContext())
306  : unwrap(packFunctionResults(type.getResults()));
307  if (!resultType)
308  return {};
309  return LLVM::LLVMType::getFunctionTy(resultType, argTypes, isVariadic);
310 }

337 // Convert a MemRef to an LLVM type. The result is a MemRef descriptor which
338 // contains:
339 // 1. the pointer to the data buffer, followed by
340 // 2. a lowered `index`-type integer containing the distance between the
341 // beginning of the buffer and the first element to be accessed through the
342 // view, followed by
343 // 3. an array containing as many `index`-type integers as the rank of the
344 // MemRef: the array represents the size, in number of elements, of the memref
345 // along the given dimension. For constant MemRef dimensions, the
346 // corresponding size entry is a constant whose runtime value must match the
347 // static value, followed by
348 // 4. a second array containing as many `index`-type integers as the rank of
349 // the MemRef: the second array represents the "stride" (in tensor abstraction
350 // sense), i.e. the number of consecutive elements of the underlying buffer.
351 // TODO: add assertions for the static cases.
352 //
353 // template <typename Elem, size_t Rank>
354 // struct {
355 // Elem *allocatedPtr;
356 // Elem *alignedPtr;
357 // int64_t offset;
358 // int64_t sizes[Rank]; // omitted when rank == 0
359 // int64_t strides[Rank]; // omitted when rank == 0
360 // };
361 static constexpr unsigned kAllocatedPtrPosInMemRefDescriptor = 0;
362 static constexpr unsigned kAlignedPtrPosInMemRefDescriptor = 1;
363 static constexpr unsigned kOffsetPosInMemRefDescriptor = 2;
364 static constexpr unsigned kSizePosInMemRefDescriptor = 3;
365 static constexpr unsigned kStridePosInMemRefDescriptor = 4;
366 Type LLVMTypeConverter::convertMemRefType(MemRefType type) {
367  int64_t offset;
368  SmallVector<int64_t, 4> strides;
369  bool strideSuccess = succeeded(getStridesAndOffset(type, strides, offset));
370  assert(strideSuccess &&
371  "Non-strided layout maps must have been normalized away");
372  (void)strideSuccess;
373  LLVM::LLVMType elementType = unwrap(convertType(type.getElementType()));
374  if (!elementType)
375  return {};
376  auto ptrTy = elementType.getPointerTo(type.getMemorySpace());
377  auto indexTy = getIndexType();
378  auto rank = type.getRank();
379  if (rank > 0) {
380  auto arrayTy = LLVM::LLVMType::getArrayTy(indexTy, type.getRank());
381  return LLVM::LLVMType::getStructTy(ptrTy, ptrTy, indexTy, arrayTy, arrayTy);
382  }
383  return LLVM::LLVMType::getStructTy(ptrTy, ptrTy, indexTy);
384 }

402 // Convert an n-D vector type to an LLVM vector type via (n-1)-D array type when
403 // n > 1.
404 // For example, `vector<4 x f32>` converts to `!llvm.type<"<4 x float>">` and
405 // `vector<4 x 8 x 16 f32>` converts to `!llvm<"[4 x [8 x <16 x float>]]">`.
406 Type LLVMTypeConverter::convertVectorType(VectorType type) {
407  auto elementType = unwrap(convertType(type.getElementType()));
408  if (!elementType)
409  return {};
410  auto vectorType =
411  LLVM::LLVMType::getVectorTy(elementType, type.getShape().back());
412  auto shape = type.getShape();
413  for (int i = shape.size() - 2; i >= 0; --i)
414  vectorType = LLVM::LLVMType::getArrayTy(vectorType, shape[i]);
415  return vectorType;
416 }

497 /// Builds IR creating a MemRef descriptor that represents `type` and
498 /// populates it with static shape and stride information extracted from the
499 /// type.
500 MemRefDescriptor
501 MemRefDescriptor::fromStaticShape(OpBuilder &builder, Location loc,
502  LLVMTypeConverter &typeConverter,
503  MemRefType type, Value memory) {
504  assert(type.hasStaticShape() && "unexpected dynamic shape");
505 
506  // Extract all strides and offsets and verify they are static.
507  int64_t offset;
508  SmallVector<int64_t, 4> strides;
509  auto result = getStridesAndOffset(type, strides, offset);
510  (void)result;
511  assert(succeeded(result) && "unexpected failure in stride computation");
512  assert(offset != MemRefType::getDynamicStrideOrOffset() &&
513  "expected static offset");
514  assert(!llvm::is_contained(strides, MemRefType::getDynamicStrideOrOffset()) &&
515  "expected static strides");
516 
517  auto convertedType = typeConverter.convertType(type);
518  assert(convertedType && "unexpected failure in memref type conversion");
519 
520  auto descr = MemRefDescriptor::undef(builder, loc, convertedType);
521  descr.setAllocatedPtr(builder, loc, memory);
522  descr.setAlignedPtr(builder, loc, memory);
523  descr.setConstantOffset(builder, loc, offset);
524 
525  // Fill in sizes and strides
526  for (unsigned i = 0, e = type.getRank(); i != e; ++i) {
527  descr.setConstantSize(builder, loc, i, type.getDimSize(i));
528  descr.setConstantStride(builder, loc, i, strides[i]);
529  }
530  return descr;
531 }

656 /// composing that descriptor, in the following order:
657 /// - allocated pointer;
658 /// - aligned pointer;
659 /// - offset;

661 /// - <rank> shapes;

664  LLVMTypeConverter &converter, MemRefType type,
665  ValueRange values) {

667  auto d = MemRefDescriptor::undef(builder, loc, llvmType);

671  d.setOffset(builder, loc, values[kOffsetPosInMemRefDescriptor]);

675  d.setSize(builder, loc, i, values[kSizePosInMemRefDescriptor + i]);
676  d.setStride(builder, loc, i, values[kSizePosInMemRefDescriptor + rank + i]);

682 /// Builds IR extracting individual elements of a MemRef descriptor structure
683 /// and returning them as `results` list.
684 void MemRefDescriptor::unpack(OpBuilder &builder, Location loc, Value packed,
685  MemRefType type,
686  SmallVectorImpl<Value> &results) {

698 }

793 void UnrankedMemRefDescriptor::computeSizes(
794  OpBuilder &builder, Location loc, LLVMTypeConverter &typeConverter,
795  ArrayRef<UnrankedMemRefDescriptor> values, SmallVectorImpl<Value> &sizes) {
796  if (values.empty())
797  return;
798 
799  // Cache the index type.
800  LLVM::LLVMType indexType = typeConverter.getIndexType();
801 
802  // Initialize shared constants.
803  Value one = createIndexAttrConstant(builder, loc, indexType, 1);
804  Value two = createIndexAttrConstant(builder, loc, indexType, 2);
805  Value pointerSize = createIndexAttrConstant(
806  builder, loc, indexType, ceilDiv(typeConverter.getPointerBitwidth(), 8));
807  Value indexSize =
808  createIndexAttrConstant(builder, loc, indexType,
809  ceilDiv(typeConverter.getIndexTypeBitwidth(), 8));
810 
811  sizes.reserve(sizes.size() + values.size());
812  for (UnrankedMemRefDescriptor desc : values) {
813  // Emit IR computing the memory necessary to store the descriptor. This
814  // assumes the descriptor to be
815  // { type*, type*, index, index[rank], index[rank] }
816  // and densely packed, so the total size is
817  // 2 * sizeof(pointer) + (1 + 2 * rank) * sizeof(index).
818  // TODO: consider including the actual size (including eventual padding due
819  // to data layout) into the unranked descriptor.
820  Value doublePointerSize =
821  builder.create<LLVM::MulOp>(loc, indexType, two, pointerSize);
822 
823  // (1 + 2 * rank) * sizeof(index)
824  Value rank = desc.rank(builder, loc);
825  Value doubleRank = builder.create<LLVM::MulOp>(loc, indexType, two, rank);
826  Value doubleRankIncremented =
827  builder.create<LLVM::AddOp>(loc, indexType, doubleRank, one);
828  Value rankIndexSize = builder.create<LLVM::MulOp>(
829  loc, indexType, doubleRankIncremented, indexSize);
830 
831  // Total allocation size.
832  Value allocationSize = builder.create<LLVM::AddOp>(
833  loc, indexType, doublePointerSize, rankIndexSize);
834  sizes.push_back(allocationSize);
835  }
836 }

930 Value ConvertToLLVMPattern::getCumulativeSizeInBytes(
931  Location loc, Type elementType, ArrayRef<Value> sizes,
932  ConversionPatternRewriter &rewriter) const {
933  // Compute the total number of memref elements.
934  Value cumulativeSizeInBytes =
935  sizes.empty() ? createIndexConstant(rewriter, loc, 1) : sizes.front();
936  for (unsigned i = 1, e = sizes.size(); i < e; ++i)
937  cumulativeSizeInBytes = rewriter.create<LLVM::MulOp>(
938  loc, getIndexType(), ArrayRef<Value>{cumulativeSizeInBytes, sizes[i]});
939 
940  // Compute the size of an individual element. This emits the MLIR equivalent
941  // of the following sizeof(...) implementation in LLVM IR:
942  // %0 = getelementptr %elementType* null, %indexType 1
943  // %1 = ptrtoint %elementType* %0 to %indexType
944  // which is a common pattern of getting the size of a type in bytes.
945  auto convertedPtrType = typeConverter.convertType(elementType)
946  .cast<LLVM::LLVMType>()
947  .getPointerTo();
948  auto nullPtr = rewriter.create<LLVM::NullOp>(loc, convertedPtrType);
949  auto gep = rewriter.create<LLVM::GEPOp>(
950  loc, convertedPtrType,
951  ArrayRef<Value>{nullPtr, createIndexConstant(rewriter, loc, 1)});
952  auto elementSize =
953  rewriter.create<LLVM::PtrToIntOp>(loc, getIndexType(), gep);
954  return rewriter.create<LLVM::MulOp>(
955  loc, getIndexType(), ArrayRef<Value>{cumulativeSizeInBytes, elementSize});
956 }

973 /// Creates an auxiliary function with pointer-to-memref-descriptor-struct
974 /// arguments instead of unpacked arguments. This function can be called from C
975 /// by passing a pointer to a C struct corresponding to a memref descriptor.
976 /// Internally, the auxiliary function unpacks the descriptor into individual
977 /// components and forwards them to `newFuncOp`.
978 static void wrapForExternalCallers(OpBuilder &rewriter, Location loc,
979  LLVMTypeConverter &typeConverter,
980  FuncOp funcOp, LLVM::LLVMFuncOp newFuncOp) {
981  auto type = funcOp.getType();
982  SmallVector<NamedAttribute, 4> attributes;
983  filterFuncAttributes(funcOp.getAttrs(), /*filterArgAttrs=*/false, attributes);
984  auto wrapperFuncOp = rewriter.create<LLVM::LLVMFuncOp>(
985  loc, llvm::formatv("_mlir_ciface_{0}", funcOp.getName()).str(),
986  typeConverter.convertFunctionTypeCWrapper(type), LLVM::Linkage::External,
987  attributes);
988 
989  OpBuilder::InsertionGuard guard(rewriter);
990  rewriter.setInsertionPointToStart(wrapperFuncOp.addEntryBlock());
991 
992  SmallVector<Value, 8> args;
993  for (auto &en : llvm::enumerate(type.getInputs())) {
994  Value arg = wrapperFuncOp.getArgument(en.index());
995  if (auto memrefType = en.value().dyn_cast<MemRefType>()) {
996  Value loaded = rewriter.create<LLVM::LoadOp>(loc, arg);
997  MemRefDescriptor::unpack(rewriter, loc, loaded, memrefType, args);
998  continue;
999  }
1000  if (en.value().isa<UnrankedMemRefType>()) {
1001  Value loaded = rewriter.create<LLVM::LoadOp>(loc, arg);
1002  UnrankedMemRefDescriptor::unpack(rewriter, loc, loaded, args);
1003  continue;
1004  }
1005 
1006  args.push_back(wrapperFuncOp.getArgument(en.index()));
1007  }
1008  auto call = rewriter.create<LLVM::CallOp>(loc, newFuncOp, args);
1009  rewriter.create<LLVM::ReturnOp>(loc, call.getResults());
1010 }

1012 /// Creates an auxiliary function with pointer-to-memref-descriptor-struct
1013 /// arguments instead of unpacked arguments. Creates a body for the (external)
1014 /// `newFuncOp` that allocates a memref descriptor on stack, packs the
1015 /// individual arguments into this descriptor and passes a pointer to it into
1016 /// the auxiliary function. This auxiliary external function is now compatible
1017 /// with functions defined in C using pointers to C structs corresponding to a
1018 /// memref descriptor.
1019 static void wrapExternalFunction(OpBuilder &builder, Location loc,
1020  LLVMTypeConverter &typeConverter,
1021  FuncOp funcOp, LLVM::LLVMFuncOp newFuncOp) {
1022  OpBuilder::InsertionGuard guard(builder);
1023 
1024  LLVM::LLVMType wrapperType =
1025  typeConverter.convertFunctionTypeCWrapper(funcOp.getType());
1026  // This conversion can only fail if it could not convert one of the argument
1027  // types. But since it has been applies to a non-wrapper function before, it
1028  // should have failed earlier and not reach this point at all.
1029  assert(wrapperType && "unexpected type conversion failure");
1030 
1031  SmallVector<NamedAttribute, 4> attributes;
1032  filterFuncAttributes(funcOp.getAttrs(), /*filterArgAttrs=*/false, attributes);
1033 
1034  // Create the auxiliary function.
1035  auto wrapperFunc = builder.create<LLVM::LLVMFuncOp>(
1036  loc, llvm::formatv("_mlir_ciface_{0}", funcOp.getName()).str(),
1037  wrapperType, LLVM::Linkage::External, attributes);
1038 
1039  builder.setInsertionPointToStart(newFuncOp.addEntryBlock());
1040 
1041  // Get a ValueRange containing arguments.
1042  FunctionType type = funcOp.getType();
1043  SmallVector<Value, 8> args;
1044  args.reserve(type.getNumInputs());
1045  ValueRange wrapperArgsRange(newFuncOp.getArguments());
1046 
1047  // Iterate over the inputs of the original function and pack values into
1048  // memref descriptors if the original type is a memref.
1049  for (auto &en : llvm::enumerate(type.getInputs())) {
1050  Value arg;
1051  int numToDrop = 1;
1052  auto memRefType = en.value().dyn_cast<MemRefType>();
1053  auto unrankedMemRefType = en.value().dyn_cast<UnrankedMemRefType>();
1054  if (memRefType || unrankedMemRefType) {
1055  numToDrop = memRefType
1056  ? MemRefDescriptor::getNumUnpackedValues(memRefType)
1057  : UnrankedMemRefDescriptor::getNumUnpackedValues();
1058  Value packed =
1059  memRefType
1060  ? MemRefDescriptor::pack(builder, loc, typeConverter, memRefType,
1061  wrapperArgsRange.take_front(numToDrop))
1062  : UnrankedMemRefDescriptor::pack(
1063  builder, loc, typeConverter, unrankedMemRefType,
1064  wrapperArgsRange.take_front(numToDrop));
1065 
1066  auto ptrTy = packed.getType().cast<LLVM::LLVMType>().getPointerTo();
1067  Value one = builder.create<LLVM::ConstantOp>(
1068  loc, typeConverter.convertType(builder.getIndexType()),
1069  builder.getIntegerAttr(builder.getIndexType(), 1));
1070  Value allocated =
1071  builder.create<LLVM::AllocaOp>(loc, ptrTy, one, /*alignment=*/0);
1072  builder.create<LLVM::StoreOp>(loc, packed, allocated);
1073  arg = allocated;
1074  } else {
1075  arg = wrapperArgsRange[0];
1076  }
1077 
1078  args.push_back(arg);
1079  wrapperArgsRange = wrapperArgsRange.drop_front(numToDrop);
1080  }
1081  assert(wrapperArgsRange.empty() && "did not map some of the arguments");
1082 
1083  auto call = builder.create<LLVM::CallOp>(loc, wrapperFunc, args);
1084  builder.create<LLVM::ReturnOp>(loc, call.getResults());
1085 }

1109  ConversionPatternRewriter &rewriter) const {

1111  // LLVMTypeConverter provided to this legalization pattern.
1112  auto varargsAttr = funcOp.getAttrOfType<BoolAttr>("std.varargs");
1113  TypeConverter::SignatureConversion result(funcOp.getNumArguments());
1114  auto llvmType = typeConverter.convertFunctionSignature(
1115  funcOp.getType(), varargsAttr && varargsAttr.getValue(), result);

1118  // converting a given original argument.
1119  SmallVector<NamedAttribute, 4> attributes;
1120  filterFuncAttributes(funcOp.getAttrs(), /*filterArgAttrs=*/true,
1121  attributes);
1122  for (unsigned i = 0, e = funcOp.getNumArguments(); i < e; ++i) {
1123  auto attr = impl::getArgAttrDict(funcOp, i);
1124  if (!attr)
1125  continue;
1126 
1127  auto mapping = result.getInputMapping(i);
1128  assert(mapping.hasValue() && "unexpected deletion of function argument");
1129 
1130  SmallString<8> name;
1131  for (size_t j = 0; j < mapping->size; ++j) {
1132  impl::getArgAttrName(mapping->inputNo + j, name);
1133  attributes.push_back(rewriter.getNamedAttr(name, attr));
1134  }
1135  }

1138  // functions have linkage.
1139  auto newFuncOp = rewriter.create<LLVM::LLVMFuncOp>(
1140  funcOp.getLoc(), funcOp.getName(), llvmType, LLVM::Linkage::External,
1141  attributes);
1142  rewriter.inlineRegionBefore(funcOp.getBody(), newFuncOp.getBody(),
1143  newFuncOp.end());

1249 // For >1-D vector types, extracts the necessary information to iterate over all
1250 // 1-D subvectors in the underlying llrepresentation of the n-D vector
1251 // Iterates on the llvm array type until we hit a non-array type (which is
1252 // asserted to be an llvm vector type).
1253 static NDVectorTypeInfo extractNDVectorTypeInfo(VectorType vectorType,
1254  LLVMTypeConverter &converter) {
1255  assert(vectorType.getRank() > 1 && "expected >1D vector type");
1256  NDVectorTypeInfo info;
1257  info.llvmArrayTy =
1258  converter.convertType(vectorType).dyn_cast<LLVM::LLVMType>();
1259  if (!info.llvmArrayTy)
1260  return info;
1261  info.arraySizes.reserve(vectorType.getRank() - 1);
1262  auto llvmTy = info.llvmArrayTy;
1263  while (llvmTy.isArrayTy()) {
1264  info.arraySizes.push_back(llvmTy.getArrayNumElements());
1265  llvmTy = llvmTy.getArrayElementType();
1266  }
1267  if (!llvmTy.isVectorTy())
1268  return info;
1269  info.llvmVectorTy = llvmTy;
1270  return info;

1273 // Express `linearIndex` in terms of coordinates of `basis`.
1274 // Returns the empty vector when linearIndex is out of the range [0, P] where
1275 // P is the product of all the basis coordinates.
1276 //
1277 // Prerequisites:
1278 // Basis is an array of nonnegative integers (signed type inherited from
1279 // vector shape type).
1280 static SmallVector<int64_t, 4> getCoordinates(ArrayRef<int64_t> basis,
1281  unsigned linearIndex) {
1282  SmallVector<int64_t, 4> res;
1283  res.reserve(basis.size());
1284  for (unsigned basisElement : llvm::reverse(basis)) {
1285  res.push_back(linearIndex % basisElement);
1286  linearIndex = linearIndex / basisElement;
1287  }
1288  if (linearIndex > 0)
1289  return {};
1290  std::reverse(res.begin(), res.end());
1291  return res;
1292 }

1294 // Iterate of linear index, convert to coords space and insert splatted 1-D
1295 // vector in each position.
1296 template <typename Lambda>
1297 void nDVectorIterate(const NDVectorTypeInfo &info, OpBuilder &builder,
1298  Lambda fun) {
1299  unsigned ub = 1;
1300  for (auto s : info.arraySizes)
1301  ub *= s;
1302  for (unsigned linearIndex = 0; linearIndex < ub; ++linearIndex) {
1303  auto coords = getCoordinates(info.arraySizes, linearIndex);
1304  // Linear index is out of bounds, we are done.
1305  if (coords.empty())
1306  break;
1307  assert(coords.size() == info.arraySizes.size());
1308  auto position = builder.getI64ArrayAttr(coords);
1309  fun(position);
1310  }
1311 }

1314 /// Replaces the given operation "op" with a new operation of type "targetOp"
1315 /// and given operands.
1316 LogicalResult LLVM::detail::oneToOneRewrite(
1317  Operation *op, StringRef targetOp, ValueRange operands,
1318  LLVMTypeConverter &typeConverter, ConversionPatternRewriter &rewriter) {
1319  unsigned numResults = op->getNumResults();
1320 
1321  Type packedType;
1322  if (numResults != 0) {
1323  packedType = typeConverter.packFunctionResults(op->getResultTypes());
1324  if (!packedType)
1325  return failure();
1326  }
1327 
1328  // Create the operation through state since we don't know its C++ type.
1329  OperationState state(op->getLoc(), targetOp);
1330  state.addTypes(packedType);
1331  state.addOperands(operands);
1332  state.addAttributes(op->getAttrs());
1333  Operation *newOp = rewriter.createOperation(state);
1334 
1335  // If the operation produced 0 or 1 result, return them immediately.
1336  if (numResults == 0)
1337  return rewriter.eraseOp(op), success();
1338  if (numResults == 1)
1339  return rewriter.replaceOp(op, newOp->getResult(0)), success();
1340 
1341  // Otherwise, it had been converted to an operation producing a structure.
1342  // Extract individual results from the structure and return them as list.
1343  SmallVector<Value, 4> results;
1344  results.reserve(numResults);
1345  for (unsigned i = 0; i < numResults; ++i) {
1346  auto type = typeConverter.convertType(op->getResult(i).getType());
1347  results.push_back(rewriter.create<LLVM::ExtractValueOp>(
1348  op->getLoc(), type, newOp->getResult(0), rewriter.getI64ArrayAttr(i)));
1349  }
1350  rewriter.replaceOp(op, results);
1351  return success();
1352 }

1354 static LogicalResult handleMultidimensionalVectors(
1355  Operation *op, ValueRange operands, LLVMTypeConverter &typeConverter,
1356  std::function<Value(LLVM::LLVMType, ValueRange)> createOperand,
1357  ConversionPatternRewriter &rewriter) {
1358  auto vectorType = op->getResult(0).getType().dyn_cast<VectorType>();
1359  if (!vectorType)
1360  return failure();
1361  auto vectorTypeInfo = extractNDVectorTypeInfo(vectorType, typeConverter);
1362  auto llvmVectorTy = vectorTypeInfo.llvmVectorTy;
1363  auto llvmArrayTy = operands[0].getType().cast<LLVM::LLVMType>();
1364  if (!llvmVectorTy || llvmArrayTy != vectorTypeInfo.llvmArrayTy)
1365  return failure();
1366 
1367  auto loc = op->getLoc();
1368  Value desc = rewriter.create<LLVM::UndefOp>(loc, llvmArrayTy);
1369  nDVectorIterate(vectorTypeInfo, rewriter, [&](ArrayAttr position) {
1370  // For this unrolled `position` corresponding to the `linearIndex`^th
1371  // element, extract operand vectors
1372  SmallVector<Value, 4> extractedOperands;
1373  for (auto operand : operands)
1374  extractedOperands.push_back(rewriter.create<LLVM::ExtractValueOp>(
1375  loc, llvmVectorTy, operand, position));
1376  Value newVal = createOperand(llvmVectorTy, extractedOperands);
1377  desc = rewriter.create<LLVM::InsertValueOp>(loc, llvmArrayTy, desc, newVal,
1378  position);
1379  });
1380  rewriter.replaceOp(op, desc);
1381  return success();
1382 }

1707  // Returns bump = (alignment - (input % alignment))% alignment, which is the
1708  // increment necessary to align `input` to `alignment` boundary.

1710  // and two bit shifts: (ptr + align - 1)/align, align is always power of 2.

1712  Value alignment) const {

1715  Value shift = b.create<LLVM::URemOp>(loc, diff, alignment);

1719  /// Creates and populates the memref descriptor struct given all its fields.
1720  /// This method also performs any post allocation alignment needed for heap
1721  /// allocations when `accessAlignment` is non null. This is used with
1722  /// allocators that do not support alignment.
1723  MemRefDescriptor createMemRefDescriptor(
1724  Location loc, ConversionPatternRewriter &rewriter, MemRefType memRefType,
1725  Value allocatedTypePtr, Value allocatedBytePtr, Value accessAlignment,
1726  uint64_t offset, ArrayRef<int64_t> strides, ArrayRef<Value> sizes) const {
1727  auto elementPtrType = this->getElementPtrType(memRefType);
1728  auto structType = typeConverter.convertType(memRefType);
1729  auto memRefDescriptor = MemRefDescriptor::undef(rewriter, loc, structType);
1730 
1731  // Field 1: Allocated pointer, used for malloc/free.
1732  memRefDescriptor.setAllocatedPtr(rewriter, loc, allocatedTypePtr);
1733 
1734  // Field 2: Actual aligned pointer to payload.
1735  Value alignedBytePtr = allocatedTypePtr;
1736  if (accessAlignment) {
1737  // offset = (align - (ptr % align))% align
1738  Value intVal = rewriter.create<LLVM::PtrToIntOp>(
1739  loc, this->getIndexType(), allocatedBytePtr);
1740  Value offset = createBumpToAlign(loc, rewriter, intVal, accessAlignment);
1741  Value aligned = rewriter.create<LLVM::GEPOp>(
1742  loc, allocatedBytePtr.getType(), allocatedBytePtr, offset);
1743  alignedBytePtr = rewriter.create<LLVM::BitcastOp>(
1744  loc, elementPtrType, ArrayRef<Value>(aligned));
1745  }
1746  memRefDescriptor.setAlignedPtr(rewriter, loc, alignedBytePtr);
1747 
1748  // Field 3: Offset in aligned pointer.
1749  memRefDescriptor.setOffset(rewriter, loc,
1750  createIndexConstant(rewriter, loc, offset));
1751 
1752  if (memRefType.getRank() == 0)
1753  // No size/stride descriptor in memref, return the descriptor value.
1754  return memRefDescriptor;
1755 
1756  // Fields 4 and 5: sizes and strides of the strided MemRef.
1757  // Store all sizes in the descriptor. Only dynamic sizes are passed in as
1758  // operands to AllocOp.
1759  Value runningStride = nullptr;
1760  // Iterate strides in reverse order, compute runningStride and strideValues.
1761  auto nStrides = strides.size();
1762  SmallVector<Value, 4> strideValues(nStrides, nullptr);
1763  for (unsigned i = 0; i < nStrides; ++i) {
1764  int64_t index = nStrides - 1 - i;
1765  if (strides[index] == MemRefType::getDynamicStrideOrOffset())
1766  // Identity layout map is enforced in the match function, so we compute:
1767  // `runningStride *= sizes[index + 1]`
1768  runningStride = runningStride
1769  ? rewriter.create<LLVM::MulOp>(loc, runningStride,
1770  sizes[index + 1])
1771  : createIndexConstant(rewriter, loc, 1);
1772  else
1773  runningStride = createIndexConstant(rewriter, loc, strides[index]);
1774  strideValues[index] = runningStride;
1775  }
1776  // Fill size and stride descriptors in memref.
1777  for (auto indexedSize : llvm::enumerate(sizes)) {
1778  int64_t index = indexedSize.index();
1779  memRefDescriptor.setSize(rewriter, loc, index, indexedSize.value());
1780  memRefDescriptor.setStride(rewriter, loc, index, strideValues[index]);
1781  }
1782  return memRefDescriptor;
1783  }

1832  /// Allocates the underlying buffer using the right call. `allocatedBytePtr`
1833  /// is set to null for stack allocations. `accessAlignment` is set if
1834  /// alignment is needed post allocation (for eg. in conjunction with malloc).
1835  Value allocateBuffer(Location loc, Value cumulativeSize, Operation *op,
1836  MemRefType memRefType, Value one, Value &accessAlignment,
1837  Value &allocatedBytePtr,
1838  ConversionPatternRewriter &rewriter) const {
1839  auto elementPtrType = this->getElementPtrType(memRefType);
1840 
1841  // With alloca, one gets a pointer to the element type right away.
1842  // For stack allocations.
1843  if (auto allocaOp = dyn_cast<AllocaOp>(op)) {
1844  allocatedBytePtr = nullptr;
1845  accessAlignment = nullptr;
1846  return rewriter.create<LLVM::AllocaOp>(
1847  loc, elementPtrType, cumulativeSize,
1848  allocaOp.alignment() ? allocaOp.alignment().getValue().getSExtValue()
1849  : 0);
1850  }
1851 
1852  // Heap allocations.
1853  AllocOp allocOp = cast<AllocOp>(op);
1854 
1855  Optional<int64_t> allocationAlignment = getAllocationAlignment(allocOp);
1856  // Whether to use std lib function aligned_alloc that supports alignment.
1857  bool useAlignedAlloc = allocationAlignment.hasValue();
1858 
1859  // Insert the malloc/aligned_alloc declaration if it is not already present.
1860  const auto *allocFuncName = useAlignedAlloc ? "aligned_alloc" : "malloc";
1861  auto module = allocOp.getParentOfType<ModuleOp>();
1862  auto allocFunc = module.lookupSymbol<LLVM::LLVMFuncOp>(allocFuncName);
1863  if (!allocFunc) {
1864  OpBuilder::InsertionGuard guard(rewriter);
1865  rewriter.setInsertionPointToStart(
1866  op->getParentOfType<ModuleOp>().getBody());
1867  SmallVector<LLVM::LLVMType, 2> callArgTypes = {getIndexType()};
1868  // aligned_alloc(size_t alignment, size_t size)
1869  if (useAlignedAlloc)
1870  callArgTypes.push_back(getIndexType());
1871  allocFunc = rewriter.create<LLVM::LLVMFuncOp>(
1872  rewriter.getUnknownLoc(), allocFuncName,
1873  LLVM::LLVMType::getFunctionTy(getVoidPtrType(), callArgTypes,
1874  /*isVarArg=*/false));
1875  }
1876 
1877  // Allocate the underlying buffer and store a pointer to it in the MemRef
1878  // descriptor.
1879  SmallVector<Value, 2> callArgs;
1880  if (useAlignedAlloc) {
1881  // Use aligned_alloc.
1882  assert(allocationAlignment && "allocation alignment should be present");
1883  auto alignedAllocAlignmentValue = rewriter.create<LLVM::ConstantOp>(
1884  loc, typeConverter.convertType(rewriter.getIntegerType(64)),
1885  rewriter.getI64IntegerAttr(allocationAlignment.getValue()));
1886  // aligned_alloc requires size to be a multiple of alignment; we will pad
1887  // the size to the next multiple if necessary.
1888  if (!isMemRefSizeMultipleOf(memRefType, allocationAlignment.getValue())) {
1889  Value bump = createBumpToAlign(loc, rewriter, cumulativeSize,
1890  alignedAllocAlignmentValue);
1891  cumulativeSize =
1892  rewriter.create<LLVM::AddOp>(loc, cumulativeSize, bump);
1893  }
1894  callArgs = {alignedAllocAlignmentValue, cumulativeSize};
1895  } else {
1896  // Adjust the allocation size to consider alignment.
1897  if (allocOp.alignment()) {
1898  accessAlignment = createIndexConstant(
1899  rewriter, loc, allocOp.alignment().getValue().getSExtValue());
1900  cumulativeSize = rewriter.create<LLVM::SubOp>(
1901  loc,
1902  rewriter.create<LLVM::AddOp>(loc, cumulativeSize, accessAlignment),
1903  one);
1904  }
1905  callArgs.push_back(cumulativeSize);
1906  }
1907  auto allocFuncSymbol = rewriter.getSymbolRefAttr(allocFunc);
1908  allocatedBytePtr = rewriter
1909  .create<LLVM::CallOp>(loc, getVoidPtrType(),
1910  allocFuncSymbol, callArgs)
1911  .getResult(0);
1912  // For heap allocations, the allocated pointer is a cast of the byte pointer
1913  // to the type pointer.
1914  return rewriter.create<LLVM::BitcastOp>(loc, elementPtrType,
1915  allocatedBytePtr);
1916  }

1918  // An `alloc` is converted into a definition of a memref descriptor value and
1919  // a call to `malloc` to allocate the underlying data buffer. The memref
1920  // descriptor is of the LLVM structure type where:
1921  // 1. the first element is a pointer to the allocated (typed) data buffer,
1922  // 2. the second element is a pointer to the (typed) payload, aligned to the
1923  // specified alignment,
1924  // 3. the remaining elements serve to store all the sizes and strides of the
1925  // memref using LLVM-converted `index` type.
1926  //
1927  // Alignment is performed by allocating `alignment - 1` more bytes than
1928  // requested and shifting the aligned pointer relative to the allocated
1929  // memory. If alignment is unspecified, the two pointers are equal.
1930 
1931  // An `alloca` is converted into a definition of a memref descriptor value and
1932  // an llvm.alloca to allocate the underlying data buffer.
1933  void rewrite(Operation *op, ArrayRef<Value> operands,
1934  ConversionPatternRewriter &rewriter) const override {
1935  MemRefType memRefType = cast<AllocLikeOp>(op).getType();
1936  auto loc = op->getLoc();
1937 
1938  // Get actual sizes of the memref as values: static sizes are constant
1939  // values and dynamic sizes are passed to 'alloc' as operands. In case of
1940  // zero-dimensional memref, assume a scalar (size 1).
1941  SmallVector<Value, 4> sizes;
1942  this->getMemRefDescriptorSizes(loc, memRefType, operands, rewriter, sizes);
1943 
1944  Value cumulativeSize = this->getCumulativeSizeInBytes(
1945  loc, memRefType.getElementType(), sizes, rewriter);
1946  // Allocate the underlying buffer.
1947  // Value holding the alignment that has to be performed post allocation
1948  // (in conjunction with allocators that do not support alignment, eg.
1949  // malloc); nullptr if no such adjustment needs to be performed.
1950  Value accessAlignment;
1951  // Byte pointer to the allocated buffer.
1952  Value allocatedBytePtr;
1953  Value allocatedTypePtr =
1954  allocateBuffer(loc, cumulativeSize, op, memRefType,
1955  createIndexConstant(rewriter, loc, 1), accessAlignment,
1956  allocatedBytePtr, rewriter);
1957 
1958  int64_t offset;
1959  SmallVector<int64_t, 4> strides;
1960  auto successStrides = getStridesAndOffset(memRefType, strides, offset);
1961  (void)successStrides;
1962  assert(succeeded(successStrides) && "unexpected non-strided memref");
1963  assert(offset != MemRefType::getDynamicStrideOrOffset() &&
1964  "unexpected dynamic offset");
1965 
1966  // 0-D memref corner case: they have size 1.
1967  assert(
1968  ((memRefType.getRank() == 0 && strides.empty() && sizes.size() == 1) ||
1969  (strides.size() == sizes.size())) &&
1970  "unexpected number of strides");
1971 
1972  // Create the MemRef descriptor.
1973  auto memRefDescriptor = createMemRefDescriptor(
1974  loc, rewriter, memRefType, allocatedTypePtr, allocatedBytePtr,
1975  accessAlignment, offset, strides, sizes);
1976 
1977  // Return the final value of the descriptor.
1978  rewriter.replaceOp(op, {memRefDescriptor});
1979  }
1980 
1981 protected:
1982  /// The minimum alignment to use with aligned_alloc (has to be a power of 2).
1983  uint64_t kMinAlignedAllocAlignment = 16UL;
1984 };

1993 /// Copies the shaped descriptor part to (if `toDynamic` is set) or from
1994 /// (otherwise) the dynamically allocated memory for any operands that were
1995 /// unranked descriptors originally.
1996 static LogicalResult copyUnrankedDescriptors(OpBuilder &builder, Location loc,
1997  LLVMTypeConverter &typeConverter,
1998  TypeRange origTypes,
1999  SmallVectorImpl<Value> &operands,
2000  bool toDynamic) {
2001  assert(origTypes.size() == operands.size() &&
2002  "expected as may original types as operands");
2003 
2004  // Find operands of unranked memref type and store them.
2005  SmallVector<UnrankedMemRefDescriptor, 4> unrankedMemrefs;
2006  for (unsigned i = 0, e = operands.size(); i < e; ++i)
2007  if (origTypes[i].isa<UnrankedMemRefType>())
2008  unrankedMemrefs.emplace_back(operands[i]);
2009 
2010  if (unrankedMemrefs.empty())
2011  return success();
2012 
2013  // Compute allocation sizes.
2014  SmallVector<Value, 4> sizes;
2015  UnrankedMemRefDescriptor::computeSizes(builder, loc, typeConverter,
2016  unrankedMemrefs, sizes);
2017 
2018  // Get frequently used types.
2019  MLIRContext *context = builder.getContext();
2020  auto voidType = LLVM::LLVMType::getVoidTy(context);
2021  auto voidPtrType = LLVM::LLVMType::getInt8PtrTy(context);
2022  auto i1Type = LLVM::LLVMType::getInt1Ty(context);
2023  LLVM::LLVMType indexType = typeConverter.getIndexType();
2024 
2025  // Find the malloc and free, or declare them if necessary.
2026  auto module = builder.getInsertionPoint()->getParentOfType<ModuleOp>();
2027  auto mallocFunc = module.lookupSymbol<LLVM::LLVMFuncOp>("malloc");
2028  if (!mallocFunc && toDynamic) {
2029  OpBuilder::InsertionGuard guard(builder);
2030  builder.setInsertionPointToStart(module.getBody());
2031  mallocFunc = builder.create<LLVM::LLVMFuncOp>(
2032  builder.getUnknownLoc(), "malloc",
2033  LLVM::LLVMType::getFunctionTy(
2034  voidPtrType, llvm::makeArrayRef(indexType), /*isVarArg=*/false));
2035  }
2036  auto freeFunc = module.lookupSymbol<LLVM::LLVMFuncOp>("free");
2037  if (!freeFunc && !toDynamic) {
2038  OpBuilder::InsertionGuard guard(builder);
2039  builder.setInsertionPointToStart(module.getBody());
2040  freeFunc = builder.create<LLVM::LLVMFuncOp>(
2041  builder.getUnknownLoc(), "free",
2042  LLVM::LLVMType::getFunctionTy(voidType, llvm::makeArrayRef(voidPtrType),
2043  /*isVarArg=*/false));
2044  }
2045 
2046  // Initialize shared constants.
2047  Value zero =
2048  builder.create<LLVM::ConstantOp>(loc, i1Type, builder.getBoolAttr(false));
2049 
2050  unsigned unrankedMemrefPos = 0;
2051  for (unsigned i = 0, e = operands.size(); i < e; ++i) {
2052  Type type = origTypes[i];
2053  if (!type.isa<UnrankedMemRefType>())
2054  continue;
2055  Value allocationSize = sizes[unrankedMemrefPos++];
2056  UnrankedMemRefDescriptor desc(operands[i]);
2057 
2058  // Allocate memory, copy, and free the source if necessary.
2059  Value memory =
2060  toDynamic
2061  ? builder.create<LLVM::CallOp>(loc, mallocFunc, allocationSize)
2062  .getResult(0)
2063  : builder.create<LLVM::AllocaOp>(loc, voidPtrType, allocationSize,
2064  /*alignment=*/0);
2065 
2066  Value source = desc.memRefDescPtr(builder, loc);
2067  builder.create<LLVM::MemcpyOp>(loc, memory, source, allocationSize, zero);
2068  if (!toDynamic)
2069  builder.create<LLVM::CallOp>(loc, freeFunc, source);
2070 
2071  // Create a new descriptor. The same descriptor can be returned multiple
2072  // times, attempting to modify its pointer can lead to memory leaks
2073  // (allocated twice and overwritten) or double frees (the caller does not
2074  // know if the descriptor points to the same memory).
2075  Type descriptorType = typeConverter.convertType(type);
2076  if (!descriptorType)
2077  return failure();
2078  auto updatedDesc =
2079  UnrankedMemRefDescriptor::undef(builder, loc, descriptorType);
2080  Value rank = desc.rank(builder, loc);
2081  updatedDesc.setRank(builder, loc, rank);
2082  updatedDesc.setMemRefDescPtr(builder, loc, memory);
2083 
2084  operands[i] = updatedDesc;
2085  }
2086 
2087  return success();
2088 }
