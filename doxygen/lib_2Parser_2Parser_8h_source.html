64  /// Encode the specified source location information into an attribute for
65  /// attachment to the IR.
66  Location getEncodedSourceLocation(llvm::SMLoc loc) {
67  // If there are no active nested parsers, we can get the encoded source
68  // location directly.
69  if (state.parserDepth == 0)
70  return state.lex.getEncodedSourceLocation(loc);
71  // Otherwise, we need to re-encode it to point to the top level buffer.
72  return state.symbols.topLevelLexer->getEncodedSourceLocation(
73  remapLocationToTopLevelBuffer(loc));
74  }
75 
76  /// Remaps the given SMLoc to the top level lexer of the parser. This is used
77  /// to adjust locations of potentially nested parsers to ensure that they can
78  /// be emitted properly as diagnostics.
79  llvm::SMLoc remapLocationToTopLevelBuffer(llvm::SMLoc loc) {
80  // If there are no active nested parsers, we can return location directly.
81  SymbolState &symbols = state.symbols;
82  if (state.parserDepth == 0)
83  return loc;
84  assert(symbols.topLevelLexer && "expected valid top-level lexer");
85 
86  // Otherwise, we need to remap the location to the main parser. This is
87  // simply offseting the location onto the location of the last nested
88  // parser.
89  size_t offset = loc.getPointer() - state.lex.getBufferBegin();
90  auto *rawLoc =
91  symbols.nestedParserLocs[state.parserDepth - 1].getPointer() + offset;
92  return llvm::SMLoc::getFromPointer(rawLoc);
93  }

103  /// If the current token has the specified kind, consume it and return true.
104  /// If not, return false.
105  bool consumeIf(Token::Kind kind) {
106  if (state.curToken.isNot(kind))
107  return false;
108  consumeToken(kind);
109  return true;
110  }
111 
112  /// Advance the current lexer onto the next token.
113  void consumeToken() {
114  assert(state.curToken.isNot(Token::eof, Token::error) &&
115  "shouldn't advance past EOF or errors");
116  state.curToken = state.lex.lexToken();
117  }
118 
119  /// Advance the current lexer onto the next token, asserting what the expected
120  /// current token is. This is preferred to the above method because it leads
121  /// to more self-documenting code with better checking.
122  void consumeToken(Token::Kind kind) {
123  assert(state.curToken.is(kind) && "consumed an unexpected token");
124  consumeToken();
125  }

234  ///
235  /// trailing-location ::= (`loc` `(` location `)`)?
236  ///
237  ParseResult parseOptionalTrailingLocation(Location &loc) {
238  // If there is a 'loc' we parse a trailing location.
239  if (!getToken().is(Token::kw_loc))
240  return success();
241 
242  // Parse the location.
243  LocationAttr directLoc;
244  if (parseLocation(directLoc))
245  return failure();
246  loc = directLoc;
247  return success();
248  }
