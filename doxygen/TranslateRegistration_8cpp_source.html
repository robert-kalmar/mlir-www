35 // Deserializes the SPIR-V binary module stored in the file named as
36 // `inputFilename` and returns a module containing the SPIR-V module.
37 static OwningModuleRef deserializeModule(const llvm::MemoryBuffer *input,
38  MLIRContext *context) {
39  Builder builder(context);
40 
41  // Make sure the input stream can be treated as a stream of SPIR-V words
42  auto start = input->getBufferStart();
43  auto size = input->getBufferSize();
44  if (size % sizeof(uint32_t) != 0) {
45  emitError(UnknownLoc::get(context))
46  << "SPIR-V binary module must contain integral number of 32-bit words";
47  return {};
48  }
49 
50  auto binary = llvm::makeArrayRef(reinterpret_cast<const uint32_t *>(start),
51  size / sizeof(uint32_t));
52 
53  spirv::OwningSPIRVModuleRef spirvModule = spirv::deserialize(binary, context);
54  if (!spirvModule)
55  return {};
56 
57  OwningModuleRef module(ModuleOp::create(FileLineColLoc::get(
58  input->getBufferIdentifier(), /*line=*/0, /*column=*/0, context)));
59  module->getBody()->push_front(spirvModule.release());
60 
61  return module;
62 }

80 static LogicalResult serializeModule(ModuleOp module, raw_ostream &output) {
81  if (!module)
82  return failure();
83 
84  SmallVector<uint32_t, 0> binary;
85 
86  SmallVector<spirv::ModuleOp, 1> spirvModules;
87  module.walk([&](spirv::ModuleOp op) { spirvModules.push_back(op); });
88 
89  if (spirvModules.empty())
90  return module.emitError("found no 'spv.module' op");
91 
92  if (spirvModules.size() != 1)
93  return module.emitError("found more than one 'spv.module' op");
94 
95  if (failed(
96  spirv::serialize(spirvModules[0], binary, /*emitDebuginfo=*/false)))
97  return failure();
98 
99  output.write(reinterpret_cast<char *>(binary.data()),
100  binary.size() * sizeof(uint32_t));
101 
102  return mlir::success();
103 }

118 static LogicalResult roundTripModule(llvm::SourceMgr &sourceMgr,
119  bool emitDebugInfo, raw_ostream &output,
120  MLIRContext *context) {
121  // Parse an MLIR module from the source manager.
122  auto srcModule = OwningModuleRef(parseSourceFile(sourceMgr, context));
123  if (!srcModule)
124  return failure();
125 
126  SmallVector<uint32_t, 0> binary;
127 
128  auto spirvModules = srcModule->getOps<spirv::ModuleOp>();
129 
130  if (spirvModules.begin() == spirvModules.end())
131  return srcModule->emitError("found no 'spv.module' op");
132 
133  if (std::next(spirvModules.begin()) != spirvModules.end())
134  return srcModule->emitError("found more than one 'spv.module' op");
135 
136  if (failed(spirv::serialize(*spirvModules.begin(), binary, emitDebugInfo)))
137  return failure();
138 
139  // Then deserialize to get back a SPIR-V module.
140  spirv::OwningSPIRVModuleRef spirvModule = spirv::deserialize(binary, context);
141  if (!spirvModule)
142  return failure();
143 
144  // Wrap around in a new MLIR module.
145  OwningModuleRef dstModule(ModuleOp::create(FileLineColLoc::get(
146  /*filename=*/"", /*line=*/0, /*column=*/0, context)));
147  dstModule->getBody()->push_front(spirvModule.release());
148  dstModule->print(output);
149 
150  return mlir::success();
151 }
