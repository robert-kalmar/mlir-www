25 /// A utility class to represent the analyses that are known to be preserved.
26 class PreservedAnalyses {
27  /// A type used to represent all potential analyses.
28  struct AllAnalysesType;
29 
30 public:
31  /// Mark all analyses as preserved.
32  void preserveAll() { preservedIDs.insert(TypeID::get<AllAnalysesType>()); }
33 
34  /// Returns true if all analyses were marked preserved.
35  bool isAll() const {
36  return preservedIDs.count(TypeID::get<AllAnalysesType>());
37  }
38 
39  /// Returns true if no analyses were marked preserved.
40  bool isNone() const { return preservedIDs.empty(); }
41 
42  /// Preserve the given analyses.
43  template <typename AnalysisT> void preserve() {
44  preserve(TypeID::get<AnalysisT>());
45  }
46  template <typename AnalysisT, typename AnalysisT2, typename... OtherAnalysesT>
47  void preserve() {
48  preserve<AnalysisT>();
49  preserve<AnalysisT2, OtherAnalysesT...>();
50  }
51  void preserve(TypeID id) { preservedIDs.insert(id); }
52 
53  /// Returns if the given analysis has been marked as preserved. Note that this
54  /// simply checks for the presence of a given analysis ID and should not be
55  /// used as a general preservation checker.
56  template <typename AnalysisT> bool isPreserved() const {
57  return isPreserved(TypeID::get<AnalysisT>());
58  }
59  bool isPreserved(TypeID id) const { return preservedIDs.count(id); }
60 
61 private:
62  /// The set of analyses that are known to be preserved.
63  SmallPtrSet<TypeID, 2> preservedIDs;
64 };

67 /// Trait to check if T provides a static 'isInvalidated' method.
68 template <typename T, typename... Args>
69 using has_is_invalidated = decltype(std::declval<T &>().isInvalidated(
70  std::declval<const PreservedAnalyses &>()));
71 
72 /// Implementation of 'isInvalidated' if the analysis provides a definition.
73 template <typename AnalysisT>
74 std::enable_if_t<llvm::is_detected<has_is_invalidated, AnalysisT>::value, bool>
75 isInvalidated(AnalysisT &analysis, const PreservedAnalyses &pa) {
76  return analysis.isInvalidated(pa);
77 }
78 /// Default implementation of 'isInvalidated'.
79 template <typename AnalysisT>
80 std::enable_if_t<!llvm::is_detected<has_is_invalidated, AnalysisT>::value, bool>
81 isInvalidated(AnalysisT &analysis, const PreservedAnalyses &pa) {
82  return !pa.isPreserved<AnalysisT>();
83 }

86 /// The abstract polymorphic base class representing an analysis.
87 struct AnalysisConcept {
88  virtual ~AnalysisConcept() = default;
89 
90  /// A hook used to query analyses for invalidation. Given a preserved analysis
91  /// set, returns true if it should truly be invalidated. This allows for more
92  /// fine-tuned invalidation in cases where an analysis wasn't explicitly
93  /// marked preserved, but may be preserved(or invalidated) based upon other
94  /// properties such as analyses sets.
95  virtual bool isInvalidated(const PreservedAnalyses &pa) = 0;
96 };

98 /// A derived analysis model used to hold a specific analysis object.
99 template <typename AnalysisT> struct AnalysisModel : public AnalysisConcept {
100  template <typename... Args>

102  : analysis(std::forward<Args>(args)...) {}

105  bool isInvalidated(const PreservedAnalyses &pa) final {
106  return analysis_impl::isInvalidated(analysis, pa);

113 /// This class represents a cache of analyses for a single operation. All
114 /// computation, caching, and invalidation of analyses takes place here.
115 class AnalysisMap {
116  /// A mapping between an analysis id and an existing analysis instance.
117  using ConceptMap = DenseMap<TypeID, std::unique_ptr<AnalysisConcept>>;
118 
119  /// Utility to return the name of the given analysis class.
120  template <typename AnalysisT> static StringRef getAnalysisName() {
121  StringRef name = llvm::getTypeName<AnalysisT>();
122  if (!name.consume_front("mlir::"))
123  name.consume_front("(anonymous namespace)::");
124  return name;
125  }
126 
127 public:
128  explicit AnalysisMap(Operation *ir) : ir(ir) {}
129 
130  /// Get an analysis for the current IR unit, computing it if necessary.
131  template <typename AnalysisT> AnalysisT &getAnalysis(PassInstrumentor *pi) {
132  TypeID id = TypeID::get<AnalysisT>();
133 
134  typename ConceptMap::iterator it;
135  bool wasInserted;
136  std::tie(it, wasInserted) = analyses.try_emplace(id);
137 
138  // If we don't have a cached analysis for this function, compute it directly
139  // and add it to the cache.
140  if (wasInserted) {
141  if (pi)
142  pi->runBeforeAnalysis(getAnalysisName<AnalysisT>(), id, ir);
143 
144  it->second = std::make_unique<AnalysisModel<AnalysisT>>(ir);
145 
146  if (pi)
147  pi->runAfterAnalysis(getAnalysisName<AnalysisT>(), id, ir);
148  }
149  return static_cast<AnalysisModel<AnalysisT> &>(*it->second).analysis;
150  }
151 
152  /// Get a cached analysis instance if one exists, otherwise return null.
153  template <typename AnalysisT>
154  Optional<std::reference_wrapper<AnalysisT>> getCachedAnalysis() const {
155  auto res = analyses.find(TypeID::get<AnalysisT>());
156  if (res == analyses.end())
157  return llvm::None;
158  return {static_cast<AnalysisModel<AnalysisT> &>(*res->second).analysis};
159  }
160 
161  /// Returns the operation that this analysis map represents.
162  Operation *getOperation() const { return ir; }
163 
164  /// Clear any held analyses.
165  void clear() { analyses.clear(); }
166 
167  /// Invalidate any cached analyses based upon the given set of preserved
168  /// analyses.
169  void invalidate(const PreservedAnalyses &pa) {
170  // Remove any analyses that were invalidated.
171  for (auto it = analyses.begin(), e = analyses.end(); it != e;) {
172  auto curIt = it++;
173  if (curIt->second->isInvalidated(pa))
174  analyses.erase(curIt);
175  }
176  }
177 
178 private:
179  Operation *ir;
180  ConceptMap analyses;
181 };

207 /// This class represents an analysis manager for a particular operation
208 /// instance. It is used to manage and cache analyses on the operation as well
209 /// as those for child operations, via nested AnalysisManager instances
210 /// accessible via 'slice'. This class is intended to be passed around by value,
211 /// and cannot be constructed directly.
212 class AnalysisManager {
213  using ParentPointerT =
214  PointerUnion<const ModuleAnalysisManager *, const AnalysisManager *>;
215 
216 public:
217  using PreservedAnalyses = detail::PreservedAnalyses;
218 
219  // Query for a cached analysis on the given parent operation. The analysis may
220  // not exist and if it does it may be out-of-date.
221  template <typename AnalysisT>
222  Optional<std::reference_wrapper<AnalysisT>>
223  getCachedParentAnalysis(Operation *parentOp) const {
224  ParentPointerT curParent = parent;
225  while (auto *parentAM = curParent.dyn_cast<const AnalysisManager *>()) {
226  if (parentAM->impl->getOperation() == parentOp)
227  return parentAM->getCachedAnalysis<AnalysisT>();
228  curParent = parentAM->parent;
229  }
230  return None;
231  }
232 
233  // Query for the given analysis for the current operation.
234  template <typename AnalysisT> AnalysisT &getAnalysis() {
235  return impl->analyses.getAnalysis<AnalysisT>(getPassInstrumentor());
236  }
237 
238  // Query for a cached entry of the given analysis on the current operation.
239  template <typename AnalysisT>
240  Optional<std::reference_wrapper<AnalysisT>> getCachedAnalysis() const {
241  return impl->analyses.getCachedAnalysis<AnalysisT>();
242  }
243 
244  /// Query for an analysis of a child operation, constructing it if necessary.
245  template <typename AnalysisT> AnalysisT &getChildAnalysis(Operation *op) {
246  return slice(op).template getAnalysis<AnalysisT>();
247  }
248 
249  /// Query for a cached analysis of a child operation, or return null.
250  template <typename AnalysisT>
251  Optional<std::reference_wrapper<AnalysisT>>
252  getCachedChildAnalysis(Operation *op) const {
253  assert(op->getParentOp() == impl->getOperation());
254  auto it = impl->childAnalyses.find(op);
255  if (it == impl->childAnalyses.end())
256  return llvm::None;
257  return it->second->analyses.getCachedAnalysis<AnalysisT>();
258  }
259 
260  /// Get an analysis manager for the given child operation.
261  AnalysisManager slice(Operation *op);
262 
263  /// Invalidate any non preserved analyses,
264  void invalidate(const PreservedAnalyses &pa) { impl->invalidate(pa); }
265 
266  /// Clear any held analyses.
267  void clear() {
268  impl->analyses.clear();
269  impl->childAnalyses.clear();
270  }
271 
272  /// Returns a pass instrumentation object for the current operation. This
273  /// value may be null.
274  PassInstrumentor *getPassInstrumentor() const;
275 
276 private:
277  AnalysisManager(const AnalysisManager *parent,
278  detail::NestedAnalysisMap *impl)
279  : parent(parent), impl(impl) {}
280  AnalysisManager(const ModuleAnalysisManager *parent,
281  detail::NestedAnalysisMap *impl)
282  : parent(parent), impl(impl) {}
283 
284  /// A reference to the parent analysis manager, or the top-level module
285  /// analysis manager.
286  ParentPointerT parent;
287 
288  /// A reference to the impl analysis map within the parent analysis manager.
289  detail::NestedAnalysisMap *impl;
290 
291  /// Allow access to the constructor.
292  friend class ModuleAnalysisManager;
293 };
