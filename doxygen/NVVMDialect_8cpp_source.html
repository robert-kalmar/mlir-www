44 // <operation> ::=
45 // `llvm.nvvm.shfl.sync.bfly %dst, %val, %offset, %clamp_and_mask`
46 // ({return_value_and_is_valid})? : result_type
47 static ParseResult parseNVVMShflSyncBflyOp(OpAsmParser &parser,
48  OperationState &result) {
49  SmallVector<OpAsmParser::OperandType, 8> ops;
50  Type resultType;
51  if (parser.parseOperandList(ops) ||
52  parser.parseOptionalAttrDict(result.attributes) ||
53  parser.parseColonType(resultType) ||
54  parser.addTypeToList(resultType, result.types))
55  return failure();
56 
57  auto type = resultType.cast<LLVM::LLVMType>();
58  for (auto &attr : result.attributes) {
59  if (attr.first != "return_value_and_is_valid")
60  continue;
61  if (type.isStructTy() && type.getStructNumElements() > 0)
62  type = type.getStructElementType(0);
63  break;
64  }
65 
66  auto int32Ty = LLVM::LLVMType::getInt32Ty(parser.getBuilder().getContext());
67  return parser.resolveOperands(ops, {int32Ty, type, int32Ty, int32Ty},
68  parser.getNameLoc(), result.operands);
69 }

72 static ParseResult parseNVVMVoteBallotOp(OpAsmParser &parser,
73  OperationState &result) {
74  MLIRContext *context = parser.getBuilder().getContext();
75  auto int32Ty = LLVM::LLVMType::getInt32Ty(context);
76  auto int1Ty = LLVM::LLVMType::getInt1Ty(context);
77 
78  SmallVector<OpAsmParser::OperandType, 8> ops;
79  Type type;
80  return failure(parser.parseOperandList(ops) ||
81  parser.parseOptionalAttrDict(result.attributes) ||
82  parser.parseColonType(type) ||
83  parser.addTypeToList(type, result.types) ||
84  parser.resolveOperands(ops, {int32Ty, int1Ty},
85  parser.getNameLoc(), result.operands));
86 }

88 static LogicalResult verify(MmaOp op) {
89  MLIRContext *context = op.getContext();
90  auto f16Ty = LLVM::LLVMType::getHalfTy(context);
91  auto f16x2Ty = LLVM::LLVMType::getVectorTy(f16Ty, 2);
92  auto f32Ty = LLVM::LLVMType::getFloatTy(context);
93  auto f16x2x4StructTy = LLVM::LLVMType::getStructTy(
94  context, {f16x2Ty, f16x2Ty, f16x2Ty, f16x2Ty});
95  auto f32x8StructTy = LLVM::LLVMType::getStructTy(
96  context, {f32Ty, f32Ty, f32Ty, f32Ty, f32Ty, f32Ty, f32Ty, f32Ty});
97 
98  SmallVector<Type, 12> operand_types(op.getOperandTypes().begin(),
99  op.getOperandTypes().end());
100  if (operand_types != SmallVector<Type, 8>(8, f16x2Ty) &&
101  operand_types != SmallVector<Type, 12>{f16x2Ty, f16x2Ty, f16x2Ty, f16x2Ty,
102  f32Ty, f32Ty, f32Ty, f32Ty, f32Ty,
103  f32Ty, f32Ty, f32Ty}) {
104  return op.emitOpError(
105  "expected operands to be 4 <halfx2>s followed by either "
106  "4 <halfx2>s or 8 floats");
107  }
108  if (op.getType() != f32x8StructTy && op.getType() != f16x2x4StructTy) {
109  return op.emitOpError("expected result type to be a struct of either 4 "
110  "<halfx2>s or 8 floats");
111  }
112 
113  auto alayout = op.getAttrOfType<StringAttr>("alayout");
114  auto blayout = op.getAttrOfType<StringAttr>("blayout");
115 
116  if (!(alayout && blayout) ||
117  !(alayout.getValue() == "row" || alayout.getValue() == "col") ||
118  !(blayout.getValue() == "row" || blayout.getValue() == "col")) {
119  return op.emitOpError(
120  "alayout and blayout attributes must be set to either "
121  "\"row\" or \"col\"");
122  }
123 
124  if (operand_types == SmallVector<Type, 12>{f16x2Ty, f16x2Ty, f16x2Ty, f16x2Ty,
125  f32Ty, f32Ty, f32Ty, f32Ty, f32Ty,
126  f32Ty, f32Ty, f32Ty} &&
127  op.getType() == f32x8StructTy && alayout.getValue() == "row" &&
128  blayout.getValue() == "col") {
129  return success();
130  }
131  return op.emitOpError("unimplemented mma.sync variant");
132 }
