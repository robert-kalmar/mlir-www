118 /// Integer types can have arbitrary bitwidth up to a large fixed limit.
119 class IntegerType
120  : public Type::TypeBase<IntegerType, Type, detail::IntegerTypeStorage> {
121 public:
122  using Base::Base;
123 
124  /// Signedness semantics.
125  enum SignednessSemantics : uint32_t {
126  Signless, /// No signedness semantics
127  Signed, /// Signed integer
128  Unsigned, /// Unsigned integer
129  };
130 
131  /// Get or create a new IntegerType of the given width within the context.
132  /// The created IntegerType is signless (i.e., no signedness semantics).
133  /// Assume the width is within the allowed range and assert on failures. Use
134  /// getChecked to handle failures gracefully.
135  static IntegerType get(unsigned width, MLIRContext *context);
136 
137  /// Get or create a new IntegerType of the given width within the context.
138  /// The created IntegerType has signedness semantics as indicated via
139  /// `signedness`. Assume the width is within the allowed range and assert on
140  /// failures. Use getChecked to handle failures gracefully.
141  static IntegerType get(unsigned width, SignednessSemantics signedness,
142  MLIRContext *context);
143 
144  /// Get or create a new IntegerType of the given width within the context,
145  /// defined at the given, potentially unknown, location. The created
146  /// IntegerType is signless (i.e., no signedness semantics). If the width is
147  /// outside the allowed range, emit errors and return a null type.
148  static IntegerType getChecked(unsigned width, Location location);
149 
150  /// Get or create a new IntegerType of the given width within the context,
151  /// defined at the given, potentially unknown, location. The created
152  /// IntegerType has signedness semantics as indicated via `signedness`. If the
153  /// width is outside the allowed range, emit errors and return a null type.
154  static IntegerType getChecked(unsigned width, SignednessSemantics signedness,
155  Location location);
156 
157  /// Verify the construction of an integer type.
158  static LogicalResult
159  verifyConstructionInvariants(Location loc, unsigned width,
160  SignednessSemantics signedness);
161 
162  /// Return the bitwidth of this integer type.
163  unsigned getWidth() const;
164 
165  /// Return the signedness semantics of this integer type.
166  SignednessSemantics getSignedness() const;
167 
168  /// Return true if this is a signless integer type.
169  bool isSignless() const { return getSignedness() == Signless; }
170  /// Return true if this is a signed integer type.
171  bool isSigned() const { return getSignedness() == Signed; }
172  /// Return true if this is an unsigned integer type.
173  bool isUnsigned() const { return getSignedness() == Unsigned; }
174 
175  /// Integer representation maximal bitwidth.
176  static constexpr unsigned kMaxWidth = 4096;
177 };

225 // ShapedType
226 //===----------------------------------------------------------------------===//
227 
228 /// This is a common base class between Vector, UnrankedTensor, RankedTensor,
229 /// and MemRef types because they share behavior and semantics around shape,
230 /// rank, and fixed element type. Any type with these semantics should inherit
231 /// from ShapedType.
232 class ShapedType : public Type {
233 public:
234  using ImplType = detail::ShapedTypeStorage;
235  using Type::Type;
236 
237  // TODO: merge these two special values in a single one used everywhere.
238  // Unfortunately, uses of `-1` have crept deep into the codebase now and are
239  // hard to track.
240  static constexpr int64_t kDynamicSize = -1;
241  static constexpr int64_t kDynamicStrideOrOffset =
242  std::numeric_limits<int64_t>::min();
243 
244  /// Return the element type.
245  Type getElementType() const;
246 
247  /// If an element type is an integer or a float, return its width. Otherwise,
248  /// abort.
249  unsigned getElementTypeBitWidth() const;
250 
251  /// If it has static shape, return the number of elements. Otherwise, abort.
252  int64_t getNumElements() const;
253 
254  /// If this is a ranked type, return the rank. Otherwise, abort.
255  int64_t getRank() const;
256 
257  /// Whether or not this is a ranked type. Memrefs, vectors and ranked tensors
258  /// have a rank, while unranked tensors do not.
259  bool hasRank() const;
260 
261  /// If this is a ranked type, return the shape. Otherwise, abort.
262  ArrayRef<int64_t> getShape() const;
263 
264  /// If this is unranked type or any dimension has unknown size (<0), it
265  /// doesn't have static shape. If all dimensions have known size (>= 0), it
266  /// has static shape.
267  bool hasStaticShape() const;
268 
269  /// If this has a static shape and the shape is equal to `shape` return true.
270  bool hasStaticShape(ArrayRef<int64_t> shape) const;
271 
272  /// If this is a ranked type, return the number of dimensions with dynamic
273  /// size. Otherwise, abort.
274  int64_t getNumDynamicDims() const;
275 
276  /// If this is ranked type, return the size of the specified dimension.
277  /// Otherwise, abort.
278  int64_t getDimSize(unsigned idx) const;
279 
280  /// Returns true if this dimension has a dynamic size (for ranked types);
281  /// aborts for unranked types.
282  bool isDynamicDim(unsigned idx) const;
283 
284  /// Returns the position of the dynamic dimension relative to just the dynamic
285  /// dimensions, given its `index` within the shape.
286  unsigned getDynamicDimIndex(unsigned index) const;
287 
288  /// Get the total amount of bits occupied by a value of this type. This does
289  /// not take into account any memory layout or widening constraints, e.g. a
290  /// vector<3xi57> is reported to occupy 3x57=171 bit, even though in practice
291  /// it will likely be stored as in a 4xi64 vector register. Fail an assertion
292  /// if the size cannot be computed statically, i.e. if the type has a dynamic
293  /// shape or if its elemental type does not have a known bit width.
294  int64_t getSizeInBits() const;
295 
296  /// Methods for support type inquiry through isa, cast, and dyn_cast.
297  static bool classof(Type type);
298 
299  /// Whether the given dimension size indicates a dynamic dimension.
300  static constexpr bool isDynamic(int64_t dSize) {
301  return dSize == kDynamicSize;
302  }
303  static constexpr bool isDynamicStrideOrOffset(int64_t dStrideOrOffset) {
304  return dStrideOrOffset == kDynamicStrideOrOffset;
305  }
306 };

440 
441 /// MemRef types represent a region of memory that have a shape with a fixed
442 /// number of dimensions. Each shape element can be a non-negative integer or
443 /// unknown (represented by -1). MemRef types also have an affine map
444 /// composition, represented as an array AffineMap pointers.
445 class MemRefType : public Type::TypeBase<MemRefType, BaseMemRefType,
446  detail::MemRefTypeStorage> {
447 public:
448  /// This is a builder type that keeps local references to arguments. Arguments
449  /// that are passed into the builder must out-live the builder.
450  class Builder {
451  public:
452  // Build from another MemRefType.
453  explicit Builder(MemRefType other)
454  : shape(other.getShape()), elementType(other.getElementType()),
455  affineMaps(other.getAffineMaps()),
456  memorySpace(other.getMemorySpace()) {}
457 
458  // Build from scratch.
459  Builder(ArrayRef<int64_t> shape, Type elementType)
460  : shape(shape), elementType(elementType), affineMaps(), memorySpace(0) {
461  }
462 
463  Builder &setShape(ArrayRef<int64_t> newShape) {
464  shape = newShape;
465  return *this;
466  }
467 
468  Builder &setElementType(Type newElementType) {
469  elementType = newElementType;
470  return *this;
471  }
472 
473  Builder &setAffineMaps(ArrayRef<AffineMap> newAffineMaps) {
474  affineMaps = newAffineMaps;
475  return *this;
476  }
477 
478  Builder &setMemorySpace(unsigned newMemorySpace) {
479  memorySpace = newMemorySpace;
480  return *this;
481  }
482 
483  operator MemRefType() {
484  return MemRefType::get(shape, elementType, affineMaps, memorySpace);
485  }
486 
487  private:
488  ArrayRef<int64_t> shape;
489  Type elementType;
490  ArrayRef<AffineMap> affineMaps;
491  unsigned memorySpace;
492  };
493 
494  using Base::Base;
495 
496  /// Get or create a new MemRefType based on shape, element type, affine
497  /// map composition, and memory space. Assumes the arguments define a
498  /// well-formed MemRef type. Use getChecked to gracefully handle MemRefType
499  /// construction failures.
500  static MemRefType get(ArrayRef<int64_t> shape, Type elementType,
501  ArrayRef<AffineMap> affineMapComposition = {},
502  unsigned memorySpace = 0);
503 
504  /// Get or create a new MemRefType based on shape, element type, affine
505  /// map composition, and memory space declared at the given location.
506  /// If the location is unknown, the last argument should be an instance of
507  /// UnknownLoc. If the MemRefType defined by the arguments would be
508  /// ill-formed, emits errors (to the handler registered with the context or to
509  /// the error stream) and returns nullptr.
510  static MemRefType getChecked(ArrayRef<int64_t> shape, Type elementType,
511  ArrayRef<AffineMap> affineMapComposition,
512  unsigned memorySpace, Location location);
513 
514  ArrayRef<int64_t> getShape() const;
515 
516  /// Returns an array of affine map pointers representing the memref affine
517  /// map composition.
518  ArrayRef<AffineMap> getAffineMaps() const;
519 
520  /// Returns the memory space in which data referred to by this memref resides.
521  unsigned getMemorySpace() const;
522 
523  // TODO: merge these two special values in a single one used everywhere.
524  // Unfortunately, uses of `-1` have crept deep into the codebase now and are
525  // hard to track.
526  static int64_t getDynamicStrideOrOffset() {
527  return ShapedType::kDynamicStrideOrOffset;
528  }
529 
530 private:
531  /// Get or create a new MemRefType defined by the arguments. If the resulting
532  /// type would be ill-formed, return nullptr. If the location is provided,
533  /// emit detailed error messages.
534  static MemRefType getImpl(ArrayRef<int64_t> shape, Type elementType,
535  ArrayRef<AffineMap> affineMapComposition,
536  unsigned memorySpace, Optional<Location> location);
537  using Base::getImpl;

638 
639 /// Returns the strides of the MemRef if the layout map is in strided form.
640 /// MemRefs with layout maps in strided form include:
641 /// 1. empty or identity layout map, in which case the stride information is
642 /// the canonical form computed from sizes;
643 /// 2. single affine map layout of the form `K + k0 * d0 + ... kn * dn`,
644 /// where K and ki's are constants or symbols.
645 ///
646 /// A stride specification is a list of integer values that are either static
647 /// or dynamic (encoded with getDynamicStrideOrOffset()). Strides encode the
648 /// distance in the number of elements between successive entries along a
649 /// particular dimension. For example, `memref<42x16xf32, (64 * d0 + d1)>`
650 /// specifies a view into a non-contiguous memory region of `42` by `16` `f32`
651 /// elements in which the distance between two consecutive elements along the
652 /// outer dimension is `1` and the distance between two consecutive elements
653 /// along the inner dimension is `64`.
654 ///
655 /// If a simple strided form cannot be extracted from the composition of the
656 /// layout map, returns llvm::None.
657 ///
658 /// The convention is that the strides for dimensions d0, .. dn appear in
659 /// order to make indexing intuitive into the result.
660 LogicalResult getStridesAndOffset(MemRefType t,
661  SmallVectorImpl<int64_t> &strides,
662  int64_t &offset);
663 LogicalResult getStridesAndOffset(MemRefType t,
664  SmallVectorImpl<AffineExpr> &strides,
665  AffineExpr &offset);
666 
667 /// Given a list of strides (in which MemRefType::getDynamicStrideOrOffset()
668 /// represents a dynamic value), return the single result AffineMap which
669 /// represents the linearized strided layout map. Dimensions correspond to the
670 /// offset followed by the strides in order. Symbols are inserted for each
671 /// dynamic dimension in order. A stride cannot take value `0`.
672 ///
673 /// Examples:
674 /// =========
675 ///
676 /// 1. For offset: 0 strides: ?, ?, 1 return
677 /// (i, j, k)[M, N]->(M * i + N * j + k)
678 ///
679 /// 2. For offset: 3 strides: 32, ?, 16 return
680 /// (i, j, k)[M]->(3 + 32 * i + M * j + 16 * k)
681 ///
682 /// 3. For offset: ? strides: ?, ?, ? return
