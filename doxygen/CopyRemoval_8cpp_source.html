23 /// This pass removes the redundant Copy operations. Additionally, it
24 /// removes the leftover definition and deallocation operations by erasing the
25 /// copy operation.
26 class CopyRemovalPass : public PassWrapper<CopyRemovalPass, OperationPass<>> {
27 public:
28  void runOnOperation() override {
29  getOperation()->walk([&](CopyOpInterface copyOp) {
30  reuseCopySourceAsTarget(copyOp);
31  reuseCopyTargetAsSource(copyOp);
32  });
33  for (Operation *op : eraseList)
34  op->erase();
35  }
36 
37 private:
38  /// List of operations that need to be removed.
39  DenseSet<Operation *> eraseList;
40 
41  /// Returns the deallocation operation for `value` in `block` if it exists.
42  Operation *getDeallocationInBlock(Value value, Block *block) {
43  assert(block && "Block cannot be null");
44  auto valueUsers = value.getUsers();
45  auto it = llvm::find_if(valueUsers, [&](Operation *op) {
46  auto effects = dyn_cast<MemoryEffectOpInterface>(op);
47  return effects && op->getBlock() == block && effects.hasEffect<Free>();
48  });
49  return (it == valueUsers.end() ? nullptr : *it);
50  }
51 
52  /// Returns true if an operation between start and end operations has memory
53  /// effect.
54  bool hasMemoryEffectOpBetween(Operation *start, Operation *end) {
55  assert((start || end) && "Start and end operations cannot be null");
56  assert(start->getBlock() == end->getBlock() &&
57  "Start and end operations should be in the same block.");
58  Operation *op = start->getNextNode();
59  while (op->isBeforeInBlock(end)) {
60  if (isa<MemoryEffectOpInterface>(op))
61  return true;
62  op = op->getNextNode();
63  }
64  return false;
65  };
66 
67  /// Returns true if `val` value has at least a user between `start` and
68  /// `end` operations.
69  bool hasUsersBetween(Value val, Operation *start, Operation *end) {
70  assert((start || end) && "Start and end operations cannot be null");
71  Block *block = start->getBlock();
72  assert(block == end->getBlock() &&
73  "Start and end operations should be in the same block.");
74  return llvm::any_of(val.getUsers(), [&](Operation *op) {
75  return op->getBlock() == block && start->isBeforeInBlock(op) &&
76  op->isBeforeInBlock(end);
77  });
78  };
79 
80  bool areOpsInTheSameBlock(ArrayRef<Operation *> operations) {
81  assert(!operations.empty() &&
82  "The operations list should contain at least a single operation");
83  Block *block = operations.front()->getBlock();
84  return llvm::none_of(
85  operations, [&](Operation *op) { return block != op->getBlock(); });
86  }

88  /// Input:
89  /// func(){
90  /// %from = alloc()
91  /// write_to(%from)
92  /// %to = alloc()
93  /// copy(%from,%to)
94  /// dealloc(%from)
95  /// return %to
96  /// }
97  ///
98  /// Output:
99  /// func(){
100  /// %from = alloc()
101  /// write_to(%from)
102  /// return %from
103  /// }
104  /// Constraints:
105  /// 1) %to, copy and dealloc must all be defined and lie in the same block.
106  /// 2) This transformation cannot be applied if there is a single user/alias
107  /// of `to` value between the defining operation of `to` and the copy
108  /// operation.
109  /// 3) This transformation cannot be applied if there is a single user/alias
110  /// of `from` value between the copy operation and the deallocation of `from`.
111  /// TODO: Alias analysis is not available at the moment. Currently, we check
112  /// if there are any operations with memory effects between copy and
113  /// deallocation operations.
114  void reuseCopySourceAsTarget(CopyOpInterface copyOp) {
115  if (eraseList.count(copyOp))
116  return;
117 
118  Value from = copyOp.getSource();
119  Value to = copyOp.getTarget();
120 
121  Operation *copy = copyOp.getOperation();
122  Operation *fromDefiningOp = from.getDefiningOp();
123  Operation *fromFreeingOp = getDeallocationInBlock(from, copy->getBlock());
124  Operation *toDefiningOp = to.getDefiningOp();
125  if (!fromDefiningOp || !fromFreeingOp || !toDefiningOp ||
126  !areOpsInTheSameBlock({fromFreeingOp, toDefiningOp, copy}) ||
127  hasUsersBetween(to, toDefiningOp, copy) ||
128  hasUsersBetween(from, copy, fromFreeingOp) ||
129  hasMemoryEffectOpBetween(copy, fromFreeingOp))
130  return;
131 
132  to.replaceAllUsesWith(from);
133  eraseList.insert(copy);
134  eraseList.insert(toDefiningOp);
135  eraseList.insert(fromFreeingOp);
136  }
137 
138  /// Input:
139  /// func(){
140  /// %to = alloc()
141  /// %from = alloc()
142  /// write_to(%from)
143  /// copy(%from,%to)
144  /// dealloc(%from)
145  /// return %to
146  /// }
147  ///
148  /// Output:
149  /// func(){
150  /// %to = alloc()
151  /// write_to(%to)
152  /// return %to
153  /// }
154  /// Constraints:
155  /// 1) %from, copy and dealloc must all be defined and lie in the same block.
156  /// 2) This transformation cannot be applied if there is a single user/alias
157  /// of `to` value between the defining operation of `from` and the copy
158  /// operation.
159  /// 3) This transformation cannot be applied if there is a single user/alias
160  /// of `from` value between the copy operation and the deallocation of `from`.
161  /// TODO: Alias analysis is not available at the moment. Currently, we check
162  /// if there are any operations with memory effects between copy and
163  /// deallocation operations.
164  void reuseCopyTargetAsSource(CopyOpInterface copyOp) {
165  if (eraseList.count(copyOp))
166  return;
167 
168  Value from = copyOp.getSource();
169  Value to = copyOp.getTarget();
170 
171  Operation *copy = copyOp.getOperation();
172  Operation *fromDefiningOp = from.getDefiningOp();
173  Operation *fromFreeingOp = getDeallocationInBlock(from, copy->getBlock());
174  if (!fromDefiningOp || !fromFreeingOp ||
175  !areOpsInTheSameBlock({fromFreeingOp, fromDefiningOp, copy}) ||
176  hasUsersBetween(to, fromDefiningOp, copy) ||
177  hasUsersBetween(from, copy, fromFreeingOp) ||
178  hasMemoryEffectOpBetween(copy, fromFreeingOp))
179  return;
180 
181  from.replaceAllUsesWith(to);
182  eraseList.insert(copy);
183  eraseList.insert(fromDefiningOp);
184  eraseList.insert(fromFreeingOp);
185  }
186 };
