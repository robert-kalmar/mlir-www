98 // DmaStartOp starts a non-blocking DMA operation that transfers data from a
99 // source memref to a destination memref. The source and destination memref need
100 // not be of the same dimensionality, but need to have the same elemental type.
101 // The operands include the source and destination memref's each followed by its
102 // indices, size of the data transfer in terms of the number of elements (of the
103 // elemental type of the memref), a tag memref with its indices, and optionally
104 // at the end, a stride and a number_of_elements_per_stride arguments. The tag
105 // location is used by a DmaWaitOp to check for completion. The indices of the
106 // source memref, destination memref, and the tag memref have the same
107 // restrictions as any load/store. The optional stride arguments should be of
108 // 'index' type, and specify a stride for the slower memory space (memory space
109 // with a lower memory space id), transferring chunks of
110 // number_of_elements_per_stride every stride until %num_elements are
111 // transferred. Either both or no stride arguments should be specified.
112 //
113 // For example, a DmaStartOp operation that transfers 256 elements of a memref
114 // '%src' in memory space 0 at indices [%i, %j] to memref '%dst' in memory space
115 // 1 at indices [%k, %l], would be specified as follows:
116 //
117 // %num_elements = constant 256
118 // %idx = constant 0 : index
119 // %tag = alloc() : memref<1 x i32, (d0) -> (d0), 4>
120 // dma_start %src[%i, %j], %dst[%k, %l], %num_elements, %tag[%idx] :
121 // memref<40 x 128 x f32>, (d0) -> (d0), 0>,
122 // memref<2 x 1024 x f32>, (d0) -> (d0), 1>,
123 // memref<1 x i32>, (d0) -> (d0), 2>
124 //
125 // If %stride and %num_elt_per_stride are specified, the DMA is expected to
126 // transfer %num_elt_per_stride elements every %stride elements apart from
127 // memory space 0 until %num_elements are transferred.
128 //
129 // dma_start %src[%i, %j], %dst[%k, %l], %num_elements, %tag[%idx], %stride,
130 // %num_elt_per_stride :
131 //
132 // TODO: add additional operands to allow source and destination striding, and
133 // multiple stride levels.
134 // TODO: Consider replacing src/dst memref indices with view memrefs.
135 class DmaStartOp
136  : public Op<DmaStartOp, OpTrait::VariadicOperands, OpTrait::ZeroResult> {
137 public:
138  using Op::Op;
139 
140  static void build(OpBuilder &builder, OperationState &result, Value srcMemRef,
141  ValueRange srcIndices, Value destMemRef,
142  ValueRange destIndices, Value numElements, Value tagMemRef,
143  ValueRange tagIndices, Value stride = nullptr,
144  Value elementsPerStride = nullptr);

146  // Returns the source MemRefType for this DMA operation.
147  Value getSrcMemRef() { return getOperand(0); }
148  // Returns the rank (number of indices) of the source MemRefType.
149  unsigned getSrcMemRefRank() {
150  return getSrcMemRef().getType().cast<MemRefType>().getRank();
151  }
152  // Returns the source memref indices for this DMA operation.
153  operand_range getSrcIndices() {
154  return {getOperation()->operand_begin() + 1,
155  getOperation()->operand_begin() + 1 + getSrcMemRefRank()};
156  }
157 
158  // Returns the destination MemRefType for this DMA operations.
159  Value getDstMemRef() { return getOperand(1 + getSrcMemRefRank()); }
160  // Returns the rank (number of indices) of the destination MemRefType.
161  unsigned getDstMemRefRank() {
162  return getDstMemRef().getType().cast<MemRefType>().getRank();
163  }
164  unsigned getSrcMemorySpace() {
165  return getSrcMemRef().getType().cast<MemRefType>().getMemorySpace();
166  }
167  unsigned getDstMemorySpace() {
168  return getDstMemRef().getType().cast<MemRefType>().getMemorySpace();
169  }
170 
171  // Returns the destination memref indices for this DMA operation.
172  operand_range getDstIndices() {
173  return {getOperation()->operand_begin() + 1 + getSrcMemRefRank() + 1,
174  getOperation()->operand_begin() + 1 + getSrcMemRefRank() + 1 +
175  getDstMemRefRank()};
176  }
177 
178  // Returns the number of elements being transferred by this DMA operation.
179  Value getNumElements() {
180  return getOperand(1 + getSrcMemRefRank() + 1 + getDstMemRefRank());
181  }
182 
183  // Returns the Tag MemRef for this DMA operation.
184  Value getTagMemRef() {
185  return getOperand(1 + getSrcMemRefRank() + 1 + getDstMemRefRank() + 1);
186  }
187  // Returns the rank (number of indices) of the tag MemRefType.
188  unsigned getTagMemRefRank() {
189  return getTagMemRef().getType().cast<MemRefType>().getRank();
190  }
191 
192  // Returns the tag memref index for this DMA operation.
193  operand_range getTagIndices() {
194  unsigned tagIndexStartPos =
195  1 + getSrcMemRefRank() + 1 + getDstMemRefRank() + 1 + 1;
196  return {getOperation()->operand_begin() + tagIndexStartPos,
197  getOperation()->operand_begin() + tagIndexStartPos +
198  getTagMemRefRank()};
199  }
200 
201  /// Returns true if this is a DMA from a faster memory space to a slower one.
202  bool isDestMemorySpaceFaster() {
203  return (getSrcMemorySpace() < getDstMemorySpace());
204  }
205 
206  /// Returns true if this is a DMA from a slower memory space to a faster one.
207  bool isSrcMemorySpaceFaster() {
208  // Assumes that a lower number is for a slower memory space.
209  return (getDstMemorySpace() < getSrcMemorySpace());
210  }
211 
212  /// Given a DMA start operation, returns the operand position of either the
213  /// source or destination memref depending on the one that is at the higher
214  /// level of the memory hierarchy. Asserts failure if neither is true.
215  unsigned getFasterMemPos() {
216  assert(isSrcMemorySpaceFaster() || isDestMemorySpaceFaster());
217  return isSrcMemorySpaceFaster() ? 0 : getSrcMemRefRank() + 1;
218  }

246 // DmaWaitOp blocks until the completion of a DMA operation associated with the
247 // tag element '%tag[%index]'. %tag is a memref, and %index has to be an index
248 // with the same restrictions as any load/store index. %num_elements is the
249 // number of elements associated with the DMA operation. For example:
250 //
251 // dma_start %src[%i, %j], %dst[%k, %l], %num_elements, %tag[%index] :
252 // memref<2048 x f32>, (d0) -> (d0), 0>,
253 // memref<256 x f32>, (d0) -> (d0), 1>
254 // memref<1 x i32>, (d0) -> (d0), 2>
255 // ...
256 // ...

258 //

305 /// Determines whether MemRefCastOp casts to a more dynamic version of the
306 /// source memref. This is useful to to fold a memref_cast into a consuming op
307 /// and implement canonicalization patterns for ops in different dialects that
308 /// may consume the results of memref_cast operations. Such foldable memref_cast
309 /// operations are typically inserted as `view` and `subview` ops are
310 /// canonicalized, to preserve the type compatibility of their uses.
311 ///
312 /// Returns true when all conditions are met:
313 /// 1. source and result are ranked memrefs with strided semantics and same
314 /// element type and rank.
315 /// 2. each of the source's size, offset or stride has more static information
316 /// than the corresponding result's size, offset or stride.
317 ///
318 /// Example 1:
319 /// ```mlir
320 /// %1 = memref_cast %0 : memref<8x16xf32> to memref<?x?xf32>
321 /// %2 = consumer %1 ... : memref<?x?xf32> ...
322 /// ```
323 ///
324 /// may fold into:
325 ///
326 /// ```mlir
327 /// %2 = consumer %0 ... : memref<8x16xf32> ...
328 /// ```
329 ///
330 /// Example 2:
331 /// ```
332 /// %1 = memref_cast %0 : memref<?x16xf32, affine_map<(i, j)->(16 * i + j)>>
333 /// to memref<?x?xf32>
334 /// consumer %1 : memref<?x?xf32> ...
335 /// ```
336 ///
337 /// may fold into:
338 ///
339 /// ```
340 /// consumer %0 ... : memref<?x16xf32, affine_map<(i, j)->(16 * i + j)>>
341 /// ```
342 bool canFoldIntoConsumerOp(MemRefCastOp castOp);
