35 ArrayRef<Extension> spirv::getImpliedExtensions(Version version) {
36  // Note: the following lists are from "Appendix A: Changes" of the spec.
37 
38 #define V_1_3_IMPLIED_EXTS \
39  Extension::SPV_KHR_shader_draw_parameters, Extension::SPV_KHR_16bit_storage, \
40  Extension::SPV_KHR_device_group, Extension::SPV_KHR_multiview, \
41  Extension::SPV_KHR_storage_buffer_storage_class, \
42  Extension::SPV_KHR_variable_pointers
43 
44 #define V_1_4_IMPLIED_EXTS \
45  Extension::SPV_KHR_no_integer_wrap_decoration, \
46  Extension::SPV_GOOGLE_decorate_string, \
47  Extension::SPV_GOOGLE_hlsl_functionality1, \
48  Extension::SPV_KHR_float_controls
49 
50 #define V_1_5_IMPLIED_EXTS \
51  Extension::SPV_KHR_8bit_storage, Extension::SPV_EXT_descriptor_indexing, \
52  Extension::SPV_EXT_shader_viewport_index_layer, \
53  Extension::SPV_EXT_physical_storage_buffer, \
54  Extension::SPV_KHR_physical_storage_buffer, \
55  Extension::SPV_KHR_vulkan_memory_model
56 
57  switch (version) {
58  default:
59  return {};
60  case Version::V_1_3: {
61  // The following manual ArrayRef constructor call is to satisfy GCC 5.
62  static const Extension exts[] = {V_1_3_IMPLIED_EXTS};
63  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
64  }
65  case Version::V_1_4: {
66  static const Extension exts[] = {V_1_3_IMPLIED_EXTS, V_1_4_IMPLIED_EXTS};
67  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
68  }
69  case Version::V_1_5: {
70  static const Extension exts[] = {V_1_3_IMPLIED_EXTS, V_1_4_IMPLIED_EXTS,
71  V_1_5_IMPLIED_EXTS};
72  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
73  }
74  }
75 
76 #undef V_1_5_IMPLIED_EXTS
77 #undef V_1_4_IMPLIED_EXTS
78 #undef V_1_3_IMPLIED_EXTS
79 }

610 void ScalarType::getExtensions(SPIRVType::ExtensionArrayRefVector &extensions,
611  Optional<StorageClass> storage) {
612  // 8- or 16-bit integer/floating-point numbers will require extra extensions
613  // to appear in interface storage classes. See SPV_KHR_16bit_storage and
614  // SPV_KHR_8bit_storage for more details.
615  if (!storage)
616  return;
617 
618  switch (*storage) {
619  case StorageClass::PushConstant:
620  case StorageClass::StorageBuffer:
621  case StorageClass::Uniform:
622  if (getIntOrFloatBitWidth() == 8) {
623  static const Extension exts[] = {Extension::SPV_KHR_8bit_storage};
624  ArrayRef<Extension> ref(exts, llvm::array_lengthof(exts));
625  extensions.push_back(ref);
626  }
627  LLVM_FALLTHROUGH;
628  case StorageClass::Input:
629  case StorageClass::Output:
630  if (getIntOrFloatBitWidth() == 16) {
631  static const Extension exts[] = {Extension::SPV_KHR_16bit_storage};
632  ArrayRef<Extension> ref(exts, llvm::array_lengthof(exts));
633  extensions.push_back(ref);
634  }
635  break;
636  default:
637  break;
638  }

641 void ScalarType::getCapabilities(
642  SPIRVType::CapabilityArrayRefVector &capabilities,
643  Optional<StorageClass> storage) {
644  unsigned bitwidth = getIntOrFloatBitWidth();
645 
646  // 8- or 16-bit integer/floating-point numbers will require extra capabilities
647  // to appear in interface storage classes. See SPV_KHR_16bit_storage and
648  // SPV_KHR_8bit_storage for more details.
649 
650 #define STORAGE_CASE(storage, cap8, cap16) \
651  case StorageClass::storage: { \
652  if (bitwidth == 8) { \
653  static const Capability caps[] = {Capability::cap8}; \
654  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps)); \
655  capabilities.push_back(ref); \
656  } else if (bitwidth == 16) { \

667  if (storage) {

669  STORAGE_CASE(PushConstant, StoragePushConstant8, StoragePushConstant16);
670  STORAGE_CASE(StorageBuffer, StorageBuffer8BitAccess,
671  StorageBuffer16BitAccess);
672  STORAGE_CASE(Uniform, UniformAndStorageBuffer8BitAccess,
673  StorageUniform16);
674  case StorageClass::Input:
675  case StorageClass::Output: {

678  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps));

695  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps)); \
696  capabilities.push_back(ref); \

701  case 32:
702  case 1:
703  break;

705  WIDTH_CASE(Int, 16);
706  WIDTH_CASE(Int, 64);
707  default:
708  llvm_unreachable("invalid bitwidth to getCapabilities");
709  }
710  } else {
711  assert(isa<FloatType>());
712  switch (bitwidth) {
713  case 32:
714  break;
715  WIDTH_CASE(Float, 16);
716  WIDTH_CASE(Float, 64);
717  default:
718  llvm_unreachable("invalid bitwidth to getCapabilities");
