87 /// A utility function to check if a value is defined at the top level of an
88 /// op with trait `AffineScope`. If the value is defined in an unlinked region,
89 /// conservatively assume it is not top-level. A value of index type defined at
90 /// the top level is always a valid symbol.
91 bool mlir::isTopLevelValue(Value value) {
92  if (auto arg = value.dyn_cast<BlockArgument>()) {
93  // The block owning the argument may be unlinked, e.g. when the surrounding
94  // region has not yet been attached to an Op, at which point the parent Op
95  // is null.
96  Operation *parentOp = arg.getOwner()->getParentOp();
97  return parentOp && parentOp->hasTrait<OpTrait::AffineScope>();
98  }
99  // The defining Op may live in an unlinked block so its parent Op may be null.
100  Operation *parentOp = value.getDefiningOp()->getParentOp();
101  return parentOp && parentOp->hasTrait<OpTrait::AffineScope>();
102 }

114 /// Returns the closest region enclosing `op` that is held by an operation with
115 /// trait `AffineScope`; `nullptr` if there is no such region.
116 // TODO: getAffineScope should be publicly exposed for affine passes/utilities.

127 // A Value can be used as a dimension id iff it meets one of the following
128 // conditions:
129 // *) It is valid as a symbol.
130 // *) It is an induction variable.
131 // *) It is the result of affine apply operation with dimension id arguments.
132 bool mlir::isValidDim(Value value) {
133  // The value must be an index type.
134  if (!value.getType().isIndex())
135  return false;
136 
137  if (auto *defOp = value.getDefiningOp())
138  return isValidDim(value, getAffineScope(defOp));
139 
140  // This value has to be a block argument for an op that has the
141  // `AffineScope` trait or for an affine.for or affine.parallel.
142  auto *parentOp = value.cast<BlockArgument>().getOwner()->getParentOp();
143  return parentOp && (parentOp->hasTrait<OpTrait::AffineScope>() ||
144  isa<AffineForOp, AffineParallelOp>(parentOp));
145 }

147 // Value can be used as a dimension id iff it meets one of the following
148 // conditions:
149 // *) It is valid as a symbol.
150 // *) It is an induction variable.
151 // *) It is the result of an affine apply operation with dimension id operands.
152 bool mlir::isValidDim(Value value, Region *region) {
153  // The value must be an index type.
154  if (!value.getType().isIndex())
155  return false;
156 
157  // All valid symbols are okay.
158  if (isValidSymbol(value, region))
159  return true;
160 
161  auto *op = value.getDefiningOp();
162  if (!op) {
163  // This value has to be a block argument for an affine.for or an
164  // affine.parallel.
165  auto *parentOp = value.cast<BlockArgument>().getOwner()->getParentOp();
166  return isa<AffineForOp, AffineParallelOp>(parentOp);
167  }
168 
169  // Affine apply operation is ok if all of its operands are ok.
170  if (auto applyOp = dyn_cast<AffineApplyOp>(op))
171  return applyOp.isValidDim(region);
172  // The dim op is okay if its operand memref/tensor is defined at the top
173  // level.
174  if (auto dimOp = dyn_cast<DimOp>(op))
175  return isTopLevelValue(dimOp.memrefOrTensor());
176  return false;
177 }

195 /// Returns true if the result of the dim op is a valid symbol for `region`.
196 static bool isDimOpValidSymbol(DimOp dimOp, Region *region) {
197  // The dim op is okay if its operand memref/tensor is defined at the top
198  // level.
199  if (isTopLevelValue(dimOp.memrefOrTensor()))

214 // A value can be used as a symbol (at all its use sites) iff it meets one of
215 // the following conditions:
216 // *) It is a constant.
217 // *) Its defining op or block arg appearance is immediately enclosed by an op
218 // with `AffineScope` trait.
219 // *) It is the result of an affine.apply operation with symbol operands.
220 // *) It is a result of the dim op on a memref whose corresponding size is a
221 // valid symbol.
222 bool mlir::isValidSymbol(Value value) {
223  // The value must be an index type.
224  if (!value.getType().isIndex())
225  return false;
226 
227  // Check that the value is a top level value.
228  if (isTopLevelValue(value))
229  return true;
230 
231  if (auto *defOp = value.getDefiningOp())
232  return isValidSymbol(value, getAffineScope(defOp));
233 
234  return false;
235 }

237 /// A value can be used as a symbol for `region` iff it meets onf of the the
238 /// following conditions:
239 /// *) It is a constant.
240 /// *) It is the result of an affine apply operation with symbol arguments.
241 /// *) It is a result of the dim op on a memref whose corresponding size is
242 /// a valid symbol.
243 /// *) It is defined at the top level of 'region' or is its argument.
244 /// *) It dominates `region`'s parent op.
245 /// If `region` is null, conservatively assume the symbol definition scope does
246 /// not exist and only accept the values that would be symbols regardless of
247 /// the surrounding region structure, i.e. the first three cases above.
248 bool mlir::isValidSymbol(Value value, Region *region) {
249  // The value must be an index type.
250  if (!value.getType().isIndex())
251  return false;
252 
253  // A top-level value is a valid symbol.
254  if (region && ::isTopLevelValue(value, region))
255  return true;
256 
257  auto *defOp = value.getDefiningOp();
258  if (!defOp) {
259  // A block argument that is not a top-level value is a valid symbol if it
260  // dominates region's parent op.
261  if (region && !region->getParentOp()->isKnownIsolatedFromAbove())
262  if (auto *parentOpRegion = region->getParentOp()->getParentRegion())
263  return isValidSymbol(value, parentOpRegion);
264  return false;
265  }
266 
267  // Constant operation is ok.
268  Attribute operandCst;
269  if (matchPattern(defOp, m_Constant(&operandCst)))
270  return true;
271 
272  // Affine apply operation is ok if all of its operands are ok.
273  if (auto applyOp = dyn_cast<AffineApplyOp>(defOp))
274  return applyOp.isValidSymbol(region);
275 
276  // Dim op results could be valid symbols at any level.
277  if (auto dimOp = dyn_cast<DimOp>(defOp))
278  return isDimOpValidSymbol(dimOp, region);
279 
280  // Check for values dominating `region`'s parent op.
281  if (region && !region->getParentOp()->isKnownIsolatedFromAbove())
282  if (auto *parentRegion = region->getParentOp()->getParentRegion())
283  return isValidSymbol(value, parentRegion);
284 
285  return false;
286 }

295 /// Utility function to verify that a set of operands are valid dimension and
296 /// symbol identifiers. The operands should be laid out such that the dimension
297 /// operands are before the symbol operands. This function returns failure if
298 /// there was an invalid operand. An operation is provided to emit any necessary
299 /// errors.
300 template <typename OpTy>
301 static LogicalResult
302 verifyDimAndSymbolIdentifiers(OpTy &op, Operation::operand_range operands,
303  unsigned numDims) {
304  unsigned opIt = 0;
305  for (auto operand : operands) {
306  if (opIt++ < numDims) {
307  if (!isValidDim(operand, getAffineScope(op)))
308  return op.emitOpError("operand cannot be used as a dimension id");
309  } else if (!isValidSymbol(operand, getAffineScope(op))) {
310  return op.emitOpError("operand cannot be used as a symbol");
311  }
312  }
313  return success();
314 }

324 static ParseResult parseAffineApplyOp(OpAsmParser &parser,
325  OperationState &result) {
326  auto &builder = parser.getBuilder();
327  auto indexTy = builder.getIndexType();
328 
329  AffineMapAttr mapAttr;
330  unsigned numDims;
331  if (parser.parseAttribute(mapAttr, "map", result.attributes) ||
332  parseDimAndSymbolList(parser, result.operands, numDims) ||
333  parser.parseOptionalAttrDict(result.attributes))
334  return failure();
335  auto map = mapAttr.getValue();
336 
337  if (map.getNumDims() != numDims ||
338  numDims + map.getNumSymbols() != result.operands.size()) {
339  return parser.emitError(parser.getNameLoc(),
340  "dimension or symbol index mismatch");
341  }
342 
343  result.types.append(map.getNumResults(), indexTy);
344  return success();
345 }

463 // Support the special case of a symbol coming from an AffineApplyOp that needs
464 // to be composed into the current AffineApplyOp.
465 // This case is handled by rewriting all such symbols into dims for the purpose
466 // of allowing mathematical AffineMap composition.
467 // Returns an AffineMap where symbols that come from an AffineApplyOp have been
468 // rewritten as dims and are ordered after the original dims.
469 // TODO: This promotion makes AffineMap lose track of which
470 // symbols are represented as dims. This loss is static but can still be
471 // recovered dynamically (with `isValidSymbol`). Still this is annoying for the
472 // semi-affine map case. A dynamic canonicalization of all dims that are valid
473 // symbols (a.k.a `canonicalizePromotedSymbols`) into symbols helps and even
474 // results in better simplifications and foldings. But we should evaluate
475 // whether this behavior is what we really want after using more.
476 static AffineMap promoteComposedSymbolsAsDims(AffineMap map,
477  ArrayRef<Value> symbols) {
478  if (symbols.empty()) {
479  return map;
480  }
481 
482  // Sanity check on symbols.
483  for (auto sym : symbols) {
484  assert(isValidSymbol(sym) && "Expected only valid symbols");
485  (void)sym;
486  }
487 
488  // Extract the symbol positions that come from an AffineApplyOp and
489  // needs to be rewritten as dims.
490  auto symPositions = indicesFromAffineApplyOp(symbols);
491  if (symPositions.empty()) {
492  return map;
493  }
494 
495  // Create the new map by replacing each symbol at pos by the next new dim.
496  unsigned numDims = map.getNumDims();
497  unsigned numSymbols = map.getNumSymbols();
498  unsigned numNewDims = 0;
499  unsigned numNewSymbols = 0;
500  SmallVector<AffineExpr, 8> symReplacements(numSymbols);
501  for (unsigned i = 0; i < numSymbols; ++i) {
502  symReplacements[i] =
503  symPositions.count(i) > 0
504  ? getAffineDimExpr(numDims + numNewDims++, map.getContext())
505  : getAffineSymbolExpr(numNewSymbols++, map.getContext());
506  }
507  assert(numSymbols >= numNewDims);
508  AffineMap newMap = map.replaceDimsAndSymbols(
509  {}, symReplacements, numDims + numNewDims, numNewSymbols);
510 
511  return newMap;
512 }

514 /// The AffineNormalizer composes AffineApplyOp recursively. Its purpose is to
515 /// keep a correspondence between the mathematical `map` and the `operands` of
516 /// a given AffineApplyOp. This correspondence is maintained by iterating over
517 /// the operands and forming an `auxiliaryMap` that can be composed
518 /// mathematically with `map`. To keep this correspondence in cases where
519 /// symbols are produced by affine.apply operations, we perform a local rewrite
520 /// of symbols as dims.
521 ///
522 /// Rationale for locally rewriting symbols as dims:
523 /// ================================================
524 /// The mathematical composition of AffineMap must always concatenate symbols
525 /// because it does not have enough information to do otherwise. For example,
526 /// composing `(d0)[s0] -> (d0 + s0)` with itself must produce
527 /// `(d0)[s0, s1] -> (d0 + s0 + s1)`.
528 ///
529 /// The result is only equivalent to `(d0)[s0] -> (d0 + 2 * s0)` when
530 /// applied to the same mlir::Value for both s0 and s1.
531 /// As a consequence mathematical composition of AffineMap always concatenates
532 /// symbols.
533 ///
534 /// When AffineMaps are used in AffineApplyOp however, they may specify
535 /// composition via symbols, which is ambiguous mathematically. This corner case
536 /// is handled by locally rewriting such symbols that come from AffineApplyOp
537 /// into dims and composing through dims.
538 /// TODO: Composition via symbols comes at a significant code
539 /// complexity. Alternatively we should investigate whether we want to
540 /// explicitly disallow symbols coming from affine.apply and instead force the
541 /// user to compose symbols beforehand. The annoyances may be small (i.e. 1 or 2
542 /// extra API calls for such uses, which haven't popped up until now) and the
543 /// benefit potentially big: simpler and more maintainable code for a
544 /// non-trivial, recursive, procedure.
545 AffineApplyNormalizer::AffineApplyNormalizer(AffineMap map,
546  ArrayRef<Value> operands)
547  : AffineApplyNormalizer() {
548  static_assert(kMaxAffineApplyDepth > 0, "kMaxAffineApplyDepth must be > 0");
549  assert(map.getNumInputs() == operands.size() &&
550  "number of operands does not match the number of map inputs");
551 
552  LLVM_DEBUG(map.print(dbgs() << "\nInput map: "));
553 
554  // Promote symbols that come from an AffineApplyOp to dims by rewriting the
555  // map to always refer to:
556  // (dims, symbols coming from AffineApplyOp, other symbols).
557  // The order of operands can remain unchanged.
558  // This is a simplification that relies on 2 ordering properties:
559  // 1. rewritten symbols always appear after the original dims in the map;
560  // 2. operands are traversed in order and either dispatched to:
561  // a. auxiliaryExprs (dims and symbols rewritten as dims);
562  // b. concatenatedSymbols (all other symbols)
563  // This allows operand order to remain unchanged.
564  unsigned numDimsBeforeRewrite = map.getNumDims();
565  map = promoteComposedSymbolsAsDims(map,
566  operands.take_back(map.getNumSymbols()));
567 
568  LLVM_DEBUG(map.print(dbgs() << "\nRewritten map: "));
569 
570  SmallVector<AffineExpr, 8> auxiliaryExprs;
571  bool furtherCompose = (affineApplyDepth() <= kMaxAffineApplyDepth);
572  // We fully spell out the 2 cases below. In this particular instance a little
573  // code duplication greatly improves readability.
574  // Note that the first branch would disappear if we only supported full
575  // composition (i.e. infinite kMaxAffineApplyDepth).
576  if (!furtherCompose) {
577  // 1. Only dispatch dims or symbols.
578  for (auto en : llvm::enumerate(operands)) {
579  auto t = en.value();
580  assert(t.getType().isIndex());
581  bool isDim = (en.index() < map.getNumDims());
582  if (isDim) {
583  // a. The mathematical composition of AffineMap composes dims.
584  auxiliaryExprs.push_back(renumberOneDim(t));
585  } else {
586  // b. The mathematical composition of AffineMap concatenates symbols.
587  // We do the same for symbol operands.
588  concatenatedSymbols.push_back(t);
589  }
590  }
591  } else {
592  assert(numDimsBeforeRewrite <= operands.size());
593  // 2. Compose AffineApplyOps and dispatch dims or symbols.
594  for (unsigned i = 0, e = operands.size(); i < e; ++i) {
595  auto t = operands[i];
596  auto affineApply = t.getDefiningOp<AffineApplyOp>();
597  if (affineApply) {
598  // a. Compose affine.apply operations.
599  LLVM_DEBUG(affineApply.getOperation()->print(
600  dbgs() << "\nCompose AffineApplyOp recursively: "));
601  AffineMap affineApplyMap = affineApply.getAffineMap();
602  SmallVector<Value, 8> affineApplyOperands(
603  affineApply.getOperands().begin(), affineApply.getOperands().end());
604  AffineApplyNormalizer normalizer(affineApplyMap, affineApplyOperands);
605 
606  LLVM_DEBUG(normalizer.affineMap.print(
607  dbgs() << "\nRenumber into current normalizer: "));
608 
609  auto renumberedMap = renumber(normalizer);
610 
611  LLVM_DEBUG(
612  renumberedMap.print(dbgs() << "\nRecursive composition yields: "));
613 
614  auxiliaryExprs.push_back(renumberedMap.getResult(0));
615  } else {
616  if (i < numDimsBeforeRewrite) {
617  // b. The mathematical composition of AffineMap composes dims.
618  auxiliaryExprs.push_back(renumberOneDim(t));
619  } else {
620  // c. The mathematical composition of AffineMap concatenates symbols.
621  // Note that the map composition will put symbols already present
622  // in the map before any symbols coming from the auxiliary map, so
623  // we insert them before any symbols that are due to renumbering,
624  // and after the proper symbols we have seen already.
625  concatenatedSymbols.insert(
626  std::next(concatenatedSymbols.begin(), numProperSymbols++), t);
627  }
628  }
629  }
630  }
631 
632  // Early exit if `map` is already composed.
633  if (auxiliaryExprs.empty()) {
634  affineMap = map;
635  return;
636  }
637 
638  assert(concatenatedSymbols.size() >= map.getNumSymbols() &&
639  "Unexpected number of concatenated symbols");
640  auto numDims = dimValueToPosition.size();
641  auto numSymbols = concatenatedSymbols.size() - map.getNumSymbols();
642  auto auxiliaryMap =
643  AffineMap::get(numDims, numSymbols, auxiliaryExprs, map.getContext());
644 
645  LLVM_DEBUG(map.print(dbgs() << "\nCompose map: "));
646  LLVM_DEBUG(auxiliaryMap.print(dbgs() << "\nWith map: "));
647  LLVM_DEBUG(map.compose(auxiliaryMap).print(dbgs() << "\nResult: "));
648 
649  // TODO: Disabling simplification results in major speed gains.
650  // Another option is to cache the results as it is expected a lot of redundant
651  // work is performed in practice.
652  affineMap = simplifyAffineMap(map.compose(auxiliaryMap));
653 
654  LLVM_DEBUG(affineMap.print(dbgs() << "\nSimplified result: "));
655  LLVM_DEBUG(dbgs() << "\n");
656 }

702 // A symbol may appear as a dim in affine.apply operations. This function
703 // canonicalizes dims that are valid symbols into actual symbols.
704 template <class MapOrSet>
705 static void canonicalizePromotedSymbols(MapOrSet *mapOrSet,
706  SmallVectorImpl<Value> *operands) {
707  if (!mapOrSet || operands->empty())
708  return;
709 
710  assert(mapOrSet->getNumInputs() == operands->size() &&
711  "map/set inputs must match number of operands");
712 
713  auto *context = mapOrSet->getContext();
714  SmallVector<Value, 8> resultOperands;
715  resultOperands.reserve(operands->size());
716  SmallVector<Value, 8> remappedSymbols;
717  remappedSymbols.reserve(operands->size());
718  unsigned nextDim = 0;
719  unsigned nextSym = 0;
720  unsigned oldNumSyms = mapOrSet->getNumSymbols();
721  SmallVector<AffineExpr, 8> dimRemapping(mapOrSet->getNumDims());
722  for (unsigned i = 0, e = mapOrSet->getNumInputs(); i != e; ++i) {
723  if (i < mapOrSet->getNumDims()) {
724  if (isValidSymbol((*operands)[i])) {
725  // This is a valid symbol that appears as a dim, canonicalize it.
726  dimRemapping[i] = getAffineSymbolExpr(oldNumSyms + nextSym++, context);
727  remappedSymbols.push_back((*operands)[i]);
728  } else {
729  dimRemapping[i] = getAffineDimExpr(nextDim++, context);
730  resultOperands.push_back((*operands)[i]);
731  }
732  } else {
733  resultOperands.push_back((*operands)[i]);
734  }
735  }
736 
737  resultOperands.append(remappedSymbols.begin(), remappedSymbols.end());
738  *operands = resultOperands;
739  *mapOrSet = mapOrSet->replaceDimsAndSymbols(dimRemapping, {}, nextDim,
740  oldNumSyms + nextSym);
741 
742  assert(mapOrSet->getNumInputs() == operands->size() &&
743  "map/set inputs must match number of operands");
744 }

746 // Works for either an affine map or an integer set.
747 template <class MapOrSet>
748 static void canonicalizeMapOrSetAndOperands(MapOrSet *mapOrSet,
749  SmallVectorImpl<Value> *operands) {
750  static_assert(llvm::is_one_of<MapOrSet, AffineMap, IntegerSet>::value,
751  "Argument must be either of AffineMap or IntegerSet type");
752 
753  if (!mapOrSet || operands->empty())
754  return;
755 
756  assert(mapOrSet->getNumInputs() == operands->size() &&
757  "map/set inputs must match number of operands");
758 
759  canonicalizePromotedSymbols<MapOrSet>(mapOrSet, operands);
760 
761  // Check to see what dims are used.
762  llvm::SmallBitVector usedDims(mapOrSet->getNumDims());
763  llvm::SmallBitVector usedSyms(mapOrSet->getNumSymbols());
764  mapOrSet->walkExprs([&](AffineExpr expr) {
765  if (auto dimExpr = expr.dyn_cast<AffineDimExpr>())
766  usedDims[dimExpr.getPosition()] = true;
767  else if (auto symExpr = expr.dyn_cast<AffineSymbolExpr>())
768  usedSyms[symExpr.getPosition()] = true;
769  });
770 
771  auto *context = mapOrSet->getContext();
772 
773  SmallVector<Value, 8> resultOperands;
774  resultOperands.reserve(operands->size());
775 
776  llvm::SmallDenseMap<Value, AffineExpr, 8> seenDims;
777  SmallVector<AffineExpr, 8> dimRemapping(mapOrSet->getNumDims());
778  unsigned nextDim = 0;
779  for (unsigned i = 0, e = mapOrSet->getNumDims(); i != e; ++i) {
780  if (usedDims[i]) {
781  // Remap dim positions for duplicate operands.
782  auto it = seenDims.find((*operands)[i]);
783  if (it == seenDims.end()) {
784  dimRemapping[i] = getAffineDimExpr(nextDim++, context);
785  resultOperands.push_back((*operands)[i]);
786  seenDims.insert(std::make_pair((*operands)[i], dimRemapping[i]));
787  } else {
788  dimRemapping[i] = it->second;
789  }
790  }
791  }
792  llvm::SmallDenseMap<Value, AffineExpr, 8> seenSymbols;
793  SmallVector<AffineExpr, 8> symRemapping(mapOrSet->getNumSymbols());
794  unsigned nextSym = 0;
795  for (unsigned i = 0, e = mapOrSet->getNumSymbols(); i != e; ++i) {
796  if (!usedSyms[i])
797  continue;
798  // Handle constant operands (only needed for symbolic operands since
799  // constant operands in dimensional positions would have already been
800  // promoted to symbolic positions above).
801  IntegerAttr operandCst;
802  if (matchPattern((*operands)[i + mapOrSet->getNumDims()],
803  m_Constant(&operandCst))) {
804  symRemapping[i] =
805  getAffineConstantExpr(operandCst.getValue().getSExtValue(), context);
806  continue;
807  }
808  // Remap symbol positions for duplicate operands.
809  auto it = seenSymbols.find((*operands)[i + mapOrSet->getNumDims()]);
810  if (it == seenSymbols.end()) {
811  symRemapping[i] = getAffineSymbolExpr(nextSym++, context);
812  resultOperands.push_back((*operands)[i + mapOrSet->getNumDims()]);
813  seenSymbols.insert(std::make_pair((*operands)[i + mapOrSet->getNumDims()],
814  symRemapping[i]));
815  } else {
816  symRemapping[i] = it->second;
817  }
818  }
819  *mapOrSet = mapOrSet->replaceDimsAndSymbols(dimRemapping, symRemapping,
820  nextDim, nextSym);
821  *operands = resultOperands;
822 }

836 /// maps that supply results into them.
837 ///
838 template <typename AffineOpTy>
839 struct SimplifyAffineOp : public OpRewritePattern<AffineOpTy> {
840  using OpRewritePattern<AffineOpTy>::OpRewritePattern;
841 
842  /// Replace the affine op with another instance of it with the supplied
843  /// map and mapOperands.
844  void replaceAffineOp(PatternRewriter &rewriter, AffineOpTy affineOp,
845  AffineMap map, ArrayRef<Value> mapOperands) const;
846 
847  LogicalResult matchAndRewrite(AffineOpTy affineOp,
848  PatternRewriter &rewriter) const override {
849  static_assert(llvm::is_one_of<AffineOpTy, AffineLoadOp, AffinePrefetchOp,
850  AffineStoreOp, AffineApplyOp, AffineMinOp,
851  AffineMaxOp>::value,
852  "affine load/store/apply/prefetch/min/max op expected");
853  auto map = affineOp.getAffineMap();
854  AffineMap oldMap = map;
855  auto oldOperands = affineOp.getMapOperands();
856  SmallVector<Value, 8> resultOperands(oldOperands);
857  composeAffineMapAndOperands(&map, &resultOperands);
858  if (map == oldMap && std::equal(oldOperands.begin(), oldOperands.end(),
859  resultOperands.begin()))
860  return failure();
861 
862  replaceAffineOp(rewriter, affineOp, map, resultOperands);
863  return success();
864  }
865 };

970 // Ex:
971 // affine.dma_start %src[%i, %j], %dst[%k, %l], %tag[%index], %size,
972 // %stride, %num_elt_per_stride
973 // : memref<3076 x f32, 0>, memref<1024 x f32, 2>, memref<1 x i32>
974 //
975 ParseResult AffineDmaStartOp::parse(OpAsmParser &parser,
976  OperationState &result) {
977  OpAsmParser::OperandType srcMemRefInfo;
978  AffineMapAttr srcMapAttr;
979  SmallVector<OpAsmParser::OperandType, 4> srcMapOperands;
980  OpAsmParser::OperandType dstMemRefInfo;
981  AffineMapAttr dstMapAttr;
982  SmallVector<OpAsmParser::OperandType, 4> dstMapOperands;
983  OpAsmParser::OperandType tagMemRefInfo;
984  AffineMapAttr tagMapAttr;
985  SmallVector<OpAsmParser::OperandType, 4> tagMapOperands;
986  OpAsmParser::OperandType numElementsInfo;
987  SmallVector<OpAsmParser::OperandType, 2> strideInfo;
988 
989  SmallVector<Type, 3> types;
990  auto indexType = parser.getBuilder().getIndexType();
991 
992  // Parse and resolve the following list of operands:
993  // *) dst memref followed by its affine maps operands (in square brackets).
994  // *) src memref followed by its affine map operands (in square brackets).
995  // *) tag memref followed by its affine map operands (in square brackets).
996  // *) number of elements transferred by DMA operation.
997  if (parser.parseOperand(srcMemRefInfo) ||
998  parser.parseAffineMapOfSSAIds(srcMapOperands, srcMapAttr,
999  getSrcMapAttrName(), result.attributes) ||
1000  parser.parseComma() || parser.parseOperand(dstMemRefInfo) ||
1001  parser.parseAffineMapOfSSAIds(dstMapOperands, dstMapAttr,
1002  getDstMapAttrName(), result.attributes) ||
1003  parser.parseComma() || parser.parseOperand(tagMemRefInfo) ||
1004  parser.parseAffineMapOfSSAIds(tagMapOperands, tagMapAttr,
1005  getTagMapAttrName(), result.attributes) ||
1006  parser.parseComma() || parser.parseOperand(numElementsInfo))
1007  return failure();
1008 
1009  // Parse optional stride and elements per stride.
1010  if (parser.parseTrailingOperandList(strideInfo)) {
1011  return failure();
1012  }
1013  if (!strideInfo.empty() && strideInfo.size() != 2) {
1014  return parser.emitError(parser.getNameLoc(),
1015  "expected two stride related operands");
1016  }
1017  bool isStrided = strideInfo.size() == 2;
1018 
1019  if (parser.parseColonTypeList(types))
1020  return failure();
1021 
1022  if (types.size() != 3)
1023  return parser.emitError(parser.getNameLoc(), "expected three types");
1024 
1025  if (parser.resolveOperand(srcMemRefInfo, types[0], result.operands) ||
1026  parser.resolveOperands(srcMapOperands, indexType, result.operands) ||
1027  parser.resolveOperand(dstMemRefInfo, types[1], result.operands) ||
1028  parser.resolveOperands(dstMapOperands, indexType, result.operands) ||
1029  parser.resolveOperand(tagMemRefInfo, types[2], result.operands) ||
1030  parser.resolveOperands(tagMapOperands, indexType, result.operands) ||
1031  parser.resolveOperand(numElementsInfo, indexType, result.operands))
1032  return failure();
1033 
1034  if (isStrided) {
1035  if (parser.resolveOperands(strideInfo, indexType, result.operands))
1036  return failure();
1037  }
1038 
1039  // Check that src/dst/tag operand counts match their map.numInputs.
1040  if (srcMapOperands.size() != srcMapAttr.getValue().getNumInputs() ||
1041  dstMapOperands.size() != dstMapAttr.getValue().getNumInputs() ||
1042  tagMapOperands.size() != tagMapAttr.getValue().getNumInputs())
1043  return parser.emitError(parser.getNameLoc(),
1044  "memref operand count not equal to map.numInputs");
1045  return success();
1046 }

1254 /// Parse a for operation loop bounds.
1255 static ParseResult parseBound(bool isLower, OperationState &result,
1256  OpAsmParser &p) {
1257  // 'min' / 'max' prefixes are generally syntactic sugar, but are required if
1258  // the map has multiple results.
1259  bool failedToParsedMinMax =
1260  failed(p.parseOptionalKeyword(isLower ? "max" : "min"));
1261 
1262  auto &builder = p.getBuilder();
1263  auto boundAttrName = isLower ? AffineForOp::getLowerBoundAttrName()
1264  : AffineForOp::getUpperBoundAttrName();
1265 
1266  // Parse ssa-id as identity map.
1267  SmallVector<OpAsmParser::OperandType, 1> boundOpInfos;
1268  if (p.parseOperandList(boundOpInfos))
1269  return failure();
1270 
1271  if (!boundOpInfos.empty()) {
1272  // Check that only one operand was parsed.
1273  if (boundOpInfos.size() > 1)
1274  return p.emitError(p.getNameLoc(),
1275  "expected only one loop bound operand");
1276 
1277  // TODO: improve error message when SSA value is not of index type.
1278  // Currently it is 'use of value ... expects different type than prior uses'
1279  if (p.resolveOperand(boundOpInfos.front(), builder.getIndexType(),
1280  result.operands))
1281  return failure();
1282 
1283  // Create an identity map using symbol id. This representation is optimized
1284  // for storage. Analysis passes may expand it into a multi-dimensional map
1285  // if desired.
1286  AffineMap map = builder.getSymbolIdentityMap();
1287  result.addAttribute(boundAttrName, AffineMapAttr::get(map));
1288  return success();
1289  }
1290 
1291  // Get the attribute location.
1292  llvm::SMLoc attrLoc = p.getCurrentLocation();
1293 
1294  Attribute boundAttr;
1295  if (p.parseAttribute(boundAttr, builder.getIndexType(), boundAttrName,
1296  result.attributes))
1297  return failure();
1298 
1299  // Parse full form - affine map followed by dim and symbol list.
1300  if (auto affineMapAttr = boundAttr.dyn_cast<AffineMapAttr>()) {
1301  unsigned currentNumOperands = result.operands.size();
1302  unsigned numDims;
1303  if (parseDimAndSymbolList(p, result.operands, numDims))
1304  return failure();
1305 
1306  auto map = affineMapAttr.getValue();
1307  if (map.getNumDims() != numDims)
1308  return p.emitError(
1309  p.getNameLoc(),
1310  "dim operand count and affine map dim count must match");
1311 
1312  unsigned numDimAndSymbolOperands =
1313  result.operands.size() - currentNumOperands;
1314  if (numDims + map.getNumSymbols() != numDimAndSymbolOperands)
1315  return p.emitError(
1316  p.getNameLoc(),
1317  "symbol operand count and affine map symbol count must match");
1318 
1319  // If the map has multiple results, make sure that we parsed the min/max
1320  // prefix.
1321  if (map.getNumResults() > 1 && failedToParsedMinMax) {
1322  if (isLower) {
1323  return p.emitError(attrLoc, "lower loop bound affine map with "
1324  "multiple results requires 'max' prefix");
1325  }
1326  return p.emitError(attrLoc, "upper loop bound affine map with multiple "
1327  "results requires 'min' prefix");
1328  }
1329  return success();
1330  }
1331 
1332  // Parse custom assembly form.
1333  if (auto integerAttr = boundAttr.dyn_cast<IntegerAttr>()) {
1334  result.attributes.pop_back();
1335  result.addAttribute(
1336  boundAttrName,
1337  AffineMapAttr::get(builder.getConstantAffineMap(integerAttr.getInt())));
1338  return success();
1339  }
1340 
1341  return p.emitError(
1342  p.getNameLoc(),
1343  "expected valid affine map representation for loop bounds");
1344 }

1346 static ParseResult parseAffineForOp(OpAsmParser &parser,
1347  OperationState &result) {
1348  auto &builder = parser.getBuilder();
1349  OpAsmParser::OperandType inductionVariable;
1350  // Parse the induction variable followed by '='.
1351  if (parser.parseRegionArgument(inductionVariable) || parser.parseEqual())
1352  return failure();
1353 
1354  // Parse loop bounds.
1355  if (parseBound(/*isLower=*/true, result, parser) ||
1356  parser.parseKeyword("to", " between bounds") ||
1357  parseBound(/*isLower=*/false, result, parser))
1358  return failure();
1359 
1360  // Parse the optional loop step, we default to 1 if one is not present.
1361  if (parser.parseOptionalKeyword("step")) {
1362  result.addAttribute(
1363  AffineForOp::getStepAttrName(),
1364  builder.getIntegerAttr(builder.getIndexType(), /*value=*/1));
1365  } else {
1366  llvm::SMLoc stepLoc = parser.getCurrentLocation();
1367  IntegerAttr stepAttr;
1368  if (parser.parseAttribute(stepAttr, builder.getIndexType(),
1369  AffineForOp::getStepAttrName().data(),
1370  result.attributes))
1371  return failure();
1372 
1373  if (stepAttr.getValue().getSExtValue() < 0)
1374  return parser.emitError(
1375  stepLoc,
1376  "expected step to be representable as a positive signed integer");
1377  }
1378 
1379  // Parse the body region.
1380  Region *body = result.addRegion();
1381  if (parser.parseRegion(*body, inductionVariable, builder.getIndexType()))
1382  return failure();
1383 
1384  AffineForOp::ensureTerminator(*body, builder, result.location);
1385 
1386  // Parse the optional attribute list.
1387  return parser.parseOptionalAttrDict(result.attributes);
1388 }

1390 static void printBound(AffineMapAttr boundMap,
1391  Operation::operand_range boundOperands,
1392  const char *prefix, OpAsmPrinter &p) {
1393  AffineMap map = boundMap.getValue();
1394 
1395  // Check if this bound should be printed using custom assembly form.
1396  // The decision to restrict printing custom assembly form to trivial cases
1397  // comes from the will to roundtrip MLIR binary -> text -> binary in a
1398  // lossless way.
1399  // Therefore, custom assembly form parsing and printing is only supported for
1400  // zero-operand constant maps and single symbol operand identity maps.
1401  if (map.getNumResults() == 1) {
1402  AffineExpr expr = map.getResult(0);
1403 
1404  // Print constant bound.
1405  if (map.getNumDims() == 0 && map.getNumSymbols() == 0) {
1406  if (auto constExpr = expr.dyn_cast<AffineConstantExpr>()) {
1407  p << constExpr.getValue();
1408  return;
1409  }
1410  }
1411

1414  if (map.getNumDims() == 0 && map.getNumSymbols() == 1) {

1416  p.printOperand(*boundOperands.begin());
1417  return;
1418  }
1419  }
1420  } else {
1421  // Map has multiple results. Print 'min' or 'max' prefix.
1422  p << prefix << ' ';
1423  }
1424 
1425  // Print the map and its operands.
1426  p << boundMap;
1427  printDimAndSymbolList(boundOperands.begin(), boundOperands.end(),

1450 /// Fold the constant bounds of a loop.
1451 static LogicalResult foldLoopBounds(AffineForOp forOp) {
1452  auto foldLowerOrUpperBound = [&forOp](bool lower) {
1453  // Check to see if each of the operands is the result of a constant. If
1454  // so, get the value. If not, ignore it.
1455  SmallVector<Attribute, 8> operandConstants;
1456  auto boundOperands =
1457  lower ? forOp.getLowerBoundOperands() : forOp.getUpperBoundOperands();
1458  for (auto operand : boundOperands) {
1459  Attribute operandCst;
1460  matchPattern(operand, m_Constant(&operandCst));
1461  operandConstants.push_back(operandCst);
1462  }
1463 
1464  AffineMap boundMap =
1465  lower ? forOp.getLowerBoundMap() : forOp.getUpperBoundMap();
1466  assert(boundMap.getNumResults() >= 1 &&
1467  "bound maps should have at least one result");
1468  SmallVector<Attribute, 4> foldedResults;
1469  if (failed(boundMap.constantFold(operandConstants, foldedResults)))
1470  return failure();
1471 
1472  // Compute the max or min as applicable over the results.
1473  assert(!foldedResults.empty() && "bounds should have at least one result");
1474  auto maxOrMin = foldedResults[0].cast<IntegerAttr>().getValue();
1475  for (unsigned i = 1, e = foldedResults.size(); i < e; i++) {
1476  auto foldedResult = foldedResults[i].cast<IntegerAttr>().getValue();
1477  maxOrMin = lower ? llvm::APIntOps::smax(maxOrMin, foldedResult)
1478  : llvm::APIntOps::smin(maxOrMin, foldedResult);
1479  }
1480  lower ? forOp.setConstantLowerBound(maxOrMin.getSExtValue())
1481  : forOp.setConstantUpperBound(maxOrMin.getSExtValue());
1482  return success();
1483  };
1484 
1485  // Try to fold the lower bound.
1486  bool folded = false;
1487  if (!forOp.hasConstantLowerBound())
1488  folded |= succeeded(foldLowerOrUpperBound(/*lower=*/true));
1489 
1490  // Try to fold the upper bound.
1491  if (!forOp.hasConstantUpperBound())
1492  folded |= succeeded(foldLowerOrUpperBound(/*lower=*/false));
1493  return success(folded);
1494 }

1496 /// Canonicalize the bounds of the given loop.
1497 static LogicalResult canonicalizeLoopBounds(AffineForOp forOp) {
1498  SmallVector<Value, 4> lbOperands(forOp.getLowerBoundOperands());
1499  SmallVector<Value, 4> ubOperands(forOp.getUpperBoundOperands());
1500 
1501  auto lbMap = forOp.getLowerBoundMap();
1502  auto ubMap = forOp.getUpperBoundMap();
1503  auto prevLbMap = lbMap;
1504  auto prevUbMap = ubMap;
1505 
1506  canonicalizeMapAndOperands(&lbMap, &lbOperands);
1507  lbMap = removeDuplicateExprs(lbMap);
1508 
1509  canonicalizeMapAndOperands(&ubMap, &ubOperands);
1510  ubMap = removeDuplicateExprs(ubMap);
1511 
1512  // Any canonicalization change always leads to updated map(s).
1513  if (lbMap == prevLbMap && ubMap == prevUbMap)
1514  return failure();
1515

1517  forOp.setLowerBound(lbOperands, lbMap);
1518  if (ubMap != prevUbMap)
1519  forOp.setUpperBound(ubOperands, ubMap);
1520  return success();
1521 }

1701  OpBuilder::InsertionGuard guard(builder);
1702  if (lbs.empty()) {
1703  if (bodyBuilderFn)
1704  bodyBuilderFn(builder, loc, ValueRange());
1705  return;
1706  }

1708  // Create the loops iteratively and store the induction variables.
1709  SmallVector<Value, 4> ivs;
1710  ivs.reserve(lbs.size());
1711  for (unsigned i = 0, e = lbs.size(); i < e; ++i) {
1712  // Callback for creating the loop body, always creates the terminator.
1713  auto loopBody = [&](OpBuilder &nestedBuilder, Location nestedLoc,
1714  Value iv) {
1715  ivs.push_back(iv);
1716  // In the innermost loop, call the body builder.
1717  if (i == e - 1 && bodyBuilderFn) {
1718  OpBuilder::InsertionGuard nestedGuard(nestedBuilder);
1719  bodyBuilderFn(nestedBuilder, nestedLoc, ivs);
1720  }
1721  nestedBuilder.create<AffineYieldOp>(nestedLoc);
1722  };
1723 
1724  // Delegate actual loop creation to the callback in order to dispatch
1725  // between constant- and variable-bound loops.
1726  auto loop = loopCreatorFn(builder, loc, lbs[i], ubs[i], steps[i], loopBody);
1727  builder.setInsertionPointToStart(loop.getBody());
1728  }
1729 }
