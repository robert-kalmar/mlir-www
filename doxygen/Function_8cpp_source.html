81 LogicalResult FuncOp::verify() {
82  // If this function is external there is nothing to do.
83  if (isExternal())
84  return success();
85 
86  // Verify that the argument list of the function and the arg list of the entry
87  // block line up. The trait already verified that the number of arguments is
88  // the same between the signature and the block.
89  auto fnInputTypes = getType().getInputs();
90  Block &entryBlock = front();
91  for (unsigned i = 0, e = entryBlock.getNumArguments(); i != e; ++i)
92  if (fnInputTypes[i] != entryBlock.getArgument(i).getType())
93  return emitOpError("type of entry block argument #")
94  << i << '(' << entryBlock.getArgument(i).getType()
95  << ") must match the type of the corresponding argument in "
96  << "function signature(" << fnInputTypes[i] << ')';
97 
98  return success();
99 }

101 void FuncOp::eraseArguments(ArrayRef<unsigned> argIndices) {
102  auto oldType = getType();
103  int originalNumArgs = oldType.getNumInputs();
104  llvm::BitVector eraseIndices(originalNumArgs);
105  for (auto index : argIndices)
106  eraseIndices.set(index);
107  auto shouldEraseArg = [&](int i) { return eraseIndices.test(i); };
108 
109  // There are 3 things that need to be updated:
110  // - Function type.
111  // - Arg attrs.
112  // - Block arguments of entry block.
113 
114  // Update the function type and arg attrs.
115  SmallVector<Type, 4> newInputTypes;
116  SmallVector<MutableDictionaryAttr, 4> newArgAttrs;
117  for (int i = 0; i < originalNumArgs; i++) {
118  if (shouldEraseArg(i))
119  continue;
120  newInputTypes.emplace_back(oldType.getInput(i));
121  newArgAttrs.emplace_back(getArgAttrDict(i));
122  }
123  setType(FunctionType::get(newInputTypes, oldType.getResults(), getContext()));
124  setAllArgAttrs(newArgAttrs);
125 
126  // Update the entry block's arguments.
127  // We do this in reverse so that we erase later indices before earlier
128  // indices, to avoid shifting the later indices.
129  Block &entry = front();
130  for (int i = 0; i < originalNumArgs; i++)
131  if (shouldEraseArg(originalNumArgs - i - 1))
132  entry.eraseArgument(originalNumArgs - i - 1);
133 }

135 /// Clone the internal blocks from this function into dest and all attributes
136 /// from this function to dest.
137 void FuncOp::cloneInto(FuncOp dest, BlockAndValueMapping &mapper) {
138  // Add the attributes of this function to dest.
139  llvm::MapVector<Identifier, Attribute> newAttrs;
140  for (auto &attr : dest.getAttrs())
141  newAttrs.insert(attr);
142  for (auto &attr : getAttrs())
143  newAttrs.insert(attr);
144  dest.getOperation()->setAttrs(
145  DictionaryAttr::get(newAttrs.takeVector(), getContext()));
146 
147  // Clone the body.
148  getBody().cloneInto(&dest.getBody(), mapper);
149 }

151 /// Create a deep copy of this function and all of its blocks, remapping
152 /// any operands that use values outside of the function using the map that is
153 /// provided (leaving them alone if no entry is present). Replaces references
154 /// to cloned sub-values with the corresponding value that is copied, and adds
155 /// those mappings to the mapper.
156 FuncOp FuncOp::clone(BlockAndValueMapping &mapper) {
157  FunctionType newType = getType();
158 
159  // If the function has a body, then the user might be deleting arguments to
160  // the function by specifying them in the mapper. If so, we don't add the
161  // argument to the input type vector.
162  bool isExternalFn = isExternal();
163  if (!isExternalFn) {
164  SmallVector<Type, 4> inputTypes;
165  inputTypes.reserve(newType.getNumInputs());
166  for (unsigned i = 0, e = getNumArguments(); i != e; ++i)
167  if (!mapper.contains(getArgument(i)))
168  inputTypes.push_back(newType.getInput(i));
169  newType = FunctionType::get(inputTypes, newType.getResults(), getContext());
170  }
171 
172  // Create the new function.
173  FuncOp newFunc = cast<FuncOp>(getOperation()->cloneWithoutRegions());
174  newFunc.setType(newType);
175 
176  /// Set the argument attributes for arguments that aren't being replaced.
177  for (unsigned i = 0, e = getNumArguments(), destI = 0; i != e; ++i)
178  if (isExternalFn || !mapper.contains(getArgument(i)))
179  newFunc.setArgAttrs(destI++, getArgAttrs(i));
180 
181  /// Clone the current function into the new one and return it.
182  cloneInto(newFunc, mapper);
183  return newFunc;
184 }
185 FuncOp FuncOp::clone() {
186  BlockAndValueMapping mapper;
187  return clone(mapper);
188 }
