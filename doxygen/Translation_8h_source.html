29 /// Interface of the function that translates the sources managed by `sourceMgr`
30 /// to MLIR. The source manager has at least one buffer. The implementation
31 /// should create a new MLIR ModuleOp in the given context and return a pointer
32 /// to it, or a nullptr in case of any error.
33 using TranslateSourceMgrToMLIRFunction =
34  std::function<OwningModuleRef(llvm::SourceMgr &sourceMgr, MLIRContext *)>;

36 /// Interface of the function that translates the given string to MLIR. The
37 /// implementation should create a new MLIR ModuleOp in the given context. If
38 /// source-related error reporting is required from within the function, use
39 /// TranslateSourceMgrToMLIRFunction instead.
40 using TranslateStringRefToMLIRFunction =
41  std::function<OwningModuleRef(llvm::StringRef, MLIRContext *)>;

43 /// Interface of the function that translates MLIR to a different format and
44 /// outputs the result to a stream. It is allowed to modify the module.
45 using TranslateFromMLIRFunction =
46  std::function<LogicalResult(ModuleOp, llvm::raw_ostream &output)>;

48 /// Interface of the function that performs file-to-file translation involving
49 /// MLIR. The input file is held in the given MemoryBuffer; the output file
50 /// should be written to the given raw_ostream. The implementation should create
51 /// all MLIR constructs needed during the process inside the given context. This
52 /// can be used for round-tripping external formats through the MLIR system.
53 using TranslateFunction = std::function<LogicalResult(
54  llvm::SourceMgr &sourceMgr, llvm::raw_ostream &output, MLIRContext *)>;
