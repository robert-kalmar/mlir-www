16 // Converts the operand and result types of the Standard's CallOp, used together
17 // with the FuncOpSignatureConversion.
18 struct CallOpSignatureConversion : public OpConversionPattern<CallOp> {
19  CallOpSignatureConversion(MLIRContext *ctx, TypeConverter &converter)
20  : OpConversionPattern(ctx), converter(converter) {}
21 
22  /// Hook for derived classes to implement combined matching and rewriting.
23  LogicalResult
24  matchAndRewrite(CallOp callOp, ArrayRef<Value> operands,
25  ConversionPatternRewriter &rewriter) const override {
26  FunctionType type = callOp.getCalleeType();
27 
28  // Convert the original function results.
29  SmallVector<Type, 1> convertedResults;
30  if (failed(converter.convertTypes(type.getResults(), convertedResults)))
31  return failure();
32 
33  // Substitute with the new result types from the corresponding FuncType
34  // conversion.
35  rewriter.replaceOpWithNewOp<CallOp>(callOp, callOp.callee(),
36  convertedResults, operands);
37  return success();
38  }
39 
40  /// The type converter to use when rewriting the signature.
41  TypeConverter &converter;
42 };
43 } // end anonymous namespace
