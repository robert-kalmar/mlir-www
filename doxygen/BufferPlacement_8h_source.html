27 /// Prepares a buffer placement phase. It can place (user-defined) alloc
28 /// nodes. This simplifies the integration of the actual buffer-placement
29 /// pass. Sample usage:
30 /// BufferAssignmentPlacer baHelper(regionOp);
31 /// -> determine alloc positions
32 /// auto allocPosition = baHelper.computeAllocPosition(value);
33 /// -> place alloc
34 /// allocBuilder.setInsertionPoint(positions.getAllocPosition());
35 /// <create alloc>
36 /// Note: this class is intended to be used during legalization. In order
37 /// to move alloc and dealloc nodes into the right places you can use the
38 /// createBufferPlacementPass() function.
39 class BufferAssignmentPlacer {
40 public:
41  /// Creates a new assignment builder.
42  explicit BufferAssignmentPlacer(Operation *op);
43 
44  /// Returns the operation this analysis was constructed from.
45  Operation *getOperation() const { return operation; }
46 
47  /// Computes the actual position to place allocs for the given result.
48  OpBuilder::InsertPoint computeAllocPosition(OpResult result);
49 
50 private:
51  /// The operation this analysis was constructed from.
52  Operation *operation;
53 };

55 /// Helper conversion pattern that encapsulates a BufferAssignmentPlacer
56 /// instance. Sample usage:
57 /// class CustomConversionPattern : public
58 /// BufferAssignmentOpConversionPattern<MyOpT>
59 /// {
60 /// ... matchAndRewrite(...) {
61 /// -> Access stored BufferAssignmentPlacer
62 /// bufferAssignment->computeAllocPosition(resultOp);
63 /// }
64 /// };

66 class BufferAssignmentOpConversionPattern

70  MLIRContext *context, BufferAssignmentPlacer *bufferAssignment = nullptr,
71  TypeConverter *converter = nullptr, PatternBenefit benefit = 1)
72  : OpConversionPattern<SourceOp>(context, benefit),
73  bufferAssignment(bufferAssignment), converter(converter) {}
74 
75 protected:
76  BufferAssignmentPlacer *bufferAssignment;
77  TypeConverter *converter;
78 };

94 /// Converts the signature of the function based on whether the function is
95 /// allowed to return memref typed results or not using
96 /// `allowMemrefFunctionResults` parameter. If this option is false, then it
97 /// adds an extra function argument as an output buffer for each function result
98 /// which is going to be a memref type only after type conversion. The
99 /// other function result types remain unchanged. If
100 /// `allowMemrefFunctionResults` is true, the types are converted in place.
101 /// Any changes in function signature need to be applied
102 /// to return and caller operations. `BufferAssignmentReturnOpConverter` and
103 /// `BufferAssignmentCallOpConverter` are two helper function that match the
104 /// return and caller operation with the new function signature. Furthermore,
105 /// `BufferAssignmentTypeConverter` is a helper `TypeConverter` for converting
106 /// tensor typed values to memref typed ones.
107 template <bool allowMemrefFunctionResults>
108 class BufferAssignmentFuncOpConverter
109  : public BufferAssignmentOpConversionPattern<FuncOp> {
110 public:
111  using BufferAssignmentOpConversionPattern<
112  FuncOp>::BufferAssignmentOpConversionPattern;
113 
114  /// Performs the actual signature rewriting step.
115  LogicalResult
116  matchAndRewrite(mlir::FuncOp funcOp, ArrayRef<Value> operands,
117  ConversionPatternRewriter &rewriter) const final {
118  if (!converter)
119  return funcOp.emitError("The type converter has not been defined for "
120  "BufferAssignmentFuncOpConverter");
121  auto funcType = funcOp.getType();
122 
123  // Convert function arguments using the provided TypeConverter.
124  TypeConverter::SignatureConversion conversion(funcType.getNumInputs());
125  for (auto argType : llvm::enumerate(funcType.getInputs()))
126  conversion.addInputs(argType.index(),
127  converter->convertType(argType.value()));
128 
129  // If allowMemrefFunctionResults is false and a function result type is not
130  // a memref but it would be a memref after type conversion, a new argument
131  // should be appended to the function arguments list for this result.
132  // Otherwise, it remains unchanged as a function result.
133  SmallVector<Type, 2> newResultTypes;
134  newResultTypes.reserve(funcOp.getNumResults());
135  for (Type resType : funcType.getResults()) {
136  Type convertedType = converter->convertType(resType);
137  if (!allowMemrefFunctionResults &&
138  BufferAssignmentTypeConverter::isConvertedMemref(convertedType,
139  resType))
140  conversion.addInputs(convertedType);
141  else
142  newResultTypes.push_back(convertedType);
143  }
144  if (failed(rewriter.convertRegionTypes(&funcOp.getBody(), *converter,
145  &conversion)))
146  return failure();
147 
148  // Update the signature of the function.
149  rewriter.updateRootInPlace(funcOp, [&] {
150  funcOp.setType(rewriter.getFunctionType(conversion.getConvertedTypes(),
151  newResultTypes));
152  });
153  return success();
154  }
155 };

157 /// Rewrites the `ReturnOp` to conform with the changed function signature.
158 /// if allowMemrefFunctionResults is false, operands that correspond to return
159 /// values and have been rewritten from illegal typed results to memref
160 /// arguments are dropped. In their place, a corresponding copy operation from
161 /// the operand to the output function argument is inserted. Otherwise, the
162 /// memref typed operands are returned.
163 /// Note: If this pattern rewriter is used with BufferAssignmentFuncOpConverter,
164 /// allowMemrefFunctionResults must be set/unset for both.
165 template <typename ReturnOpSourceTy, typename ReturnOpTargetTy,
166  typename CopyOpTy, bool allowMemrefFunctionResults>
167 class BufferAssignmentReturnOpConverter
168  : public BufferAssignmentOpConversionPattern<ReturnOpSourceTy> {
169 public:
170  using BufferAssignmentOpConversionPattern<
171  ReturnOpSourceTy>::BufferAssignmentOpConversionPattern;
172 
173  /// Performs the actual return-op conversion step.
174  LogicalResult
175  matchAndRewrite(ReturnOpSourceTy returnOp, ArrayRef<Value> operands,
176  ConversionPatternRewriter &rewriter) const final {
177  // If the memref typed results can be returned as function results, the new
178  // `ReturnOp` should only return the type converted operands.
179  if (allowMemrefFunctionResults) {
180  rewriter.replaceOpWithNewOp<ReturnOpTargetTy>(returnOp, operands);
181  return success();
182  }
183 
184  // Split the operands by their kinds whether they are converted memref or
185  // not.
186  SmallVector<Value, 2> needCopyOperands, newOperands;
187  unsigned operandsSize = operands.size();
188  needCopyOperands.reserve(operandsSize);
189  newOperands.reserve(operandsSize);
190  for (auto operand : llvm::enumerate(operands))
191  if (BufferAssignmentTypeConverter::isConvertedMemref(
192  operand.value().getType(),
193  returnOp.getOperand(operand.index()).getType()))
194  needCopyOperands.push_back(operand.value());
195  else
196  newOperands.push_back(operand.value());
197 
198  Block &entryBlock = returnOp.getParentRegion()->front();
199  unsigned numFuncArgs = entryBlock.getNumArguments();
200 
201  // Find the index of the first destination buffer.
202  assert(needCopyOperands.size() <= numFuncArgs &&
203  "The number of operands of return operation is more than the "
204  "number of function arguments.");
205  unsigned destArgNum = numFuncArgs - needCopyOperands.size();
206  rewriter.setInsertionPoint(returnOp);
207  for (Value operand : needCopyOperands) {
208  // Insert a `CopyOp` for each converted memref-type operand.
209  rewriter.create<CopyOpTy>(returnOp.getLoc(), operand,
210  entryBlock.getArgument(destArgNum));
211  ++destArgNum;
212  }
213 
214  // Insert the new target Return operation.
215  rewriter.replaceOpWithNewOp<ReturnOpTargetTy>(returnOp, newOperands);
216  return success();
217  }
218 };

220 /// Rewrites the `CallOp` to match its operands and results with the signature
221 /// of the callee after rewriting the callee with
222 /// BufferAssignmentFuncOpConverter. If allowMemrefFunctionResults is false, a
223 /// buffer is allocated as an output buffer only for each memref typed result
224 /// that has been rewritten. The new allocated buffer is passed through the
225 /// operands list of the new `CallOp`.
226 /// Note: If this pattern rewriter is used with BufferAssignmentFuncOpConverter,
227 /// allowMemrefFunctionResults must be set/unset for both.
228 template <bool allowMemrefFunctionResults>
229 class BufferAssignmentCallOpConverter
230  : public BufferAssignmentOpConversionPattern<CallOp> {
231 public:
232  using BufferAssignmentOpConversionPattern<
233  CallOp>::BufferAssignmentOpConversionPattern;
234 
235  LogicalResult
236  matchAndRewrite(CallOp callOp, ArrayRef<Value> operands,
237  ConversionPatternRewriter &rewriter) const final {
238  if (!converter)
239  return callOp.emitError("The type converter has not been defined for "
240  "BufferAssignmentCallOpConverter");
241  Location loc = callOp.getLoc();
242 
243  // If the memref typed results can be returned as function results, there is
244  // no need to create output buffers. It is only required to convert the type
245  // of operands and results in place for creating the new `CallOp`.
246  if (allowMemrefFunctionResults) {
247  SmallVector<Type, 2> resultTypes;
248  resultTypes.reserve(callOp.getNumResults());
249  for (Type type : callOp.getResultTypes())
250  resultTypes.push_back(converter->convertType(type));
251  rewriter.replaceOpWithNewOp<CallOp>(callOp, callOp.getCallee(),
252  resultTypes, operands);
253  return success();
254  }
255 
256  SmallVector<Value, 2> newOperands, replacingValues;
257  SmallVector<Type, 2> newResultTypes;
258  unsigned numResults = callOp.getNumResults();
259  newOperands.reserve(numResults + operands.size());
260  newOperands.append(operands.begin(), operands.end());
261  newResultTypes.reserve(numResults);
262  replacingValues.reserve(numResults);
263 
264  // For each memref result of `CallOp` which has not been a memref before
265  // the type conversion, a new buffer is allocated and passed to the operands
266  // list of the new `CallOp`. Otherwise, it remains as a caller result.
267  for (Value result : callOp.getResults()) {
268  Type currType = result.getType();
269  Type newType = converter->convertType(result.getType());
270  if (BufferAssignmentTypeConverter::isConvertedMemref(newType, currType)) {
271  OpBuilder::InsertionGuard guard(rewriter);
272  rewriter.restoreInsertionPoint(bufferAssignment->computeAllocPosition(
273  result.dyn_cast<OpResult>()));
274  Value alloc =
275  rewriter.create<AllocOp>(loc, newType.dyn_cast<MemRefType>());
276  newOperands.push_back(alloc);
277  replacingValues.push_back(alloc);
278  } else {
279  newResultTypes.push_back(currType);
280 
281  // No replacing is required.
282  replacingValues.push_back(nullptr);
283  }
284  }
285 
286  // Creating the new `CallOp`.
287  rewriter.create<CallOp>(loc, callOp.getCallee(), newResultTypes,
288  newOperands);
289 
290  // Replacing the results of the old `CallOp`.
291  rewriter.replaceOp(callOp, replacingValues);
292  return success();
293  }
294 };
295 } // end namespace detail
