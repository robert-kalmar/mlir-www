48 /// A utility class for folding operations, and unifying duplicated constants
49 /// generated along the way.
50 class OperationFolder {
51 public:
52  OperationFolder(MLIRContext *ctx) : interfaces(ctx) {}
53 
54  /// Tries to perform folding on the given `op`, including unifying
55  /// deduplicated constants. If successful, replaces `op`'s uses with
56  /// folded results, and returns success. `preReplaceAction` is invoked on `op`
57  /// before it is replaced. 'processGeneratedConstants' is invoked for any new
58  /// operations generated when folding. If the op was completely folded it is
59  /// erased. If it is just updated in place, `inPlaceUpdate` is set to true.
60  LogicalResult
61  tryToFold(Operation *op,
62  function_ref<void(Operation *)> processGeneratedConstants = nullptr,
63  function_ref<void(Operation *)> preReplaceAction = nullptr,
64  bool *inPlaceUpdate = nullptr);
65 
66  /// Notifies that the given constant `op` should be remove from this
67  /// OperationFolder's internal bookkeeping.
68  ///
69  /// Note: this method must be called if a constant op is to be deleted
70  /// externally to this OperationFolder. `op` must be a constant op.
71  void notifyRemoval(Operation *op);
72 
73  /// Create an operation of specific op type with the given builder,
74  /// and immediately try to fold it. This function populates 'results' with
75  /// the results after folding the operation.
76  template <typename OpTy, typename... Args>
77  void create(OpBuilder &builder, SmallVectorImpl<Value> &results,
78  Location location, Args &&... args) {
79  // The op needs to be inserted only if the fold (below) fails, or the number
80  // of results produced by the successful folding is zero (which is treated
81  // as an in-place fold). Using create methods of the builder will insert the
82  // op, so not using it here.
83  OperationState state(location, OpTy::getOperationName());
84  OpTy::build(builder, state, std::forward<Args>(args)...);
85  Operation *op = Operation::create(state);
86 
87  if (failed(tryToFold(builder, op, results)) || results.empty()) {
88  builder.insert(op);
89  results.assign(op->result_begin(), op->result_end());
90  return;
91  }
92  op->destroy();
93  }
94 
95  /// Overload to create or fold a single result operation.
96  template <typename OpTy, typename... Args>
97  typename std::enable_if<OpTy::template hasTrait<OpTrait::OneResult>(),
98  Value>::type
99  create(OpBuilder &builder, Location location, Args &&... args) {
100  SmallVector<Value, 1> results;
101  create<OpTy>(builder, results, location, std::forward<Args>(args)...);
102  return results.front();
103  }
104 
105  /// Overload to create or fold a zero result operation.
106  template <typename OpTy, typename... Args>
107  typename std::enable_if<OpTy::template hasTrait<OpTrait::ZeroResult>(),
108  OpTy>::type
109  create(OpBuilder &builder, Location location, Args &&... args) {
110  auto op = builder.create<OpTy>(location, std::forward<Args>(args)...);
111  SmallVector<Value, 0> unused;
112  (void)tryToFold(op.getOperation(), unused);
113 
114  // Folding cannot remove a zero-result operation, so for convenience we
115  // continue to return it.
116  return op;
117  }
118 
119  /// Clear out any constants cached inside of the folder.
120  void clear();
121 
122  /// Get or create a constant using the given builder. On success this returns
123  /// the constant operation, nullptr otherwise.
124  Value getOrCreateConstant(OpBuilder &builder, Dialect *dialect,
125  Attribute value, Type type, Location loc);
126 
127 private:
128  /// This map keeps track of uniqued constants by dialect, attribute, and type.
129  /// A constant operation materializes an attribute with a type. Dialects may
130  /// generate different constants with the same input attribute and type, so we
131  /// also need to track per-dialect.
132  using ConstantMap =
133  DenseMap<std::tuple<Dialect *, Attribute, Type>, Operation *>;
134 
135  /// Tries to perform folding on the given `op`. If successful, populates
136  /// `results` with the results of the folding.
137  LogicalResult tryToFold(
138  OpBuilder &builder, Operation *op, SmallVectorImpl<Value> &results,
139  function_ref<void(Operation *)> processGeneratedConstants = nullptr);
140 
141  /// Try to get or create a new constant entry. On success this returns the
142  /// constant operation, nullptr otherwise.
143  Operation *tryGetOrCreateConstant(ConstantMap &uniquedConstants,
144  Dialect *dialect, OpBuilder &builder,
145  Attribute value, Type type, Location loc);
146 
147  /// A mapping between an insertion region and the constants that have been
148  /// created within it.
149  DenseMap<Region *, ConstantMap> foldScopes;
150 
151  /// This map tracks all of the dialects that an operation is referenced by;
152  /// given that many dialects may generate the same constant.
153  DenseMap<Operation *, SmallVector<Dialect *, 2>> referencedDialects;
154 
155  /// A collection of dialect folder interfaces.
156  DialectInterfaceCollection<OpFolderDialectInterface> interfaces;
157 };
