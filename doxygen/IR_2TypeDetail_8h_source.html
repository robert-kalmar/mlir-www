82 /// Function Type Storage and Uniquing.
83 struct FunctionTypeStorage : public TypeStorage {
84  FunctionTypeStorage(unsigned numInputs, unsigned numResults,
85  Type const *inputsAndResults)
86  : numInputs(numInputs), numResults(numResults),
87  inputsAndResults(inputsAndResults) {}
88 
89  /// The hash key used for uniquing.
90  using KeyTy = std::pair<TypeRange, TypeRange>;
91  bool operator==(const KeyTy &key) const {
92  return key == KeyTy(getInputs(), getResults());
93  }
94 
95  /// Construction.
96  static FunctionTypeStorage *construct(TypeStorageAllocator &allocator,
97  const KeyTy &key) {
98  TypeRange inputs = key.first, results = key.second;
99 
100  // Copy the inputs and results into the bump pointer.
101  SmallVector<Type, 16> types;
102  types.reserve(inputs.size() + results.size());
103  types.append(inputs.begin(), inputs.end());
104  types.append(results.begin(), results.end());
105  auto typesList = allocator.copyInto(ArrayRef<Type>(types));
106 
107  // Initialize the memory using placement new.
108  return new (allocator.allocate<FunctionTypeStorage>())
109  FunctionTypeStorage(inputs.size(), results.size(), typesList.data());
110  }
111 
112  ArrayRef<Type> getInputs() const {
113  return ArrayRef<Type>(inputsAndResults, numInputs);
114  }
115  ArrayRef<Type> getResults() const {
116  return ArrayRef<Type>(inputsAndResults + numInputs, numResults);
117  }
118 
119  unsigned numInputs;
120  unsigned numResults;
121  Type const *inputsAndResults;
122 };

210 struct MemRefTypeStorage : public ShapedTypeStorage {
211  MemRefTypeStorage(unsigned shapeSize, Type elementType,
212  const int64_t *shapeElements, const unsigned numAffineMaps,
213  AffineMap const *affineMapList, const unsigned memorySpace)
214  : ShapedTypeStorage(elementType), shapeElements(shapeElements),
215  shapeSize(shapeSize), numAffineMaps(numAffineMaps),
216  affineMapList(affineMapList), memorySpace(memorySpace) {}
217 
218  /// The hash key used for uniquing.
219  // MemRefs are uniqued based on their shape, element type, affine map
220  // composition, and memory space.
221  using KeyTy =
222  std::tuple<ArrayRef<int64_t>, Type, ArrayRef<AffineMap>, unsigned>;
223  bool operator==(const KeyTy &key) const {
224  return key == KeyTy(getShape(), elementType, getAffineMaps(), memorySpace);
225  }
226 
227  /// Construction.
228  static MemRefTypeStorage *construct(TypeStorageAllocator &allocator,
229  const KeyTy &key) {
230  // Copy the shape into the bump pointer.
231  ArrayRef<int64_t> shape = allocator.copyInto(std::get<0>(key));
232 
233  // Copy the affine map composition into the bump pointer.
234  ArrayRef<AffineMap> affineMapComposition =
235  allocator.copyInto(std::get<2>(key));
236 
237  // Initialize the memory using placement new.
238  return new (allocator.allocate<MemRefTypeStorage>())
239  MemRefTypeStorage(shape.size(), std::get<1>(key), shape.data(),
240  affineMapComposition.size(),
241  affineMapComposition.data(), std::get<3>(key));
242  }
243 
244  ArrayRef<int64_t> getShape() const {
245  return ArrayRef<int64_t>(shapeElements, shapeSize);
246  }
247 
248  ArrayRef<AffineMap> getAffineMaps() const {
249  return ArrayRef<AffineMap>(affineMapList, numAffineMaps);
250  }
251 
252  /// An array of integers which stores the shape dimension sizes.
253  const int64_t *shapeElements;
254  /// The number of shape elements.
255  unsigned shapeSize;
256  /// The number of affine maps in the 'affineMapList' array.
257  const unsigned numAffineMaps;
258  /// List of affine maps in the memref's layout/index map composition.
259  AffineMap const *affineMapList;
260  /// Memory space in which data referenced by memref resides.
261  const unsigned memorySpace;
262 };

307 /// A type representing a collection of other types.
308 struct TupleTypeStorage final
309  : public TypeStorage,
310  public llvm::TrailingObjects<TupleTypeStorage, Type> {
311  using KeyTy = TypeRange;
312 
313  TupleTypeStorage(unsigned numTypes) : numElements(numTypes) {}
314 
315  /// Construction.
316  static TupleTypeStorage *construct(TypeStorageAllocator &allocator,
317  TypeRange key) {
318  // Allocate a new storage instance.
319  auto byteSize = TupleTypeStorage::totalSizeToAlloc<Type>(key.size());
320  auto rawMem = allocator.allocate(byteSize, alignof(TupleTypeStorage));
321  auto result = ::new (rawMem) TupleTypeStorage(key.size());
322 
323  // Copy in the element types into the trailing storage.
324  std::uninitialized_copy(key.begin(), key.end(),
325  result->getTrailingObjects<Type>());
326  return result;
327  }
328 
329  bool operator==(const KeyTy &key) const { return key == getTypes(); }
330 
331  /// Return the number of held types.
332  unsigned size() const { return numElements; }
333 
334  /// Return the held types.
335  ArrayRef<Type> getTypes() const {
336  return {getTrailingObjects<Type>(), size()};
337  }
338 
339  /// The number of tuple elements.
340  unsigned numElements;
