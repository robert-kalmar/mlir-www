25 /// Converts `atomic_rmw` that cannot be lowered to a simple atomic op with
26 /// AtomicRMWOpLowering pattern, e.g. with "minf" or "maxf" attributes, to
27 /// `generic_atomic_rmw` with the expanded code.
28 ///
29 /// %x = atomic_rmw "maxf" %fval, %F[%i] : (f32, memref<10xf32>) -> f32
30 ///
31 /// will be lowered to
32 ///
33 /// %x = std.generic_atomic_rmw %F[%i] : memref<10xf32> {
34 /// ^bb0(%current: f32):
35 /// %cmp = cmpf "ogt", %current, %fval : f32
36 /// %new_value = select %cmp, %current, %fval : f32
37 /// atomic_yield %new_value : f32
38 /// }
39 struct AtomicRMWOpConverter : public OpRewritePattern<AtomicRMWOp> {
40 public:
41  using OpRewritePattern::OpRewritePattern;
42 
43  LogicalResult matchAndRewrite(AtomicRMWOp op,
44  PatternRewriter &rewriter) const final {
45  CmpFPredicate predicate;
46  switch (op.kind()) {
47  case AtomicRMWKind::maxf:
48  predicate = CmpFPredicate::OGT;
49  break;
50  case AtomicRMWKind::minf:
51  predicate = CmpFPredicate::OLT;
52  break;
53  default:
54  return failure();
55  }
56 
57  auto loc = op.getLoc();
58  auto genericOp =
59  rewriter.create<GenericAtomicRMWOp>(loc, op.memref(), op.indices());
60  OpBuilder bodyBuilder = OpBuilder::atBlockEnd(genericOp.getBody());
61 
62  Value lhs = genericOp.getCurrentValue();
63  Value rhs = op.value();
64  Value cmp = bodyBuilder.create<CmpFOp>(loc, predicate, lhs, rhs);
65  Value select = bodyBuilder.create<SelectOp>(loc, cmp, lhs, rhs);
66  bodyBuilder.create<AtomicYieldOp>(loc, select);
67 
68  rewriter.replaceOp(op, genericOp.getResult());
69  return success();
70  }
71 };

73 struct ExpandAtomic : public ExpandAtomicBase<ExpandAtomic> {
74  void runOnFunction() override {
75  OwningRewritePatternList patterns;
76  patterns.insert<AtomicRMWOpConverter>(&getContext());
77 
78  ConversionTarget target(getContext());
79  target.addLegalOp<GenericAtomicRMWOp>();
80  target.addDynamicallyLegalOp<AtomicRMWOp>([](AtomicRMWOp op) {
81  return op.kind() != AtomicRMWKind::maxf &&
82  op.kind() != AtomicRMWKind::minf;
83  });
84  if (failed(mlir::applyPartialConversion(getFunction(), target, patterns)))
85  signalPassFailure();
86  }
87 };
