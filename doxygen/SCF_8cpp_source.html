26  // destination regions (e.g. while/conditional bodies). Always allow it.

29  return true;

32  // pure.

36  }
37  // Handle the given inlined terminator by replacing it with a new operation
38  // as necessary. Required when the region has only one block.
39  void handleTerminator(Operation *op,
40  ArrayRef<Value> valuesToRepl) const final {
41  auto retValOp = dyn_cast<YieldOp>(op);
42  if (!retValOp)
43  return;
44 
45  for (auto retValue : llvm::zip(valuesToRepl, retValOp.getOperands())) {
46  std::get<0>(retValue).replaceAllUsesWith(std::get<1>(retValue));
47  }
48  }

73 void ForOp::build(OpBuilder &builder, OperationState &result, Value lb,
74  Value ub, Value step, ValueRange iterArgs,
75  BodyBuilderFn bodyBuilder) {
76  result.addOperands({lb, ub, step});
77  result.addOperands(iterArgs);
78  for (Value v : iterArgs)
79  result.addTypes(v.getType());
80  Region *bodyRegion = result.addRegion();
81  bodyRegion->push_back(new Block);
82  Block &bodyBlock = bodyRegion->front();
83  bodyBlock.addArgument(builder.getIndexType());
84  for (Value v : iterArgs)
85  bodyBlock.addArgument(v.getType());
86 
87  // Create the default terminator if the builder is not provided and if the
88  // iteration arguments are not provided. Otherwise, leave this to the caller
89  // because we don't know which values to return from the loop.
90  if (iterArgs.empty() && !bodyBuilder) {
91  ForOp::ensureTerminator(*bodyRegion, builder, result.location);
92  } else if (bodyBuilder) {
93  OpBuilder::InsertionGuard guard(builder);
94  builder.setInsertionPointToStart(&bodyBlock);
95  bodyBuilder(builder, result.location, bodyBlock.getArgument(0),
96  bodyBlock.getArguments().drop_front());
97  }
98 }

100 static LogicalResult verify(ForOp op) {
101  if (auto cst = op.step().getDefiningOp<ConstantIndexOp>())
102  if (cst.getValue() <= 0)
103  return op.emitOpError("constant step operand must be positive");
104 
105  // Check that the body defines as single block argument for the induction
106  // variable.
107  auto *body = op.getBody();
108  if (!body->getArgument(0).getType().isIndex())
109  return op.emitOpError(
110  "expected body first argument to be an index argument for "
111  "the induction variable");
112 
113  auto opNumResults = op.getNumResults();
114  if (opNumResults == 0)
115  return success();
116  // If ForOp defines values, check that the number and types of
117  // the defined values match ForOp initial iter operands and backedge
118  // basic block arguments.
119  if (op.getNumIterOperands() != opNumResults)
120  return op.emitOpError(
121  "mismatch in number of loop-carried values and defined values");
122  if (op.getNumRegionIterArgs() != opNumResults)
123  return op.emitOpError(
124  "mismatch in number of basic block args and defined values");
125  auto iterOperands = op.getIterOperands();
126  auto iterArgs = op.getRegionIterArgs();
127  auto opResults = op.getResults();
128  unsigned i = 0;
129  for (auto e : llvm::zip(iterOperands, iterArgs, opResults)) {
130  if (std::get<0>(e).getType() != std::get<2>(e).getType())
131  return op.emitOpError() << "types mismatch between " << i
132  << "th iter operand and defined value";
133  if (std::get<1>(e).getType() != std::get<2>(e).getType())
134  return op.emitOpError() << "types mismatch between " << i
135  << "th iter region arg and defined value";
136 
137  i++;
138  }
139 
140  return RegionBranchOpInterface::verifyTypes(op);
141 }

166 static ParseResult parseForOp(OpAsmParser &parser, OperationState &result) {
167  auto &builder = parser.getBuilder();
168  OpAsmParser::OperandType inductionVariable, lb, ub, step;
169  // Parse the induction variable followed by '='.
170  if (parser.parseRegionArgument(inductionVariable) || parser.parseEqual())
171  return failure();
172 
173  // Parse loop bounds.
174  Type indexType = builder.getIndexType();
175  if (parser.parseOperand(lb) ||
176  parser.resolveOperand(lb, indexType, result.operands) ||
177  parser.parseKeyword("to") || parser.parseOperand(ub) ||
178  parser.resolveOperand(ub, indexType, result.operands) ||
179  parser.parseKeyword("step") || parser.parseOperand(step) ||
180  parser.resolveOperand(step, indexType, result.operands))
181  return failure();
182 
183  // Parse the optional initial iteration arguments.
184  SmallVector<OpAsmParser::OperandType, 4> regionArgs, operands;
185  SmallVector<Type, 4> argTypes;
186  regionArgs.push_back(inductionVariable);
187 
188  if (succeeded(parser.parseOptionalKeyword("iter_args"))) {
189  // Parse assignment list and results type list.
190  if (parser.parseAssignmentList(regionArgs, operands) ||
191  parser.parseArrowTypeList(result.types))
192  return failure();
193  // Resolve input operands.
194  for (auto operand_type : llvm::zip(operands, result.types))
195  if (parser.resolveOperand(std::get<0>(operand_type),
196  std::get<1>(operand_type), result.operands))
197  return failure();
198  }
199  // Induction variable.
200  argTypes.push_back(indexType);
201  // Loop carried variables
202  argTypes.append(result.types.begin(), result.types.end());
203  // Parse the body region.
204  Region *body = result.addRegion();
205  if (regionArgs.size() != argTypes.size())
206  return parser.emitError(
207  parser.getNameLoc(),
208  "mismatch in number of loop-carried values and defined values");
209 
210  if (parser.parseRegion(*body, regionArgs, argTypes))
211  return failure();
212 
213  ForOp::ensureTerminator(*body, builder, result.location);
214 
215  // Parse the optional attribute list.
216  if (parser.parseOptionalAttrDict(result.attributes))
217  return failure();
218 
219  return success();
220 }

255 /// Given the region at `index`, or the parent operation if `index` is None,
256 /// return the successor regions. These are the regions that may be selected
257 /// during the flow of control. `operands` is a set of optional attributes that
258 /// correspond to a constant value for each operand, or null if that operand is
259 /// not a constant.
260 void ForOp::getSuccessorRegions(Optional<unsigned> index,
261  ArrayRef<Attribute> operands,
262  SmallVectorImpl<RegionSuccessor> &regions) {
263  // If the predecessor is the ForOp, branch into the body using the iterator
264  // arguments.
265  if (!index.hasValue()) {
266  regions.push_back(RegionSuccessor(&getLoopBody(), getRegionIterArgs()));
267  return;
268  }
269 
270  // Otherwise, the loop may branch back to itself or the parent operation.
271  assert(index.getValue() == 0 && "expected loop region");
272  regions.push_back(RegionSuccessor(&getLoopBody(), getRegionIterArgs()));
273  regions.push_back(RegionSuccessor(getResults()));
274 }

276 ValueVector mlir::scf::buildLoopNest(
277  OpBuilder &builder, Location loc, ValueRange lbs, ValueRange ubs,
278  ValueRange steps, ValueRange iterArgs,
279  function_ref<ValueVector(OpBuilder &, Location, ValueRange, ValueRange)>
280  bodyBuilder) {
281  assert(lbs.size() == ubs.size() &&
282  "expected the same number of lower and upper bounds");
283  assert(lbs.size() == steps.size() &&
284  "expected the same number of lower bounds and steps");
285 
286  // If there are no bounds, call the body-building function and return early.
287  if (lbs.empty()) {
288  ValueVector results =
289  bodyBuilder ? bodyBuilder(builder, loc, ValueRange(), iterArgs)
290  : ValueVector();
291  assert(results.size() == iterArgs.size() &&
292  "loop nest body must return as many values as loop has iteration "
293  "arguments");
294  return results;
295  }
296 
297  // First, create the loop structure iteratively using the body-builder
298  // callback of `ForOp::build`. Do not create `YieldOp`s yet.
299  OpBuilder::InsertionGuard guard(builder);
300  SmallVector<scf::ForOp, 4> loops;
301  SmallVector<Value, 4> ivs;
302  loops.reserve(lbs.size());
303  ivs.reserve(lbs.size());
304  ValueRange currentIterArgs = iterArgs;
305  Location currentLoc = loc;
306  for (unsigned i = 0, e = lbs.size(); i < e; ++i) {
307  auto loop = builder.create<scf::ForOp>(
308  currentLoc, lbs[i], ubs[i], steps[i], currentIterArgs,
309  [&](OpBuilder &nestedBuilder, Location nestedLoc, Value iv,
310  ValueRange args) {
311  ivs.push_back(iv);
312  // It is safe to store ValueRange args because it points to block
313  // arguments of a loop operation that we also own.
314  currentIterArgs = args;
315  currentLoc = nestedLoc;
316  });
317  // Set the builder to point to the body of the newly created loop. We don't
318  // do this in the callback because the builder is reset when the callback
319  // returns.
320  builder.setInsertionPointToStart(loop.getBody());
321  loops.push_back(loop);
322  }
323 
324  // For all loops but the innermost, yield the results of the nested loop.
325  for (unsigned i = 0, e = loops.size() - 1; i < e; ++i) {
326  builder.setInsertionPointToEnd(loops[i].getBody());
327  builder.create<scf::YieldOp>(loc, loops[i + 1].getResults());
328  }
329 
330  // In the body of the innermost loop, call the body building function if any
331  // and yield its results.
332  builder.setInsertionPointToStart(loops.back().getBody());
333  ValueVector results = bodyBuilder
334  ? bodyBuilder(builder, currentLoc, ivs,
335  loops.back().getRegionIterArgs())
336  : ValueVector();
337  assert(results.size() == iterArgs.size() &&
338  "loop nest body must return as many values as loop has iteration "
339  "arguments");
340  builder.setInsertionPointToEnd(loops.back().getBody());
341  builder.create<scf::YieldOp>(loc, results);
342 
343  // Return the results of the outermost loop.
344  return ValueVector(loops.front().result_begin(), loops.front().result_end());
345 }

419 static ParseResult parseIfOp(OpAsmParser &parser, OperationState &result) {
420  // Create the regions for 'then'.
421  result.regions.reserve(2);
422  Region *thenRegion = result.addRegion();
423  Region *elseRegion = result.addRegion();
424 
425  auto &builder = parser.getBuilder();
426  OpAsmParser::OperandType cond;
427  Type i1Type = builder.getIntegerType(1);
428  if (parser.parseOperand(cond) ||
429  parser.resolveOperand(cond, i1Type, result.operands))
430  return failure();
431  // Parse optional results type list.
432  if (parser.parseOptionalArrowTypeList(result.types))
433  return failure();
434  // Parse the 'then' region.
435  if (parser.parseRegion(*thenRegion, /*arguments=*/{}, /*argTypes=*/{}))
436  return failure();
437  IfOp::ensureTerminator(*thenRegion, parser.getBuilder(), result.location);
438 
439  // If we find an 'else' keyword then parse the 'else' region.
440  if (!parser.parseOptionalKeyword("else")) {
441  if (parser.parseRegion(*elseRegion, /*arguments=*/{}, /*argTypes=*/{}))
442  return failure();
443  IfOp::ensureTerminator(*elseRegion, parser.getBuilder(), result.location);
444  }
445 
446  // Parse the optional attribute list.
447  if (parser.parseOptionalAttrDict(result.attributes))
448  return failure();
449  return success();
450 }

477 /// Given the region at `index`, or the parent operation if `index` is None,
478 /// return the successor regions. These are the regions that may be selected
479 /// during the flow of control. `operands` is a set of optional attributes that
480 /// correspond to a constant value for each operand, or null if that operand is
481 /// not a constant.
482 void IfOp::getSuccessorRegions(Optional<unsigned> index,
483  ArrayRef<Attribute> operands,
484  SmallVectorImpl<RegionSuccessor> &regions) {
485  // The `then` and the `else` region branch back to the parent operation.
486  if (index.hasValue()) {
487  regions.push_back(RegionSuccessor(getResults()));
488  return;
489  }
490 
491  // Don't consider the else region if it is empty.
492  Region *elseRegion = &this->elseRegion();
493  if (elseRegion->empty())
494  elseRegion = nullptr;
495 
496  // Otherwise, the successor is dependent on the condition.
497  bool condition;
498  if (auto condAttr = operands.front().dyn_cast_or_null<IntegerAttr>()) {
499  condition = condAttr.getValue().isOneValue();
500  } else {
501  // If the condition isn't constant, both regions may be executed.
502  regions.push_back(RegionSuccessor(&thenRegion()));
503  regions.push_back(RegionSuccessor(elseRegion));
504  return;
505  }
506 
507  // Add the successor regions using the condition.
508  regions.push_back(RegionSuccessor(condition ? &thenRegion() : elseRegion));
509 }

567 static LogicalResult verify(ParallelOp op) {
568  // Check that there is at least one value in lowerBound, upperBound and step.
569  // It is sufficient to test only step, because it is ensured already that the
570  // number of elements in lowerBound, upperBound and step are the same.
571  Operation::operand_range stepValues = op.step();
572  if (stepValues.empty())
573  return op.emitOpError(
574  "needs at least one tuple element for lowerBound, upperBound and step");
575 
576  // Check whether all constant step values are positive.
577  for (Value stepValue : stepValues)
578  if (auto cst = stepValue.getDefiningOp<ConstantIndexOp>())
579  if (cst.getValue() <= 0)
580  return op.emitOpError("constant step operand must be positive");
581 
582  // Check that the body defines the same number of block arguments as the
583  // number of tuple elements in step.
584  Block *body = op.getBody();
585  if (body->getNumArguments() != stepValues.size())
586  return op.emitOpError()
587  << "expects the same number of induction variables: "
588  << body->getNumArguments()
589  << " as bound and step values: " << stepValues.size();
590  for (auto arg : body->getArguments())
591  if (!arg.getType().isIndex())
592  return op.emitOpError(
593  "expects arguments for the induction variable to be of index type");
594 
595  // Check that the yield has no results
596  Operation *yield = body->getTerminator();
597  if (yield->getNumOperands() != 0)
598  return yield->emitOpError() << "not allowed to have operands inside '"
599  << ParallelOp::getOperationName() << "'";
600 
601  // Check that the number of results is the same as the number of ReduceOps.
602  SmallVector<ReduceOp, 4> reductions(body->getOps<ReduceOp>());
603  auto resultsSize = op.results().size();
604  auto reductionsSize = reductions.size();
605  auto initValsSize = op.initVals().size();
606  if (resultsSize != reductionsSize)
607  return op.emitOpError()
608  << "expects number of results: " << resultsSize
609  << " to be the same as number of reductions: " << reductionsSize;
610  if (resultsSize != initValsSize)
611  return op.emitOpError()
612  << "expects number of results: " << resultsSize
613  << " to be the same as number of initial values: " << initValsSize;
614 
615  // Check that the types of the results and reductions are the same.
616  for (auto resultAndReduce : llvm::zip(op.results(), reductions)) {

628 static ParseResult parseParallelOp(OpAsmParser &parser,
629  OperationState &result) {
630  auto &builder = parser.getBuilder();
631  // Parse an opening `(` followed by induction variables followed by `)`
632  SmallVector<OpAsmParser::OperandType, 4> ivs;
633  if (parser.parseRegionArgumentList(ivs, /*requiredOperandCount=*/-1,
634  OpAsmParser::Delimiter::Paren))
635  return failure();
636 
637  // Parse loop bounds.
638  SmallVector<OpAsmParser::OperandType, 4> lower;
639  if (parser.parseEqual() ||
640  parser.parseOperandList(lower, ivs.size(),
641  OpAsmParser::Delimiter::Paren) ||
642  parser.resolveOperands(lower, builder.getIndexType(), result.operands))
643  return failure();
644 
645  SmallVector<OpAsmParser::OperandType, 4> upper;
646  if (parser.parseKeyword("to") ||
647  parser.parseOperandList(upper, ivs.size(),
648  OpAsmParser::Delimiter::Paren) ||
649  parser.resolveOperands(upper, builder.getIndexType(), result.operands))
650  return failure();
651 
652  // Parse step values.
653  SmallVector<OpAsmParser::OperandType, 4> steps;
654  if (parser.parseKeyword("step") ||
655  parser.parseOperandList(steps, ivs.size(),
656  OpAsmParser::Delimiter::Paren) ||
657  parser.resolveOperands(steps, builder.getIndexType(), result.operands))
658  return failure();
659 
660  // Parse init values.
661  SmallVector<OpAsmParser::OperandType, 4> initVals;
662  if (succeeded(parser.parseOptionalKeyword("init"))) {
663  if (parser.parseOperandList(initVals, /*requiredOperandCount=*/-1,
664  OpAsmParser::Delimiter::Paren))
665  return failure();
666  }
667 
668  // Parse optional results in case there is a reduce.
669  if (parser.parseOptionalArrowTypeList(result.types))
670  return failure();
671 
672  // Now parse the body.
673  Region *body = result.addRegion();
674  SmallVector<Type, 4> types(ivs.size(), builder.getIndexType());
675  if (parser.parseRegion(*body, ivs, types))
676  return failure();
677 
678  // Set `operand_segment_sizes` attribute.
679  result.addAttribute(
680  ParallelOp::getOperandSegmentSizeAttr(),
681  builder.getI32VectorAttr({static_cast<int32_t>(lower.size()),
682  static_cast<int32_t>(upper.size()),
683  static_cast<int32_t>(steps.size()),
684  static_cast<int32_t>(initVals.size())}));
685 
686  // Parse attributes.
687  if (parser.parseOptionalAttrDict(result.attributes))
688  return failure();
689 
690  if (!initVals.empty())
691  parser.resolveOperands(initVals, result.types, parser.getNameLoc(),
692  result.operands);
693  // Add a terminator if none was parsed.
694  ForOp::ensureTerminator(*body, builder, result.location);
695 
696  return success();
697 }

733 // Collapse loop dimensions that perform a single iteration.
734 struct CollapseSingleIterationLoops : public OpRewritePattern<ParallelOp> {
735  using OpRewritePattern<ParallelOp>::OpRewritePattern;
736 
737  LogicalResult matchAndRewrite(ParallelOp op,
738  PatternRewriter &rewriter) const override {
739  BlockAndValueMapping mapping;
740  // Compute new loop bounds that omit all single-iteration loop dimensions.
741  SmallVector<Value, 2> newLowerBounds;
742  SmallVector<Value, 2> newUpperBounds;
743  SmallVector<Value, 2> newSteps;
744  newLowerBounds.reserve(op.lowerBound().size());
745  newUpperBounds.reserve(op.upperBound().size());
746  newSteps.reserve(op.step().size());
747  for (auto dim : llvm::zip(op.lowerBound(), op.upperBound(), op.step(),
748  op.getInductionVars())) {
749  Value lowerBound, upperBound, step, iv;
750  std::tie(lowerBound, upperBound, step, iv) = dim;
751  // Collect the statically known loop bounds.
752  auto lowerBoundConstant =
753  dyn_cast_or_null<ConstantIndexOp>(lowerBound.getDefiningOp());
754  auto upperBoundConstant =
755  dyn_cast_or_null<ConstantIndexOp>(upperBound.getDefiningOp());
756  auto stepConstant =
757  dyn_cast_or_null<ConstantIndexOp>(step.getDefiningOp());
758  // Replace the loop induction variable by the lower bound if the loop
759  // performs a single iteration. Otherwise, copy the loop bounds.
760  if (lowerBoundConstant && upperBoundConstant && stepConstant &&
761  (upperBoundConstant.getValue() - lowerBoundConstant.getValue()) > 0 &&
762  (upperBoundConstant.getValue() - lowerBoundConstant.getValue()) <=
763  stepConstant.getValue()) {
764  mapping.map(iv, lowerBound);
765  } else {
766  newLowerBounds.push_back(lowerBound);
767  newUpperBounds.push_back(upperBound);
768  newSteps.push_back(step);
769  }
770  }
771  // Exit if all or none of the loop dimensions perform a single iteration.
772  if (newLowerBounds.size() == 0 ||
773  newLowerBounds.size() == op.lowerBound().size())
774  return failure();
775  // Replace the parallel loop by lower-dimensional parallel loop.
776  auto newOp =
777  rewriter.create<ParallelOp>(op.getLoc(), newLowerBounds, newUpperBounds,
778  newSteps, op.initVals(), nullptr);
779  // Clone the loop body and remap the block arguments of the collapsed loops
780  // (inlining does not support a cancellable block argument mapping).
781  rewriter.cloneRegionBefore(op.region(), newOp.region(),
782  newOp.region().begin(), mapping);
783  rewriter.replaceOp(op, newOp.getResults());
784  return success();
785  }
786 };
787 } // namespace

812 static LogicalResult verify(ReduceOp op) {
813  // The region of a ReduceOp has two arguments of the same type as its operand.
814  auto type = op.operand().getType();
815  Block &block = op.reductionOperator().front();
816  if (block.empty())
817  return op.emitOpError("the block inside reduce should not be empty");
818  if (block.getNumArguments() != 2 ||
819  llvm::any_of(block.getArguments(), [&](const BlockArgument &arg) {
820  return arg.getType() != type;
821  }))
822  return op.emitOpError()
823  << "expects two arguments to reduce block of type " << type;
824 
825  // Check that the block is terminated by a ReduceReturnOp.
826  if (!isa<ReduceReturnOp>(block.getTerminator()))
827  return op.emitOpError("the block inside reduce should be terminated with a "
828  "'scf.reduce.return' op");
829 
830  return success();
831 }
