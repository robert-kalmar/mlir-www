41 // Create a call to ROCm-Device-Library function
42 // Currently this routine will work only for calling ROCDL functions that
43 // take a single int32 argument. It is likely that the interface of this
44 // function will change to make it more generic.
45 static llvm::Value *createDeviceFunctionCall(llvm::IRBuilder<> &builder,
46  StringRef fn_name, int parameter) {
47  llvm::Module *module = builder.GetInsertBlock()->getModule();
48  llvm::FunctionType *function_type = llvm::FunctionType::get(
49  llvm::Type::getInt64Ty(module->getContext()), // return type.
50  llvm::Type::getInt32Ty(module->getContext()), // parameter type.
51  false); // no variadic arguments.
52  llvm::Function *fn = dyn_cast<llvm::Function>(
53  module->getOrInsertFunction(fn_name, function_type).getCallee());
54  llvm::Value *fn_op0 = llvm::ConstantInt::get(
55  llvm::Type::getInt32Ty(module->getContext()), parameter);
56  return builder.CreateCall(fn, ArrayRef<llvm::Value *>(fn_op0));
57 }

78 std::unique_ptr<llvm::Module>
79 mlir::translateModuleToROCDLIR(Operation *m, llvm::LLVMContext &llvmContext,
80  StringRef name) {
81  // lower MLIR (with RODL Dialect) to LLVM IR (with ROCDL intrinsics)
82  auto llvmModule = LLVM::ModuleTranslation::translateModule<ModuleTranslation>(
83  m, llvmContext, name);
84 
85  // foreach GPU kernel
86  // 1. Insert AMDGPU_KERNEL calling convention.
87  // 2. Insert amdgpu-flat-workgroup-size(1, 1024) attribute.
88  for (auto func :
89  ModuleTranslation::getModuleBody(m).getOps<LLVM::LLVMFuncOp>()) {
90  if (!func.getAttrOfType<UnitAttr>(gpu::GPUDialect::getKernelFuncAttrName()))
91  continue;
92 
93  auto *llvmFunc = llvmModule->getFunction(func.getName());
94 
95  llvmFunc->setCallingConv(llvm::CallingConv::AMDGPU_KERNEL);
96 
97  llvmFunc->addFnAttr("amdgpu-flat-work-group-size", "1, 1024");
98  }
99 
100  return llvmModule;
101 }
