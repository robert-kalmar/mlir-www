27 inline bool isRowMajorMatmul(ArrayAttr indexingMaps) {
28  auto context = indexingMaps.getContext();
29  AffineExpr m, n, k;
30  bindDims(context, m, n, k);
31  auto mapA = AffineMapAttr::get(AffineMap::get(3, 0, {m, k}, context));
32  auto mapB = AffineMapAttr::get(AffineMap::get(3, 0, {k, n}, context));
33  auto mapC = AffineMapAttr::get(AffineMap::get(3, 0, {m, n}, context));
34  auto maps = ArrayAttr::get({mapA, mapB, mapC}, context);
35  return indexingMaps == maps;
36 }

115 /// Returns the iterator of a certain type.
116 inline unsigned getNumIterators(StringRef name, ArrayAttr iteratorTypes) {
117  auto names = getAllIteratorTypeNames();
118  (void)names;
119  assert(llvm::is_contained(names, name));
120  return llvm::count_if(iteratorTypes, [name](Attribute a) {
121  return a.cast<StringAttr>().getValue() == name;
122  });
123 }
124 
125 inline unsigned getNumIterators(ArrayAttr iteratorTypes) {
126  unsigned res = 0;
127  for (auto n : getAllIteratorTypeNames())
128  res += getNumIterators(n, iteratorTypes);
129  return res;
130 }
