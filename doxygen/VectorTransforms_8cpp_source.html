72 // Helper to construct an affine map with one index removed.
73 static AffineMap adjustMap(AffineMap map, int64_t index,
74  PatternRewriter &rewriter) {
75  auto *ctx = rewriter.getContext();
76  SmallVector<AffineExpr, 4> results;
77  for (int64_t i = 0, e = map.getNumResults(); i < e; ++i) {
78  int64_t idx = map.getResult(i).cast<AffineDimExpr>().getPosition();
79  if (idx == index)

89 // Helper to drop dimension from vector type.
90 static Type adjustType(VectorType tp, int64_t index) {
91  int64_t rank = tp.getRank();
92  Type eltType = tp.getElementType();
93  if (rank == 1) {
94  assert(index == 0 && "index for scalar result out of bounds");
95  return eltType;
96  }
97  SmallVector<int64_t, 4> adjustedShape;
98  for (int64_t i = 0; i < rank; ++i) {
99  // Omit dimension at the given index.
100  if (i == index)
101  continue;
102  // Otherwise, add dimension back.
103  adjustedShape.push_back(tp.getDimSize(i));
104  }
105  return VectorType::get(adjustedShape, eltType);
106 }

110 static Value reshapeLoad(Location loc, Value val, VectorType type,
111  int64_t index, int64_t pos,
112  PatternRewriter &rewriter) {

114  return val;
115  Type lowType = adjustType(type, 0);
116  // At extraction dimension?
117  if (index == 0) {
118  auto posAttr = rewriter.getI64ArrayAttr(pos);
119  return rewriter.create<vector::ExtractOp>(loc, lowType, val, posAttr);
120  }
121  // Unroll leading dimensions.
122  VectorType vType = lowType.cast<VectorType>();
123  VectorType resType = adjustType(type, index).cast<VectorType>();
124  Value result =
125  rewriter.create<ConstantOp>(loc, resType, rewriter.getZeroAttr(resType));
126  for (int64_t d = 0, e = resType.getDimSize(0); d < e; d++) {
127  auto posAttr = rewriter.getI64ArrayAttr(d);
128  Value ext = rewriter.create<vector::ExtractOp>(loc, vType, val, posAttr);
129  Value load = reshapeLoad(loc, ext, vType, index - 1, pos, rewriter);
130  result =
131  rewriter.create<vector::InsertOp>(loc, resType, load, result, posAttr);
132  }
133  return result;

138 static Value reshapeStore(Location loc, Value val, Value result,
139  VectorType type, int64_t index, int64_t pos,
140  PatternRewriter &rewriter) {
141  // Unmodified?
142  if (index == -1)
143  return val;
144  // At insertion dimension?
145  if (index == 0) {
146  auto posAttr = rewriter.getI64ArrayAttr(pos);
147  return rewriter.create<vector::InsertOp>(loc, type, val, result, posAttr);
148  }
149  // Unroll leading dimensions.
150  Type lowType = adjustType(type, 0);
151  VectorType vType = lowType.cast<VectorType>();
152  Type insType = adjustType(vType, 0);
153  for (int64_t d = 0, e = type.getDimSize(0); d < e; d++) {
154  auto posAttr = rewriter.getI64ArrayAttr(d);
155  Value ext = rewriter.create<vector::ExtractOp>(loc, vType, result, posAttr);
156  Value ins = rewriter.create<vector::ExtractOp>(loc, insType, val, posAttr);
157  Value sto = reshapeStore(loc, ins, ext, vType, index - 1, pos, rewriter);
158  result = rewriter.create<vector::InsertOp>(loc, type, sto, result, posAttr);
159  }
160  return result;

188 // Returns a tuple type with vector element types for each resulting slice
189 // of 'vectorType' unrolled by 'sizes' and 'strides'.
190 // TODO: Move this to a utility function and share it with
191 // Extract/InsertSlicesOp verification.
192 static TupleType generateExtractSlicesOpResultType(VectorType vectorType,
193  ArrayRef<int64_t> sizes,
194  ArrayRef<int64_t> strides,
195  OpBuilder &builder) {
196  assert(llvm::all_of(strides, [](int64_t s) { return s == 1; }));
197  assert(static_cast<int64_t>(sizes.size()) == vectorType.getRank());
198  assert(static_cast<int64_t>(strides.size()) == vectorType.getRank());
199 
200  // Compute shape ratio of 'shape' and 'sizes'.
201  auto shape = vectorType.getShape();
202  auto maybeDimSliceCounts = shapeRatio(shape, sizes);
203  assert(maybeDimSliceCounts.hasValue());
204  auto sliceDimCounts = *maybeDimSliceCounts;
205 
206  // Compute strides w.r.t number of slices in each dimension.
207  auto sliceStrides = computeStrides(sliceDimCounts);
208  int64_t sliceCount = computeMaxLinearIndex(sliceDimCounts);
209  SmallVector<Type, 4> vectorTypes(sliceCount);
210  for (unsigned i = 0; i < sliceCount; ++i) {
211  auto vectorOffsets = delinearize(sliceStrides, i);
212  auto elementOffsets =
213  computeElementOffsetsFromVectorSliceOffsets(sizes, vectorOffsets);
214  auto sliceSizes = computeSliceSizes(shape, sizes, elementOffsets);
215  // Create Vector type and add to 'vectorTypes[i]'.
216  vectorTypes[i] = VectorType::get(sliceSizes, vectorType.getElementType());
217  }
218  return TupleType::get(vectorTypes, builder.getContext());

231 // Populates 'state' with unrolled shape, unroll factors, basis and
232 // num unrolled instances for 'vectorType'.
233 static void initUnrolledVectorState(VectorType vectorType, Value initValue,
234  const DenseMap<int64_t, int64_t> &indexMap,
235  ArrayRef<int64_t> targetShape,
236  UnrolledVectorState &state,
237  OpBuilder &builder) {
238  // Compute unrolled shape of 'vectorType'.
239  state.unrolledShape.resize(vectorType.getRank());
240  getMappedElements(indexMap, targetShape, state.unrolledShape);
241  // Compute unroll factors for unrolled shape.
242  auto maybeUnrollFactors =
243  shapeRatio(vectorType.getShape(), state.unrolledShape);
244  assert(maybeUnrollFactors.hasValue());
245  state.unrollFactors = *maybeUnrollFactors;
246  // Compute 'basis' and 'numInstances' based on 'state.unrollFactors'.
247  state.basis = computeStrides(state.unrollFactors);
248  state.numInstances = computeMaxLinearIndex(state.unrollFactors);
249  state.slicesTuple = nullptr;
250  if (initValue != nullptr) {
251  // Create ExtractSlicesOp.
252  SmallVector<int64_t, 4> sizes(state.unrolledShape);
253  SmallVector<int64_t, 4> strides(state.unrollFactors.size(), 1);
254  auto tupleType =
255  generateExtractSlicesOpResultType(vectorType, sizes, strides, builder);
256  state.slicesTuple = builder.create<vector::ExtractSlicesOp>(
257  initValue.getLoc(), tupleType, initValue, sizes, strides);
258  }
259 }

274 // Returns an unrolled vector at 'vectorOffsets' within the vector
275 // represented by 'state'. The vector is created from a slice of 'initValue'
276 // if not present in 'cache'.
277 static Value getOrCreateUnrolledVectorSlice(
278  Location loc, UnrolledVectorState &state, ArrayRef<int64_t> vectorOffsets,
279  ArrayRef<int64_t> offsets, DenseMap<int64_t, int64_t> &indexMap,
280  Value initValue, SmallVectorImpl<Value> &cache, OpBuilder &builder) {
281  // Compute slice offsets.
282  SmallVector<int64_t, 4> sliceOffsets(state.unrolledShape.size());
283  getMappedElements(indexMap, offsets, sliceOffsets);
284  // TODO: Support non-1 strides.
285  SmallVector<int64_t, 4> sliceStrides(state.unrolledShape.size(), 1);
286  // Compute linear index of 'sliceOffsets' w.r.t 'state.basis'.
287  int64_t sliceLinearIndex =
288  getUnrolledVectorLinearIndex(state, vectorOffsets, indexMap);
289  assert(sliceLinearIndex < static_cast<int64_t>(cache.size()));
290  auto valueSlice = cache[sliceLinearIndex];
291  if (valueSlice == nullptr) {
292  // Return tuple element at 'sliceLinearIndex'.
293  auto tupleIndex = builder.getI64IntegerAttr(sliceLinearIndex);
294  auto initValueType = initValue.getType().cast<VectorType>();
295  auto vectorType =
296  VectorType::get(state.unrolledShape, initValueType.getElementType());
297  // Initialize 'cache' with slice from 'initValue'.
298  valueSlice = builder.create<vector::TupleGetOp>(
299  loc, vectorType, state.slicesTuple, tupleIndex);
300  // Store value back to 'cache'.
301  cache[sliceLinearIndex] = valueSlice;
302  }
303  return valueSlice;
304 }

365 // TODO: Generalize this to support structured ops beyond
366 // vector ContractionOp, and merge it with 'unrollSingleResultVectorOp'
367 static Value unrollSingleResultStructuredOp(Operation *op,
368  ArrayRef<int64_t> iterationBounds,
369  std::vector<VectorState> &vectors,
370  unsigned resultIndex,
371  ArrayRef<int64_t> targetShape,
372  OpBuilder &builder) {
373  auto shapedType = op->getResult(0).getType().dyn_cast_or_null<ShapedType>();
374  if (!shapedType || !shapedType.hasStaticShape())
375  assert(false && "Expected a statically shaped result type");
376 
377  // Compute unroll factors for 'iterationBounds' based on 'targetShape'
378  auto maybeUnrollFactors = shapeRatio(iterationBounds, targetShape);
379  if (!maybeUnrollFactors.hasValue())
380  assert(false && "Failed to compute unroll factors for target shape");
381  auto unrollFactors = *maybeUnrollFactors;
382 
383  // Compute unrolled vector state for each vector in 'vectors'.
384  unsigned numVectors = vectors.size();
385  SmallVector<UnrolledVectorState, 3> unrolledVectorState(numVectors);
386  for (unsigned i = 0; i < numVectors; ++i) {
387  int64_t operandIndex = vectors[i].operandIndex;
388  auto operand = operandIndex >= 0 ? op->getOperand(operandIndex) : nullptr;
389  initUnrolledVectorState(vectors[i].type, operand, vectors[i].indexMap,
390  targetShape, unrolledVectorState[i], builder);
391  }
392  // Compute number of total unrolled instances.
393  auto numUnrolledInstances = computeMaxLinearIndex(unrollFactors);
394  auto sliceStrides = computeStrides(unrollFactors);
395 
396  auto &resultValueState = unrolledVectorState[resultIndex];
397  auto unrolledResultType = VectorType::get(resultValueState.unrolledShape,
398  shapedType.getElementType());
399 
400  // Initialize caches for intermediate vector results.
401  std::vector<SmallVector<Value, 4>> caches(numVectors);
402  for (unsigned i = 0; i < numVectors; ++i)
403  caches[i].resize(unrolledVectorState[i].numInstances);
404 
405  // Unroll 'numUnrolledInstances' of 'op', storing results in 'caches'.
406  for (unsigned i = 0; i < numUnrolledInstances; ++i) {
407  auto vectorOffsets = delinearize(sliceStrides, i);
408  auto elementOffsets =
409  computeElementOffsetsFromVectorSliceOffsets(targetShape, vectorOffsets);
410  // Get cached slice (or create slice) for each operand at 'offsets'.
411  SmallVector<Value, 3> operands;
412  operands.resize(op->getNumOperands());
413  for (unsigned i = 0; i < numVectors; ++i) {
414  int64_t operandIndex = vectors[i].operandIndex;
415  if (operandIndex < 0)
416  continue; // Output
417  auto operand = op->getOperand(operandIndex);
418  operands[operandIndex] = getOrCreateUnrolledVectorSlice(
419  op->getLoc(), unrolledVectorState[i], vectorOffsets, elementOffsets,
420  vectors[i].indexMap, operand, caches[i], builder);
421  }
422  // Create op on sliced vector arguments.
423  auto resultVector =
424  cloneOpWithOperandsAndTypes(builder, op->getLoc(), op, operands,
425  unrolledResultType)
426  ->getResult(0);
427 
428  // Compute linear result index.
429  int64_t linearIndex = getUnrolledVectorLinearIndex(
430  resultValueState, vectorOffsets, vectors[resultIndex].indexMap);
431  // Update result cache at 'linearIndex'.
432  caches[resultIndex][linearIndex] = resultVector;
433  }
434 
435  // Create TupleOp of unrolled result vectors.
436  SmallVector<Type, 4> vectorTupleTypes(resultValueState.numInstances);
437  SmallVector<Value, 4> vectorTupleValues(resultValueState.numInstances);
438  for (unsigned i = 0; i < resultValueState.numInstances; ++i) {
439  vectorTupleTypes[i] = caches[resultIndex][i].getType().cast<VectorType>();
440  vectorTupleValues[i] = caches[resultIndex][i];
441  }
442  TupleType tupleType = builder.getTupleType(vectorTupleTypes);
443  Value tupleOp = builder.create<vector::TupleOp>(op->getLoc(), tupleType,
444  vectorTupleValues);
445 
446  // Create InsertSlicesOp(Tuple(result_vectors)).
447  auto resultVectorType = op->getResult(0).getType().cast<VectorType>();
448  SmallVector<int64_t, 4> sizes(resultValueState.unrolledShape);
449  SmallVector<int64_t, 4> strides(resultValueState.unrollFactors.size(), 1);
450 
451  Value insertSlicesOp = builder.create<vector::InsertSlicesOp>(
452  op->getLoc(), resultVectorType, tupleOp, builder.getI64ArrayAttr(sizes),
453  builder.getI64ArrayAttr(strides));
454  return insertSlicesOp;
455 }

457 static void getVectorContractionOpUnrollState(
458  vector::ContractionOp contractionOp, ArrayRef<int64_t> targetShape,
459  std::vector<VectorState> &vectors, unsigned &resultIndex) {
460  // Get map from iteration space index to lhs/rhs/result shape index.
461  std::vector<DenseMap<int64_t, int64_t>> iterationIndexMapList;
462  contractionOp.getIterationIndexMap(iterationIndexMapList);
463  unsigned numIterators = iterationIndexMapList.size();
464  vectors.resize(numIterators);
465  unsigned accOperandIndex = vector::ContractionOp::getAccOperandIndex();
466  for (unsigned i = 0; i < numIterators; ++i) {
467  vectors[i].type = contractionOp.getOperand(i).getType().cast<VectorType>();
468  vectors[i].indexMap = iterationIndexMapList[i];
469  vectors[i].operandIndex = i;
470  vectors[i].isAcc = i == accOperandIndex ? true : false;
471  }
472 
473  if (llvm::size(contractionOp.masks()) == 2) {
474  // Add vectors for lhs/rhs vector mask arguments. Masks have the
475  // same vector shape lhs/rhs args, so copy their index maps.
476  vectors.push_back({contractionOp.getLHSVectorMaskType(),
477  vectors[0].indexMap, accOperandIndex + 1, false});
478  vectors.push_back({contractionOp.getRHSVectorMaskType(),
479  vectors[1].indexMap, accOperandIndex + 2, false});
480  }
481  // TODO: Use linalg style 'args_in'/'args_out' to partition
482  // 'vectors' instead of 'resultIndex'.
483  resultIndex = accOperandIndex;
484 }

486 static void getVectorElementwiseOpUnrollState(Operation *op,
487  ArrayRef<int64_t> targetShape,
488  std::vector<VectorState> &vectors,
489  unsigned &resultIndex) {
490  // Verify that operation and operands all have the same vector shape.
491  auto resultType = op->getResult(0).getType().dyn_cast_or_null<VectorType>();
492  assert(resultType && "Expected op with vector result type");
493  auto resultShape = resultType.getShape();
494  // Verify that all operands have the same vector type as result.
495  assert(llvm::all_of(op->getOperandTypes(),
496  [=](Type type) { return type == resultType; }));
497 
498  // Create trivial elementwise identity index map based on 'resultShape'.
499  DenseMap<int64_t, int64_t> indexMap;
500  indexMap.reserve(resultShape.size());
501  for (unsigned i = 0; i < resultShape.size(); ++i)
502  indexMap[i] = i;
503 
504  // Create VectorState each operand and single result.
505  unsigned numVectors = op->getNumOperands() + op->getNumResults();
506  vectors.resize(numVectors);
507  for (unsigned i = 0; i < op->getNumOperands(); ++i)
508  vectors[i] = {resultType, indexMap, i, false};
509  vectors[numVectors - 1] = {resultType, indexMap, -1, false};
510  resultIndex = numVectors - 1;
511 }

542 /// Generates slices of 'vectorType' according to 'sizes' and 'strides, and
543 /// calls 'fn' with linear index and indices for each slice.
544 static void generateTransferOpSlices(
545  Type memrefElementType, VectorType vectorType, TupleType tupleType,
546  ArrayRef<int64_t> sizes, ArrayRef<int64_t> strides, ArrayRef<Value> indices,
547  OpBuilder &builder, function_ref<void(unsigned, ArrayRef<Value>)> fn) {
548  // Compute strides w.r.t. to slice counts in each dimension.
549  auto maybeDimSliceCounts = shapeRatio(vectorType.getShape(), sizes);
550  assert(maybeDimSliceCounts.hasValue());
551  auto sliceDimCounts = *maybeDimSliceCounts;
552  auto sliceStrides = computeStrides(sliceDimCounts);
553 
554  int64_t numSlices = tupleType.size();
555  unsigned numSliceIndices = indices.size();
556  // Compute 'indexOffset' at which to update 'indices', which is equal
557  // to the memref rank (indices.size) minus the effective 'vectorRank'.
558  // The effective 'vectorRank', is equal to the rank of the vector type
559  // minus the rank of the memref vector element type (if it has one).
560  //
561  // For example:
562  //
563  // Given memref type 'memref<6x2x1xvector<2x4xf32>>' and vector
564  // transfer_read/write ops which read/write vectors of type
565  // 'vector<2x1x2x4xf32>'. The memref rank is 3, and the effective
566  // vector rank is 4 - 2 = 2, and so 'indexOffset' = 3 - 2 = 1.
567  //
568  unsigned vectorRank = vectorType.getRank();
569  if (auto memrefVectorElementType = memrefElementType.dyn_cast<VectorType>()) {
570  assert(vectorRank >= memrefVectorElementType.getRank());
571  vectorRank -= memrefVectorElementType.getRank();
572  }
573  unsigned indexOffset = numSliceIndices - vectorRank;
574 
575  auto *ctx = builder.getContext();
576  for (unsigned i = 0; i < numSlices; ++i) {
577  auto vectorOffsets = delinearize(sliceStrides, i);
578  auto elementOffsets =
579  computeElementOffsetsFromVectorSliceOffsets(sizes, vectorOffsets);
580  // Compute 'sliceIndices' by adding 'sliceOffsets[i]' to 'indices[i]'.
581  SmallVector<Value, 4> sliceIndices(numSliceIndices);
582  for (unsigned j = 0; j < numSliceIndices; ++j) {
583  if (j < indexOffset) {
584  sliceIndices[j] = indices[j];
585  } else {
586  auto expr = getAffineDimExpr(0, ctx) +
587  getAffineConstantExpr(elementOffsets[j - indexOffset], ctx);
588  auto map = AffineMap::get(/*dimCount=*/1, /*symbolCount=*/0, expr);
589  sliceIndices[j] = builder.create<AffineApplyOp>(
590  indices[j].getLoc(), map, ArrayRef<Value>(indices[j]));
591  }
592  }
593  // Call 'fn' to generate slice 'i' at 'sliceIndices'.
594  fn(i, sliceIndices);
595  }
596 }

598 /// Returns true if 'map' is a suffix of an identity affine map, false
599 /// otherwise. Example: affine_map<(d0, d1, d2, d3) -> (d2, d3)>
600 static bool isIdentitySuffix(AffineMap map) {
601  if (map.getNumDims() < map.getNumResults())
602  return false;
603  ArrayRef<AffineExpr> results = map.getResults();
604  Optional<int> lastPos;
605  for (unsigned i = 0, e = map.getNumResults(); i < e; ++i) {
606  auto expr = results[i].dyn_cast<AffineDimExpr>();
607  if (!expr)
608  return false;
609  int currPos = static_cast<int>(expr.getPosition());
610  if (lastPos.hasValue() && currPos != lastPos.getValue() + 1)
611  return false;
612  lastPos = currPos;
613  }
614  return true;
615 }

768 /// Returns the producer Value of the same type as 'consumerValue', by tracking
769 /// the tuple index and offsets of the consumer vector value through the
770 /// chain of operations (TupleGetOp, InsertSlicesOp, ExtractSlicesOp, TupleOp,
771 /// and ShapeCastOp) from consumer to producer. Each operation in the chain is
772 /// structured, and so the tuple index and offsets can be mapped from result to
773 /// input, while visiting each operation in the chain.
774 /// Returns nullptr on failure.
775 static Value getProducerValue(Value consumerValue) {
776  auto consumerVectorType = consumerValue.getType().cast<VectorType>();
777  // A tupleIndex == -1 indicates that 'offsets' are w.r.t a vector type.
778  int64_t tupleIndex = -1;
779  SmallVector<int64_t, 4> offsets(consumerVectorType.getRank(), 0);
780  auto *op = consumerValue.getDefiningOp();
781  while (op != nullptr) {
782  if (auto tupleGetOp = dyn_cast<vector::TupleGetOp>(op)) {
783  assert(tupleIndex == -1 && "TupleGetOp must have vector result type");
784 
785  // Update 'tupleIndex' and next defining 'op' to visit.
786  tupleIndex = tupleGetOp.getIndex();
787  op = tupleGetOp.vectors().getDefiningOp();
788  } else if (auto extractSlicesOp = dyn_cast<vector::ExtractSlicesOp>(op)) {
789  assert(tupleIndex >= 0);
790 
791  // Compute slice strides for 'extractSlicesOp'.
792  SmallVector<int64_t, 4> sizes;
793  extractSlicesOp.getSizes(sizes);
794  auto sliceStrides = computeStrides(
795  extractSlicesOp.getSourceVectorType().getShape(), sizes);
796 
797  // Compute 'elementOffsets' into 'extractSlicesOp' input vector type,
798  // of 'extractSlicesOp' result vector tuple element at 'tupleIndex'.
799  auto vectorOffsets = delinearize(sliceStrides, tupleIndex);
800  auto elementOffsets =
801  computeElementOffsetsFromVectorSliceOffsets(sizes, vectorOffsets);
802 
803  // Add 'elementOffsets' to 'offsets' so that 'offsets' are now relative
804  // to the 'extractSlicesOp' input vector type.
805  assert(offsets.size() == elementOffsets.size());
806  for (unsigned i = 0, e = offsets.size(); i < e; ++i)
807  offsets[i] += elementOffsets[i];
808 
809  // Clear 'tupleIndex' and update next defining 'op' to visit.
810  tupleIndex = -1;
811  op = extractSlicesOp.vector().getDefiningOp();
812  } else if (auto insertSlicesOp = dyn_cast<vector::InsertSlicesOp>(op)) {
813  assert(tupleIndex == -1);
814 
815  // Compute slice strides for 'insertSlicesOp'.
816  SmallVector<int64_t, 4> sizes;
817  insertSlicesOp.getSizes(sizes);
818  auto sliceStrides = computeStrides(
819  insertSlicesOp.getResultVectorType().getShape(), sizes);
820 
821  // Compute 'vectorOffsets' of 'insertSlicesOp' input vector slice,
822  // of 'insertSlicesOp' result vector type at 'offsets'.
823  SmallVector<int64_t, 4> vectorOffsets(offsets.size());
824  assert(offsets.size() == sizes.size());
825  for (unsigned i = 0, e = offsets.size(); i < e; ++i)
826  vectorOffsets[i] = offsets[i] / sizes[i];
827 
828  // Compute the source tuple element index.
829  tupleIndex = linearize(vectorOffsets, sliceStrides);
830 
831  // Subtract 'elementOffsets' from 'offsets' so that 'offsets' are now
832  // relative to input tuple element vector type at 'tupleIndex'.
833  auto elementOffsets =
834  computeElementOffsetsFromVectorSliceOffsets(sizes, vectorOffsets);
835  assert(offsets.size() == elementOffsets.size());
836  for (unsigned i = 0, e = offsets.size(); i < e; ++i) {
837  offsets[i] -= elementOffsets[i];
838  assert(offsets[i] >= 0);
839  }
840 
841  // Update next defining 'op' to visit.
842  op = insertSlicesOp.vectors().getDefiningOp();
843  } else if (auto tupleOp = dyn_cast<vector::TupleOp>(op)) {
844  assert(tupleIndex >= 0);
845 
846  // Return tuple element 'value' at 'tupleIndex' if it matches type.
847  auto value = tupleOp.getOperand(tupleIndex);
848  if (value.getType() == consumerVectorType)
849  return value;
850 
851  // Update 'tupleIndex' and next defining 'op' to visit.
852  tupleIndex = -1;
853  op = value.getDefiningOp();
854  } else if (auto shapeCastOp = dyn_cast<vector::ShapeCastOp>(op)) {
855  if (shapeCastOp.source().getType().isa<TupleType>())
856  return nullptr;
857  assert(tupleIndex == -1);
858  auto sourceVectorType = shapeCastOp.getSourceVectorType();
859  auto sourceVectorShape = sourceVectorType.getShape();
860  unsigned sourceVectorRank = sourceVectorType.getRank();
861  auto resultVectorType = shapeCastOp.getResultVectorType();
862  auto resultVectorShape = resultVectorType.getShape();
863  unsigned resultVectorRank = resultVectorType.getRank();
864 
865  int i = sourceVectorRank - 1;
866  int j = resultVectorRank - 1;
867 
868  // Check that source/result vector shape prefixes match while updating
869  // 'newOffsets'.
870  SmallVector<int64_t, 4> newOffsets(sourceVectorRank, 0);
871  for (auto it : llvm::zip(llvm::reverse(sourceVectorShape),
872  llvm::reverse(resultVectorShape))) {
873  if (std::get<0>(it) != std::get<1>(it))
874  return nullptr;
875  newOffsets[i--] = offsets[j--];
876  }
877 
878  // Check that remaining prefix of source/result vector shapes are all 1s.
879  // Currently we only support producer/consumer tracking through trivial
880  // shape cast ops. Examples:
881  // %1 = vector.shape_cast %0 : vector<1x1x2x4xf32> to vector<2x4xf32>
882  // %3 = vector.shape_cast %2 : vector<16x8xf32> to vector<1x16x8xf32>
883  assert(i == -1 || j == -1);
884  if (i >= 0 &&
885  !std::all_of(sourceVectorShape.begin(), sourceVectorShape.begin() + i,
886  [](int64_t v) { return v == 1; }))
887  return nullptr;
888  if (j >= 0 &&
889  !std::all_of(resultVectorShape.begin(), resultVectorShape.begin() + j,
890  [](int64_t v) { return v == 1; }))
891  return nullptr;
892 
893  offsets.swap(newOffsets);
894  op = shapeCastOp.source().getDefiningOp();
895  } else {
896  // Check if 'op' produces a Value with the same type as 'consumerValue'.
897  if (op->getNumResults() == 1 &&
898  op->getResult(0).getType() == consumerVectorType)
899  return op->getResult(0);
900  return nullptr;
901  }
902  }
903  return nullptr;
904 }

1082  VectorType dstType = op.getVectorType();
1083  VectorType srcType = op.getSourceType().dyn_cast<VectorType>();
1084  Type eltType = dstType.getElementType();
1085 
1086  // Determine rank of source and destination.
1087  int64_t srcRank = srcType ? srcType.getRank() : 0;
1088  int64_t dstRank = dstType.getRank();
1089 
1090  // Duplicate this rank.
1091  // For example:
1092  // %x = broadcast %y : k-D to n-D, k < n
1093  // becomes:
1094  // %b = broadcast %y : k-D to (n-1)-D
1095  // %x = [%b,%b,%b,%b] : n-D
1096  // becomes:
1097  // %b = [%y,%y] : (n-1)-D
1098  // %x = [%b,%b,%b,%b] : n-D
1099  if (srcRank < dstRank) {
1100  // Scalar to any vector can use splat.
1101  if (srcRank == 0) {
1102  rewriter.replaceOpWithNewOp<SplatOp>(op, dstType, op.source());
1103  return success();
1104  }
1105  // Duplication.
1106  VectorType resType =
1107  VectorType::get(dstType.getShape().drop_front(), eltType);
1108  Value bcst =
1109  rewriter.create<vector::BroadcastOp>(loc, resType, op.source());
1110  Value result = rewriter.create<ConstantOp>(loc, dstType,
1111  rewriter.getZeroAttr(dstType));
1112  for (int64_t d = 0, dim = dstType.getDimSize(0); d < dim; ++d)
1113  result = rewriter.create<vector::InsertOp>(loc, bcst, result, d);
1114  rewriter.replaceOp(op, result);
1115  return success();
1116  }
1117 
1118  // Find non-matching dimension, if any.
1119  assert(srcRank == dstRank);
1120  int64_t m = -1;
1121  for (int64_t r = 0; r < dstRank; r++)
1122  if (srcType.getDimSize(r) != dstType.getDimSize(r)) {
1123  m = r;
1124  break;
1125  }
1126 
1127  // All trailing dimensions are the same. Simply pass through.
1128  if (m == -1) {
1129  rewriter.replaceOp(op, op.source());
1130  return success();
1131  }
1132 
1133  // Stretching scalar inside vector (e.g. vector<1xf32>) can use splat.
1134  if (srcRank == 1) {
1135  assert(m == 0);
1136  Value ext = rewriter.create<vector::ExtractOp>(loc, op.source(), 0);
1137  rewriter.replaceOpWithNewOp<SplatOp>(op, dstType, ext);
1138  return success();
1139  }
1140 
1141  // Any non-matching dimension forces a stretch along this rank.
1142  // For example:
1143  // %x = broadcast %y : vector<4x1x2xf32> to vector<4x2x2xf32>
1144  // becomes:
1145  // %a = broadcast %y[0] : vector<1x2xf32> to vector<2x2xf32>
1146  // %b = broadcast %y[1] : vector<1x2xf32> to vector<2x2xf32>
1147  // %c = broadcast %y[2] : vector<1x2xf32> to vector<2x2xf32>
1148  // %d = broadcast %y[3] : vector<1x2xf32> to vector<2x2xf32>
1149  // %x = [%a,%b,%c,%d]
1150  // becomes:
1151  // %u = broadcast %y[0][0] : vector<2xf32> to vector <2x2xf32>
1152  // %v = broadcast %y[1][0] : vector<2xf32> to vector <2x2xf32>
1153  // %a = [%u, %v]
1154  // ..
1155  // %x = [%a,%b,%c,%d]
1156  VectorType resType =
1157  VectorType::get(dstType.getShape().drop_front(), eltType);
1158  Value result = rewriter.create<ConstantOp>(loc, dstType,
1159  rewriter.getZeroAttr(dstType));
1160  if (m == 0) {
1161  // Stetch at start.
1162  Value ext = rewriter.create<vector::ExtractOp>(loc, op.source(), 0);
1163  Value bcst = rewriter.create<vector::BroadcastOp>(loc, resType, ext);
1164  for (int64_t d = 0, dim = dstType.getDimSize(0); d < dim; ++d)
1165  result = rewriter.create<vector::InsertOp>(loc, bcst, result, d);
1166  } else {
1167  // Stetch not at start.
1168  for (int64_t d = 0, dim = dstType.getDimSize(0); d < dim; ++d) {
1169  Value ext = rewriter.create<vector::ExtractOp>(loc, op.source(), d);
1170  Value bcst = rewriter.create<vector::BroadcastOp>(loc, resType, ext);
1171  result = rewriter.create<vector::InsertOp>(loc, bcst, result, d);
1172  }
1173  }
1174  rewriter.replaceOp(op, result);
1175  return success();
1176  }

1513 public:

1517  PatternRewriter &rewriter) const override {

1520  auto resultVectorType = op.getResultVectorType();
1521  // Intended 2D/1D lowerings with better implementations.
1522  int64_t srcRank = sourceVectorType.getRank();

1524  if ((srcRank == 2 && resRank == 1) || (srcRank == 1 && resRank == 2))
1525  return failure();

1531  // x[0,0,0] = y[0,0]

1535  // within the source and result shape.

1538  Value result = rewriter.create<ConstantOp>(

1543  incIdx(resIdx, resultVectorType, resRank - 1);
1544  }
1545  Value e = rewriter.create<vector::ExtractOp>(loc, op.source(), srcIdx);
1546  result = rewriter.create<vector::InsertOp>(loc, e, result, resIdx);
1547  }

1630 /// Progressively lower a `vector.contract %a, %b, %c` with row-major matmul
1631 /// semantics to a reduction_size-unrolled sequence:
1632 /// ```
1633 /// %at = vector.transpose %a, [1, 0]
1634 /// %bRow0 = vector.extract %b[0]
1635 /// %atRow0 = vector.extract %at[0]
1636 /// %c0 = vector.outerproduct %atRow0, %bRow0, %c
1637 /// ...
1638 /// %bRowK = vector.extract %b[K]
1639 /// %atRowK = vector.extract %at[K]
1640 /// %cK = vector.outerproduct %atRowK, %bRowK, %cK-1
1641 /// ```
1642 ///
1643 /// This only kicks in when VectorTransformsOptions is set to OuterProduct but
1644 /// otherwise supports any layout permutation of the matrix-multiply.
1645 LogicalResult ContractionOpToOuterProductOpLowering::matchAndRewrite(
1646  vector::ContractionOp op, PatternRewriter &rewriter) const {
1647  // TODO: implement masks
1648  if (llvm::size(op.masks()) != 0)
1649  return failure();
1650 
1651  if (vectorTransformsOptions.vectorContractLowering !=
1652  vector::VectorContractLowering::OuterProduct)
1653  return failure();
1654 
1655  if (failed(filter(op)))
1656  return failure();
1657 
1658  Location loc = op.getLoc();
1659  int64_t reductionSize = 0;
1660  VectorType lhsType = op.getLhsType();
1661  Value lhs = op.lhs(), rhs = op.rhs(), res = op.acc();
1662 
1663  // Set up the parallel/reduction structure in right form.
1664  using MapList = ArrayRef<ArrayRef<AffineExpr>>;
1665  auto infer = [](MapList m) { return AffineMap::inferFromExprList(m); };
1666  AffineExpr m, n, k;
1667  bindDims(rewriter.getContext(), m, n, k);
1668  static constexpr std::array<int64_t, 2> perm = {1, 0};
1669  auto iteratorTypes = op.iterator_types().getValue();
1670  SmallVector<AffineMap, 4> maps = op.getIndexingMaps();
1671  if (isParallelIterator(iteratorTypes[0]) &&
1672  isParallelIterator(iteratorTypes[1]) &&
1673  isReductionIterator(iteratorTypes[2])) {
1674  //
1675  // Two outer parallel, one inner reduction (matmat flavor).
1676  //
1677  if (maps == infer({{m, k}, {k, n}, {m, n}})) {
1678  // This is the classical row-major matmul. Just permute the lhs.
1679  reductionSize = lhsType.getDimSize(1);
1680  lhs = rewriter.create<vector::TransposeOp>(loc, lhs, perm);
1681  } else if (maps == infer({{m, k}, {n, k}, {m, n}})) {
1682  // TODO: may be better to fail and use some vector<k> -> scalar reduction.
1683  reductionSize = lhsType.getDimSize(1);
1684  lhs = rewriter.create<vector::TransposeOp>(loc, lhs, perm);
1685  rhs = rewriter.create<vector::TransposeOp>(loc, rhs, perm);
1686  } else if (maps == infer({{k, m}, {k, n}, {m, n}})) {
1687  // No need to permute anything.
1688  reductionSize = lhsType.getDimSize(0);
1689  } else if (maps == infer({{k, m}, {n, k}, {m, n}})) {
1690  // Just permute the rhs.
1691  reductionSize = lhsType.getDimSize(0);
1692  rhs = rewriter.create<vector::TransposeOp>(loc, rhs, perm);
1693  } else if (maps == infer({{m, k}, {k, n}, {n, m}})) {
1694  // This is the classical row-major matmul. Just permute the lhs.
1695  reductionSize = lhsType.getDimSize(1);
1696  Value tmp = rhs;
1697  rhs = rewriter.create<vector::TransposeOp>(loc, lhs, perm);
1698  lhs = tmp;
1699  } else if (maps == infer({{m, k}, {n, k}, {n, m}})) {
1700  // TODO: may be better to fail and use some vector<k> -> scalar reduction.
1701  reductionSize = lhsType.getDimSize(1);
1702  Value tmp = rhs;
1703  rhs = rewriter.create<vector::TransposeOp>(loc, lhs, perm);
1704  lhs = rewriter.create<vector::TransposeOp>(loc, tmp, perm);
1705  } else if (maps == infer({{k, m}, {k, n}, {n, m}})) {
1706  // No need to permute anything, but still swap lhs and rhs.
1707  reductionSize = lhsType.getDimSize(0);
1708  std::swap(lhs, rhs);
1709  } else if (maps == infer({{k, m}, {n, k}, {n, m}})) {
1710  // Just permute the rhs.
1711  reductionSize = lhsType.getDimSize(0);
1712  Value tmp = lhs;
1713  lhs = rewriter.create<vector::TransposeOp>(loc, rhs, perm);
1714  rhs = tmp;
1715  } else {
1716  return failure();
1717  }
1718  } else if (isParallelIterator(iteratorTypes[0]) &&
1719  isReductionIterator(iteratorTypes[1])) {
1720  //
1721  // One outer parallel, one inner reduction (matvec flavor)
1722  //
1723  if (maps == infer({{m, n}, {n}, {m}})) {
1724  // Case mat-vec: transpose.
1725  reductionSize = lhsType.getDimSize(1);
1726  lhs = rewriter.create<vector::TransposeOp>(loc, lhs, perm);
1727  } else if (maps == infer({{n, m}, {n}, {m}})) {
1728  // Case mat-trans-vec: ready to go.
1729  reductionSize = lhsType.getDimSize(0);
1730  } else if (maps == infer({{n}, {m, n}, {m}})) {
1731  // Case vec-mat: swap and transpose.
1732  reductionSize = lhsType.getDimSize(0);
1733  std::swap(lhs, rhs);
1734  lhs = rewriter.create<vector::TransposeOp>(loc, lhs, perm);
1735  } else if (maps == infer({{n}, {n, m}, {m}})) {
1736  // Case vec-mat-trans: swap and ready to go.
1737  reductionSize = lhsType.getDimSize(0);
1738  std::swap(lhs, rhs);
1739  } else {
1740  return failure();
1741  }
1742  } else {
1743  return failure();
1744  }
1745  assert(reductionSize > 0);
1746 
1747  // Unroll outer-products along reduction.
1748  for (int64_t k = 0; k < reductionSize; ++k) {
1749  Value a = rewriter.create<vector::ExtractOp>(op.getLoc(), lhs, k);
1750  Value b = rewriter.create<vector::ExtractOp>(op.getLoc(), rhs, k);
1751  res = rewriter.create<vector::OuterProductOp>(op.getLoc(), a, b, res);
1752  }
1753  rewriter.replaceOp(op, res);
1754  return success();
1755 }

1872 /// Progressive lowering of ContractionOp.
1873 /// One:
1874 /// %x = vector.contract with at least one free/batch dimension
1875 /// is replaced by:
1876 /// %a = vector.contract with one less free/batch dimension
1877 /// %b = vector.contract with one less free/batch dimension
1878 /// ..
1879 /// %x = combine %a %b ..
1880 /// until a pure contraction is reached (no free/batch dimensions),
1881 /// which is replaced by a dot-product.
1882 ///
1883 /// This only kicks in when either VectorTransformsOptions is set
1884 /// to DOT or when other contraction patterns fail.
1885 //
1886 // TODO: break down into transpose/reshape/cast ops
1887 // when they become available to avoid code dup
1888 // TODO: investigate lowering order impact on performance
1889 LogicalResult
1890 ContractionOpLowering::matchAndRewrite(vector::ContractionOp op,
1891  PatternRewriter &rewriter) const {
1892  // TODO: implement masks.
1893  if (llvm::size(op.masks()) != 0)
1894  return failure();
1895 
1896  if (failed(filter(op)))
1897  return failure();
1898 
1899  // TODO: support mixed mode contract lowering.
1900  if (op.getLhsType().getElementType() !=
1901  getElementTypeOrSelf(op.getAccType()) ||
1902  op.getRhsType().getElementType() != getElementTypeOrSelf(op.getAccType()))
1903  return failure();
1904 
1905  // TODO: implement benefits, cost models.
1906  MLIRContext *ctx = op.getContext();
1907  ContractionOpToMatmulOpLowering pat1(vectorTransformsOptions, ctx);
1908  if (succeeded(pat1.matchAndRewrite(op, rewriter)))
1909  return success();
1910  ContractionOpToOuterProductOpLowering pat2(vectorTransformsOptions, ctx);
1911  if (succeeded(pat2.matchAndRewrite(op, rewriter)))
1912  return success();
1913  ContractionOpToDotLowering pat3(vectorTransformsOptions, ctx);
1914  if (succeeded(pat3.matchAndRewrite(op, rewriter)))
1915  return success();
1916 
1917  // Find first batch dimension in LHS/RHS, and lower when found.
1918  std::vector<std::pair<int64_t, int64_t>> batchDimMap = op.getBatchDimMap();
1919  if (!batchDimMap.empty()) {
1920  int64_t lhsIndex = batchDimMap[0].first;
1921  int64_t rhsIndex = batchDimMap[0].second;
1922  rewriter.replaceOp(op, lowerParallel(op, lhsIndex, rhsIndex, rewriter));
1923  return success();
1924  }
1925 
1926  // Collect contracting dimensions.
1927  std::vector<std::pair<int64_t, int64_t>> contractingDimMap =
1928  op.getContractingDimMap();
1929  DenseSet<int64_t> lhsContractingDimSet;
1930  DenseSet<int64_t> rhsContractingDimSet;
1931  for (auto &dimPair : contractingDimMap) {
1932  lhsContractingDimSet.insert(dimPair.first);
1933  rhsContractingDimSet.insert(dimPair.second);
1934  }
1935 
1936  // Find first free dimension in LHS, and lower when found.
1937  VectorType lhsType = op.getLhsType();
1938  for (int64_t lhsIndex = 0, e = lhsType.getRank(); lhsIndex < e; ++lhsIndex) {
1939  if (lhsContractingDimSet.count(lhsIndex) == 0) {
1940  rewriter.replaceOp(
1941  op, lowerParallel(op, lhsIndex, /*rhsIndex=*/-1, rewriter));
1942  return success();
1943  }
1944  }
1945 
1946  // Find first free dimension in RHS, and lower when found.
1947  VectorType rhsType = op.getRhsType();
1948  for (int64_t rhsIndex = 0, e = rhsType.getRank(); rhsIndex < e; ++rhsIndex) {
1949  if (rhsContractingDimSet.count(rhsIndex) == 0) {
1950  rewriter.replaceOp(
1951  op, lowerParallel(op, /*lhsIndex=*/-1, rhsIndex, rewriter));
1952  return success();
1953  }
1954  }
1955 
1956  // Lower the first remaining reduction dimension.
1957  if (!contractingDimMap.empty()) {
1958  rewriter.replaceOp(op, lowerReduction(op, rewriter));
1959  return success();
1960  }
1961 
1962  return failure();
1963 }

2018 // Lower one reduction dimension.
2019 Value ContractionOpLowering::lowerReduction(vector::ContractionOp op,
2020  PatternRewriter &rewriter) const {
2021  auto loc = op.getLoc();
2022  VectorType lhsType = op.getLhsType();
2023  VectorType rhsType = op.getRhsType();
2024  Type resType = op.getResultType();
2025  assert(!resType.isa<VectorType>());
2026  // Use iterator index 0.
2027  int64_t iterIndex = 0;
2028  SmallVector<AffineMap, 4> iMap = op.getIndexingMaps();
2029  Optional<int64_t> lookupLhs = getResultIndex(iMap[0], iterIndex);
2030  Optional<int64_t> lookupRhs = getResultIndex(iMap[1], iterIndex);
2031  assert(lookupLhs.hasValue() && "missing LHS parallel index");
2032  assert(lookupRhs.hasValue() && "missing RHS parallel index");
2033  int64_t lhsIndex = lookupLhs.getValue();
2034  int64_t rhsIndex = lookupRhs.getValue();
2035  int64_t dimSize = lhsType.getDimSize(lhsIndex);
2036  assert(dimSize == rhsType.getDimSize(rhsIndex) && "corrupt shape");
2037  // Base case.
2038  if (lhsType.getRank() == 1) {
2039  assert(rhsType.getRank() == 1 && "corrupt contraction");
2040  Value m = rewriter.create<MulFOp>(loc, op.lhs(), op.rhs());
2041  StringAttr kind = rewriter.getStringAttr("add");
2042  return rewriter.create<vector::ReductionOp>(loc, resType, kind, m,
2043  op.acc());
2044  }
2045  // Construct new iterator types and affine map array attribute.
2046  std::array<AffineMap, 3> lowIndexingMaps = {
2047  adjustMap(iMap[0], iterIndex, rewriter),
2048  adjustMap(iMap[1], iterIndex, rewriter),
2049  adjustMap(iMap[2], iterIndex, rewriter)};
2050  auto lowAffine = rewriter.getAffineMapArrayAttr(lowIndexingMaps);
2051  auto lowIter =
2052  rewriter.getArrayAttr(adjustIter(op.iterator_types(), iterIndex));
2053  // Unroll into a series of lower dimensional vector.contract ops.
2054  // By feeding the initial accumulator into the first contraction,
2055  // and the result of each contraction into the next, eventually
2056  // the sum of all reductions is computed.
2057  Value result = op.acc();
2058  for (int64_t d = 0; d < dimSize; ++d) {
2059  auto lhs = reshapeLoad(loc, op.lhs(), lhsType, lhsIndex, d, rewriter);
2060  auto rhs = reshapeLoad(loc, op.rhs(), rhsType, rhsIndex, d, rewriter);
2061  result = rewriter.create<vector::ContractionOp>(loc, lhs, rhs, result,
2062  lowAffine, lowIter);
2063  }
2064  return result;
2065 }
