34 /// Dialects are groups of MLIR operations and behavior associated with the
35 /// entire group. For example, hooks into other systems for constant folding,
36 /// default named types for asm printing, etc.
37 ///
38 /// Instances of the dialect object are global across all MLIRContext's that may
39 /// be active in the process.
40 ///

70  /// Registered fallback constant fold hook for the dialect. Like the constant
71  /// fold hook of each operation, it attempts to constant fold the operation
72  /// with the specified constant operand values - the elements in "operands"
73  /// will correspond directly to the operands of the operation, but may be null
74  /// if non-constant. If constant folding is successful, this fills in the
75  /// `results` vector. If not, this returns failure and `results` is
76  /// unspecified.
77  DialectConstantFoldHook constantFoldHook =
78  [](Operation *op, ArrayRef<Attribute> operands,
79  SmallVectorImpl<Attribute> &results) { return failure(); };
80 
81  /// Registered hook to decode opaque constants associated with this
82  /// dialect. The hook function attempts to decode an opaque constant tensor
83  /// into a tensor with non-opaque content. If decoding is successful, this
84  /// method returns false and sets 'output' attribute. If not, it returns true
85  /// and leaves 'output' unspecified. The default hook fails to decode.
86  DialectConstantDecodeHook decodeHook =
87  [](const OpaqueElementsAttr input, ElementsAttr &output) { return true; };
88 
89  /// Registered hook to extract an element from an opaque constant associated
90  /// with this dialect. If element has been successfully extracted, this
91  /// method returns that element. If not, it returns an empty attribute.
92  /// The default hook fails to extract an element.
93  DialectExtractElementHook extractElementHook =
94  [](const OpaqueElementsAttr input, ArrayRef<uint64_t> index) {
95  return Attribute();
96  };
97 
98  /// Registered hook to materialize a single constant operation from a given
99  /// attribute value with the desired resultant type. This method should use
100  /// the provided builder to create the operation without changing the
101  /// insertion position. The generated operation is expected to be constant
102  /// like, i.e. single result, zero operands, non side-effecting, etc. On
103  /// success, this hook should return the value generated to represent the
104  /// constant value. Otherwise, it should return null on failure.
105  virtual Operation *materializeConstant(OpBuilder &builder, Attribute value,
106  Type type, Location loc) {
107  return nullptr;
108  }

270 /// Utility to register a dialect. Client can register their dialect with the
271 /// global registry by calling registerDialect<MyDialect>();
272 /// Note: This method is not thread-safe.
273 template <typename ConcreteDialect> void registerDialect() {
274  Dialect::registerDialectAllocator(
275  TypeID::get<ConcreteDialect>(),
276  [](MLIRContext *ctx) { ctx->getOrCreateDialect<ConcreteDialect>(); });
277 }
