24 Operation *mlir::edsc::makeGenericLinalgOp(
25  ArrayRef<IteratorType> iteratorTypes, ArrayRef<StructuredIndexed> inputs,
26  ArrayRef<StructuredIndexed> outputs,
27  function_ref<void(ValueRange)> regionBuilder, ArrayRef<Value> otherValues,
28  ArrayRef<Attribute> otherAttributes) {
29  for (unsigned i = 0, e = outputs.size(); i + 1 < e; ++i)
30  assert(!(outputs[i].getType().isa<RankedTensorType>() &&
31  outputs[i + 1].getType().isa<MemRefType>()) &&
32  "output tensors must be passed after output buffers");
33  auto &builder = edsc::ScopedContext::getBuilderRef();
34  auto *ctx = builder.getContext();
35  unsigned nInputs = inputs.size();
36  unsigned nOutputs = outputs.size();
37 
38  SmallVector<SmallVector<AffineExpr, 4>, 4> exprsList;
39  exprsList.reserve(nInputs + nOutputs);
40  for (auto structuredIndexed : inputs)
41  exprsList.emplace_back(structuredIndexed.getExprs().begin(),
42  structuredIndexed.getExprs().end());
43  for (auto structuredIndexed : outputs)
44  exprsList.emplace_back(structuredIndexed.getExprs().begin(),
45  structuredIndexed.getExprs().end());
46  auto maps = AffineMap::inferFromExprList(exprsList);
47 
48  unsigned nViews = nInputs + nOutputs;
49  SmallVector<Value, 4> values;
50  values.reserve(nViews);
51  values.append(inputs.begin(), inputs.end());
52  std::copy_if(outputs.begin(), outputs.end(), std::back_inserter(values),
53  [](StructuredIndexed s) { return s.hasValue(); });
54  SmallVector<Type, 4> types;
55  std::copy_if(outputs.begin(), outputs.end(), std::back_inserter(types),
56  [](StructuredIndexed s) { return !s.hasValue(); });
57 
58  auto iteratorStrTypes =
59  llvm::to_vector<8>(llvm::map_range(iteratorTypes, toString));
60  // clang-format off
61  auto *op =
62  edsc::ScopedContext::getBuilderRef()
63  .create<linalg::GenericOp>(
64  edsc::ScopedContext::getLocation(),
65  types,
66  values,
67  IntegerAttr::get(IntegerType::get(64, ctx), nInputs),
68  IntegerAttr::get(IntegerType::get(64, ctx), nOutputs),
69  builder.getAffineMapArrayAttr(maps),
70  builder.getStrArrayAttr(iteratorStrTypes),
71  StringAttr() /*doc*/,
72  StringAttr() /*library_call*/,
73  IntegerAttr() /*symbol_source*/
74  /* TODO: other attributes in op */
75  )
76  .getOperation();
77  // clang-format on
78 
79  using namespace edsc;
80  SmallVector<Type, 4> blockTypes;
81  blockTypes.reserve(values.size());
82  for (auto it : llvm::enumerate(values))
83  blockTypes.push_back((it.index() < nViews)
84  ? getElementTypeOrSelf(it.value())
85  : it.value().getType());
86 
87  assert(op->getNumRegions() == 1);
88  assert(op->getRegion(0).empty());
89  OpBuilder opBuilder(op);
90  ScopedContext scope(opBuilder, op->getLoc());
91  buildInNewBlock(op->getRegion(0), blockTypes, regionBuilder);
92  assert(llvm::hasSingleElement(op->getRegion(0)));
93  return op;
94 }

112 Operation *mlir::edsc::ops::linalg_generic_pointwise(
113  UnaryPointwiseOpBuilder unaryOp, StructuredIndexed I, StructuredIndexed O) {
114  SmallVector<IteratorType, 4> iterTypes(O.getExprs().size(),
115  IteratorType::Parallel);
116  if (O.getType().isa<RankedTensorType>()) {
117  auto fun = [&unaryOp](ValueRange args) {
118  assert(args.size() == 1 && "expected 1 block arguments");
119  Value a(args[0]);
120  linalg_yield(unaryOp(a));
121  };
122  return makeGenericLinalgOp(iterTypes, {I}, {O}, fun);
123  }
124  auto fun = [&unaryOp](ValueRange args) {
125  assert(args.size() == 2 && "expected 2 block arguments");
126  Value a(args[0]);
127  linalg_yield(unaryOp(a));
128  };
129  return makeGenericLinalgOp(iterTypes, {I}, {O}, fun);
130 }

225 Operation *mlir::edsc::ops::linalg_generic_conv_nhwc(Value vI, Value vW,
226  Value vO,
227  ArrayRef<int> strides,
228  ArrayRef<int> dilations) {
229  MLIRContext *ctx = ScopedContext::getContext();
230  // TODO: some template magic to make everything rank-polymorphic.
231  assert((dilations.empty() || dilations.size() == 2) && "only 2-D conv atm");
232  assert((strides.empty() || strides.size() == 2) && "only 2-D conv atm");
233 
234  // Some short names.
235  auto par = IteratorType::Parallel;
236  auto red = IteratorType::Reduction;
237  auto s = strides;
238  auto d = dilations;
239 
240  AffineExpr b, f, h, w, kh, kw, c;
241  bindDims(ctx, b, f, h, w, kh, kw, c);
242  unsigned numDims = c.cast<AffineDimExpr>().getPosition() + 1;
243  StructuredIndexed I(vI), W(vW), O(vO);
244  // clang-format off
245  return makeGenericLinalgOp(
246  {par, par, par, par, red, red, red}, {
247  I({b,
248  // Roundtrip to flattened form to serve as canonicalization and ensure
249  // consistent ordering of subexpressions.
250  simplifyAffineExpr(s[0] * h + d[0] * kh, numDims, 0),
251  simplifyAffineExpr(s[1] * w + d[1] * kw, numDims, 0),
252  c}),
253  W({kh, kw, c, f})}, {
254  O({b, h, w, f})},
255  macRegionBuilder);
256  // clang-format on
257 }
