30 void tblgen::OpMethodSignature::writeDefTo(raw_ostream &os,
31  StringRef namePrefix) const {
32  // We need to remove the default values for parameters in method definition.
33  // TODO: We are using '=' and ',' as delimiters for parameter
34  // initializers. This is incorrect for initializer list with more than one
35  // element. Change to a more robust approach.
36  auto removeParamDefaultValue = [](StringRef params) {
37  std::string result;
38  std::pair<StringRef, StringRef> parts;
39  while (!params.empty()) {
40  parts = params.split("=");
41  result.append(result.empty() ? "" : ", ");
42  result += parts.first;
43  params = parts.second.split(",").second;
44  }
45  return result;
46  };
47 
48  os << returnType << (elideSpaceAfterType(returnType) ? "" : " ") << namePrefix
49  << (namePrefix.empty() ? "" : "::") << methodName << "("
50  << removeParamDefaultValue(parameters) << ")";
51 }

173 void tblgen::Class::writeDeclTo(raw_ostream &os) const {
174  bool hasPrivateMethod = false;
175  os << "class " << className << " {\n";
176  os << "public:\n";
177  for (const auto &method :
178  llvm::concat<const OpMethod>(constructors, methods)) {
179  if (!method.isPrivate()) {
180  method.writeDeclTo(os);
181  os << '\n';
182  } else {
183  hasPrivateMethod = true;
184  }
185  }
186  os << '\n';
187  os << "private:\n";
188  if (hasPrivateMethod) {
189  for (const auto &method :
190  llvm::concat<const OpMethod>(constructors, methods)) {
191  if (method.isPrivate()) {
192  method.writeDeclTo(os);
193  os << '\n';

195  }
196  os << '\n';
197  }
198  for (const auto &field : fields)
199  os.indent(2) << field << ";\n";
200  os << "};\n";
