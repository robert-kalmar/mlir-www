22 /// A very primitive alias analysis which just records for each view, either:
23 /// 1. The base buffer, or
24 /// 2. The block argument view
25 /// that it indexes into.
26 /// This does not perform inter-block or inter-procedural analysis and assumes
27 /// that different block argument views do not alias.
28 class Aliases {
29 public:
30  /// Returns true if v1 and v2 alias.
31  bool alias(Value v1, Value v2) { return find(v1) == find(v2); }
32 
33 private:
34  /// Returns the base buffer or block argument into which the view `v` aliases.
35  /// This lazily records the new aliases discovered while walking back the
36  /// use-def chain.
37  Value find(Value v);
38 
39  DenseMap<Value, Value> aliases;
40 };

42 /// Data structure for holding a dependence graph that operates on LinalgOp and
43 /// views as SSA values.
44 class LinalgDependenceGraph {
45 public:
46  struct LinalgOpView {
47  Operation *op;
48  Value view;
49  };
50  struct LinalgDependenceGraphElem {
51  // dependentOpView may be either:
52  // 1. src in the case of dependencesIntoGraphs.
53  // 2. dst in the case of dependencesFromDstGraphs.
54  LinalgOpView dependentOpView;
55  // View in the op that is used to index in the graph:
56  // 1. src in the case of dependencesFromDstGraphs.
57  // 2. dst in the case of dependencesIntoGraphs.
58  Value indexingView;
59  };
60  using LinalgDependences = SmallVector<LinalgDependenceGraphElem, 8>;
61  using DependenceGraph = DenseMap<Operation *, LinalgDependences>;
