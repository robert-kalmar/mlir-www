184 /// Parses optional memory access attributes attached to a memory access
185 /// operand/pointer. Specifically, parses the following syntax:
186 /// (`[` memory-access `]`)?
187 /// where:
188 /// memory-access ::= `"None"` | `"Volatile"` | `"Aligned", `
189 /// integer-literal | `"NonTemporal"`
190 static ParseResult parseMemoryAccessAttributes(OpAsmParser &parser,
191  OperationState &state) {
192  // Parse an optional list of attributes staring with '['
193  if (parser.parseOptionalLSquare()) {
194  // Nothing to do
195  return success();
196  }
197 
198  spirv::MemoryAccess memoryAccessAttr;
199  if (parseEnumStrAttr(memoryAccessAttr, parser, state,
200  kMemoryAccessAttrName)) {
201  return failure();
202  }
203 
204  if (spirv::bitEnumContains(memoryAccessAttr, spirv::MemoryAccess::Aligned)) {
205  // Parse integer attribute for alignment.
206  Attribute alignmentAttr;
207  Type i32Type = parser.getBuilder().getIntegerType(32);
208  if (parser.parseComma() ||
209  parser.parseAttribute(alignmentAttr, i32Type, kAlignmentAttrName,
210  state.attributes)) {
211  return failure();
212  }
213  }
214  return parser.parseRSquare();
215 }

307 static LogicalResult verifyCastOp(Operation *op,
308  bool requireSameBitWidth = true) {
309  Type operandType = op->getOperand(0).getType();
310  Type resultType = op->getResult(0).getType();
311 
312  // ODS checks that result type and operand type have the same shape.
313  if (auto vectorType = operandType.dyn_cast<VectorType>()) {
314  operandType = vectorType.getElementType();
315  resultType = resultType.cast<VectorType>().getElementType();
316  }
317 
318  if (auto coopMatrixType =
319  operandType.dyn_cast<spirv::CooperativeMatrixNVType>()) {
320  operandType = coopMatrixType.getElementType();
321  resultType =
322  resultType.cast<spirv::CooperativeMatrixNVType>().getElementType();
323  }
324 
325  auto operandTypeBitWidth = operandType.getIntOrFloatBitWidth();
326  auto resultTypeBitWidth = resultType.getIntOrFloatBitWidth();
327  auto isSameBitWidth = operandTypeBitWidth == resultTypeBitWidth;
328 
329  if (requireSameBitWidth) {
330  if (!isSameBitWidth) {
331  return op->emitOpError(
332  "expected the same bit widths for operand type and result "
333  "type, but provided ")
334  << operandType << " and " << resultType;
335  }
336  return success();
337  }
338 
339  if (isSameBitWidth) {
340  return op->emitOpError(
341  "expected the different bit widths for operand type and result "
342  "type, but provided ")
343  << operandType << " and " << resultType;
344  }
345  return success();

350  // ODS checks for attributes values. Just need to verify that if the

352  // present.
353  auto *op = memoryOp.getOperation();

432 template <typename BarrierOp>
433 static LogicalResult verifyMemorySemantics(BarrierOp op) {
434  // According to the SPIR-V specification:
435  // "Despite being a mask and allowing multiple bits to be combined, it is
436  // invalid for more than one of these four bits to be set: Acquire, Release,
437  // AcquireRelease, or SequentiallyConsistent. Requesting both Acquire and
438  // Release semantics is done by setting the AcquireRelease bit, not by setting
439  // two bits."
440  auto memorySemantics = op.memory_semantics();
441  auto atMostOneInSet = spirv::MemorySemantics::Acquire |
442  spirv::MemorySemantics::Release |
443  spirv::MemorySemantics::AcquireRelease |
444  spirv::MemorySemantics::SequentiallyConsistent;
445 
446  auto bitCount = llvm::countPopulation(
447  static_cast<uint32_t>(memorySemantics & atMostOneInSet));
448  if (bitCount > 1) {
449  return op.emitError("expected at most one of these four memory constraints "
450  "to be set: `Acquire`, `Release`,"
451  "`AcquireRelease` or `SequentiallyConsistent`");
452  }
453  return success();
454 }

554 /// Walks the given type hierarchy with the given indices, potentially down
555 /// to component granularity, to select an element type. Returns null type and
556 /// emits errors with the given loc on failure.
557 static Type
558 getElementType(Type type, ArrayRef<int32_t> indices,
559  function_ref<InFlightDiagnostic(StringRef)> emitErrorFn) {
560  if (indices.empty()) {
561  emitErrorFn("expected at least one index for spv.CompositeExtract");
562  return nullptr;
563  }
564 
565  for (auto index : indices) {
566  if (auto cType = type.dyn_cast<spirv::CompositeType>()) {
567  if (cType.hasCompileTimeKnownNumElements() &&
568  (index < 0 ||
569  static_cast<uint64_t>(index) >= cType.getNumElements())) {
570  emitErrorFn("index ") << index << " out of bounds for " << type;
571  return nullptr;
572  }
573  type = cType.getElementType(index);
574  } else {
575  emitErrorFn("cannot extract from non-composite type ")
576  << type << " with index " << index;
577  return nullptr;
578  }
579  }
580  return type;
581 }

883 static Type getElementPtrType(Type type, ValueRange indices, Location baseLoc) {
884  auto ptrType = type.dyn_cast<spirv::PointerType>();
885  if (!ptrType) {
886  emitError(baseLoc, "'spv.AccessChain' op expected a pointer "
887  "to composite type, but provided ")
888  << type;
889  return nullptr;
890  }
891 
892  auto resultType = ptrType.getPointeeType();
893  auto resultStorageClass = ptrType.getStorageClass();
894  int32_t index = 0;
895 
896  for (auto indexSSA : indices) {
897  auto cType = resultType.dyn_cast<spirv::CompositeType>();
898  if (!cType) {
899  emitError(baseLoc,
900  "'spv.AccessChain' op cannot extract from non-composite type ")
901  << resultType << " with index " << index;
902  return nullptr;
903  }
904  index = 0;
905  if (resultType.isa<spirv::StructType>()) {
906  Operation *op = indexSSA.getDefiningOp();
907  if (!op) {
908  emitError(baseLoc, "'spv.AccessChain' op index must be an "
909  "integer spv.constant to access "
910  "element of spv.struct");
911  return nullptr;
912  }
913 
914  // TODO: this should be relaxed to allow
915  // integer literals of other bitwidths.
916  if (failed(extractValueFromConstOp(op, index))) {
917  emitError(baseLoc,
918  "'spv.AccessChain' index must be an integer spv.constant to "
919  "access element of spv.struct, but provided ")
920  << op->getName();
921  return nullptr;
922  }
923  if (index < 0 || static_cast<uint64_t>(index) >= cType.getNumElements()) {
924  emitError(baseLoc, "'spv.AccessChain' op index ")
925  << index << " out of bounds for " << resultType;
926  return nullptr;
927  }
928  }
929  resultType = cType.getElementType(index);
930  }
931  return spirv::PointerType::get(resultType, resultStorageClass);
932 }

1467 static LogicalResult verify(spirv::ConstantOp constOp) {
1468  auto opType = constOp.getType();
1469  auto value = constOp.value();
1470  auto valueType = value.getType();
1471 
1472  // ODS already generates checks to make sure the result type is valid. We just
1473  // need to additionally check that the value's attribute type is consistent
1474  // with the result type.
1475  if (value.isa<IntegerAttr, FloatAttr>()) {
1476  if (valueType != opType)
1477  return constOp.emitOpError("result type (")
1478  << opType << ") does not match value type (" << valueType << ")";
1479  return success();
1480  }
1481  if (value.isa<DenseIntOrFPElementsAttr, SparseElementsAttr>()) {
1482  if (valueType == opType)
1483  return success();
1484  auto arrayType = opType.dyn_cast<spirv::ArrayType>();
1485  auto shapedType = valueType.dyn_cast<ShapedType>();
1486  if (!arrayType) {
1487  return constOp.emitOpError(
1488  "must have spv.array result type for array value");
1489  }
1490 
1491  int numElements = arrayType.getNumElements();
1492  auto opElemType = arrayType.getElementType();
1493  while (auto t = opElemType.dyn_cast<spirv::ArrayType>()) {
1494  numElements *= t.getNumElements();
1495  opElemType = t.getElementType();
1496  }
1497  if (!opElemType.isIntOrFloat())
1498  return constOp.emitOpError("only support nested array result type");
1499 
1500  auto valueElemType = shapedType.getElementType();
1501  if (valueElemType != opElemType) {
1502  return constOp.emitOpError("result element type (")
1503  << opElemType << ") does not match value element type ("
1504  << valueElemType << ")";
1505  }
1506 
1507  if (numElements != shapedType.getNumElements()) {
1508  return constOp.emitOpError("result number of elements (")
1509  << numElements << ") does not match value number of elements ("
1510  << shapedType.getNumElements() << ")";
1511  }
1512  return success();
1513  }
1514  if (auto attayAttr = value.dyn_cast<ArrayAttr>()) {
1515  auto arrayType = opType.dyn_cast<spirv::ArrayType>();
1516  if (!arrayType)
1517  return constOp.emitOpError(
1518  "must have spv.array result type for array value");
1519  Type elemType = arrayType.getElementType();
1520  for (Attribute element : attayAttr.getValue()) {
1521  if (element.getType() != elemType)
1522  return constOp.emitOpError("has array element whose type (")
1523  << element.getType()
1524  << ") does not match the result element type (" << elemType
1525  << ')';
1526  }
1527  return success();
1528  }
1529  return constOp.emitOpError("cannot have value of type ") << valueType;
1530 }

2126 static LogicalResult verify(spirv::LoopOp loopOp) {
2127  auto *op = loopOp.getOperation();
2128 
2129  // We need to verify that the blocks follow the following layout:
2130  //
2131  // +-------------+
2132  // | entry block |
2133  // +-------------+
2134  // |
2135  // v
2136  // +-------------+
2137  // | loop header | <-----+
2138  // +-------------+ |
2139  // |
2140  // ... |
2141  // \ | / |
2142  // v |
2143  // +---------------+ |
2144  // | loop continue | -----+
2145  // +---------------+
2146  //
2147  // ...
2148  // \ | /
2149  // v
2150  // +-------------+
2151  // | merge block |
2152  // +-------------+
2153 
2154  auto &region = op->getRegion(0);
2155  // Allow empty region as a degenerated case, which can come from
2156  // optimizations.
2157  if (region.empty())
2158  return success();
2159 
2160  // The last block is the merge block.
2161  Block &merge = region.back();
2162  if (!isMergeBlock(merge))
2163  return loopOp.emitOpError(
2164  "last block must be the merge block with only one 'spv._merge' op");
2165 
2166  if (std::next(region.begin()) == region.end())
2167  return loopOp.emitOpError(
2168  "must have an entry block branching to the loop header block");
2169  // The first block is the entry block.
2170  Block &entry = region.front();
2171 
2172  if (std::next(region.begin(), 2) == region.end())
2173  return loopOp.emitOpError(
2174  "must have a loop header block branched from the entry block");
2175  // The second block is the loop header block.
2176  Block &header = *std::next(region.begin(), 1);
2177 
2178  if (!hasOneBranchOpTo(entry, header))
2179  return loopOp.emitOpError(
2180  "entry block must only have one 'spv.Branch' op to the second block");
2181 
2182  if (std::next(region.begin(), 3) == region.end())
2183  return loopOp.emitOpError(
2184  "requires a loop continue block branching to the loop header block");
2185  // The second to last block is the loop continue block.
2186  Block &cont = *std::prev(region.end(), 2);
2187 
2188  // Make sure that we have a branch from the loop continue block to the loop
2189  // header block.
2190  if (llvm::none_of(
2191  llvm::seq<unsigned>(0, cont.getNumSuccessors()),
2192  [&](unsigned index) { return cont.getSuccessor(index) == &header; }))
2193  return loopOp.emitOpError("second to last block must be the loop continue "
2194  "block that branches to the loop header block");
2195 
2196  // Make sure that no other blocks (except the entry and loop continue block)
2197  // branches to the loop header block.
2198  for (auto &block : llvm::make_range(std::next(region.begin(), 2),
2199  std::prev(region.end(), 2))) {
2200  for (auto i : llvm::seq<unsigned>(0, block.getNumSuccessors())) {
2201  if (block.getSuccessor(i) == &header) {
2202  return loopOp.emitOpError("can only have the entry and loop continue "
2203  "block branching to the loop header block");
2204  }
2205  }
2206  }
2207 
2208  return success();
2209 }
