20 /// This class acts as an owning reference to an op, and will automatically
21 /// destroy the held op on destruction if the held op is valid.
22 ///
23 /// Note that OpBuilder and related functionality should be highly preferred
24 /// instead, and this should only be used in situations where existing solutions
25 /// are not viable.
26 template <typename OpTy>
27 class OwningOpRefBase {
28 public:
29  OwningOpRefBase(std::nullptr_t = nullptr) {}
30  OwningOpRefBase(OpTy op) : op(op) {}
31  OwningOpRefBase(OwningOpRefBase &&other) : op(other.release()) {}
32  ~OwningOpRefBase() {
33  if (op)
34  op.erase();
35  }
36 
37  // Assign from another op reference.
38  OwningOpRefBase &operator=(OwningOpRefBase &&other) {
39  if (op)
40  op.erase();
41  op = other.release();
42  return *this;
43  }
44 
45  /// Allow accessing the internal op.
46  OpTy get() const { return op; }
47  OpTy operator*() const { return op; }
48  OpTy *operator->() { return &op; }
49  explicit operator bool() const { return op; }
50 
51  /// Release the referenced op.
52  OpTy release() {
53  OpTy released;
54  std::swap(released, op);
55  return released;
56  }
57 
58 private:
59  OpTy op;
60 };
