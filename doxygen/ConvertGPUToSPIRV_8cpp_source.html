100 static Optional<int32_t> getLaunchConfigIndex(Operation *op) {
101  auto dimAttr = op->getAttrOfType<StringAttr>("dimension");
102  if (!dimAttr) {
103  return {};
104  }
105  if (dimAttr.getValue() == "x") {
106  return 0;
107  } else if (dimAttr.getValue() == "y") {
108  return 1;
109  } else if (dimAttr.getValue() == "z") {
110  return 2;
111  }
112  return {};
113 }

115 template <typename SourceOp, spirv::BuiltIn builtin>
116 LogicalResult LaunchConfigConversion<SourceOp, builtin>::matchAndRewrite(
117  SourceOp op, ArrayRef<Value> operands,
118  ConversionPatternRewriter &rewriter) const {
119  auto index = getLaunchConfigIndex(op);
120  if (!index)
121  return failure();
122 
123  // SPIR-V invocation builtin variables are a vector of type <3xi32>
124  auto spirvBuiltin = spirv::getBuiltinVariableValue(op, builtin, rewriter);
125  rewriter.replaceOpWithNewOp<spirv::CompositeExtractOp>(
126  op, rewriter.getIntegerType(32), spirvBuiltin,
127  rewriter.getI32ArrayAttr({index.getValue()}));
128  return success();
129 }

141 LogicalResult WorkGroupSizeConversion::matchAndRewrite(
142  gpu::BlockDimOp op, ArrayRef<Value> operands,
143  ConversionPatternRewriter &rewriter) const {
144  auto index = getLaunchConfigIndex(op);
145  if (!index)
146  return failure();
147 
148  auto workGroupSizeAttr = spirv::lookupLocalWorkGroupSize(op);
149  auto val = workGroupSizeAttr.getValue<int32_t>(index.getValue());
150  auto convertedType = typeConverter.convertType(op.getResult().getType());
151  if (!convertedType)
152  return failure();
153  rewriter.replaceOpWithNewOp<spirv::ConstantOp>(
154  op, convertedType, IntegerAttr::get(convertedType, val));
155  return success();
156 }

162 // Legalizes a GPU function as an entry SPIR-V function.
163 static spirv::FuncOp
164 lowerAsEntryFunction(gpu::GPUFuncOp funcOp, SPIRVTypeConverter &typeConverter,
165  ConversionPatternRewriter &rewriter,
166  spirv::EntryPointABIAttr entryPointInfo,
167  ArrayRef<spirv::InterfaceVarABIAttr> argABIInfo) {
168  auto fnType = funcOp.getType();
169  if (fnType.getNumResults()) {
170  funcOp.emitError("SPIR-V lowering only supports entry functions"
171  "with no return values right now");
172  return nullptr;
173  }
174  if (!argABIInfo.empty() && fnType.getNumInputs() != argABIInfo.size()) {
175  funcOp.emitError(
176  "lowering as entry functions requires ABI info for all arguments "
177  "or none of them");
178  return nullptr;
179  }
180  // Update the signature to valid SPIR-V types and add the ABI
181  // attributes. These will be "materialized" by using the
182  // LowerABIAttributesPass.
183  TypeConverter::SignatureConversion signatureConverter(fnType.getNumInputs());
184  {
185  for (auto argType : enumerate(funcOp.getType().getInputs())) {
186  auto convertedType = typeConverter.convertType(argType.value());
187  signatureConverter.addInputs(argType.index(), convertedType);
188  }
189  }
190  auto newFuncOp = rewriter.create<spirv::FuncOp>(
191  funcOp.getLoc(), funcOp.getName(),
192  rewriter.getFunctionType(signatureConverter.getConvertedTypes(),
193  llvm::None));
194  for (const auto &namedAttr : funcOp.getAttrs()) {
195  if (namedAttr.first == impl::getTypeAttrName() ||
196  namedAttr.first == SymbolTable::getSymbolAttrName())
197  continue;
198  newFuncOp.setAttr(namedAttr.first, namedAttr.second);
199  }
200 
201  rewriter.inlineRegionBefore(funcOp.getBody(), newFuncOp.getBody(),
202  newFuncOp.end());
203  if (failed(rewriter.convertRegionTypes(&newFuncOp.getBody(), typeConverter,
204  &signatureConverter)))
205  return nullptr;
206  rewriter.eraseOp(funcOp);
207 
208  spirv::setABIAttrs(newFuncOp, entryPointInfo, argABIInfo);
209  return newFuncOp;
210 }

212 /// Populates `argABI` with spv.interface_var_abi attributes for lowering
213 /// gpu.func to spv.func if no arguments have the attributes set
214 /// already. Returns failure if any argument has the ABI attribute set already.
215 static LogicalResult
216 getDefaultABIAttrs(MLIRContext *context, gpu::GPUFuncOp funcOp,
217  SmallVectorImpl<spirv::InterfaceVarABIAttr> &argABI) {
218  spirv::TargetEnvAttr targetEnv = spirv::lookupTargetEnvOrDefault(funcOp);
219  if (!spirv::needsInterfaceVarABIAttrs(targetEnv))
220  return success();
221 
222  for (auto argIndex : llvm::seq<unsigned>(0, funcOp.getNumArguments())) {
223  if (funcOp.getArgAttrOfType<spirv::InterfaceVarABIAttr>(
224  argIndex, spirv::getInterfaceVarABIAttrName()))
225  return failure();
226  // Vulkan's interface variable requirements needs scalars to be wrapped in a
227  // struct. The struct held in storage buffer.
228  Optional<spirv::StorageClass> sc;
229  if (funcOp.getArgument(argIndex).getType().isIntOrIndexOrFloat())
230  sc = spirv::StorageClass::StorageBuffer;
231  argABI.push_back(spirv::getInterfaceVarABIAttr(0, argIndex, sc, context));
232  }
233  return success();
234 }

236 LogicalResult GPUFuncOpConversion::matchAndRewrite(
237  gpu::GPUFuncOp funcOp, ArrayRef<Value> operands,
238  ConversionPatternRewriter &rewriter) const {
239  if (!gpu::GPUDialect::isKernel(funcOp))
240  return failure();
241 
242  SmallVector<spirv::InterfaceVarABIAttr, 4> argABI;
243  if (failed(getDefaultABIAttrs(rewriter.getContext(), funcOp, argABI))) {
244  argABI.clear();
245  for (auto argIndex : llvm::seq<unsigned>(0, funcOp.getNumArguments())) {
246  // If the ABI is already specified, use it.
247  auto abiAttr = funcOp.getArgAttrOfType<spirv::InterfaceVarABIAttr>(
248  argIndex, spirv::getInterfaceVarABIAttrName());
249  if (!abiAttr) {
250  funcOp.emitRemark(
251  "match failure: missing 'spv.interface_var_abi' attribute at "
252  "argument ")
253  << argIndex;
254  return failure();
255  }
256  argABI.push_back(abiAttr);
257  }
258  }
259 
260  auto entryPointAttr = spirv::lookupEntryPointABI(funcOp);
261  if (!entryPointAttr) {
262  funcOp.emitRemark("match failure: missing 'spv.entry_point_abi' attribute");
263  return failure();
264  }
265  spirv::FuncOp newFuncOp = lowerAsEntryFunction(
266  funcOp, typeConverter, rewriter, entryPointAttr, argABI);
267  if (!newFuncOp)
268  return failure();
269  newFuncOp.removeAttr(Identifier::get(gpu::GPUDialect::getKernelFuncAttrName(),
270  rewriter.getContext()));
271  return success();
272 }

278 LogicalResult GPUModuleConversion::matchAndRewrite(
279  gpu::GPUModuleOp moduleOp, ArrayRef<Value> operands,
280  ConversionPatternRewriter &rewriter) const {
281  spirv::TargetEnvAttr targetEnv = spirv::lookupTargetEnvOrDefault(moduleOp);
282  spirv::AddressingModel addressingModel = spirv::getAddressingModel(targetEnv);
283  FailureOr<spirv::MemoryModel> memoryModel = spirv::getMemoryModel(targetEnv);
284  if (failed(memoryModel))
285  return moduleOp.emitRemark("match failure: could not selected memory model "
286  "based on 'spv.target_env'");
287 
288  auto spvModule = rewriter.create<spirv::ModuleOp>(
289  moduleOp.getLoc(), addressingModel, memoryModel.getValue());
290 
291  // Move the region from the module op into the SPIR-V module.
292  Region &spvModuleRegion = spvModule.body();
293  rewriter.inlineRegionBefore(moduleOp.body(), spvModuleRegion,
294  spvModuleRegion.begin());
295  // The spv.module build method adds a block with a terminator. Remove that
296  // block. The terminator of the module op in the remaining block will be
297  // legalized later.
298  rewriter.eraseBlock(&spvModuleRegion.back());
299  rewriter.eraseOp(moduleOp);
300  return success();
301 }
