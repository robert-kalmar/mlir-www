25 /// All memrefs passed across functions with non-trivial layout maps are
26 /// converted to ones with trivial identity layout ones.
27 
28 // Input :-
29 // #tile = affine_map<(i) -> (i floordiv 4, i mod 4)>
30 // func @matmul(%A: memref<16xf64, #tile>, %B: index, %C: memref<16xf64>) ->
31 // (memref<16xf64, #tile>) {
32 // affine.for %arg3 = 0 to 16 {
33 // %a = affine.load %A[%arg3] : memref<16xf64, #tile>
34 // %p = mulf %a, %a : f64
35 // affine.store %p, %A[%arg3] : memref<16xf64, #tile>
36 // }
37 // %c = alloc() : memref<16xf64, #tile>
38 // %d = affine.load %c[0] : memref<16xf64, #tile>
39 // return %A: memref<16xf64, #tile>
40 // }
41 
42 // Output :-
43 // func @matmul(%arg0: memref<4x4xf64>, %arg1: index, %arg2: memref<16xf64>)
44 // -> memref<4x4xf64> {
45 // affine.for %arg3 = 0 to 16 {
46 // %2 = affine.load %arg0[%arg3 floordiv 4, %arg3 mod 4] : memref<4x4xf64>
47 // %3 = mulf %2, %2 : f64
48 // affine.store %3, %arg0[%arg3 floordiv 4, %arg3 mod 4] : memref<4x4xf64>
49 // }
50 // %0 = alloc() : memref<16xf64, #map0>
51 // %1 = affine.load %0[0] : memref<16xf64, #map0>
52 // return %arg0 : memref<4x4xf64>
53 // }

68 void NormalizeMemRefs::runOnOperation() {
69  ModuleOp moduleOp = getOperation();
70  // We traverse each function within the module in order to normalize the
71  // memref type arguments.
72  // TODO: Handle external functions.
73  moduleOp.walk([&](FuncOp funcOp) {
74  if (areMemRefsNormalizable(funcOp))
75  runOnFunction(funcOp);
76  });
77 }

79 // Return true if this operation dereferences one or more memref's.
80 // TODO: Temporary utility, will be replaced when this is modeled through
81 // side-effects/op traits.
82 static bool isMemRefDereferencingOp(Operation &op) {
83  return isa<AffineReadOpInterface, AffineWriteOpInterface, AffineDmaStartOp,
84  AffineDmaWaitOp>(op);
85 }
86 
87 // Check whether all the uses of oldMemRef are either dereferencing uses or the
88 // op is of type : DeallocOp, CallOp. Only if these constraints are satisfied
89 // will the value become a candidate for replacement.
90 static bool isMemRefNormalizable(Value::user_range opUsers) {
91  if (llvm::any_of(opUsers, [](Operation *op) {
92  if (isMemRefDereferencingOp(*op))
93  return false;
94  return !isa<DeallocOp, CallOp>(*op);
95  }))
96  return false;
97  return true;
98 }
99 
100 // Check whether all the uses of AllocOps, CallOps and function arguments of a
101 // function are either of dereferencing type or of type: DeallocOp, CallOp. Only
102 // if these constraints are satisfied will the function become a candidate for
103 // normalization.
104 bool NormalizeMemRefs::areMemRefsNormalizable(FuncOp funcOp) {
105  if (funcOp
106  .walk([&](AllocOp allocOp) -> WalkResult {
107  Value oldMemRef = allocOp.getResult();
108  if (!isMemRefNormalizable(oldMemRef.getUsers()))
109  return WalkResult::interrupt();
110  return WalkResult::advance();
111  })
112  .wasInterrupted())
113  return false;
114 
115  if (funcOp
116  .walk([&](CallOp callOp) -> WalkResult {
117  for (unsigned resIndex :
118  llvm::seq<unsigned>(0, callOp.getNumResults())) {
119  Value oldMemRef = callOp.getResult(resIndex);
120  if (oldMemRef.getType().isa<MemRefType>())
121  if (!isMemRefNormalizable(oldMemRef.getUsers()))
122  return WalkResult::interrupt();
123  }
124  return WalkResult::advance();
125  })
126  .wasInterrupted())
127  return false;
128 
129  for (unsigned argIndex : llvm::seq<unsigned>(0, funcOp.getNumArguments())) {
130  BlockArgument oldMemRef = funcOp.getArgument(argIndex);
131  if (oldMemRef.getType().isa<MemRefType>())
132  if (!isMemRefNormalizable(oldMemRef.getUsers()))
133  return false;
134  }
135 
136  return true;
137 }

139 // Fetch the updated argument list and result of the function and update the
140 // function signature.
141 void NormalizeMemRefs::updateFunctionSignature(FuncOp funcOp) {
142  FunctionType functionType = funcOp.getType();
143  SmallVector<Type, 8> argTypes;
144  SmallVector<Type, 4> resultTypes;
145 
146  for (const auto &arg : llvm::enumerate(funcOp.getArguments()))
147  argTypes.push_back(arg.value().getType());
148 
149  resultTypes = llvm::to_vector<4>(functionType.getResults());
150  // We create a new function type and modify the function signature with this
151  // new type.
152  FunctionType newFuncType = FunctionType::get(/*inputs=*/argTypes,
153  /*results=*/resultTypes,
154  /*context=*/&getContext());
155 
156  // TODO: Handle ReturnOps to update function results the caller site.
157  funcOp.setType(newFuncType);
158 }

160 void NormalizeMemRefs::runOnFunction(FuncOp funcOp) {
161  // Turn memrefs' non-identity layouts maps into ones with identity. Collect
162  // alloc ops first and then process since normalizeMemRef replaces/erases ops
163  // during memref rewriting.
164  SmallVector<AllocOp, 4> allocOps;
165  funcOp.walk([&](AllocOp op) { allocOps.push_back(op); });
166  for (AllocOp allocOp : allocOps)
167  normalizeMemRef(allocOp);
168 
169  // We use this OpBuilder to create new memref layout later.
170  OpBuilder b(funcOp);
171 
172  // Walk over each argument of a function to perform memref normalization (if
173  // any).
174  for (unsigned argIndex : llvm::seq<unsigned>(0, funcOp.getNumArguments())) {
175  Type argType = funcOp.getArgument(argIndex).getType();
176  MemRefType memrefType = argType.dyn_cast<MemRefType>();
177  // Check whether argument is of MemRef type. Any other argument type can
178  // simply be part of the final function signature.
179  if (!memrefType)
180  continue;
181  // Fetch a new memref type after normalizing the old memref to have an
182  // identity map layout.
183  MemRefType newMemRefType = normalizeMemRefType(memrefType, b,
184  /*numSymbolicOperands=*/0);
185  if (newMemRefType == memrefType) {
186  // Either memrefType already had an identity map or the map couldn't be
187  // transformed to an identity map.
188  continue;
189  }
190 
191  // Insert a new temporary argument with the new memref type.
192  BlockArgument newMemRef =
193  funcOp.front().insertArgument(argIndex, newMemRefType);
194  BlockArgument oldMemRef = funcOp.getArgument(argIndex + 1);
195  AffineMap layoutMap = memrefType.getAffineMaps().front();
196  // Replace all uses of the old memref.
197  if (failed(replaceAllMemRefUsesWith(oldMemRef, /*newMemRef=*/newMemRef,
198  /*extraIndices=*/{},
199  /*indexRemap=*/layoutMap,
200  /*extraOperands=*/{},
201  /*symbolOperands=*/{},
202  /*domInstFilter=*/nullptr,
203  /*postDomInstFilter=*/nullptr,
204  /*allowNonDereferencingOps=*/true,
205  /*handleDeallocOp=*/true))) {
206  // If it failed (due to escapes for example), bail out. Removing the
207  // temporary argument inserted previously.
208  funcOp.front().eraseArgument(argIndex);
209  continue;
210  }
211 
212  // All uses for the argument with old memref type were replaced
213  // successfully. So we remove the old argument now.
214  funcOp.front().eraseArgument(argIndex + 1);
215  }
216 
217  updateFunctionSignature(funcOp);
218 }
