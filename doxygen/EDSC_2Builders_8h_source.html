26 /// Helper class to transparently handle builder insertion points by RAII.
27 /// As its name indicates, a ScopedContext is means to be used locally in a
28 /// scoped fashion. This abstracts away all the boilerplate related to
29 /// checking proper usage of captures, NestedBuilders as well as handling the
30 /// setting and restoring of insertion points.
31 class ScopedContext {
32 public:
33  ScopedContext(OpBuilder &b, Location location);
34 
35  /// Sets the insertion point of the builder to 'newInsertPt' for the duration
36  /// of the scope. The existing insertion point of the builder is restored on
37  /// destruction.
38  ScopedContext(OpBuilder &b, OpBuilder::InsertPoint newInsertPt,
39  Location location);
40  ~ScopedContext();
41 
42  static MLIRContext *getContext();
43  static OpBuilder &getBuilderRef();
44  static Location getLocation();
45 
46 private:
47  /// Only NestedBuilder (which is used to create an operation with a body)
48  /// may access private members in order to implement scoping.
49  friend class NestedBuilder;
50 
51  ScopedContext() = delete;
52  ScopedContext(const ScopedContext &) = delete;
53  ScopedContext &operator=(const ScopedContext &) = delete;
54 
55  static ScopedContext *&getCurrentScopedContext();
56 
57  /// Top level OpBuilder.
58  OpBuilder &builder;
59  /// Guard to the previous insertion point.
60  OpBuilder::InsertionGuard guard;
61  /// Current location.
62  Location location;
63  /// Parent context we return into.
64  ScopedContext *enclosingScopedContext;
65 };

120 /// A StructuredIndexed represents an indexable quantity that is either:
121 /// 1. a captured value, which is suitable for buffer and tensor operands, or;
122 /// 2. a captured type, which is suitable for tensor return values.
123 ///
124 /// A StructuredIndexed itself is indexed and passed to `makeGenericLinalgOp`.
125 /// It enable an idiomatic syntax for index expressions such as:
126 ///
127 /// ```
128 /// StructuredIndexed A(buffer_or_tensor_value), B(buffer_or_tensor_value),
129 /// C(buffer_value_or_tensor_type);
130 /// makeGenericLinalgOp({A({m, n}), B({k, n})}, {C({m, n})}, ... );
131 /// ```
132 struct StructuredIndexed {
133  StructuredIndexed(Value v) : value(v) {}
134  StructuredIndexed(Type t) : type(t) {}
135  StructuredIndexed operator()(ArrayRef<AffineExpr> indexings) {
136  return value ? StructuredIndexed(value, indexings)
137  : StructuredIndexed(type, indexings);
138  }
139 
140  StructuredIndexed(Value v, ArrayRef<AffineExpr> indexings)
141  : value(v), exprs(indexings.begin(), indexings.end()) {
142  assert((v.getType().isa<MemRefType, RankedTensorType, VectorType>()) &&
143  "MemRef, RankedTensor or Vector expected");
144  }
145  StructuredIndexed(Type t, ArrayRef<AffineExpr> indexings)
146  : type(t), exprs(indexings.begin(), indexings.end()) {
147  assert((t.isa<MemRefType, RankedTensorType, VectorType>()) &&
148  "MemRef, RankedTensor or Vector expected");
149  }
150 
151  bool hasValue() const { return (bool)value; }
152  Value getValue() const {
153  assert(value && "StructuredIndexed Value not set.");
154  return value;
155  }
156  Type getType() const {
157  assert((value || type) && "StructuredIndexed Value and Type not set.");
158  return value ? value.getType() : type;
159  }
160  ArrayRef<AffineExpr> getExprs() const { return exprs; }
161  operator Value() const { return getValue(); }
162  operator Type() const { return getType(); }
163 
164 private:
165  // Only one of Value or type may be set.
166  Type type;
167  Value value;
168  SmallVector<AffineExpr, 4> exprs;
169 };

171 /// A TemplatedIndexedValue brings an index notation over the template Load and
172 /// Store parameters. Assigning to an IndexedValue emits an actual `Store`

176 class TemplatedIndexedValue {

182  TemplatedIndexedValue operator()() { return *this; }

185  TemplatedIndexedValue res(value);
186  res.indices.push_back(index);
187  return res;
188  }
189  template <typename... Args>
190  TemplatedIndexedValue operator()(Value index, Args... indices) {
191  return TemplatedIndexedValue(value, index).append(indices...);

194  return TemplatedIndexedValue(value, indices);
