24 OpTrait OpTrait::create(const llvm::Init *init) {
25  auto def = cast<llvm::DefInit>(init)->getDef();
26  if (def->isSubClassOf("PredOpTrait"))
27  return OpTrait(Kind::Pred, def);
28  if (def->isSubClassOf("GenInternalOpTrait"))
29  return OpTrait(Kind::Internal, def);
30  if (def->isSubClassOf("OpInterfaceTrait"))
31  return OpTrait(Kind::Interface, def);
32  assert(def->isSubClassOf("NativeOpTrait"));
33  return OpTrait(Kind::Native, def);
34 }
