26 /** Opaque type declarations.
27  *
28  * Types are exposed to C bindings as structs containing opaque pointers. They
29  * are not supposed to be inspected from C. This allows the underlying
30  * representation to change without affecting the API users. The use of structs
31  * instead of typedefs enables some type safety as structs are not implicitly
32  * convertible to each other.
33  *
34  * Instaces of these types may or may not own the underlying object (most often
35  * only point to an IR fragment without owning it). The ownership semantics is
36  * defined by how an instance of the type was obtained.
37  */
38 /*============================================================================*/
39 
40 #define DEFINE_C_API_STRUCT(name, storage) \
41  struct name { \
42  storage *ptr; \
43  }; \
44  typedef struct name name

112 /** An auxiliary class for constructing operations.
113  *
114  * This class contains all the information necessary to construct the operation.
115  * It owns the MlirRegions it has pointers to and does not own anything else.
116  * By default, the state can be constructed from a name and location, the latter
117  * being also used to access the context, and has no other components. These
118  * components can be added progressively until the operation is constructed.
119  * Users are not expected to rely on the internals of this class and should use
120  * mlirOperationState* functions instead.
121  */
122 struct MlirOperationState {
123  const char *name;
124  MlirLocation location;
125  unsigned nResults;
126  MlirType *results;
127  unsigned nOperands;
128  MlirValue *operands;
129  unsigned nRegions;
130  MlirRegion *regions;
131  unsigned nSuccessors;
132  MlirBlock *successors;
133  unsigned nAttributes;
134  MlirNamedAttribute *attributes;
135 };
136 typedef struct MlirOperationState MlirOperationState;
