26 /// A simple compositional matcher for AffineExpr
27 ///
28 /// Example usage:
29 ///
30 /// ```c++
31 /// AffineExprMatcher x, C, m;
32 /// AffineExprMatcher pattern1 = ((x % C) * m) + x;
33 /// AffineExprMatcher pattern2 = x + ((x % C) * m);
34 /// if (pattern1.match(expr) || pattern2.match(expr)) {
35 /// ...
36 /// }
37 /// ```
38 class AffineExprMatcherStorage;
39 class AffineExprMatcher {
40 public:
41  AffineExprMatcher();
42  AffineExprMatcher(const AffineExprMatcher &other);
43 
44  AffineExprMatcher operator+(AffineExprMatcher other) {
45  return AffineExprMatcher(AffineExprKind::Add, *this, other);
46  }
47  AffineExprMatcher operator*(AffineExprMatcher other) {
48  return AffineExprMatcher(AffineExprKind::Mul, *this, other);
49  }
50  AffineExprMatcher floorDiv(AffineExprMatcher other) {
51  return AffineExprMatcher(AffineExprKind::FloorDiv, *this, other);
52  }
53  AffineExprMatcher ceilDiv(AffineExprMatcher other) {
54  return AffineExprMatcher(AffineExprKind::CeilDiv, *this, other);
55  }
56  AffineExprMatcher operator%(AffineExprMatcher other) {
57  return AffineExprMatcher(AffineExprKind::Mod, *this, other);
58  }
59 
60  AffineExpr match(AffineExpr expr);
61  AffineExpr matched();
62  Optional<int> getMatchedConstantValue();
63 
64 private:
65  AffineExprMatcher(AffineExprKind k, AffineExprMatcher a, AffineExprMatcher b);

67  // A shared_ptr allows multiple references to same matcher storage without

100 AffineExpr AffineExprMatcher::match(AffineExpr expr) {
101  if (kind > AffineExprKind::LAST_AFFINE_BINARY_OP) {
102  if (storage->matched)
103  if (storage->matched != expr)
104  return AffineExpr();
105  storage->matched = expr;
106  return storage->matched;
107  }
108  if (kind != expr.getKind()) {
109  return AffineExpr();
110  }
111  if (auto bin = expr.dyn_cast<AffineBinaryOpExpr>()) {
112  if (!storage->subExprs.empty() &&
113  !storage->subExprs[0].match(bin.getLHS())) {
114  return AffineExpr();
115  }
116  if (!storage->subExprs.empty() &&
117  !storage->subExprs[1].match(bin.getRHS())) {
118  return AffineExpr();
119  }
120  if (storage->matched)
121  if (storage->matched != expr)
122  return AffineExpr();
123  storage->matched = expr;
124  return storage->matched;
125  }
126  llvm_unreachable("binary expected");
127 }

150 void SDBMExpr::print(raw_ostream &os) const {
151  struct Printer : public SDBMVisitor<Printer> {
152  Printer(raw_ostream &ostream) : prn(ostream) {}
153 
154  void visitSum(SDBMSumExpr expr) {
155  visit(expr.getLHS());
156  prn << " + ";
157  visit(expr.getRHS());
158  }
159  void visitDiff(SDBMDiffExpr expr) {
160  visit(expr.getLHS());
161  prn << " - ";
162  visit(expr.getRHS());
163  }
164  void visitDim(SDBMDimExpr expr) { prn << 'd' << expr.getPosition(); }
165  void visitSymbol(SDBMSymbolExpr expr) { prn << 's' << expr.getPosition(); }
166  void visitStripe(SDBMStripeExpr expr) {
167  SDBMDirectExpr lhs = expr.getLHS();
168  bool isTerm = lhs.isa<SDBMTermExpr>();
169  if (!isTerm)
170  prn << '(';
171  visit(lhs);
172  if (!isTerm)
173  prn << ')';
174  prn << " # ";
175  visitConstant(expr.getStripeFactor());
176  }
177  void visitNeg(SDBMNegExpr expr) {
178  bool isSum = expr.getVar().isa<SDBMSumExpr>();
179  prn << '-';
180  if (isSum)
181  prn << '(';
182  visit(expr.getVar());
183  if (isSum)
184  prn << ')';
185  }
186  void visitConstant(SDBMConstantExpr expr) { prn << expr.getValue(); }
187 
188  raw_ostream &prn;
189  };
190  Printer printer(os);
191  printer.visit(*this);
192 }

200 // Helper class to perform negation of an SDBM expression.
201 struct SDBMNegator : public SDBMVisitor<SDBMNegator, SDBMExpr> {
202  // Any term expression is wrapped into a negation expression.
203  // -(x) = -x
204  SDBMExpr visitDirect(SDBMDirectExpr expr) { return SDBMNegExpr::get(expr); }
205  // A negation expression is unwrapped.
206  // -(-x) = x
207  SDBMExpr visitNeg(SDBMNegExpr expr) { return expr.getVar(); }
208  // The value of the constant is negated.
209  SDBMExpr visitConstant(SDBMConstantExpr expr) {
210  return SDBMConstantExpr::get(expr.getDialect(), -expr.getValue());
211  }
212 
213  // Terms of a difference are interchanged. Since only the LHS of a diff
214  // expression is allowed to be a sum with a constant, we need to recreate the
215  // sum with the negated value:
216  // -((x + C) - y) = (y - C) - x.
217  SDBMExpr visitDiff(SDBMDiffExpr expr) {
218  // If the LHS is just a term, we can do straightforward interchange.
219  if (auto term = expr.getLHS().dyn_cast<SDBMTermExpr>())
220  return SDBMDiffExpr::get(expr.getRHS(), term);
221 
222  auto sum = expr.getLHS().cast<SDBMSumExpr>();
223  auto cst = visitConstant(sum.getRHS()).cast<SDBMConstantExpr>();
224  return SDBMDiffExpr::get(SDBMSumExpr::get(expr.getRHS(), cst),
225  sum.getLHS());
226  }
227 };
228 } // namespace

236 SDBMSumExpr SDBMSumExpr::get(SDBMTermExpr lhs, SDBMConstantExpr rhs) {
237  assert(lhs && "expected SDBM variable expression");
238  assert(rhs && "expected SDBM constant");
239 
240  // If LHS of a sum is another sum, fold the constant RHS parts.
241  if (auto lhsSum = lhs.dyn_cast<SDBMSumExpr>()) {
242  lhs = lhsSum.getLHS();
243  rhs = SDBMConstantExpr::get(rhs.getDialect(),
244  rhs.getValue() + lhsSum.getRHS().getValue());
245  }
246 
247  StorageUniquer &uniquer = lhs.getDialect()->getUniquer();
248  return uniquer.get<detail::SDBMBinaryExprStorage>(
249  TypeID::get<detail::SDBMBinaryExprStorage>(),
250  /*initFn=*/{}, static_cast<unsigned>(SDBMExprKind::Add), lhs, rhs);
251 }

300 // Given a direct expression `expr`, add the given constant to it and pass the
301 // resulting expression to `builder` before returning its result. If the
302 // expression is already a sum expression, update its constant and extract the
303 // LHS if the constant becomes zero. Otherwise, construct a sum expression.
304 template <typename Result>
305 static Result addConstantAndSink(SDBMDirectExpr expr, int64_t constant,
306  bool negated,
307  function_ref<Result(SDBMDirectExpr)> builder) {
308  SDBMDialect *dialect = expr.getDialect();
309  if (auto sumExpr = expr.dyn_cast<SDBMSumExpr>()) {
310  if (negated)
311  constant = sumExpr.getRHS().getValue() - constant;
312  else
313  constant += sumExpr.getRHS().getValue();
314 
315  if (constant != 0) {
316  auto sum = SDBMSumExpr::get(sumExpr.getLHS(),
317  SDBMConstantExpr::get(dialect, constant));
318  return builder(sum);
319  } else {
320  return builder(sumExpr.getLHS());
321  }
322  }
323  if (constant != 0)
324  return builder(SDBMSumExpr::get(
325  expr.cast<SDBMTermExpr>(),
326  SDBMConstantExpr::get(dialect, negated ? -constant : constant)));
327  return expr;
328 }

330 // Construct an expression lhs + constant while maintaining the canonical form
331 // of the SDBM expressions, in particular sink the constant expression to the
332 // nearest sum expression in the left subtree of the expression tree.
333 static SDBMExpr addConstant(SDBMVaryingExpr lhs, int64_t constant) {
334  if (auto lhsDiff = lhs.dyn_cast<SDBMDiffExpr>())
335  return addConstantAndSink<SDBMExpr>(
336  lhsDiff.getLHS(), constant, /*negated=*/false,
337  [lhsDiff](SDBMDirectExpr e) {
338  return SDBMDiffExpr::get(e, lhsDiff.getRHS());
339  });
340  if (auto lhsNeg = lhs.dyn_cast<SDBMNegExpr>())
341  return addConstantAndSink<SDBMExpr>(
342  lhsNeg.getVar(), constant, /*negated=*/true,
343  [](SDBMDirectExpr e) { return SDBMNegExpr::get(e); });
344  if (auto lhsSum = lhs.dyn_cast<SDBMSumExpr>())
345  return addConstantAndSink<SDBMExpr>(lhsSum, constant, /*negated=*/false,
346  [](SDBMDirectExpr e) { return e; });
347  if (constant != 0)
348  return SDBMSumExpr::get(lhs.cast<SDBMTermExpr>(),
349  SDBMConstantExpr::get(lhs.getDialect(), constant));
350  return lhs;
351 }

353 // Build a difference expression given a direct expression and a negation
354 // expression.
355 static SDBMExpr buildDiffExpr(SDBMDirectExpr lhs, SDBMNegExpr rhs) {
356  // Fold (x + C) - (x + D) = C - D.
357  if (lhs.getTerm() == rhs.getVar().getTerm())
358  return SDBMConstantExpr::get(
359  lhs.getDialect(), lhs.getConstant() - rhs.getVar().getConstant());
360 
361  return SDBMDiffExpr::get(
362  addConstantAndSink<SDBMDirectExpr>(lhs, -rhs.getVar().getConstant(),
363  /*negated=*/false,
364  [](SDBMDirectExpr e) { return e; }),
365  rhs.getVar().getTerm());
366 }

368 // Try folding an expression (lhs + rhs) where at least one of the operands
369 // contains a negated variable, i.e. is a negation or a difference expression.
370 static SDBMExpr foldSumDiff(SDBMExpr lhs, SDBMExpr rhs) {
371  // If exactly one of LHS, RHS is a negation expression, we can construct
372  // a difference expression, which is a special kind in SDBM.
373  auto lhsDirect = lhs.dyn_cast<SDBMDirectExpr>();
374  auto rhsDirect = rhs.dyn_cast<SDBMDirectExpr>();
375  auto lhsNeg = lhs.dyn_cast<SDBMNegExpr>();
376  auto rhsNeg = rhs.dyn_cast<SDBMNegExpr>();
377 
378  if (lhsDirect && rhsNeg)
379  return buildDiffExpr(lhsDirect, rhsNeg);
380  if (lhsNeg && rhsDirect)
381  return buildDiffExpr(rhsDirect, lhsNeg);
382 
383  // If a subexpression appears in a diff expression on the LHS(RHS) of a
384  // sum expression where it also appears on the RHS(LHS) with the opposite
385  // sign, we can simplify it away and obtain the SDBM form.
386  auto lhsDiff = lhs.dyn_cast<SDBMDiffExpr>();
387  auto rhsDiff = rhs.dyn_cast<SDBMDiffExpr>();
388 
389  // -(x + A) + ((x + B) - y) = -(y + (A - B))
390  if (lhsNeg && rhsDiff &&
391  lhsNeg.getVar().getTerm() == rhsDiff.getLHS().getTerm()) {
392  int64_t constant =
393  lhsNeg.getVar().getConstant() - rhsDiff.getLHS().getConstant();
394  // RHS of the diff is a term expression, its sum with a constant is a direct
395  // expression.
396  return SDBMNegExpr::get(
397  addConstant(rhsDiff.getRHS(), constant).cast<SDBMDirectExpr>());
398  }
399 
400  // (x + A) + ((y + B) - x) = (y + B) + A.
401  if (lhsDirect && rhsDiff && lhsDirect.getTerm() == rhsDiff.getRHS())
402  return addConstant(rhsDiff.getLHS(), lhsDirect.getConstant());
403 
404  // ((x + A) - y) + (-(x + B)) = -(y + (B - A)).
405  if (lhsDiff && rhsNeg &&
406  lhsDiff.getLHS().getTerm() == rhsNeg.getVar().getTerm()) {
407  int64_t constant =
408  rhsNeg.getVar().getConstant() - lhsDiff.getLHS().getConstant();
409  // RHS of the diff is a term expression, its sum with a constant is a direct
410  // expression.
411  return SDBMNegExpr::get(
412  addConstant(lhsDiff.getRHS(), constant).cast<SDBMDirectExpr>());
413  }
414 
415  // ((x + A) - y) + (y + B) = (x + A) + B.
416  if (rhsDirect && lhsDiff && rhsDirect.getTerm() == lhsDiff.getRHS())
417  return addConstant(lhsDiff.getLHS(), rhsDirect.getConstant());
418 
419  return {};
420 }

422 Optional<SDBMExpr> SDBMExpr::tryConvertAffineExpr(AffineExpr affine) {
423  struct Converter : public AffineExprVisitor<Converter, SDBMExpr> {
424  SDBMExpr visitAddExpr(AffineBinaryOpExpr expr) {
425  auto lhs = visit(expr.getLHS()), rhs = visit(expr.getRHS());
426  if (!lhs || !rhs)
427  return {};
428 
429  // In a "add" AffineExpr, the constant always appears on the right. If
430  // there were two constants, they would have been folded away.
431  assert(!lhs.isa<SDBMConstantExpr>() && "non-canonical affine expression");
432 
433  // If RHS is a constant, we can always extend the SDBM expression to
434  // include it by sinking the constant into the nearest sum expression.
435  if (auto rhsConstant = rhs.dyn_cast<SDBMConstantExpr>()) {
436  int64_t constant = rhsConstant.getValue();
437  auto varying = lhs.dyn_cast<SDBMVaryingExpr>();
438  assert(varying && "unexpected uncanonicalized sum of constants");
439  return addConstant(varying, constant);
440  }
441 
442  // Try building a difference expression if one of the values is negated,
443  // or check if a difference on either hand side cancels out the outer term
444  // so as to remain correct within SDBM. Return null otherwise.
445  return foldSumDiff(lhs, rhs);
446  }

448  SDBMExpr visitMulExpr(AffineBinaryOpExpr expr) {
449  // Attempt to recover a stripe expression "x # C = (x floordiv C) * C".
450  AffineExprMatcher x, C;
451  AffineExprMatcher pattern = (x.floorDiv(C)) * C;
452  if (pattern.match(expr)) {
453  if (SDBMExpr converted = visit(x.matched())) {
454  if (auto varConverted = converted.dyn_cast<SDBMTermExpr>())
455  // TODO: return varConverted.stripe(C.getConstantValue());
456  return SDBMStripeExpr::get(
457  varConverted,
458  SDBMConstantExpr::get(dialect,
459  C.getMatchedConstantValue().getValue()));
460  }
461  }
462 
463  auto lhs = visit(expr.getLHS()), rhs = visit(expr.getRHS());
464  if (!lhs || !rhs)
465  return {};
466 
467  // In a "mul" AffineExpr, the constant always appears on the right. If
468  // there were two constants, they would have been folded away.
469  assert(!lhs.isa<SDBMConstantExpr>() && "non-canonical affine expression");
470  auto rhsConstant = rhs.dyn_cast<SDBMConstantExpr>();
471  if (!rhsConstant)
472  return {};
473 
474  // The only supported "multiplication" expression is an SDBM is dimension
475  // negation, that is a product of dimension and constant -1.
476  if (rhsConstant.getValue() != -1)
477  return {};
478 
479  if (auto lhsVar = lhs.dyn_cast<SDBMTermExpr>())
480  return SDBMNegExpr::get(lhsVar);
481  if (auto lhsDiff = lhs.dyn_cast<SDBMDiffExpr>())
482  return SDBMNegator().visitDiff(lhsDiff);
483 
484  // Other multiplications are not allowed in SDBM.
485  return {};
486  }
487 
488  SDBMExpr visitModExpr(AffineBinaryOpExpr expr) {
489  auto lhs = visit(expr.getLHS()), rhs = visit(expr.getRHS());
490  if (!lhs || !rhs)
491  return {};
492 
493  // 'mod' can only be converted to SDBM if its LHS is a direct expression
494  // and its RHS is a constant. Then it `x mod c = x - x stripe c`.
495  auto rhsConstant = rhs.dyn_cast<SDBMConstantExpr>();
496  auto lhsVar = lhs.dyn_cast<SDBMDirectExpr>();
497  if (!lhsVar || !rhsConstant)
498  return {};
499  return SDBMDiffExpr::get(lhsVar,
500  SDBMStripeExpr::get(lhsVar, rhsConstant));
501  }
502 
503  // `a floordiv b = (a stripe b) / b`, but we have no division in SDBM
504  SDBMExpr visitFloorDivExpr(AffineBinaryOpExpr expr) { return {}; }
505  SDBMExpr visitCeilDivExpr(AffineBinaryOpExpr expr) { return {}; }
506 
507  // Dimensions, symbols and constants are converted trivially.
508  SDBMExpr visitConstantExpr(AffineConstantExpr expr) {
509  return SDBMConstantExpr::get(dialect, expr.getValue());
510  }
511  SDBMExpr visitDimExpr(AffineDimExpr expr) {
512  return SDBMDimExpr::get(dialect, expr.getPosition());
513  }
514  SDBMExpr visitSymbolExpr(AffineSymbolExpr expr) {
515  return SDBMSymbolExpr::get(dialect, expr.getPosition());
516  }
517 
518  SDBMDialect *dialect;
519  } converter;
520  converter.dialect = affine.getContext()->getRegisteredDialect<SDBMDialect>();
521 
522  if (auto result = converter.visit(affine))
523  return result;
524  return None;
525 }

673 SDBMExpr mlir::ops_assertions::operator+(SDBMExpr lhs, SDBMExpr rhs) {
674  if (auto folded = foldSumDiff(lhs, rhs))
675  return folded;
676  assert(!(lhs.isa<SDBMNegExpr>() && rhs.isa<SDBMNegExpr>()) &&
677  "a sum of negated expressions is a negation of a sum of variables and "
678  "not a correct SDBM");
679 
680  // Fold (x - y) + (y - x) = 0.
681  auto lhsDiff = lhs.dyn_cast<SDBMDiffExpr>();
682  auto rhsDiff = rhs.dyn_cast<SDBMDiffExpr>();
683  if (lhsDiff && rhsDiff) {
684  if (lhsDiff.getLHS() == rhsDiff.getRHS() &&
685  lhsDiff.getRHS() == rhsDiff.getLHS())
686  return SDBMConstantExpr::get(lhs.getDialect(), 0);
687  }
688 
689  // If LHS is a constant and RHS is not, swap the order to get into a supported
690  // sum case. From now on, RHS must be a constant.
691  auto lhsConstant = lhs.dyn_cast<SDBMConstantExpr>();
692  auto rhsConstant = rhs.dyn_cast<SDBMConstantExpr>();
693  if (!rhsConstant && lhsConstant) {
694  std::swap(lhs, rhs);
695  std::swap(lhsConstant, rhsConstant);
696  }
697  assert(rhsConstant && "at least one operand must be a constant");
698 
699  // Constant-fold if LHS is also a constant.
700  if (lhsConstant)
701  return SDBMConstantExpr::get(lhs.getDialect(), lhsConstant.getValue() +
702  rhsConstant.getValue());
703  return addConstant(lhs.cast<SDBMVaryingExpr>(), rhsConstant.getValue());
704 }

706 SDBMExpr mlir::ops_assertions::operator-(SDBMExpr lhs, SDBMExpr rhs) {
707  // Fold x - x == 0.
708  if (lhs == rhs)
709  return SDBMConstantExpr::get(lhs.getDialect(), 0);
710 
711  // LHS and RHS may be constants.
712  auto lhsConstant = lhs.dyn_cast<SDBMConstantExpr>();
713  auto rhsConstant = rhs.dyn_cast<SDBMConstantExpr>();
714 
715  // Constant fold if both LHS and RHS are constants.
716  if (lhsConstant && rhsConstant)
717  return SDBMConstantExpr::get(lhs.getDialect(), lhsConstant.getValue() -
718  rhsConstant.getValue());
719 
720  // Replace a difference with a sum with a negated value if one of LHS and RHS
721  // is a constant:
722  // x - C == x + (-C);
723  // C - x == -x + C.
724  // This calls into operator+ for further simplification.
725  if (rhsConstant)
726  return lhs + (-rhsConstant);
727  if (lhsConstant)
728  return -rhs + lhsConstant;
729 
730  return buildDiffExpr(lhs.cast<SDBMDirectExpr>(), (-rhs).cast<SDBMNegExpr>());
731 }

736

739  return expr;
740 
741  return SDBMStripeExpr::get(expr.cast<SDBMDirectExpr>(), constantFactor);
