31 void ConvertSPIRVToLLVMPass::runOnOperation() {
32  MLIRContext *context = &getContext();
33  ModuleOp module = getOperation();
34  LLVMTypeConverter converter(&getContext());
35 
36  OwningRewritePatternList patterns;
37 
38  populateSPIRVToLLVMTypeConversion(converter);
39 
40  populateSPIRVToLLVMModuleConversionPatterns(context, converter, patterns);
41  populateSPIRVToLLVMConversionPatterns(context, converter, patterns);
42  populateSPIRVToLLVMFunctionConversionPatterns(context, converter, patterns);
43 
44  // Currently pulls in Std to LLVM conversion patterns
45  // that help with testing. This allows to convert
46  // function arguments to LLVM.
47  populateStdToLLVMConversionPatterns(converter, patterns);
48 
49  ConversionTarget target(getContext());
50  target.addIllegalDialect<spirv::SPIRVDialect>();
51  target.addLegalDialect<LLVM::LLVMDialect>();
52 
53  // set `ModuleOp` and `ModuleTerminatorOp` as legal for `spv.module`
54  // conversion.
55  target.addLegalOp<ModuleOp>();
56  target.addLegalOp<ModuleTerminatorOp>();
57  if (failed(applyPartialConversion(module, target, patterns)))
58  signalPassFailure();
59 }
